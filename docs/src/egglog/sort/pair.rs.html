3#[derive(Clone, Debug, PartialEq, Eq, Hash)]
4pub struct PairContainer {
5    do_rebuild_first: bool,
6    do_rebuild_second: bool,
7    pub first: Value,
8    pub second: Value,
9}
10
11impl PairContainer {
12    /// Build a [`PairContainer`] with explicit per-element rebuild flags.
13    /// `do_rebuild_*` must be `true` exactly when the corresponding element is an
14    /// eq-sort or eq-container-sort (so congruence rebuilds it), matching what
15    /// the `pair` primitive computes. Used by callers (e.g. the proof-mode
16    /// find-or-refl primitive) that build pairs outside the `pair` primitive.
17    pub fn new(
18        do_rebuild_first: bool,
19        do_rebuild_second: bool,
20        first: Value,
21        second: Value,
22    ) -> Self {
23        PairContainer {
24            do_rebuild_first,
25            do_rebuild_second,
26            first,
27            second,
28        }
29    }
30}
31
32impl ContainerValue for PairContainer {
33    fn rebuild_contents(&mut self, rebuilder: &dyn Rebuilder) -> bool {
34        let mut changed = false;
35        if self.do_rebuild_first {
36            let new = rebuilder.rebuild_val(self.first);
37            changed |= self.first != new;
38            self.first = new;
39        }
40        if self.do_rebuild_second {
41            let new = rebuilder.rebuild_val(self.second);
42            changed |= self.second != new;
43            self.second = new;
44        }
45        changed
46    }
47    fn iter(&self) -> impl Iterator<Item = Value> + '_ {
48        [self.first, self.second].into_iter()
49    }
50}

52/// A pair of two values supporting these primitives:
53/// - `pair`
54/// - `pair-first`
55/// - `pair-second`
56#[derive(Clone, Debug)]
57pub struct PairSort {
58    name: String,
59    first: ArcSort,
60    second: ArcSort,
61}
62
63impl PairSort {
64    pub fn first(&self) -> ArcSort {
65        self.first.clone()
66    }
67
68    pub fn second(&self) -> ArcSort {
69        self.second.clone()
70    }
71
72    /// Build a [`PairContainer`] for this sort, deriving the per-element
73    /// rebuild flags exactly as the `pair` primitive does. Used by callers that
74    /// must construct a pair value outside the `pair` primitive (e.g. the
75    /// proof-mode find-or-refl primitive) while keeping rebuild behavior
76    /// identical.
77    pub fn make_container(&self, first: Value, second: Value) -> PairContainer {
78        PairContainer::new(
79            self.first.is_eq_sort() || self.first.is_eq_container_sort(),
80            self.second.is_eq_sort() || self.second.is_eq_container_sort(),
81            first,
82            second,
83        )
84    }
85}
86
87impl Presort for PairSort {
88    fn presort_name() -> &'static str {
89        "Pair"
90    }
91
92    fn reserved_primitives() -> Vec<&'static str> {
93        vec!["pair", "pair-first", "pair-second"]
94    }
95
96    fn make_sort(
97        typeinfo: &mut TypeInfo,
98        name: String,
99        args: &[Expr],
100    ) -> Result<ArcSort, TypeError> {
101        if let [Expr::Var(a_span, a), Expr::Var(b_span, b)] = args {
102            let a = typeinfo
103                .get_sort_by_name(a)
104                .ok_or(TypeError::UndefinedSort(a.clone(), a_span.clone()))?;
105            let b = typeinfo
106                .get_sort_by_name(b)
107                .ok_or(TypeError::UndefinedSort(b.clone(), b_span.clone()))?;
108
109            let out = Self {
110                name,
111                first: a.clone(),
112                second: b.clone(),
113            };
114            Ok(out.to_arcsort())
115        } else {
116            panic!("Pair sort requires exactly two arguments")
117        }
118    }
119}

132    fn is_eq_container_sort(&self) -> bool {
133        self.first.is_eq_sort()
134            || self.second.is_eq_sort()
135            || self.first.is_eq_container_sort()
136            || self.second.is_eq_container_sort()

139    fn inner_values(
140        &self,
141        container_values: &ContainerValues,
142        value: Value,
143    ) -> Vec<(ArcSort, Value)> {
144        let val = container_values
145            .get_val::<PairContainer>(value)
146            .unwrap()
147            .clone();
148        vec![
149            (self.first.clone(), val.first),
150            (self.second.clone(), val.second),
151        ]
152    }

154    fn register_primitives(&self, eg: &mut EGraph) {
155        let arc = self.clone().to_arcsort();
156
157        add_primitive!(eg, "pair" = {self.clone(): PairSort} |x: # (self.first()), y: # (self.second())| -> @PairContainer (arc) {
158            PairContainer {
159                do_rebuild_first: self.ctx.first.is_eq_sort() || self.ctx.first.is_eq_container_sort(),
160                do_rebuild_second: self.ctx.second.is_eq_sort() || self.ctx.second.is_eq_container_sort(),
161                first: x,
162                second: y,
163            }
164        });
165
166        add_primitive!(eg, "pair-first"  = |xs: @PairContainer (arc)| -> # (self.first())  { xs.first  });
167        add_primitive!(eg, "pair-second" = |xs: @PairContainer (arc)| -> # (self.second()) { xs.second });
168    }

179    }
