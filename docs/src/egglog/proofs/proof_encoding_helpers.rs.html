245    pub(crate) fn format_prooflist(&self, proofs: &[String]) -> String {
246        let pcons = &self.proof_names().pcons;
247        let pnil = &self.proof_names().pnil;
248
249        let mut prooflist = format!("({pnil})");
250        for proof in proofs.iter().rev() {
251            prooflist = format!("({pcons} {proof} {prooflist})");

363    /// True for the *declaration* of a non-eq-sort global, lowered by
364    /// `remove_globals` to a 0-arg `Custom` function (e.g.
365    /// `(let $N (bigrat 3 1))`). Such a function has no eclass and never
366    /// participates in UF/congruence, so the term encoder leaves it as a plain
367    /// 0-arg key-value store rather than building a term/view: the value is
368    /// `set` directly and read back via the 0-arg lookup `($N)`. The rule
369    /// referencing it was marked `:unsafe-seminaive` by `remove_globals`
370    /// (a RHS function lookup), which permits the read.
371    ///
372    /// Recognized by the `internal_let` flag (set only by `remove_globals` on
373    /// lifted globals — never on user functions, so a 0-arg user function like
374    /// `(function MyMap () i64 :merge …)` is correctly NOT matched), restricted
375    /// to the 0-arg non-eq-sort `Custom` shape. `is_global` is NOT usable here:
376    /// at term-encode time the global's `global_sorts` entry is not yet
377    /// registered (that happens later, during the post-encoding typecheck). The
378    /// name is recorded in `pass_through_globals` so the read sites — which only
379    /// have a `FuncType` (no `internal_let`) — recognize it via
380    /// `is_pass_through_global`.
381    pub(crate) fn is_non_eq_sort_global_decl(&self, fdecl: &ResolvedFunctionDecl) -> bool {
382        fdecl.internal_let
383            && fdecl.subtype == FunctionSubtype::Custom
384            && fdecl.schema.input.is_empty()
385            && !fdecl.resolved_schema.output().is_eq_sort()
386    }

404    /// True when PROOF-mode canonicalize-at-creation should be emitted.
405    ///
406    /// Enabled only when (a) canon-at-creation is on, (b) proofs are enabled,
407    /// (c) the backend supports inline table lookups — i.e. the native bridge —
408    /// and (d) the encoding is NOT native-UF.
409    ///
410    /// The proof-mode canon read uses `@uf_find_or_refl_S` / `@uf_find_leader_S`
411    /// ReadPrims that look up the RELATIONAL union-find `@UF_Sf` (schema
412    /// `(S) Pair(S, Proof)`) mid-rule. Those ReadPrims are registered only by the
413    /// relational `declare_sort` (via the pair sort's `:internal-uf-pair-prims`),
414    /// so under `--native-uf` they are unbound; native-UF's `@UF_Sf` is a
415    /// different `(S S) Proof` displaced-union-find schema anyway. So this
416    /// mechanism is RELATIONAL-UF only: under `--native-uf` proof mode we fall
417    /// back to the pre-STEP-0 path where the rebuild rules
418    /// (`rebuilding_rules_native_uf_proof`) canonicalize the views.
419    ///
420    /// The dataflow/SQL backends (duckdb/feldera/flowlog) likewise cannot reenter
421    /// a table lookup mid-rule (`supports_inline_table_lookups() == false`) and
422    /// run proofs without canon-at-creation, so this is also gated off there —
423    /// keeping their proof-mode encoding byte-identical to before this feature.
424    pub(crate) fn proof_canon_at_creation(&self) -> bool {
425        self.egraph.proof_state.proofs_enabled
426            && self.egraph.backend.supports_inline_table_lookups()
427            && !self.native_uf()
428    }

471    /// True for the RELATIONAL native-merge path: `--native-merge` WITHOUT
472    /// `--native-uf`, in term (non-proof) mode. In this mode the union-find is a
473    /// SINGLE self-referential `:merge` function `@UF_Sf : (S) -> S` (no `@UF_S`
474    /// edge table, no `singleparent` / `uf_function_index` rules), and the
475    /// constructor view's congruence union goes into that `@UF_Sf` via the same
476    /// native self-referential merge. Used to gate the `declare_sort` / `union` /
477    /// rebuild-schedule / congruence-target changes so the plain `--term-encoding`,
478    /// proof, and `--native-uf` paths stay byte-identical.

483    /// True when a CUSTOM `:merge` function's merge body is a pure VALUE-FOLD: a
484    /// fold of PRIMITIVES / literals over `old`/`new` that yields a value, with
485    /// NO constructor / function calls (no e-node creation). Such a merge can run
486    /// natively in the backend's FD-conflict merge callback (lowered to a
487    /// `MergeFn` tree of `Primitive`/`Const`/`Old`/`New`), so we can drop the
488    /// rule-encoded `@merge_rule` + `@merge_cleanup` and FD-key the view on its
489    /// children with the fold as the merge.
490    ///
491    /// A `ResolvedCall::Func` anywhere in the body disqualifies it: a constructor
492    /// call CREATES an e-node (needs `add_term_and_view`'s fresh-id + term/view
493    /// rows + congruence threading, which cannot run inside a merge callback —
494    /// TERM-BUILDING), and a non-constructor function call would need a mid-merge
495    /// table lookup. Both stay rule-encoded. `Values` (tuple merge) is allowed
496    /// only if every column is itself a value-fold.
497    pub(crate) fn merge_is_value_fold(expr: &ResolvedExpr) -> bool {
498        match expr {
499            GenericExpr::Lit(..) => true,
500            GenericExpr::Var(..) => true,
501            GenericExpr::Call(_, ResolvedCall::Func(_), _) => false,
502            GenericExpr::Call(_, ResolvedCall::Primitive(_), args)
503            | GenericExpr::Call(_, ResolvedCall::Values(_), args) => {
504                args.iter().all(Self::merge_is_value_fold)
505            }
506        }
507    }

509    /// True when this custom-`:merge` function should run its merge NATIVELY as a
510    /// value-fold instead of via the rule-encoded `@merge_rule`/`@merge_cleanup`.
511    ///
512    /// Gated narrowly: native-merge on, term (NOT proof) mode, the backend can run
513    /// a complex merge callback (`supports_complex_merge()` — bridge only), the
514    /// function is a custom `:merge` whose body is a value-fold, AND the output
515    /// sort is NOT an eq-sort (a value-fold yields a plain value; an eq-sort
516    /// output would imply e-node creation and is handled by congruence/term
517    /// rules). Everything outside this predicate keeps the existing rule encoding
518    /// byte-identical.
519    pub(crate) fn native_value_fold_merge(&self, fdecl: &ResolvedFunctionDecl) -> bool {
520        self.native_merge()
521            && !self.egraph.proof_state.proofs_enabled
522            && self.egraph.backend.supports_value_fold_merge()
523            && fdecl.subtype == FunctionSubtype::Custom
524            && !fdecl.resolved_schema.output().is_eq_sort()
525            && fdecl.merge.as_ref().is_some_and(Self::merge_is_value_fold)
526    }

528    /// True when a CUSTOM `:merge` function's merge body BUILDS TERMS: it contains
529    /// at least one constructor `Func` call (e.g. `(C2 (C1 old new) (C2 old new))`)
530    /// and every leaf is a `Lit` / `Var` / nested constructor / primitive. Unlike
531    /// [`Self::merge_is_value_fold`], a `ResolvedCall::Func` is ALLOWED here (it is
532    /// what distinguishes a term-building merge from a value fold). Used by the
533    /// PR #933 `:merge`-multiple-actions lowering — such a body is re-expressed as a
534    /// `MergeFn::Seq` of `Construct` / `TableInsert` / union variants and runs
535    /// inline in the FD-view merge callback.
536    pub(crate) fn merge_is_term_build(expr: &ResolvedExpr) -> bool {
537        match expr {
538            // A bare `old`/`new`/literal body builds no term — that is a value
539            // fold (or a degenerate `old`/`new` merge), not a term build.
540            GenericExpr::Lit(..) | GenericExpr::Var(..) => false,
541            // A top-level constructor call is a term build. Its arguments may be
542            // vars, literals, or further nested constructor / primitive calls.
543            GenericExpr::Call(_, ResolvedCall::Func(_), args) => {
544                args.iter().all(Self::term_build_arg_ok)
545            }
546            // A `(values ...)` tuple merge is not (yet) a term build; a top-level
547            // primitive whose args build terms is also out of scope for now.
548            GenericExpr::Call(_, ResolvedCall::Primitive(_), _)
549            | GenericExpr::Call(_, ResolvedCall::Values(_), _) => false,
550        }
551    }
552
553    /// Whether an argument of a term-building constructor call is expressible in
554    /// the [`MergeFn::Seq`] lowering: a var (`old`/`new`/`oldN`/`newN`), a literal,
555    /// a nested constructor call, or a primitive over such args.
556    fn term_build_arg_ok(expr: &ResolvedExpr) -> bool {
557        match expr {
558            GenericExpr::Lit(..) | GenericExpr::Var(..) => true,
559            GenericExpr::Call(_, ResolvedCall::Func(_), args) => {
560                args.iter().all(Self::term_build_arg_ok)
561            }
562            GenericExpr::Call(_, ResolvedCall::Primitive(_), args) => {
563                args.iter().all(Self::term_build_arg_ok)
564            }
565            GenericExpr::Call(_, ResolvedCall::Values(_), _) => false,
566        }
567    }

569    /// True when this custom-`:merge` function should run its term-building merge
570    /// NATIVELY as a `MergeFn::Seq` instead of via the rule-encoded
571    /// `@merge_rule`/`@merge_cleanup`. Gated like [`Self::native_value_fold_merge`]
572    /// but for an EQ-SORT output (a term-building merge mints e-nodes) whose body
573    /// is a term build. The PR #933 `:merge`-multiple-actions port.
574    ///
575    /// Works in BOTH term and proof mode on the bridge. In PROOF mode the
576    /// `MergeFn::Seq` also threads the `Merge` view-proof terms through the A2
577    /// tuple proof views / `___TreeProof` / `@UF_S` self-edges (see
578    /// `lower_term_build_expr`), so it requires the bridge-only A2
579    /// `UnionIntoUfWithProof` toolkit — hence the proof-mode arm gates on
580    /// `supports_complex_merge()` (bridge only) rather than
581    /// `supports_term_build_merge()` (which the flowlog/feldera dataflow backends
582    /// force `true` for the NON-proof term path). Proof + term-build on
583    /// duckdb/feldera/flowlog therefore stays on the rule (`@merge_rule`) path.
584    pub(crate) fn native_term_build_merge(&self, fdecl: &ResolvedFunctionDecl) -> bool {
585        let backend_supports = if self.egraph.proof_state.proofs_enabled {
586            self.egraph.backend.supports_complex_merge()
587        } else {
588            self.egraph.backend.supports_term_build_merge()
589        };
590        self.native_merge()
591            && backend_supports
592            && fdecl.subtype == FunctionSubtype::Custom
593            && fdecl.resolved_schema.output().is_eq_sort()
594            && fdecl.merge.as_ref().is_some_and(Self::merge_is_term_build)
595    }

597    /// True when `view_name` is a native-merge FD view (keyed on children, eclass
598    /// as the function output). Consulted by `update_view` /
599    /// `query_view_and_get_proof` to choose the FD `(children) -> ...` form.
600    ///
601    /// TERM mode: a single-output `(children) -> eclass` view with a `UnionId` /
602    /// `UnionIntoUf` merge (congruence inline at FD-conflict).
603    /// PROOF mode: a TUPLE-output `(children) -> (eclass, Proof)` view with a
604    /// `Columns([UnionIntoUfWithProof, EclassMinProof])` merge — the merge both
605    /// composes the congruence edge proof into `@UF_Sf` and keeps the surviving
606    /// (min) eclass's view proof. See `is_native_merge_proof_view`.
607    pub(crate) fn is_native_merge_view(&self, view_name: &str) -> bool {
608        self.egraph
609            .proof_state
610            .native_merge_views
611            .contains(view_name)

614    /// True when `view_name` is a native-merge view AND it is rendered with the
615    /// FD-keyed `(children) -> eclass` view SHAPE (function output). Gates the view
616    /// DECLARATION + the children-keyed `view_delete` / function-output
617    /// `view_body_atom` / children-keyed `view_subsume` / `set_view` forms.
618    ///
619    /// Returns `false` ONLY for a pure CONSTRUCTOR-CONGRUENCE view on a
620    /// `native_merge_views_coexist()` backend (DuckDB): such a view keeps the
621    /// BASELINE all-columns Unit-relation shape — `(function @<C>View (children...,
622    /// eclass) Unit :merge old)` — so its rows COEXIST and the rebuild / delete /
623    /// subsume key on the FULL `(children, eclass)` row (a children-only delete on
624    /// coexisting rows would self-wipe the canonical row, and a full-key delete can't
625    /// be expressed on an FD function). It STAYS in `native_merge_views` (so the
626    /// `@congruence_rule*` is still dropped and the view -> UF association is still
627    /// registered), and congruence is done by the host pass `emit_native_congruence`.
628    ///
629    /// VALUE-FOLD and TERM-BUILD views ALWAYS keep the FD shape (even on DuckDB):
630    /// their merge does real work at the FD conflict (a fold / e-node minting) and is
631    /// resolved natively in-SQL by `mergefn_to_sql` / `emit_term_build_merges`, which
632    /// REQUIRE the `(children) -> value`/`eclass` function form. So this returns
633    /// `true` for them regardless of `native_merge_views_coexist()`.
634    pub(crate) fn uses_fd_native_merge_view(&self, view_name: &str) -> bool {
635        if !self.is_native_merge_view(view_name) {
636            return false;
637        }
638        // VALUE-FOLD / TERM-BUILD views always use the FD shape.
639        let is_value_fold = self
640            .egraph
641            .proof_state
642            .native_value_merge_views
643            .contains_key(view_name);
644        let is_term_build = self
645            .egraph
646            .proof_state
647            .native_term_build_views
648            .contains_key(view_name);
649        if is_value_fold || is_term_build {
650            return true;
651        }
652        // Pure constructor-congruence view: FD shape unless the backend coexists.
653        !self.native_merge_views_coexist()
654    }

910/// Reads a file and checks that its commands support the proof encoding.
911pub fn file_supports_proofs(path: &Path) -> bool {
912    let contents = match std::fs::read_to_string(path) {
913        Ok(contents) => contents,
914        Err(_) => return false,
915    };
916
917    let canonical = match std::fs::canonicalize(path) {
918        Ok(canonical) => canonical,
919        Err(_) => return false,
920    };
921
922    let mut egraph = EGraph::default();
923    let filename = canonical.to_string_lossy().into_owned();
924    // Run the proof-encoding gate exactly as the pipeline does: on the
925    // typechecked program BEFORE global removal. Checking it after
926    // `resolve_program` (which removes globals) false-negatives on let-global
927    // files, because global removal injects `:unsafe-seminaive` rules that the
928    // gate's `UnsafeSeminaive` arm rejects — rules the real proof pipeline
929    // never gates (it gates the user program first, then removes globals).
930    egraph
931        .check_program_proof_gate(Some(filename), &contents)
932        .is_ok()
933}

983fn expr_primitives_have_validators(expr: &ResolvedExpr) -> bool {

985    use crate::core::ResolvedCall;

1013/// Checks whether a resolved command supports proof encoding.
1014/// Returns Ok(()) if supported, or Err with the reason if not.
1015///
1016/// `proofs_enabled` indicates whether the proof-tracking phase is
1017/// actually going to run. Some checks below only matter when proofs
1018/// are produced (e.g. the validator requirement: a validator is the
1019/// hook that certifies a proof for a primitive call, so primitives
1020/// without one can't produce a proof — but they're fine in plain
1021/// term-encoding mode where no proofs are emitted). Callers pass
1022/// `false` when term encoding is on but proofs are not, so the
1023/// stricter proof-only checks are skipped.
1024pub(crate) fn command_supports_proof_encoding(
1025    command: &ResolvedCommand,
1026    type_info: &TypeInfo,
1027    proofs_enabled: bool,
1028) -> Result<(), ProofEncodingUnsupportedReason> {
1029    // `:unsafe-seminaive` rules perform arbitrary RHS reads against the
1030    // live database and skip the action-lookup typecheck. The term/proof
1031    // encoding can't represent that, so reject such rules in *both*
1032    // plain term-encoding and proof modes (ungated by `proofs_enabled`).
1033    if let GenericCommand::Rule { rule } = command
1034        && rule.unsafe_seminaive
1035    {
1036        return Err(ProofEncodingUnsupportedReason::UnsafeSeminaive);
1037    }
1038
1039    // Check all expressions for primitives without validators —
1040    // proof-mode only. Validators are how a primitive certifies its
1041    // own result inside a generated proof; absent proofs, missing
1042    // validators don't matter.
1043    if proofs_enabled {
1044        let mut all_primitives_have_validators = true;
1045        command.clone().visit_exprs(&mut |expr| {
1046            if !expr_primitives_have_validators(&expr) {
1047                all_primitives_have_validators = false;
1048            }
1049            expr
1050        });
1051
1052        if !all_primitives_have_validators {
1053            return Err(ProofEncodingUnsupportedReason::PrimitiveWithoutValidator);
1054        }
1055    }
1056
1057    // Check actions (not queries) for function lookups —
1058    // proof-mode only. Global function calls (which get desugared to
1059    // constructors) are fine; arbitrary lookups break proof
1060    // generation but not plain term encoding.
1061    if proofs_enabled {
1062        let mut has_function_lookup_in_action = false;
1063        command.clone().visit_actions(&mut |action| {
1064            has_function_lookup_in_action |= action_has_function_lookup(&action, type_info);
1065            action
1066        });
1067
1068        if has_function_lookup_in_action {
1069            return Err(ProofEncodingUnsupportedReason::FunctionLookupInAction);
1070        }
1071    }
1072
1073    // Now check command-specific constraints
1074    match command {
1075        GenericCommand::Sort {
1076            presort_and_args: Some(_),
1077            ..
1078        } => Err(ProofEncodingUnsupportedReason::SortWithPresort),
1079        GenericCommand::Sort { uf: Some(_), .. } => {
1080            Err(ProofEncodingUnsupportedReason::SortWithUfAnnotation)
1081        }
1082        GenericCommand::Sort {
1083            proof_func: Some(_),
1084            ..
1085        } => Err(ProofEncodingUnsupportedReason::SortWithProofFuncAnnotation),
1086        GenericCommand::UserDefined(..) if proofs_enabled => {
1087            // Proof generation can't reason about user-defined
1088            // commands (they're opaque to the term encoder). Plain
1089            // term encoding is fine: the user-defined command runs
1090            // through whatever the experimental extension installed
1091            // (e.g. egglog-experimental's `run-schedule` rebinding,
1092            // `multi-extract`), and term encoding doesn't need to
1093            // rewrite it.
1094            Err(ProofEncodingUnsupportedReason::UserDefinedCommand)
1095        }
1096        GenericCommand::UserDefined(..) => Ok(()),
1097        GenericCommand::Input { .. } if proofs_enabled => {
1098            Err(ProofEncodingUnsupportedReason::InputCommand)
1099        }
1100        GenericCommand::Input { .. } => Ok(()),
1101        // Extract commands can't have non-global function lookups
1102        // because instrument_action_expr doesn't support them
1103        // (global function calls are fine - they get desugared to constructors)
1104        GenericCommand::Extract(_, expr, variants) => {
1105            if type_info.expr_has_function_lookup(expr).is_some()
1106                || type_info.expr_has_function_lookup(variants).is_some()
1107            {
1108                Err(ProofEncodingUnsupportedReason::FunctionLookupInAction)
1109            } else {
1110                Ok(())
1111            }
1112        }
1113        // no-merge on a non-global function — proof-mode only.
1114        // Term encoding without proofs is fine with this.
1115        // To add support: https://github.com/egraphs-good/egglog/issues/774
1116        GenericCommand::Function {
1117            merge: None, name, ..
1118        } if proofs_enabled => {
1119            if type_info.is_global(name) {
1120                Ok(())
1121            } else {
1122                Err(ProofEncodingUnsupportedReason::NoMergeOnNonGlobalFunction)
1123            }
1124        }
1125        _ => Ok(()),
1126    }
1127}
