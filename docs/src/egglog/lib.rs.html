146/// Output from a command.
147#[derive(Clone, Debug)]
148#[allow(clippy::large_enum_variant)]
149pub enum CommandOutput {
150    /// The size of a function
151    PrintFunctionSize(usize),
152    /// The name of all functions and their sizes
153    PrintAllFunctionsSize(Vec<(String, usize)>),
154    /// The best term found after extracting
155    ExtractBest(TermDag, DefaultCost, TermId),
156    /// The variants of a function found after extracting. Like normal extraction, but has to choose one extraction per e-node in the e-class.
157    ExtractVariants(TermDag, Vec<TermId>),
158    /// A high-level proof witnessing constructor existence
159    ProveExists {
160        proof_store: ProofStore,
161        proof_id: ProofId,
162    },
163    /// The report from all runs
164    OverallStatistics(RunReport),
165    /// A printed function and all its values
166    PrintFunction(Function, TermDag, Vec<(TermId, TermId)>, PrintFunctionMode),
167    /// The report from a single run
168    RunSchedule(RunReport),
169    /// A user defined output
170    UserDefined(Arc<dyn UserDefinedCommandOutput>),
171}

173impl std::fmt::Display for CommandOutput {
174    /// Format the command output for display, ending with a newline.
175    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
176        match self {
177            CommandOutput::PrintFunctionSize(size) => writeln!(f, "{size}"),
178            CommandOutput::PrintAllFunctionsSize(names_and_sizes) => {
179                write!(f, "(")?;
180                for (i, (name, size)) in names_and_sizes.iter().enumerate() {
181                    // indent except for the first line
182                    if i > 0 {
183                        write!(f, " ")?;
184                    }
185                    // write the pair of funciton symbol and size
186                    write!(f, "({name} {size})")?;
187                    // add a newline except at the end
188                    if i < names_and_sizes.len() - 1 {
189                        writeln!(f)?;
190                    }
191                }
192                writeln!(f, ")")
193            }
194            CommandOutput::ExtractBest(termdag, _cost, term) => {
195                writeln!(f, "{}", termdag.to_string(*term))
196            }
197            CommandOutput::ExtractVariants(termdag, terms) => {
198                writeln!(f, "(")?;
199                for expr in terms {
200                    writeln!(f, "   {}", termdag.to_string(*expr))?;
201                }
202                writeln!(f, ")")
203            }
204            CommandOutput::ProveExists {
205                proof_store,
206                proof_id,
207            } => writeln!(f, "{}", proof_store.proof_to_string(*proof_id)),
208            CommandOutput::OverallStatistics(run_report) => {
209                write!(f, "Overall statistics:\n{run_report}")
210            }
211            CommandOutput::PrintFunction(function, termdag, terms_and_outputs, mode) => {
212                let out_is_unit = function.schema.output.name() == UnitSort.name();
213                if *mode == PrintFunctionMode::CSV {
214                    let mut wtr = Writer::from_writer(vec![]);
215                    for (term_id, output) in terms_and_outputs {
216                        let term = termdag.get(*term_id);
217                        match term {
218                            Term::App(name, children) => {
219                                let mut values = vec![name.clone()];
220                                for child_id in children {
221                                    values.push(termdag.to_string(*child_id));
222                                }
223
224                                if !out_is_unit {
225                                    values.push(termdag.to_string(*output));
226                                }
227                                wtr.write_record(&values).map_err(|_| std::fmt::Error)?;
228                            }
229                            _ => panic!("Expect function_to_dag to return a list of apps."),
230                        }
231                    }
232                    let csv_bytes = wtr.into_inner().map_err(|_| std::fmt::Error)?;
233                    f.write_str(&String::from_utf8(csv_bytes).map_err(|_| std::fmt::Error)?)
234                } else {
235                    writeln!(f, "(")?;
236                    for (term, output) in terms_and_outputs.iter() {
237                        write!(f, "   {}", termdag.to_string(*term))?;
238                        if !out_is_unit {
239                            write!(f, " -> {}", termdag.to_string(*output))?;
240                        }
241                        writeln!(f)?;
242                    }
243                    writeln!(f, ")")
244                }
245            }
246            CommandOutput::RunSchedule(_report) => Ok(()),
247            CommandOutput::UserDefined(output) => {
248                write!(f, "{}", *output)
249            }
250        }
251    }
252}

254/// The main interface for an e-graph in egglog.
255///
256/// An [`EGraph`] maintains a collection of equivalence classes of terms and provides
257/// operations for adding facts, running rules, and extracting optimal terms.
258///
259/// # Examples
260///
261/// ```
262/// use egglog::*;
263///
264/// let mut egraph = EGraph::default();
265/// egraph.parse_and_run_program(None, "(datatype Math (Num i64) (Add Math Math))").unwrap();
266/// ```
267#[derive(Clone)]
268pub struct EGraph {
269    backend: Box<dyn egglog_backend_trait::Backend>,
270    pub parser: Parser,
271    names: check_shadowing::Names,
272    /// pushed_egraph forms a linked list of pushed egraphs.
273    /// Pop reverts the egraph to the last pushed egraph.
274    pushed_egraph: Option<Box<Self>>,
275    functions: IndexMap<String, Function>,
276    rulesets: IndexMap<String, Ruleset>,
277    pub fact_directory: Option<PathBuf>,
278    pub seminaive: bool,
279    pub no_decomp: bool,
280    type_info: TypeInfo,
281    /// The run report unioned over all runs so far.
282    overall_run_report: RunReport,
283    schedulers: DenseIdMap<SchedulerId, SchedulerRecord>,
284    commands: IndexMap<String, Arc<dyn UserDefinedCommand>>,
285    strict_mode: bool,
286    warned_about_global_prefix: bool,
287    /// Registry for command-level macros
288    command_macros: CommandMacroRegistry,
289    proof_state: EncodingState,
290    /// In proof mode, this is the program before proof instrumentation and the version we use for proof checking.
291    proof_check_program: Vec<ResolvedNCommand>,
292}

351#[derive(Clone, Debug)]
352pub struct ResolvedSchema {
353    pub input: Vec<ArcSort>,
354    /// The first (primary) output sort. See [`ResolvedSchema::outputs`].
355    pub output: ArcSort,
356    /// Additional output sorts for tuple-output functions. Empty for ordinary functions.
357    pub extra_outputs: Vec<ArcSort>,
358}
359
360impl ResolvedSchema {
361    /// All output (value-column) sorts: the primary output followed by any extras.
362    pub fn outputs(&self) -> impl Iterator<Item = &ArcSort> {
363        std::iter::once(&self.output).chain(self.extra_outputs.iter())
364    }
365
366    /// Get the type at position `index`, counting the output columns as positions
367    /// `input.len()`, `input.len() + 1`, ...
368    pub fn get_by_pos(&self, index: usize) -> Option<&ArcSort> {
369        if index < self.input.len() {
370            self.input.get(index)
371        } else {
372            let out_idx = index - self.input.len();
373            if out_idx == 0 {
374                Some(&self.output)
375            } else {
376                self.extra_outputs.get(out_idx - 1)
377            }
378        }
379    }
380}

398    /// Build a frontend `EGraph` around the given backend trait object.
399    /// This is the shared backbone of [`EGraph::default`] and
400    /// [`EGraph::with_duckdb_backend`].
401    fn with_backend(backend: Box<dyn egglog_backend_trait::Backend>) -> Self {
402        let mut parser = Parser::default();
403        let proof_state = EncodingState::new(&mut parser.symbol_gen);
404        let mut eg = Self {
405            backend,
406            parser,
407            names: Default::default(),
408            pushed_egraph: Default::default(),
409            functions: Default::default(),
410            rulesets: Default::default(),
411            fact_directory: None,
412            seminaive: true,
413            no_decomp: false,
414            overall_run_report: Default::default(),
415            type_info: Default::default(),
416            schedulers: Default::default(),
417            commands: Default::default(),
418            strict_mode: false,
419            warned_about_global_prefix: false,
420            command_macros: Default::default(),
421            proof_state,
422            proof_check_program: vec![],
423        };
424        add_base_sort(&mut eg, UnitSort, span!()).unwrap();
425        add_base_sort(&mut eg, StringSort, span!()).unwrap();
426        add_base_sort(&mut eg, BoolSort, span!()).unwrap();
427        add_base_sort(&mut eg, I64Sort, span!()).unwrap();
428        add_base_sort(&mut eg, F64Sort, span!()).unwrap();
429        add_base_sort(&mut eg, BigIntSort, span!()).unwrap();
430        add_base_sort(&mut eg, BigRatSort, span!()).unwrap();
431        eg.type_info.add_presort::<MapSort>(span!()).unwrap();
432        eg.type_info.add_presort::<SetSort>(span!()).unwrap();
433        eg.type_info.add_presort::<VecSort>(span!()).unwrap();
434        eg.type_info.add_presort::<FunctionSort>(span!()).unwrap();
435        eg.type_info.add_presort::<MultiSetSort>(span!()).unwrap();
436        eg.type_info.add_presort::<PairSort>(span!()).unwrap();
437
438        // Add != with a validator that computes inequality result
439        let neq_validator = |termdag: &mut TermDag, args: &[TermId]| -> Option<TermId> {
440            if args.len() == 2 && args[0] != args[1] {
441                // Return unit literal for successful inequality
442                Some(termdag.lit(Literal::Unit))
443            } else {
444                None
445            }
446        };
447        add_primitive_with_validator!(
448            &mut eg,
449            "!=" = |a: #, b: #| -?> () {
450                (a != b).then_some(())
451            },
452            neq_validator
453        );
454
455        add_primitive_with_validator!(
456            &mut eg,
457            "bool-!=" = |a: #, b: #| -> bool {
458                (a != b)
459            },
460            |termdag: &mut TermDag, args: &[TermId]| -> Option<TermId> {
461                if args.len() == 2 {
462                    Some(termdag.lit(Literal::Bool(args[0] != args[1])))
463                } else {
464                    None
465                }
466            }
467        );
468
469        add_primitive!(&mut eg, "value-eq" = |a: #, b: #| -?> () {
470            (a == b).then_some(())
471        });
472        add_primitive!(&mut eg, "ordering-min" = |a: #, b: #| -> # {
473            if a < b { a } else { b }
474        });
475        add_primitive!(&mut eg, "ordering-max" = |a: #, b: #| -> # {
476            if a > b { a } else { b }
477        });
478
479        eg.rulesets
480            .insert("".into(), Ruleset::Rules(Default::default()));
481
482        eg
483    }

485    /// Construct an `EGraph` whose backend is the DuckDB-backed
486    /// [`egglog_bridge_duckdb::EGraph`] driven through the
487    /// [`egglog_backend_trait::Backend`] trait.
488    ///
489    /// This is the entry point exercised by `tests/files.rs`'s duckdb
490    /// path after Phase 2 Commit 14. It performs the same sort /
491    /// primitive registration as [`EGraph::default`] but the underlying
492    /// storage and rule execution route through DuckDB.
493    ///
494    /// Term encoding (or proof-tracking term encoding) is enabled
495    /// unconditionally: the DuckDB backend is term-encoding only.
496    pub fn with_duckdb_backend(config: DuckBackendConfig) -> anyhow::Result<Self> {
497        let mut db = egglog_bridge_duckdb::EGraph::new()?;
498        if config.fast_rebuild {
499            db.enable_fast_rebuild();
500        }
501        if config.proofs {
502            // Switch the eq-sort constructor hash-cons to the bare term
503            // table so each application keeps its own id; reusing the
504            // canonicalized view's leader id breaks proof extraction.
505            db.enable_proofs();
506        }
507        let backend: Box<dyn egglog_backend_trait::Backend> = Box::new(db);
508        let mut eg = Self::with_backend(backend);
509
510        // Term encoding requires a separate typechecker EGraph for
511        // re-typechecking after the encoder runs. The duckdb backend
512        // cannot be cloned (its `Connection` is not trivially
513        // cloneable), so we instead build a bridge-backed typechecker
514        // that mirrors the same sort + primitive registration done
515        // above.
516        let typechecker = if config.proofs {
517            EGraph::new_with_proofs()
518        } else {
519            EGraph::default().with_term_encoding_enabled()
520        };
521        eg.proof_state.original_typechecking = Some(Box::new(typechecker));
522        if config.proofs {
523            eg.proof_state.proofs_enabled = true;
524        }
525        Ok(eg)
526    }

531    ///
532    /// This is the Milestone-3 entry point exercised by `tests/files.rs`'s
533    /// `feldera` treatment: real `.egg` files run parse → desugar → typecheck →
534    /// term-encode → Feldera and are checked against the shared snapshot

536    ///
537    /// Like the DuckDB backend, the Feldera backend is **term-encoding only**:
538    /// the encoder lowers eq-sorts to view-table congruence + the `@uf` / `@uff`
539    /// rebuild rulesets, which the Feldera host interpreter runs faithfully (see

545    /// [`EGraph::with_feldera_backend`] with explicit knobs. Feldera has been
546    /// migrated OFF native-UF onto the fast RELATIONAL term-encoding, so there is
547    /// no `native_uf` knob: `--feldera` runs the relational encoding — RULE-ENCODED
548    /// congruence (`@congruence_rule*`) canonicalized by the `@rebuild_rule*`
549    /// rebuild rules, which lower as ORDINARY rules through Feldera's general DBSP
550    /// rule engine (`view ⋈ @UF_Sf` incremental joins over the real materialized
551    /// `@UF_Sf` flat index). [`FelderaBackendConfig::fast_rebuild`] is a NO-OP
552    /// (the δuf-scoping rebuild optimization was removed with native-UF), and
553    /// [`FelderaBackendConfig::proofs`] runs the relational proof encoding.
554    pub fn with_feldera_backend_config(config: FelderaBackendConfig) -> anyhow::Result<Self> {
555        let db = egglog_bridge_feldera::EGraph::new();
556        // `config.fast_rebuild` is now a no-op on Feldera: the relational general
557        // rebuild (every `@rebuild_rule*` lowered as an ordinary DBSP rule) is
558        // always used — there is no δuf-scoping fast-rebuild variant to engage.
559        let _ = config.fast_rebuild;
560        let backend: Box<dyn egglog_backend_trait::Backend> = Box::new(db);
561        let mut eg = Self::with_backend(backend);
562
563        // Term encoding requires a separate bridge-backed typechecker EGraph for
564        // re-typechecking after the encoder runs (the Feldera backend, like
565        // DuckDB, is not the typechecker). Mirror the DuckDB setup, including
566        // building a proof-enabled typechecker when proofs are requested so the
567        // CLI never reaches the backend-cloning `with_proofs_enabled` path.
568        let typechecker = if config.proofs {
569            EGraph::new_with_proofs()
570        } else {
571            EGraph::default().with_term_encoding_enabled()
572        };
573        eg.proof_state.original_typechecking = Some(Box::new(typechecker));
574        if config.proofs {
575            eg.proof_state.proofs_enabled = true;
576        }
577        Ok(eg)
578    }

601    /// migrated OFF native-UF onto the fast RELATIONAL term-encoding (the LAST
602    /// backend, mirroring the DuckDB / Feldera migrations), so there is no
603    /// `native_uf` knob: `--flowlog` runs the relational encoding — RULE-ENCODED
604    /// congruence (`@congruence_rule*`) canonicalized by the `@rebuild_rule*`
605    /// rebuild rules, which lower as ORDINARY rules through FlowLog's general DD
606    /// rule engine (the sound `step_symmetric` path).
607    /// [`FlowlogBackendConfig::fast_rebuild`] is a NO-OP (the only flowlog
608    /// fast-rebuild variant was the unsound δUF path, now deleted), and
609    /// [`FlowlogBackendConfig::proofs`] runs the relational proof encoding.
610    pub fn with_flowlog_backend_config(config: FlowlogBackendConfig) -> anyhow::Result<Self> {
611        let mut db = egglog_bridge_flowlog::EGraph::new_interpret();
612        // `config.fast_rebuild` is a no-op on FlowLog: the relational general
613        // rebuild (every `@rebuild_rule*` lowered as an ordinary DD rule) is
614        // always used — there is no δUF-scoping fast-rebuild variant to engage.
615        let _ = config.fast_rebuild;
616        if config.wcoj {
617            db.enable_wcoj();
618        }
619        let backend: Box<dyn egglog_backend_trait::Backend> = Box::new(db);
620        let mut eg = Self::with_backend(backend);
621
622        // Canonicalize-at-creation is always-on for all term-encoding backends
623        // (set in `EncodingState::new`); no per-backend override is needed here.
624
625        // Term encoding requires a separate bridge-backed typechecker EGraph for
626        // re-typechecking after the encoder runs. Mirror the DuckDB/Feldera setup,
627        // including building a proof-enabled typechecker when proofs are requested
628        // so the CLI never reaches the backend-cloning `with_proofs_enabled` path.
629        let typechecker = if config.proofs {
630            EGraph::new_with_proofs()
631        } else {
632            EGraph::default().with_term_encoding_enabled()
633        };
634        eg.proof_state.original_typechecking = Some(Box::new(typechecker));
635        if config.proofs {
636            eg.proof_state.proofs_enabled = true;
637        }
638        Ok(eg)
639    }

916    /// Set the number of threads used for parallel operations.
917    ///
918    /// This is a helper that simply configures the global rayon thread pool. It can only be called
919    /// once per process; subsequent calls will be ignored.
920    ///
921    /// # Panics
922    ///
923    /// Panics on wasm if `num_threads > 1`.
924    pub fn set_num_threads(num_threads: usize) {
925        #[cfg(target_family = "wasm")]
926        if num_threads > 1 {
927            panic!("cannot use more than 1 thread on wasm");
928        }
929        #[cfg(not(target_family = "wasm"))]
930        {
931            // Configure the global rayon pool exactly once per process. rayon's
932            // `build_global` only succeeds before the pool is first used, so the
933            // FIRST `set_num_threads` call wins and later ones are silently
934            // ignored (matching this method's doc). The `Once` guard avoids a
935            // spurious GlobalPoolAlreadyInitialized warning when it's called more
936            // than once -- e.g. egglog-experimental's main configures the pool
937            // before constructing the egraph, then `cli` calls this again.
938            use std::sync::Once;
939            static INIT: Once = Once::new();
940            INIT.call_once(|| {
941                match rayon::ThreadPoolBuilder::new()
942                    .num_threads(num_threads)
943                    .build_global()
944                {
945                    Ok(()) => {
946                        log::info!("Initialized global thread pool with {num_threads} threads")
947                    }
948                    Err(e) => log::warn!(
949                        "Failed to initialize global thread pool with {num_threads} threads: {e:?}"
950                    ),
951                }
952            });
953        }
954    }

1081    /// egraph.

1083        match self.pushed_egraph.take() {

1088                // generated after pop don't collide with ones generated before pop.

1098    /// lookup for eq-sort children, EXACTLY as the rule-encoded merge body's
1099    /// `canon_arg` does (`(__UF_Sf x)`). In `--native-merge` (relational, no
1100    /// `--native-uf`) the `@UF_Sf` index is a plain `function (S) S :merge
1101    /// (ordering-min old new)` declared with `DefaultVal::Identity` (lookup-or-self
1102    /// on miss). Expressed as a `MergeFn::Function` against that table; the bridge's
1103    /// resolved `Function` honors the `Identity` default by returning the looked-up
1104    /// key on a miss. Returns the inner `MergeFn` unchanged when the sort has no
1105    /// `@UF_Sf` index (non-eq-sort, or proof mode where canon-at-creation is off).
1106    fn canon_mergefn(
1107        &self,
1108        sort_name: &str,
1109        inner: egglog_bridge::MergeFn,
1110    ) -> egglog_bridge::MergeFn {
1111        if self.proof_state.proofs_enabled {
1112            return inner;
1113        }
1114        match self
1115            .proof_state
1116            .uf_function
1117            .get(sort_name)
1118            .and_then(|uf_name| self.functions.get(uf_name))
1119        {
1120            Some(uf_func) => egglog_bridge::MergeFn::Function(uf_func.backend_id, vec![inner]),
1121            None => inner,
1122        }
1123    }

1130    /// self-referential function `@UF_Sf : (S) -> S` (the WHOLE union-find).
1131    /// To union `a, b`: `(set (@UF_Sf a) b)`. On an FD conflict — `@UF_Sf a` is
1132    /// already `old`, a new write proposes `new` — the merge must
1133    ///   1. return `ordering-min(old, new)` as the surviving leader of `a`, AND
1134    ///   2. `(set (@UF_Sf ordering-max(old, new)) ordering-min(old, new))` —
1135    ///      union the two parents into ITS OWN table (respecting its own merge),
1136    ///      so the displaced `max(old,new)` is NOT orphaned (asserting `a~old`
1137    ///      AND `a~new` ⇒ `old~new` must be recorded; otherwise the orphan's
1138    ///      identity-on-miss lookup would make it its own leader — WRONG).
1139    ///
1140    /// This expression is the merge for BOTH `@UF_Sf` itself (the recursive
1141    /// parent-union via `TableInsert` into `uf_self_id`) AND a CONSTRUCTOR's
1142    /// `@<F>View` eclass column (congruence: on a view FD-conflict the two output
1143    /// eclasses are unioned into `@UF_Sf` and the surviving view eclass is the
1144    /// min). `uf_self_id` is the backend id of `@UF_Sf` for this sort.
1145    ///
1146    /// The `ordering-min` / `ordering-max` primitives are resolved to their
1147    /// backend `ExternalFunctionId`s via the primitives registry (the `#`
1148    /// placeholder overload is sort-generic, doing a plain value comparison), the
1149    /// same id `translate_expr_to_mergefn` would produce for a
1150    /// `(ordering-min old new)` call.
1151    fn build_uf_self_merge(
1152        &self,
1153        uf_self_id: egglog_bridge::FunctionId,
1154    ) -> Result<egglog_bridge::MergeFn, Error> {
1155        use egglog_bridge::MergeFn;
1156        let resolve = |name: &str| -> Result<ExternalFunctionId, Error> {
1157            self.type_info
1158                .current_primitive_external_id(name, crate::Context::Write)
1159                .ok_or_else(|| {
1160                    Error::BackendError(format!(
1161                        "native-merge UF merge: primitive `{name}` not resolvable in write context"
1162                    ))
1163                })
1164        };
1165        let min_id = resolve("ordering-min")?;
1166        let max_id = resolve("ordering-max")?;
1167        let min = MergeFn::Primitive(min_id, vec![MergeFn::Old, MergeFn::New]);
1168        let max = MergeFn::Primitive(max_id, vec![MergeFn::Old, MergeFn::New]);
1169        Ok(MergeFn::IfEq {
1170            // old == new: inert, no side write (matches the rule's `(!= old new)`
1171            // guard — the singleparent/congruence rule never fires when equal).
1172            a: Box::new(MergeFn::Old),
1173            b: Box::new(MergeFn::New),
1174            then: Box::new(MergeFn::Old),
1175            els: Box::new(MergeFn::Seq(vec![
1176                // (set (@UF_Sf max) min) — recursive parent-union into ITS OWN
1177                // table (respects this very merge); strictly `min < max`, so the
1178                // chain of self-inserts is finite (union-by-min, idempotent).
1179                MergeFn::TableInsert(uf_self_id, vec![max, min.clone()]),
1180                // surviving value of this column = min.
1181                min,
1182            ])),
1183        })
1184    }

1203    /// ast)` view-proof term, mirroring `add_term_and_view`'s
1204    /// `Justification::Merge` proof. `ast` is the already-built `MergeFn`
1205    /// evaluating to the wrapped `(___AstS term)` AST node. `old_pf`/`new_pf` are
1206    /// the conflicting view rows' proof columns. The `___Merge` constructor is
1207    /// hash-consed via `MergeFn::Function`, the same mint path `UnionIntoUfWithProof`
1208    /// uses for `Trans`/`Sym`.
1209    fn build_merge_proof(

1211        merge_id: egglog_bridge::FunctionId,
1212        fn_name: &str,

1214        new_pf: egglog_bridge::MergeFn,
1215        ast: egglog_bridge::MergeFn,

1219            self.base_values(),
1220            &Literal::String(fn_name.into()),
1221        ));
1222        MergeFn::Function(merge_id, vec![name_const, old_pf, new_pf, ast])

1225    /// Lower a term-building custom `:merge` body (e.g. `(C2 (C1 old new) (C2 old
1226    /// new))`) into a `MergeFn::Seq` of `Construct` / `TableInsert` / union
1227    /// variants — the PR #933 `:merge`-multiple-actions port. This re-expresses the
1228    /// SAME action sequence the rule-encoded `@merge_rule` head produced (via
1229    /// `add_term_and_view`): for each constructor call it mints the term
1230    /// (`Construct`), writes the canonicalized `@<C>View` row (`TableInsert`,
1231    /// respecting the view's own congruence merge), and stages the self-union edge,
1232    /// returning the minted (canonicalized) eclass. The outermost call's value is
1233    /// the last `Seq` element (the merged eclass written into the FD `@<F>View`).
1234    ///
1235    /// `output_sort` is the merged function's output sort; `fn_name` its ORIGINAL
1236    /// name (the view's `:internal-term-constructor`). The whole Seq is guarded by
1237    /// `IfEq(canon(old), canon(new), …)`: the rule-encoded `@merge_rule` fires only
1238    /// on `(!= old new)` view rows (and its view rows are canonical under
1239    /// canon-at-creation), so when `old`/`new` canonicalize to the SAME eclass the
1240    /// rule encoding builds NO helper term. Without this guard the native merge
1241    /// callback fires on every FD conflict — including ones whose two values later
1242    /// collapse to one eclass — and `Construct` over-materializes a spurious
1243    /// `(C old old)` helper term that the rule encoding never keeps (e.g.
1244    /// `tests/web-demo/rw-analysis.egg`'s `merge-val`). The guard reproduces the
1245    /// rule encoding's materialization exactly: equal => return that value, no mint.
1246    ///
1247    /// PROOF MODE: the view is the A2 TUPLE `(children) -> (eclass, Proof)`. We
1248    /// return `MergeFn::Columns([eclass_branch, proof_branch])`:
1249    ///   * `eclass_branch` is the guarded eclass `Seq` (col0). Its side-effects also
1250    ///     thread the per-constructor `(___Merge "fn" old_pf new_pf (___AstS minted))`
1251    ///     view-proof into the inner `@<C>View` tuple proof column, the
1252    ///     `(___TreeProof minted)` term-proof, and the `@UF_S` self-edge — exactly
1253    ///     what the dropped `@merge_rule` head wrote per `add_term_and_view`.
1254    ///   * `proof_branch` is the result view-row proof (col1):
1255    ///     `(___Merge "fn" old_pf new_pf (___Ast_fview (fn keys merged_eclass)))`,
1256    ///     guarded the same way (equal => keep the surviving old proof column).
1257    ///     `merged_eclass` re-evaluates the outermost `Construct` (hash-consed, so
1258    ///     it returns the same minted id col0 produced).
1259    fn translate_term_build_merge_to_seq(
1260        &self,
1261        expr: &ResolvedExpr,
1262        output_sort: &str,
1263        fn_name: &str,
1264    ) -> Result<egglog_bridge::MergeFn, Error> {
1265        use egglog_bridge::MergeFn;
1266
1267        if self.proof_state.proofs_enabled {
1268            // --- PROOF mode: A2 tuple view `(children) -> (eclass, proof)`. ---
1269            // The conflicting rows' proof column is index 1 in the tuple view.
1270            let old_pf = MergeFn::OldCol(1);
1271            let new_pf = MergeFn::NewCol(1);
1272            let merge_id = self.proof_fn_id(&self.proof_state.proof_names.merge_fn_constructor)?;
1273
1274            // Column 0: the eclass-producing guarded Seq, whose side-effects also
1275            // write the proof columns/tables for each minted constructor.
1276            let mut side_effects = Vec::new();
1277            let (outer_eclass, outer_construct) = self.lower_term_build_expr_proof(
1278                expr,
1279                &mut side_effects,
1280                fn_name,
1281                &old_pf,
1282                &new_pf,
1283            )?;
1284            side_effects.push(outer_eclass);
1285            let eclass_body = MergeFn::Seq(side_effects);
1286
1287            let canon_old = self.canon_mergefn(output_sort, MergeFn::Old);
1288            let canon_new = self.canon_mergefn(output_sort, MergeFn::New);
1289            let eclass_branch = MergeFn::IfEq {
1290                a: Box::new(canon_old.clone()),
1291                b: Box::new(canon_new.clone()),
1292                then: Box::new(canon_old.clone()),
1293                els: Box::new(eclass_body),
1294            };
1295
1296            // Column 1: the result view-row proof
1297            // `(___Merge "fn" old_pf new_pf (___Ast_fview (fn keys merged_eclass)))`.
1298            // The view sort's AST constructor wraps the term-table view application
1299            // `(fn keys merged_eclass)` (NOT the bare eclass) — matching the oracle's
1300            // `(___Merge "f" ___p12 ___p22 (___Ast___view5 (f c0_ ___v48)))`.
1301            let term_table = self
1302                .functions
1303                .get(fn_name)
1304                .map(|f| f.backend_id)
1305                .ok_or_else(|| {
1306                    Error::BackendError(format!(
1307                        "term-build merge (proof): term table `{fn_name}` not declared"
1308                    ))
1309                })?;
1310            let view_sort = self
1311                .proof_state
1312                .proof_names
1313                .fn_to_term_sort
1314                .get(fn_name)
1315                .ok_or_else(|| {
1316                    Error::BackendError(format!(
1317                        "term-build merge (proof): no view sort recorded for `{fn_name}`"
1318                    ))
1319                })?;
1320            let ast_view_name = self
1321                .proof_state
1322                .proof_names
1323                .sort_to_ast_constructor
1324                .get(view_sort)
1325                .ok_or_else(|| {
1326                    Error::BackendError(format!(
1327                        "term-build merge (proof): no AST constructor for view sort `{view_sort}`"
1328                    ))
1329                })?;
1330            let ast_view_id = self.proof_fn_id(ast_view_name)?;
1331
1332            // (fn keys merged_eclass): the merge keys are the view's input columns
1333            // (OldCol(i)/NewCol(i) agree pre-merge since the FD keys are equal); use
1334            // OldCol for each key, then the re-evaluated outermost eclass.
1335            // The term table `f` of a CUSTOM function is `(orig_inputs..., output)
1336            // -> view_sort` (the original output sort is appended as a key). The
1337            // `(f keys merged_eclass)` view term takes the original input keys plus
1338            // the merged eclass, so the number of merge keys is the term table's
1339            // input arity MINUS that appended output column. The keys are the
1340            // MERGE's input (key) columns — read via `KeyCol(i)` (`cur[i]`), NOT
1341            // `OldCol`, which indexes the VALUE columns (`cur[n_keys + i]`).
1342            let n_keys = self.functions[fn_name].schema.input.len().saturating_sub(1);
1343            let mut fview_args: Vec<MergeFn> = (0..n_keys).map(MergeFn::KeyCol).collect();
1344            fview_args.push(outer_construct);
1345            let fview_term = MergeFn::Function(term_table, fview_args);
1346            let ast_view = MergeFn::Function(ast_view_id, vec![fview_term]);
1347            let result_pf = self.build_merge_proof(merge_id, fn_name, old_pf, new_pf, ast_view);
1348
1349            let proof_branch = MergeFn::IfEq {
1350                a: Box::new(canon_old),
1351                b: Box::new(canon_new),
1352                // equal => keep the surviving (old) view-row proof, no new proof.
1353                then: Box::new(MergeFn::OldCol(1)),
1354                els: Box::new(result_pf),
1355            };
1356
1357            return Ok(MergeFn::Columns(vec![eclass_branch, proof_branch]));
1358        }
1359
1360        // --- TERM (non-proof) mode: single-output view `(children) -> eclass`. ---
1361        let mut side_effects = Vec::new();
1362        let value = self.lower_term_build_expr(expr, &mut side_effects)?;
1363        side_effects.push(value);
1364        let body = MergeFn::Seq(side_effects);
1365
1366        // Guard on `canon(old) != canon(new)` for eq-sort outputs (a term-building
1367        // merge always has an eq-sort output, but the canon lookup is a no-op for
1368        // any sort without a `@UF_Sf` index). The condition operands are the
1369        // canonicalized old/new function values; the `then` branch returns
1370        // `canon(old)` (== `canon(new)`) without minting, exactly as the
1371        // rule-encoded merge rule would (it simply never fires).
1372        let canon_old = self.canon_mergefn(output_sort, MergeFn::Old);
1373        let canon_new = self.canon_mergefn(output_sort, MergeFn::New);
1374        Ok(MergeFn::IfEq {
1375            a: Box::new(canon_old.clone()),
1376            b: Box::new(canon_new),
1377            then: Box::new(canon_old),
1378            els: Box::new(body),
1379        })
1380    }

1382    /// Recursively lower one node of a term-building merge body, pushing the
1383    /// `@<C>View` `TableInsert` + self-union side-effects (innermost first, matching
1384    /// the desugared `add_term_and_view` order) onto `side_effects`, and returning
1385    /// the `MergeFn` that EVALUATES TO this node's value (a canonicalized eclass for
1386    /// a constructor call, a leaf `MergeFn` for a var/literal/primitive).
1387    fn lower_term_build_expr(
1388        &self,
1389        expr: &ResolvedExpr,
1390        side_effects: &mut Vec<egglog_bridge::MergeFn>,
1391    ) -> Result<egglog_bridge::MergeFn, Error> {
1392        use egglog_bridge::MergeFn;
1393        match expr {
1394            // Leaves: a var / literal carries the OLD / NEW value (or a constant)
1395            // verbatim. A `child` of a constructor is canonicalized by the caller.
1396            GenericExpr::Lit(..) | GenericExpr::Var(..) => self.translate_expr_to_mergefn(expr),
1397            GenericExpr::Call(_, ResolvedCall::Primitive(_), _) => {
1398                // A primitive over (canon-free) leaves: reuse the value-fold
1399                // translator (no e-node creation, no side-effects).
1400                self.translate_expr_to_mergefn(expr)
1401            }
1402            GenericExpr::Call(_, ResolvedCall::Func(f), args) => {
1403                let func = &self.functions[&f.name];
1404                let term_table = func.backend_id;
1405                let out_sort = &func.decl.schema.output;
1406                // The view function `@<C>View` for this constructor (the FD
1407                // `(children) -> eclass` table the merge body's `(set (@CView ...)
1408                // ...)` writes). Looked up by the constructor's ORIGINAL name.
1409                let view_name = self
1410                    .proof_state
1411                    .proof_names
1412                    .view_name
1413                    .get(&*f.name)
1414                    .ok_or_else(|| {
1415                        Error::BackendError(format!(
1416                            "term-build merge: no view table recorded for constructor `{}`",
1417                            f.name
1418                        ))
1419                    })?;
1420                let view_table = self.functions[view_name].backend_id;
1421
1422                // Lower each child first (innermost-first side-effect order),
1423                // canonicalizing eq-sort children via `@UF_Sf` find-or-self EXACTLY
1424                // as `add_term_and_view`'s `canon_arg` does.
1425                let mut canon_args = Vec::with_capacity(args.len());
1426                for (i, arg) in args.iter().enumerate() {
1427                    let child = self.lower_term_build_expr(arg, side_effects)?;
1428                    let child = if i < func.decl.schema.input.len()
1429                        && self
1430                            .type_info
1431                            .get_sort_by_name(&func.decl.schema.input[i])
1432                            .is_some_and(|s| s.is_eq_sort())
1433                    {
1434                        self.canon_mergefn(&func.decl.schema.input[i], child)
1435                    } else {
1436                        child
1437                    };
1438                    canon_args.push(child);
1439                }
1440
1441                // Mint the term: `(let fv (C canon_args..))`. `Construct` mints the
1442                // FreshId output on miss (idempotent on a present key), returning
1443                // the eclass `fv`.
1444                let construct = MergeFn::Construct(term_table, canon_args.clone(), vec![]);
1445
1446                // The view's eclass column is the CANONICALIZED `fv`
1447                // (`(__UF_Sf fv)` in the desugared body).
1448                let canon_fv = self.canon_mergefn(out_sort, construct.clone());
1449
1450                // Side-effect 1: write the `@<C>View` row `(set (@CView
1451                // canon_args..) canon_fv)` — `TableInsert` respects the view's own
1452                // congruence merge, which is the key to the nested-view congruence
1453                // resolving correctly (PR #933).
1454                let mut view_row = canon_args;
1455                view_row.push(canon_fv.clone());
1456                side_effects.push(MergeFn::TableInsert(view_table, view_row));
1457
1458                // Side-effect 2: the self-union edge `(set (@UF_S (max fv fv) (min
1459                // fv fv)) ())` — a self-loop edge, mirroring `add_term_and_view`'s
1460                // `union(fv, fv)`. Staged into the per-sort RELATIONAL `@UF_S`
1461                // parent table via `UnionIntoParentTable` (with `cur == new`, so it
1462                // is the inert self-edge the desugared body writes).
1463                if let Some(parent_table) = self
1464                    .proof_state
1465                    .uf_parent
1466                    .get(out_sort)
1467                    .and_then(|pname| self.functions.get(pname))
1468                    .map(|pf| pf.backend_id)
1469                {
1470                    side_effects.push(MergeFn::TableInsert(
1471                        parent_table,
1472                        vec![
1473                            construct.clone(),
1474                            construct.clone(),
1475                            MergeFn::Const(self.backend.base_values().get::<()>(())),
1476                        ],
1477                    ));
1478                }
1479
1480                // This node evaluates to its canonicalized eclass (used as a child
1481                // of an enclosing constructor, or as the final merged value).
1482                Ok(canon_fv)
1483            }
1484            GenericExpr::Call(span, ResolvedCall::Values(_), _) => Err(Error::TypeError(
1485                TypeError::TupleMergeNotValues("<merge>".to_owned(), span.clone()),
1486            )),
1487        }
1488    }

1490    /// PROOF-mode analogue of [`Self::lower_term_build_expr`]. Recursively lowers
1491    /// one node of a term-building merge body, pushing the A2 tuple `@<C>View`
1492    /// `TableInsert` + `(___TreeProof minted)` term-proof + `@UF_S` self-edge
1493    /// side-effects, and returning a pair `(eclass_mergefn, construct_mergefn)`:
1494    ///   * `eclass_mergefn` is the value used as a child of an enclosing
1495    ///     constructor / the final merged value.
1496    ///   * `construct_mergefn` is the RAW (un-canonicalized) `Construct` for this
1497    ///     node — used by the caller to build the `(fn keys construct)` result AST.
1498    ///     (`canon_mergefn` is a no-op in proof mode, so the two are equal for a
1499    ///     constructor node, but we return the construct explicitly to make the
1500    ///     result-AST shape — which wraps the raw minted term — unambiguous.)
1501    ///
1502    /// Mirrors the dropped `@merge_rule` head (`add_term_and_view` with
1503    /// `Justification::Merge`): each minted constructor's view-row proof is
1504    /// `(___Merge "fn" old_pf new_pf (___AstS minted))`, written to the inner
1505    /// view's tuple proof column, to `(___TreeProof minted)`, and as the `@UF_S`
1506    /// self-edge proof. No `Trans`/`Sym`/`Congr` canon-at-creation wrapper is
1507    /// emitted: in proof mode `canon_mergefn` is a no-op (children pass through
1508    /// raw), so the wrapper the rule oracle emits would collapse to no-op refl
1509    /// `Congr`s; the rule-based `@rebuild_rule*` re-canonicalizes the views (and
1510    /// composes the congruence edge proofs) after the merge, exactly as for an
1511    /// ordinary `add_term_and_view` insert.
1512    fn lower_term_build_expr_proof(
1513        &self,
1514        expr: &ResolvedExpr,
1515        side_effects: &mut Vec<egglog_bridge::MergeFn>,
1516        fn_name: &str,
1517        old_pf: &egglog_bridge::MergeFn,
1518        new_pf: &egglog_bridge::MergeFn,
1519    ) -> Result<(egglog_bridge::MergeFn, egglog_bridge::MergeFn), Error> {
1520        use egglog_bridge::MergeFn;
1521        match expr {
1522            // A leaf `old`/`new` refers to the OLD/NEW ECLASS — value COLUMN 0 of
1523            // the A2 tuple view. Use the column-ABSOLUTE `OldCol(0)`/`NewCol(0)`,
1524            // NOT the `self_col`-relative `Old`/`New` (`translate_expr_to_mergefn`
1525            // maps `old` -> `Old`): inside the proof column's merge (`self_col` =
1526            // 1) `Old` would read the PROOF column, not the eclass. Using the
1527            // absolute column makes a `Construct` reusable across both the eclass
1528            // and proof columns (so they mint the SAME id).
1529            GenericExpr::Var(span, resolved_var) => {
1530                let name = resolved_var.name.as_str();
1531                let v = if name == "old" {
1532                    MergeFn::OldCol(0)
1533                } else if name == "new" {
1534                    MergeFn::NewCol(0)
1535                } else {
1536                    return Err(TypeError::Unbound(resolved_var.name.clone(), span.clone()).into());
1537                };
1538                Ok((v.clone(), v))
1539            }
1540            GenericExpr::Lit(..) => {
1541                let v = self.translate_expr_to_mergefn(expr)?;
1542                Ok((v.clone(), v))
1543            }
1544            GenericExpr::Call(_, ResolvedCall::Primitive(_), _) => {
1545                let v = self.translate_expr_to_mergefn(expr)?;
1546                Ok((v.clone(), v))
1547            }
1548            GenericExpr::Call(_, ResolvedCall::Func(f), args) => {
1549                let func = &self.functions[&f.name];
1550                let term_table = func.backend_id;
1551                let out_sort = func.decl.schema.output.clone();
1552                let view_name = self
1553                    .proof_state
1554                    .proof_names
1555                    .view_name
1556                    .get(&*f.name)
1557                    .ok_or_else(|| {
1558                        Error::BackendError(format!(
1559                            "term-build merge (proof): no view table recorded for constructor `{}`",
1560                            f.name
1561                        ))
1562                    })?
1563                    .clone();
1564                let view_table = self.functions[&view_name].backend_id;
1565
1566                // Lower each child first (innermost-first side-effect order). In
1567                // proof mode `canon_mergefn` is a no-op, so children pass through
1568                // their raw value.
1569                let mut child_eclasses = Vec::with_capacity(args.len());
1570                for arg in args.iter() {
1571                    let (child_eclass, _child_construct) = self.lower_term_build_expr_proof(
1572                        arg,
1573                        side_effects,
1574                        fn_name,
1575                        old_pf,
1576                        new_pf,
1577                    )?;
1578                    child_eclasses.push(child_eclass);
1579                }
1580
1581                // Mint the term `(C children..)` (hash-consed, idempotent).
1582                let construct = MergeFn::Construct(term_table, child_eclasses.clone(), vec![]);
1583
1584                // The constructor's view-row proof
1585                // `(___Merge "fn" old_pf new_pf (___AstS construct))`.
1586                let merge_id =
1587                    self.proof_fn_id(&self.proof_state.proof_names.merge_fn_constructor)?;
1588                let ast_name = self
1589                    .proof_state
1590                    .proof_names
1591                    .sort_to_ast_constructor
1592                    .get(out_sort.as_str())
1593                    .ok_or_else(|| {
1594                        Error::BackendError(format!(
1595                            "term-build merge (proof): no AST constructor for sort `{out_sort}`"
1596                        ))
1597                    })?
1598                    .clone();
1599                let ast_id = self.proof_fn_id(&ast_name)?;
1600                let ast = MergeFn::Function(ast_id, vec![construct.clone()]);
1601                let merge_pf =
1602                    self.build_merge_proof(merge_id, fn_name, old_pf.clone(), new_pf.clone(), ast);
1603
1604                // Side-effect: `(set (___TreeProof construct) merge_pf)`.
1605                if let Some(tp_id) = self
1606                    .proof_state
1607                    .proof_func_parent
1608                    .get(out_sort.as_str())
1609                    .and_then(|n| self.functions.get(n))
1610                    .map(|fdef| fdef.backend_id)
1611                {
1612                    side_effects.push(MergeFn::TableInsert(
1613                        tp_id,
1614                        vec![construct.clone(), merge_pf.clone()],
1615                    ));
1616                }
1617
1618                // Side-effect: write the A2 tuple `@<C>View` row
1619                // `(set (@CView children..) (values construct merge_pf))`.
1620                // `TableInsert` runs the view's own
1621                // `Columns([UnionIntoUfWithProof, EclassMinProof])` merge — inline
1622                // congruence with proof composition — on an FD conflict.
1623                let mut view_row = child_eclasses;
1624                view_row.push(construct.clone());
1625                view_row.push(merge_pf.clone());
1626                side_effects.push(MergeFn::TableInsert(view_table, view_row));
1627
1628                // Side-effect: the `@UF_S` self-edge `(set (@UF_S construct
1629                // construct) merge_pf)` — `add_term_and_view`'s `union(fv, fv,
1630                // Proof(merge_pf))`, a self-loop that registers `fv` in the
1631                // union-find with its term proof.
1632                //
1633                //   * RELATIONAL proof path (no `--native-uf`): `@UF_S` (the
1634                //     `uf_parent` table) is a plain `(S S) -> Proof :merge old`
1635                //     function, so the self-edge is a `TableInsert` of `[fv, fv,
1636                //     merge_pf]` — byte-for-byte the `(set (@UF_S fv fv) merge_pf)`
1637                //     row the rule oracle writes.
1638                //   * `--native-uf`: `@UF_Sf` is a `:impl displaced-union-find`
1639                //     function — it cannot be `TableInsert`ed (a `TableAction`
1640                //     rejects a UF table), and `union(fv, fv)` is a genuine no-op
1641                //     for the union-find (same id, no leader change, no onchange
1642                //     proof to compose). The id is registered by the UF the first
1643                //     time it appears as a leader; the term proof is already in
1644                //     `@TreeProof`. So the native-UF self-edge is omitted — it
1645                //     would change nothing.
1646                if !self.proof_state.native_uf
1647                    && let Some(uf_id) = self
1648                        .proof_state
1649                        .uf_parent
1650                        .get(out_sort.as_str())
1651                        .and_then(|n| self.functions.get(n))
1652                        .map(|fdef| fdef.backend_id)
1653                {
1654                    side_effects.push(MergeFn::TableInsert(
1655                        uf_id,
1656                        vec![construct.clone(), construct.clone(), merge_pf],
1657                    ));
1658                }
1659
1660                Ok((construct.clone(), construct))
1661            }
1662            GenericExpr::Call(span, ResolvedCall::Values(_), _) => Err(Error::TypeError(
1663                TypeError::TupleMergeNotValues("<merge>".to_owned(), span.clone()),
1664            )),
1665        }
1666    }

1668    /// DuckDB native VALUE-FOLD `:merge`: register each primitive in `expr` under
1669    /// its sort-specific duck name (`*`->`bigint-mul`, `to-string`->`bigint-to-string`,
1670    /// …) in the duckdb registry, so the in-SQL fold built by
1671    /// [`Self::translate_expr_to_mergefn`] resolves the SAME UDF/op the rule path
1672    /// would. `set_external_func_name` also registers the matching builtin UDF.
1673    /// No-op on any non-DuckDB backend (their merge trees keep the original names,
1674    /// which their host interpreters resolve directly).
1675    fn register_duck_merge_prim_names(&mut self, expr: &ResolvedExpr) {
1676        // Collect the (external_id, duck_name) rewrites under an immutable borrow,
1677        // then apply them with the mutable downcast — avoids borrow conflicts.
1678        let mut renames: Vec<(ExternalFunctionId, String)> = Vec::new();
1679        fn collect(expr: &ResolvedExpr, out: &mut Vec<(ExternalFunctionId, String)>) {
1680            if let GenericExpr::Call(_, ResolvedCall::Primitive(p), args) = expr {
1681                let in0 = p.input().first().map(|s| s.name());
1682                if let Some(duck_name) = duck_prim_rename(p.name(), p.output().name(), in0) {
1683                    out.push((p.external_id(crate::Context::Write), duck_name.to_owned()));
1684                }
1685                for a in args {
1686                    collect(a, out);
1687                }
1688            } else if let GenericExpr::Call(_, _, args) = expr {
1689                for a in args {
1690                    collect(a, out);
1691                }
1692            }
1693        }
1694        collect(expr, &mut renames);
1695        if renames.is_empty() {
1696            return;
1697        }
1698        if let Some(duck) = self
1699            .backend
1700            .as_any_mut()
1701            .downcast_mut::<egglog_bridge_duckdb::EGraph>()
1702        {
1703            for (id, name) in renames {
1704                duck.set_external_func_name(id, name);
1705            }
1706        }
1707    }

1709    fn translate_expr_to_mergefn(
1710        &self,
1711        expr: &ResolvedExpr,
1712    ) -> Result<egglog_bridge::MergeFn, Error> {
1713        match expr {
1714            GenericExpr::Lit(_, literal) => {
1715                // Backend-agnostic: `self.base_values()` resolves on bridge,
1716                // duckdb, feldera, AND flowlog (the value-fold native-merge path
1717                // builds the `MergeFn` tree on flowlog, where `self.bridge()`
1718                // would panic on the downcast).
1719                let val = literal_to_value(self.base_values(), literal);
1720                Ok(egglog_bridge::MergeFn::Const(val))
1721            }
1722            GenericExpr::Var(span, resolved_var) => {
1723                let name = resolved_var.name.as_str();
1724                // Single-output merges use `old`/`new`; tuple-output merges use `old0`, `new0`,
1725                // `old1`, ... to refer to the old/new value of a specific output column.
1726                if name == "old" {
1727                    Ok(egglog_bridge::MergeFn::Old)
1728                } else if name == "new" {
1729                    Ok(egglog_bridge::MergeFn::New)
1730                } else if let Some(i) = name.strip_prefix("old").and_then(|s| s.parse().ok()) {
1731                    Ok(egglog_bridge::MergeFn::OldCol(i))
1732                } else if let Some(i) = name.strip_prefix("new").and_then(|s| s.parse().ok()) {
1733                    Ok(egglog_bridge::MergeFn::NewCol(i))
1734                } else {
1735                    // NB: type-checking should already catch unbound variables here.
1736                    Err(TypeError::Unbound(resolved_var.name.clone(), span.clone()).into())
1737                }
1738            }
1739            GenericExpr::Call(_, ResolvedCall::Func(f), args) => {
1740                let translated_args = args
1741                    .iter()
1742                    .map(|arg| self.translate_expr_to_mergefn(arg))
1743                    .collect::<Result<Vec<_>, _>>()?;
1744                Ok(egglog_bridge::MergeFn::Function(
1745                    self.functions[&f.name].backend_id,
1746                    translated_args,
1747                ))
1748            }
1749            GenericExpr::Call(_, ResolvedCall::Primitive(p), args) => {
1750                let mut translated_args = args
1751                    .iter()
1752                    .map(|arg| self.translate_expr_to_mergefn(arg))
1753                    .collect::<Result<Vec<_>, _>>()?;
1754                if p.name() == "unstable-fn" {
1755                    let Some(GenericExpr::Lit(_, Literal::String(name))) = args.first() else {
1756                        return Err(Error::BackendError(
1757                            "expected string literal after `unstable-fn`".into(),
1758                        ));
1759                    };
1760                    let bridge = self
1761                        .backend
1762                        .as_any()
1763                        .downcast_ref::<egglog_bridge::EGraph>()
1764                        .expect("unstable-fn requires the bridge backend");
1765                    let panic_id = bridge.action_registry().read().unwrap().default_panic_id();
1766                    let resolved = resolve_function_container_target_with_context(
1767                        bridge,
1768                        &self.functions,
1769                        &self.type_info,
1770                        name,
1771                        p,
1772                        panic_id,
1773                    );
1774                    translated_args[0] =
1775                        egglog_bridge::MergeFn::Const(self.backend.base_values().get(resolved));
1776                }
1777                Ok(egglog_bridge::MergeFn::Primitive(
1778                    p.external_id(crate::Context::Write),
1779                    translated_args,
1780                ))
1781            }
1782            // A top-level `(values ...)` tuple merge is decomposed per column in
1783            // `declare_function`; reaching here means a `values` was nested inside another merge
1784            // expression, which is not supported.
1785            GenericExpr::Call(span, ResolvedCall::Values(_), _) => Err(Error::TypeError(
1786                TypeError::TupleMergeNotValues("<merge>".to_owned(), span.clone()),
1787            )),
1788        }
1789    }

1791    fn declare_function(&mut self, decl: &ResolvedFunctionDecl) -> Result<(), Error> {
1792        let get_sort = |name: &String| match self.type_info.get_sort_by_name(name) {
1793            Some(sort) => Ok(sort.clone()),
1794            None => Err(Error::TypeError(TypeError::UndefinedSort(
1795                name.to_owned(),
1796                decl.span.clone(),
1797            ))),
1798        };
1799
1800        let input = decl
1801            .schema
1802            .input
1803            .iter()
1804            .map(get_sort)
1805            .collect::<Result<Vec<_>, _>>()?;
1806        let output = get_sort(&decl.schema.output)?;
1807        let extra_outputs = decl
1808            .schema
1809            .extra_outputs
1810            .iter()
1811            .map(get_sort)
1812            .collect::<Result<Vec<_>, _>>()?;
1813        let num_outputs = 1 + extra_outputs.len();
1814
1815        use egglog_bridge::{DefaultVal, MergeFn};
1816        let (backend_id, can_subsume) = match &decl.impl_kind {
1817            FunctionImpl::Default => {
1818                let can_subsume = match decl.subtype {
1819                    FunctionSubtype::Constructor => true,
1820                    // View tables (functions with term_constructor) need subsumption support
1821                    FunctionSubtype::Custom => decl.term_constructor.is_some(),
1822                };
1823
1824                // Canonicalize-at-creation: the flat UF-index functions
1825                // (`@UF_Sf`) are declared with identity-on-miss lookup semantics
1826                // so the encoder can emit `find_UFold` lookups (lookup-or-self
1827                // against the frozen UF_old table) at constructor-creation time.
1828                // Detected by name against the encoder's `uf_function` map.
1829                // Always-on for term-encoding backends, but skipped in PROOF mode
1830                // (where the encoder emits no `find_UFold` lookup, so the
1831                // Identity default would be inert at best and a behavior change
1832                // at worst — proof mode stays baseline).
1833                let is_uf_index = !self.proof_state.proofs_enabled
1834                    && self
1835                        .proof_state
1836                        .uf_function
1837                        .values()
1838                        .any(|n| n == &*decl.name);
1839                // Native VALUE-FOLD `:merge` on DuckDB: the merge MergeFn tree built
1840                // below by `translate_expr_to_mergefn` carries each primitive's
1841                // *original* (pre-overload) name in the backend registry. DuckDB
1842                // resolves a `:merge` in-SQL (not via a host interpreter), so it
1843                // needs the SAME sort-specific names the rule path emits
1844                // (`*`->`bigint-mul`, `to-string`->`bigint-to-string`, …) and the
1845                // matching UDFs registered. The rule-builder's `rename_prim` hook
1846                // never runs for a native (non-rule-encoded) merge, so do that
1847                // registration here over the stashed merge expr.
1848                if let Some(value_merge) = self
1849                    .proof_state
1850                    .native_value_merge_views
1851                    .get(&*decl.name)
1852                    .cloned()
1853                {
1854                    self.register_duck_merge_prim_names(&value_merge);
1855                }
1856                let backend_id = self.backend.add_table(egglog_bridge::FunctionConfig {
1857                    schema: input
1858                        .iter()
1859                        .chain([&output])
1860                        .chain(extra_outputs.iter())
1861                        .map(|sort| sort.column_ty(&*self.backend))
1862                        .collect(),
1863                    default: if is_uf_index {
1864                        DefaultVal::Identity
1865                    } else {
1866                        match decl.subtype {
1867                            FunctionSubtype::Constructor => DefaultVal::FreshId,
1868                            FunctionSubtype::Custom => DefaultVal::Fail,
1869                        }
1870                    },
1871                    // Native-merge (`--native-merge`): the encoder declares a
1872                    // CONSTRUCTOR's `@<F>View` FD-keyed `(children) -> eclass` as a
1873                    // plain `function` (Custom subtype, no fresh-id minting) but
1874                    // records its name here so the backing table gets the
1875                    // side-effecting congruence merge — congruence done inline at
1876                    // FD-conflict instead of by a self-join rule. Checked first so
1877                    // it overrides the Custom `:merge` placeholder.
1878                    //
1879                    // The merge names the per-sort UF-backed function (`@UF_Sf`)
1880                    // that owns the view's eclass (OUTPUT) column via
1881                    // `MergeFn::UnionIntoUf`, so the FD-conflict congruence union
1882                    // lands in EXACTLY that union-find (not the global `$uf`) —
1883                    // the same view→UF association the uniform
1884                    // `register_native_merge_view` contract records. The native
1885                    // bridge resolves the named UF directly; the dataflow/SQL
1886                    // backends ignore the payload (mapping it to a plain min) and
1887                    // route the union via their host-pass `native_merge_uf`
1888                    // association populated by `register_native_merge_view`. When
1889                    // the output sort has no `@UF_Sf` yet (it always does in
1890                    // native-UF mode, since the sort is declared before its
1891                    // constructors), fall back to the global `UnionId`.
1892                    merge: if self
1893                        .proof_state
1894                        .native_merge_uf_functions
1895                        .contains(&*decl.name)
1896                    {
1897                        // Relational native-merge `@UF_Sf : (S) -> S` (the WHOLE
1898                        // union-find). Attach the NATIVE self-referential `:merge`
1899                        // (recursive parent-union via `TableInsert` into ITS OWN
1900                        // table), overriding the source `:merge (ordering-min old
1901                        // new)`. The function's own backend id is needed for the
1902                        // self `TableInsert`; it is the id the about-to-run
1903                        // `add_table` will return (peeked deterministically). Note:
1904                        // `is_uf_index` above also gives `@UF_Sf` the
1905                        // `DefaultVal::Identity` lookup-or-self default.
1906                        let uf_self_id = self.backend.peek_next_function_id();
1907                        self.build_uf_self_merge(uf_self_id)?
1908                    } else if let Some(term_build_merge) =
1909                        self.proof_state.native_term_build_views.get(&*decl.name)
1910                    {
1911                        // Native TERM-BUILD custom `:merge` (`--native-merge`,
1912                        // bridge): the encoder declared this `@<F>View` FD-keyed
1913                        // `(children) -> eclass` (term) / `(children) -> (eclass,
1914                        // Proof)` (A2 proof) and stashed the term-building merge
1915                        // expr here. Lower it to a `MergeFn::Seq` of `Construct` /
1916                        // `TableInsert` / union variants (PR #933 multi-action
1917                        // merge) so the term build runs NATIVELY in the FD-conflict
1918                        // merge callback, replacing the rule-encoded `@merge_rule` /
1919                        // `@merge_cleanup`. In proof mode the Seq additionally
1920                        // threads the `___Merge` view-proof terms (returned as a
1921                        // `MergeFn::Columns([eclass, proof])`). Checked FIRST (a
1922                        // term-build view is also in `native_merge_views` for the FD
1923                        // set/query/delete forms, but it needs the Seq merge, not
1924                        // the congruence `UnionId`/A2 congruence merge).
1925                        let term_build_merge = term_build_merge.clone();
1926                        // The ORIGINAL custom-function name (`f`): the view decl's
1927                        // `:internal-term-constructor`. It names every `___Merge`
1928                        // proof and resolves the result-AST view term `(f keys …)`.
1929                        let fn_name = decl.term_constructor.clone().ok_or_else(|| {
1930                            Error::BackendError(format!(
1931                                "term-build merge: view `{}` has no term_constructor",
1932                                decl.name
1933                            ))
1934                        })?;
1935                        self.translate_term_build_merge_to_seq(
1936                            &term_build_merge,
1937                            &decl.schema.output,
1938                            &fn_name,
1939                        )?
1940                    } else if let Some(value_merge) =
1941                        self.proof_state.native_value_merge_views.get(&*decl.name)
1942                    {
1943                        // Native VALUE-FOLD custom `:merge` (`--native-merge`,
1944                        // term, non-proof, bridge): the encoder declared this
1945                        // `@<F>View` FD-keyed `(children) -> value` and stashed the
1946                        // function's merge expr here. Lower it to a `MergeFn` tree
1947                        // of `Primitive`/`Const`/`Old`/`New` (the value-fold
1948                        // classifier guarantees no `Func` calls) so the fold runs
1949                        // NATIVELY in the backend's FD-conflict merge callback,
1950                        // replacing the rule-encoded `@merge_rule`/`@merge_cleanup`.
1951                        // Checked FIRST: a value-fold view is also in
1952                        // `native_merge_views` (for the FD set/query/delete forms),
1953                        // but it needs the FOLD merge, not the congruence `UnionId`.
1954                        let value_merge = value_merge.clone();
1955                        self.translate_expr_to_mergefn(&value_merge)?
1956                    } else if self.proof_state.native_merge_views.contains(&*decl.name)
1957                        // A `native_merge_views_coexist()` backend (DuckDB) renders a
1958                        // constructor-congruence view in the BASELINE all-columns shape
1959                        // with a plain `:merge old` (congruence is done by the host
1960                        // pass `emit_native_congruence` in term mode /
1961                        // `emit_native_congruence_proof` in proof mode — which reads the
1962                        // `@<F>ViewProof` side-table and writes the relational `@UF_S`),
1963                        // so it falls through to the baseline merge below in BOTH term
1964                        // and proof mode rather than taking the FD `UnionId` /
1965                        // `UnionIntoUf` / `UnionIntoParentTableWithProof` merge here.
1966                        && !self.backend.native_merge_views_coexist()
1967                    {
1968                        let uf_id = self
1969                            .proof_state
1970                            .uf_function
1971                            .get(&decl.schema.output)
1972                            .and_then(|uf_name| self.functions.get(uf_name))
1973                            .map(|uf_func| uf_func.backend_id);
1974                        if self.proof_state.proofs_enabled
1975                            && !self.backend.supports_tuple_output_views()
1976                        {
1977                            // 2-TABLE PROOF-mode native-merge view (single-output
1978                            // backends, e.g. FlowLog): the eclass view is
1979                            // SINGLE-output `(children) -> eclass`; the per-row view
1980                            // proof lives in a separate `@<F>ViewProof` SIDE-TABLE
1981                            // `(children..., eclass) -> proof`. The merge
1982                            // `UnionIntoParentTableWithProof` reads both colliding
1983                            // rows' proofs from the side-table, composes the same
1984                            // `Trans(larger_pf, Sym(smaller_pf))` edge proof the A2
1985                            // tuple merge does, and writes it into the RELATIONAL
1986                            // proof-UF `@UF_S` (`uf_parent`) — never the displaced
1987                            // `@UF_Sf` (FlowLog has no proof-mode native-UF; the CLI
1988                            // does not force `--native-uf` for this combination). The
1989                            // relational singleparent / path_compress rules then
1990                            // canonicalize the proof-UF, exactly as for the
1991                            // rule-encoded `@congruence_rule`.
1992                            let parent_table = self
1993                                .proof_state
1994                                .uf_parent
1995                                .get(&decl.schema.output)
1996                                .and_then(|uf_name| self.functions.get(uf_name))
1997                                .map(|uf_func| uf_func.backend_id)
1998                                .expect(
1999                                    "2-table native-merge proof view requires the output \
2000                                     sort's @UF_S parent table (relational proof path)",
2001                                );
2002                            // The side-table is declared just before this view in
2003                            // `term_and_view`, so it is already in `self.functions`.
2004                            // The side-table map is keyed by the ORIGINAL function
2005                            // name (`f`), which the view carries as its
2006                            // `:internal-term-constructor` (`term_constructor`).
2007                            let orig_name = decl.term_constructor.clone().ok_or_else(|| {
2008                                Error::BackendError(format!(
2009                                    "2-table native-merge proof view `{}` has no \
2010                                     term_constructor",
2011                                    decl.name
2012                                ))
2013                            })?;
2014                            let side_name = self
2015                                .proof_state
2016                                .proof_names
2017                                .proof_side_table_name
2018                                .get(&orig_name)
2019                                .expect(
2020                                    "2-table native-merge proof view requires its \
2021                                     @<F>ViewProof side-table (declared before the view)",
2022                                )
2023                                .clone();
2024                            let proof_side_table = self.functions[&side_name].backend_id;
2025                            let trans = self.functions
2026                                [&self.proof_state.proof_names.eq_trans_constructor]
2027                                .backend_id;
2028                            let sym = self.functions
2029                                [&self.proof_state.proof_names.eq_sym_constructor]
2030                                .backend_id;
2031                            MergeFn::UnionIntoParentTableWithProof {
2032                                parent_table,
2033                                proof_side_table,
2034                                trans,
2035                                sym,
2036                            }
2037                        } else if self.proof_state.proofs_enabled {
2038                            // A2 PROOF-mode native-merge view: TUPLE output
2039                            // `(children) -> (eclass, proof)` (col0=eclass,
2040                            // col1=proof). The per-column merge does congruence
2041                            // INLINE: `UnionIntoUfWithProof` composes the
2042                            // congruence edge proof `Trans(larger_pf,
2043                            // Sym(smaller_pf))` and writes the 4-column edge row
2044                            // `[larger, smaller, edge_proof, ts]` (exactly what the
2045                            // rule-encoded `@congruence_rule*` would have written via
2046                            // `union(new, old, Proof(Trans(prf1, Sym(prf2))))`) into
2047                            // the proof-mode union-find, returning the surviving
2048                            // (min) eclass; `EclassMinProof` keeps the min eclass's
2049                            // view proof.
2050                            //
2051                            // WHICH 2-key `(S S) -> Proof` table receives the edge
2052                            // depends on whether native-UF is on (both share the
2053                            // `[larger, smaller, proof, ts]` row shape, so only the
2054                            // target id differs):
2055                            //   * `--native-uf`: the per-sort `@UF_Sf`
2056                            //     (`uf_function`) — a provenance-tracking
2057                            //     `displaced-union-find` whose onchange callback
2058                            //     drives the rebuild.
2059                            //   * WITHOUT `--native-uf` (the relational proof path,
2060                            //     e.g. plain `--proofs --native-merge` on the
2061                            //     bridge): the relational `@UF_S` PARENT table
2062                            //     (`uf_parent`), keyed `(S S) -> Proof :merge old`,
2063                            //     which the singleparent / path_compress /
2064                            //     uf_function_index rules then canonicalize. (The
2065                            //     1-key `@UF_Sf` function index is `(S) -> Pair` and
2066                            //     would mis-slot the proof id into its timestamp
2067                            //     column — the source of the sort-order panic.)
2068                            // The proof constructors must exist (declared with the
2069                            // Proof datatype before any constructor's view).
2070                            let uf = if self.proof_state.native_uf {
2071                                uf_id.expect(
2072                                    "native-merge proof view requires the output sort's @UF_Sf \
2073                                     (declared before its constructors in native-UF mode)",

2262    /// Native-merge (`--native-merge`): tell the backend, for each FD-keyed
2263    /// constructor view (`(children) -> eclass`), which union-find its eclass
2264    /// (OUTPUT) column belongs to, so the view's `MergeFn::UnionId` merge can
2265    /// inject the union on an FD conflict. Run before every `(run ...)` (like
2266    /// `register_nativerb_views`) so views declared between runs are registered.
2267    /// The backend records the view -> UF association and uses it in the merge.
2268    fn register_native_merge_views(&mut self) {
2269        // The view->UF backend association only exists to route the FD-conflict
2270        // congruence union into a UF-BACKED `@UF_Sf` (the `--native-uf` path).
2271        // In the RELATIONAL native-merge path (`--native-merge` without
2272        // `--native-uf`) the view's merge is `MergeFn::UnionIntoParentTable`,
2273        // which writes the union edge straight into the relational `@UF_S` parent
2274        // table — fully self-contained, no backend association needed (and
2275        // `register_native_merge_view` would assert a non-existent UF-backed
2276        // function). So there is nothing to register here — UNLESS this is the
2277        // proof-mode COEXIST path (DuckDB `--proofs --native-merge`): there the view
2278        // keeps the all-columns COEXIST shape with a plain `:merge old`, so the
2279        // backend host pass (`emit_native_congruence_proof`) needs the
2280        // proof-congruence config (relational `@UF_S` parent, `@<F>ViewProof`
2281        // side-table, Trans/Sym) registered explicitly. Handle that first.
2282        if self.proof_state.proofs_enabled && self.backend.native_merge_views_coexist() {
2283            self.register_native_merge_proof_views();
2284            return;
2285        }
2286        if !self.proof_state.native_uf {
2287            return;
2288        }
2289        let mut to_register: Vec<(String, egglog_bridge::FunctionId, String)> = Vec::new();
2290        for func in self.functions.values() {

2293                continue;
2294            }
2295            // Value-fold views have a primitive (non-eq-sort) output and no UF to
2296            // union into; their fold merge is fully self-contained in the backend
2297            // table. Skip the view->UF registration for them.
2298            if self
2299                .proof_state
2300                .native_value_merge_views
2301                .contains_key(&view_name)
2302            {
2303                continue;
2304            }
2305            if self
2306                .proof_state
2307                .native_merge_registered
2308                .contains(&view_name)
2309            {
2310                continue;
2311            }
2312            // The eclass column's sort owns the UF to union into. For the FD-shaped
2313            // view `(children) -> eclass` (collapse-at-insert backends) the eclass is
2314            // the view's OUTPUT. For a `native_merge_views_coexist()` backend (DuckDB)
2315            // the view keeps the BASELINE all-columns Unit-relation shape `(children...,
2316            // eclass) -> Unit`, so the eclass is the LAST INPUT column (output is Unit).
2317            let eclass_sort =
2318                if self.backend.native_merge_views_coexist() && !self.proof_state.proofs_enabled {

2320                        .schema
2321                        .input
2322                        .last()
2323                        .cloned()
2324                        .unwrap_or_else(|| func.decl.schema.output.clone())
2325                } else {
2326                    func.decl.schema.output.clone()

2337            self.proof_state.native_merge_registered.insert(view_name);
2338        }

2725    /// Runs a ruleset for an iteration.
2726    ///
2727    /// This applies every match it finds (under semi-naive).
2728    /// See [`EGraph::step_rules_with_scheduler`] for more fine-grained control.
2729    ///
2730    /// This will return an error if an egglog primitive returns None in an action.
2731    pub fn step_rules(&mut self, ruleset: &str) -> Result<RunReport, Error> {
2732        fn collect(
2733            ruleset: &str,
2734            rulesets: &IndexMap<String, Ruleset>,
2735            ids: &mut Vec<egglog_bridge::RuleId>,
2736            names: &mut Vec<String>,
2737        ) {
2738            match &rulesets[ruleset] {
2739                Ruleset::Rules(rules) => {
2740                    for (n, (_, id)) in rules.iter() {
2741                        ids.push(*id);
2742                        names.push(n.clone());
2743                    }
2744                }
2745                Ruleset::Combined(sub_rulesets) => {
2746                    for sub_ruleset in sub_rulesets {
2747                        collect(sub_ruleset, rulesets, ids, names);
2748                    }
2749                }
2750            }
2751        }
2752
2753        let mut rule_ids = Vec::new();
2754        let mut rule_names = Vec::new();
2755        collect(ruleset, &self.rulesets, &mut rule_ids, &mut rule_names);
2756
2757        // Seminaive-encoding hook: bump next_ts before the iteration if the
2758        // program defines it. No-op for programs that don't use the encoding.
2759        let new_ts = self.bump_next_ts_global();
2760
2761        let iteration_report = self
2762            .backend
2763            .run_rules(&rule_ids)
2764            .map_err(|e| Error::BackendError(e.to_string()))?;
2765
2766        // After the iteration, set last_run_at_<src> to the bumped value for
2767        // every source rule that ran. The source name is the rule name with
2768        // any `@<digits>` variant suffix stripped.
2769        if let Some(ts) = new_ts {
2770            self.update_last_run_at_globals(&rule_names, ts);
2771        }
2772
2773        Ok(RunReport::singleton(ruleset, iteration_report))
2774    }

2813    fn add_rule(&mut self, rule: ast::ResolvedRule) -> Result<String, Error> {
2814        // Disable union_to_set optimization in proof or term encoding mode, since
2815        // it expects only `union` on constructors (not set).
2816        let core_rule = rule.to_canonicalized_core_rule(
2817            &self.type_info,
2818            &mut self.parser.symbol_gen,
2819            self.proof_state.original_typechecking.is_none(),
2820        )?;
2821        let (query, actions) = (&core_rule.body, &core_rule.head);
2822
2823        // The `:naive` rule option opts a single rule out of seminaive
2824        // evaluation. This widens primitive-context selection from
2825        // Pure/Write to Read/Full, so primitives that read or write the
2826        // database can run inside this rule.
2827        let seminaive = self.seminaive && !rule.naive;
2828        // The `:unsafe-seminaive` rule option decouples evaluation from
2829        // primitive context: the rule is still evaluated seminaive-ly
2830        // (delta matching, via `new_rule(seminaive)` below), but its
2831        // query/action are compiled with the permissive Read/Full
2832        // contexts so the RHS can perform arbitrary database reads
2833        // (read-primitives and, together with the typecheck skip,
2834        // function-table lookups). Unsound in general — see the field
2835        // docs on `GenericRule::unsafe_seminaive`.
2836        let context_seminaive = seminaive && !rule.unsafe_seminaive;
2837        // The `:no-decomp` rule option (and the global `--no-decomp`
2838        // flag) skips tree-decomposition in query planning, forcing
2839        // the single-bag fast path.
2840        let no_decomp = self.no_decomp || rule.no_decomp;
2841
2842        // Fast-rebuild (`--fast-rebuild`, term mode): if this is a relational
2843        // rebuild rule whose view atom must be excluded from seminaive
2844        // delta-focus, resolve the recorded view function to its backend table
2845        // id now (before the `&mut self.backend` borrow) so we can pass it to
2846        // the rule builder below. Dropping the view focus drops the
2847        // always-empty `δview ⋈ uf_old` term, leaving only `view ⋈ δuf`.
2848        let focus_exclude_id = self
2849            .proof_state
2850            .rebuild_view_exclude
2851            .get(&rule.name)
2852            .and_then(|view_name| self.functions.get(view_name))
2853            .map(|f| f.backend_id);
2854
2855        let rule_id = {
2856            // Construct the trait-routed rule builder. The lifetime
2857            // separately borrows `&mut self.backend` for the builder
2858            // and `&self.backend` for column-ty / pool reads — these
2859            // overlap, so we route the read-only borrow through a raw
2860            // pointer reborrow inside an unsafe block. Both pointers
2861            // are derived from the same `Box<dyn Backend>` and are
2862            // never aliased mutably while the builder is alive: the
2863            // builder only borrows `&mut Self`'s internal state, never
2864            // `BaseValuePool` mutably.
2865            let backend_ptr: *const dyn egglog_backend_trait::Backend = &*self.backend;
2866            let mut rb = self.backend.new_rule(&rule.name, seminaive);
2867            rb.set_no_decomp(no_decomp);
2868            if let Some(view_id) = focus_exclude_id {
2869                rb.set_focus_exclude_table(view_id);
2870            }
2871            let mut translator = BackendRule::new(
2872                rb,
2873                // SAFETY: the rule builder's internal `&mut Backend`
2874                // borrow does not touch the `BaseValuePool` / column-
2875                // ty lookups, so the disjoint reborrow is sound for
2876                // the duration of `query`/`actions`/`build`.
2877                unsafe { &*backend_ptr },
2878                &self.functions,
2879                &self.type_info,
2880                context_seminaive,
2881            );
2882            translator.query(query, false);
2883            translator.actions(actions)?;
2884            translator.build()
2885        };
2886
2887        if let Some(rules) = self.rulesets.get_mut(&rule.ruleset) {
2888            match rules {
2889                Ruleset::Rules(rules) => {
2890                    match rules.entry(rule.name.clone()) {
2891                        indexmap::map::Entry::Occupied(_) => {
2892                            let name = rule.name;
2893                            panic!("Rule '{name}' was already present")
2894                        }
2895                        indexmap::map::Entry::Vacant(e) => e.insert((core_rule, rule_id)),
2896                    };
2897                    Ok(rule.name)
2898                }
2899                Ruleset::Combined(_) => Err(Error::CombinedRulesetError(rule.ruleset, rule.span)),
2900            }
2901        } else {
2902            Err(Error::NoSuchRuleset(rule.ruleset, rule.span))
2903        }
2904    }

2993        let span = expr.span();

2995        // Resolve against the *pre-term-encoding* typechecker. Term
2996        // encoding lifts `Action::Expr(...)` into UF/View table
2997        // updates and drops the return value (literals become a bare
2998        // string that's never re-parsed), so routing eval_expr
2999        // through it loses the very thing we want to evaluate.

3001        // original_typechecking (or self.type_info if it's absent)

3004        // backend, which still carries the user-facing function

3040    /// Rewrite the arguments of every `get-size!` primitive call inside `expr`
3041    /// to count the canonical `@<F>View` tables instead of the monotonic
3042    /// hash-cons TERM tables, matching `instrument_get_size` in the term encoder.
3043    /// Recurses into all sub-expressions.
3044    ///
3045    /// - explicit-filter form `(get-size! "F" ...)`: each user-table name `"F"`
3046    ///   with a view becomes `"@FView"`; names without a view (e.g. pass-through
3047    ///   globals) are left as-is.
3048    /// - no-argument `(get-size!)` (count-all): the runtime primitive sums all
3049    ///   non-`@` tables, which under term encoding are the term tables (they
3050    ///   over-count). Replace it with an explicit filter listing every view name
3051    ///   plus the pass-through globals — the canonical egraph size — so the
3052    ///   `@`-view tables are counted despite the usual `@` exclusion.

3061                // backend also counts).

3063                    .values()
3064                    .cloned()

3070                    .map(|n| ResolvedExpr::Lit(span.clone(), Literal::String(n)))

3072            } else {

3083        if let ResolvedExpr::Call(_, _, args) = expr {

3087        }

3182    fn check_facts(&mut self, span: &Span, facts: &[ResolvedFact]) -> Result<(), Error> {
3183        // Refresh the `get-size!` size snapshot the registry-free wrapper reads
3184        // on backends without an `ActionRegistry` (flowlog/feldera). `check_facts`
3185        // is the `:until` evaluation point (called once per `run` iteration via
3186        // `run_rules`), so the snapshot tracks the current table sizes each time
3187        // the `:until` condition (e.g. `(<= 4000 (get-size!))`) is checked. No-op
3188        // on backends with an `ActionRegistry` (bridge reads sizes live).
3189        self.refresh_get_size_snapshot();
3190        let fresh_name = self.parser.symbol_gen.fresh("check_facts");
3191        let fresh_ruleset = self.parser.symbol_gen.fresh("check_facts_ruleset");
3192        let rule = ast::ResolvedRule {
3193            span: span.clone(),
3194            head: ResolvedActions::default(),
3195            body: facts.to_vec(),
3196            name: fresh_name.clone(),
3197            ruleset: fresh_ruleset.clone(),
3198            unsafe_seminaive: false,
3199            // `:naive` so the body is typechecked/compiled in the Read/Full
3200            // primitive contexts, not Pure/Write — required for a read-primitive
3201            // like `get-size!` (a `ReadPrim`, valid in Read/Full only) to resolve
3202            // in a `:until (… (get-size!))` condition. A one-shot existence check
3203            // matches the whole DB anyway (no seminaive delta to ride).
3204            naive: true,
3205            no_decomp: false,
3206        };
3207        let core_rule = rule.to_canonicalized_core_rule(
3208            &self.type_info,
3209            &mut self.parser.symbol_gen,
3210            self.proof_state.original_typechecking.is_none(),
3211        )?;
3212        let query = core_rule.body;
3213
3214        // Backend-agnostic check: build the body via the trait, then
3215        // ask the rule builder whether it matches. The DuckDB impl
3216        // runs `SELECT … LIMIT 1` against the compiled body; the
3217        // bridge backend falls back to the side-channel sentinel
3218        // pattern (see below).
3219        let is_duck = self
3220            .backend
3221            .as_any()
3222            .downcast_ref::<egglog_bridge_duckdb::EGraph>()
3223            .is_some();
3224
3225        if is_duck {
3226            let backend_ptr: *const dyn egglog_backend_trait::Backend = &*self.backend;
3227            let rb = self.backend.new_rule("check_facts", false);
3228            let mut translator = BackendRule::new(
3229                rb,
3230                // SAFETY: see `add_rule` for the disjoint-reborrow rationale.
3231                unsafe { &*backend_ptr },
3232                &self.functions,
3233                &self.type_info,
3234                false,
3235            );
3236            translator.query(&query, true);
3237            let matched = translator
3238                .into_rb()
3239                .build_check()
3240                .map_err(|e| Error::BackendError(format!("check_facts: {e}")))?;
3241            if !matched {
3242                return Err(Error::CheckError(
3243                    facts.iter().map(|f| f.clone().make_unresolved()).collect(),
3244                    span.clone(),
3245                ));
3246            }
3247            return Ok(());
3248        }
3249
3250        let ext_sc = egglog_bridge::SideChannel::default();
3251        let ext_sc_ref = ext_sc.clone();
3252        let ext_id = self
3253            .backend
3254            .register_external_func(Box::new(make_external_func(move |_, _| {
3255                *ext_sc_ref.lock().unwrap() = Some(());
3256                Some(Value::new_const(0))
3257            })));
3258
3259        let id = {
3260            let backend_ptr: *const dyn egglog_backend_trait::Backend = &*self.backend;
3261            let rb = self.backend.new_rule("check_facts", false);
3262            let mut translator = BackendRule::new(
3263                rb,
3264                // SAFETY: see `add_rule` for the disjoint-reborrow rationale.
3265                unsafe { &*backend_ptr },
3266                &self.functions,
3267                &self.type_info,
3268                false,
3269            );
3270            translator.query(&query, true);
3271            translator.rb.call_external_func(
3272                ext_id,
3273                &[],
3274                egglog_bridge::ColumnTy::Id,
3275                Box::new(|| "this function will never panic".to_string()),
3276            );
3277            translator.build()
3278        };
3279        let run_result = self.backend.run_rules(&[id]);
3280        self.backend.free_rule(id);
3281        self.backend.free_external_func(ext_id);
3282        run_result.map_err(|e| Error::BackendError(e.to_string()))?;
3283
3284        let ext_sc_val = ext_sc.lock().unwrap().take();
3285        let matched = matches!(ext_sc_val, Some(()));
3286
3287        if !matched {
3288            Err(Error::CheckError(
3289                facts.iter().map(|f| f.clone().make_unresolved()).collect(),
3290                span.clone(),
3291            ))
3292        } else {
3293            Ok(())
3294        }
3295    }

3575    fn input_file(&mut self, func_name: &str, file: String) -> Result<(), Error> {
3576        let function_type = self
3577            .type_info
3578            .get_func_type(func_name)
3579            .unwrap_or_else(|| panic!("Unrecognized function name {func_name}"));
3580        let (func_backend_id, func_schema_input, func_schema_output) = {
3581            let func = self.functions.get(func_name).unwrap();
3582            (
3583                func.backend_id,
3584                func.schema.input.clone(),
3585                func.schema.output.clone(),
3586            )
3587        };
3588
3589        let mut filename = self.fact_directory.clone().unwrap_or_default();
3590        filename.push(file.as_str());
3591
3592        // check that the function uses supported types
3593
3594        for t in &func_schema_input {
3595            match t.name() {
3596                "i64" | "f64" | "String" => {}
3597                s => panic!("Unsupported type {s} for input"),
3598            }
3599        }
3600
3601        if function_type.subtype != FunctionSubtype::Constructor {
3602            match func_schema_output.name() {
3603                "i64" | "String" | "Unit" => {}
3604                s => panic!("Unsupported type {s} for input"),
3605            }
3606        }
3607
3608        log::info!("Opening file '{filename:?}'...");
3609        let mut f = File::open(filename).unwrap();
3610        let mut contents = String::new();
3611        f.read_to_string(&mut contents).unwrap();
3612
3613        // Can also do a row-major Vec<Value>
3614        let mut parsed_contents: Vec<Vec<Value>> = Vec::with_capacity(contents.lines().count());
3615
3616        let mut row_schema = func_schema_input;
3617        if function_type.subtype == FunctionSubtype::Custom {
3618            row_schema.push(func_schema_output);
3619        }

3621        log::debug!("{row_schema:?}");
3622
3623        // Use the trait-level base value pool so this works on both
3624        // the bridge backend and the DuckDB backend. `pool_get<T>`
3625        // takes a `&dyn BaseValuePool` and converts a typed value
3626        // into the runtime `Value` representation; both backends
3627        // implement the trait, so neither needs a downcast.
3628        use egglog_backend_trait::pool_get;
3629        let pool = self.backend.base_value_pool();
3630        let unit_val = pool_get::<()>(pool, ());
3631
3632        for line in contents.lines() {
3633            let mut it = line.split('\t').map(|s| s.trim());
3634
3635            let mut row: Vec<Value> = Vec::with_capacity(row_schema.len());
3636
3637            for sort in row_schema.iter() {
3638                if let Some(raw) = it.next() {
3639                    let val = match sort.name() {
3640                        "i64" => {
3641                            if let Ok(i) = raw.parse::<i64>() {
3642                                pool_get::<i64>(self.backend.base_value_pool(), i)
3643                            } else {
3644                                return Err(Error::InputFileFormatError(file));
3645                            }
3646                        }
3647                        "f64" => {
3648                            if let Ok(f) = raw.parse::<f64>() {
3649                                pool_get::<F>(
3650                                    self.backend.base_value_pool(),
3651                                    core_relations::Boxed::new(f.into()),
3652                                )
3653                            } else {
3654                                return Err(Error::InputFileFormatError(file));
3655                            }
3656                        }
3657                        "String" => {
3658                            pool_get::<S>(self.backend.base_value_pool(), raw.to_string().into())
3659                        }
3660                        "Unit" => unit_val,
3661                        _ => panic!("Unreachable"),
3662                    };
3663                    row.push(val);
3664                } else {
3665                    break;
3666                }
3667            }
3668
3669            if row.is_empty() {
3670                continue;
3671            }
3672
3673            if row.len() != row_schema.len() || it.next().is_some() {
3674                return Err(Error::InputFileFormatError(file));
3675            }
3676
3677            parsed_contents.push(row);
3678        }
3679
3680        log::debug!("Successfully loaded file.");
3681
3682        let num_facts = parsed_contents.len();
3683
3684        if function_type.subtype != FunctionSubtype::Constructor {
3685            self.backend.insert_rows(func_backend_id, &parsed_contents);
3686        } else {
3687            self.backend
3688                .lookup_constructor_rows(func_backend_id, &parsed_contents);
3689        }
3690
3691        self.backend.flush_updates();
3692
3693        log::info!("Read {num_facts} facts into {func_name} from '{file}'.");
3694        Ok(())
3695    }

3755    /// Compile the given facts as a rule body, check if any row
3756    /// matches, and return the boolean answer. Used by
3757    /// [`prelude::query`] on the duckdb backend to answer
3758    /// `:until`-style existence queries from `run-schedule` without
3759    /// the bridge-only `rust_rule` machinery.
3760    ///
3761    /// Unlike [`Self::check_facts`] (private, returns
3762    /// `Err(CheckError)` on a non-match), this returns the boolean
3763    /// directly so the caller can decide what to do with it.
3764    pub fn run_check_facts_any(&mut self, facts: Vec<Fact>) -> Result<bool, Error> {
3765        // Refresh the row-count snapshot that the duckdb `get-size!`
3766        // UDF reads from. The snapshot lives in the duckdb EGraph
3767        // (out of reach for the bridge backend), so down-cast and
3768        // call its public refresh helper. The bridge backend handles
3769        // `get-size!` via the regular ExternalFunction path, which
3770        // reads `ExecutionState::table_ids()` live — no snapshot
3771        // needed there.
3772        if let Some(duck) = self
3773            .backend
3774            .as_any()
3775            .downcast_ref::<egglog_bridge_duckdb::EGraph>()
3776        {
3777            duck.refresh_table_sizes()
3778                .map_err(|e| Error::BackendError(format!("refresh_table_sizes: {e}")))?;
3779        }
3780
3781        // Route through the *full* resolve pipeline (including term
3782        // encoding when enabled). The facts may reference user-level
3783        // function names like `bad-merge?`; under term encoding those
3784        // get rewritten to view-table references (`@bad-merge?View`)
3785        // which is the schema the duckdb backend actually has. Using
3786        // `resolve_command_before_proofs` here previously bypassed
3787        // term encoding and produced body atoms with the wrong
3788        // arity at duck rule-build time.
3789        let fresh_name = self.parser.symbol_gen.fresh("query_check");
3790        let fresh_ruleset = self.parser.symbol_gen.fresh("query_check_ruleset");
3791        let rule = ast::Rule {
3792            span: span!(),
3793            head: ast::Actions::default(),
3794            body: facts,
3795            name: fresh_name.clone(),
3796            ruleset: fresh_ruleset.clone(),
3797            unsafe_seminaive: false,
3798            naive: false,
3799            no_decomp: false,
3800        };
3801        let cmd = ast::Command::Rule { rule };
3802        let resolved = self.resolve_command(cmd)?;
3803        // Extract the resolved rule's body. After term encoding,
3804        // there may be multiple commands (rule + auxiliary
3805        // function/insert declarations); we only want the rule.
3806        let mut resolved_body: Option<Vec<ResolvedFact>> = None;
3807        for r in resolved.desugared {
3808            if let ResolvedNCommand::NormRule { rule } = r {
3809                resolved_body = Some(rule.body.clone());
3810                break;
3811            }
3812        }
3813        let resolved_body = resolved_body.ok_or_else(|| {
3814            Error::BackendError("run_check_facts_any: resolver did not yield a rule".into())
3815        })?;
3816
3817        // Same backend-agnostic shape as `check_facts`'s duckdb
3818        // branch, but split out into a returns-bool form.
3819        let core_rule = ast::ResolvedRule {
3820            span: span!(),
3821            head: ResolvedActions::default(),
3822            body: resolved_body,
3823            name: fresh_name,
3824            ruleset: fresh_ruleset,
3825            unsafe_seminaive: false,
3826            naive: false,
3827            no_decomp: false,
3828        }
3829        .to_canonicalized_core_rule(
3830            &self.type_info,
3831            &mut self.parser.symbol_gen,
3832            self.proof_state.original_typechecking.is_none(),
3833        )?;
3834        let query = core_rule.body;
3835
3836        let backend_ptr: *const dyn egglog_backend_trait::Backend = &*self.backend;
3837        let rb = self.backend.new_rule("query_check", false);
3838        let mut translator = BackendRule::new(
3839            rb,
3840            // SAFETY: see `add_rule` for the disjoint-reborrow rationale.
3841            unsafe { &*backend_ptr },
3842            &self.functions,
3843            &self.type_info,
3844            false,
3845        );
3846        translator.query(&query, true);
3847        let matched = translator
3848            .into_rb()
3849            .build_check()
3850            .map_err(|e| Error::BackendError(format!("run_check_facts_any: {e}")))?;
3851        Ok(matched)
3852    }

3854    fn resolve_command_before_proofs(
3855        &mut self,
3856        command: Command,
3857    ) -> Result<Vec<ResolvedNCommand>, Error> {
3858        let desugared = desugar_command(command, &mut self.parser, self.proof_state.proof_testing)?;
3859        if let Some(original_typechecking) = self.proof_state.original_typechecking.as_mut() {
3860            // Typecheck using the original egraph
3861            // TODO this is ugly- we don't need an entire e-graph just for type information.
3862            let typechecked = original_typechecking.typecheck_program(&desugared)?;
3863
3864            // This proof-encoding gate intentionally runs on the *user*
3865            // program, before global removal (see `resolve_command`). Global
3866            // removal injects `:unsafe-seminaive` rules (for `Custom` globals
3867            // referenced in a rule head); gating here keeps those internal
3868            // rules exempt from the `:unsafe-seminaive` rejection below.
3869            for command in &typechecked {
3870                if let Err(reason) = command_supports_proof_encoding(
3871                    &command.to_command(),
3872                    &self.type_info,
3873                    self.proof_state.proofs_enabled,
3874                ) {
3875                    let command_text = format!("{}", command.to_command());
3876                    return Err(Error::UnsupportedProofCommand {
3877                        command: command_text,
3878                        reason,
3879                    });
3880                }
3881            }
3882
3883            Ok(proof_form(typechecked, &mut self.parser.symbol_gen))
3884        } else {
3885            let mut typechecked = self.typecheck_program(&desugared)?;
3886
3887            typechecked = remove_globals::remove_globals(
3888                typechecked,
3889                // Native backend: lower every global to a 0-arg `Custom`
3890                // function (no constructor lifting).
3891                false,
3892            );

3900    /// Desugars, typechecks, and removes globals from a single [`Command`].
3901    /// Leverages previous type information in the [`EGraph`] to do so, adding new type information.
3902    /// When will_run is true, adds to `desugared_commands_run_so_far`, which is used for proof checking.
3903    fn resolve_command(&mut self, command: Command) -> Result<ResolvedNCommands, Error> {
3904        let resolved_before_proofs = self.resolve_command_before_proofs(command)?;
3905
3906        // Add term encoding when it is enabled
3907        if self.proof_state.original_typechecking.is_none() {
3908            Ok(ResolvedNCommands {
3909                desugared: resolved_before_proofs,
3910                desugared_before_proofs: vec![],
3911            })
3912        } else {
3913            // Remove globals for actual execution (but NOT from
3914            // desugared_commands). Term/proof encoding lifts eq-sort
3915            // globals to constructors so they get a real eclass; non-eq-sort
3916            // globals become 0-arg `Custom` functions that the encoder
3917            // leaves as plain key-value stores.
3918            let typechecked_no_globals =
3919                remove_globals::remove_globals(resolved_before_proofs.clone(), true);
3920            for command in &typechecked_no_globals {
3921                self.names.check_shadowing(command)?;
3922            }
3923
3924            let term_encoding_added =
3925                ProofInstrumentor::add_term_encoding(self, typechecked_no_globals);
3926            let mut new_typechecked = vec![];
3927            for new_cmd in term_encoding_added {
3928                let desugared =
3929                    desugar_command(new_cmd, &mut self.parser, self.proof_state.proof_testing)?;
3930                for cmd in &desugared {
3931                    log::debug!("Desugared term encoding: {}", cmd.to_command());
3932                }
3933
3934                // Now typecheck using self, adding term type information.
3935                // The term encoder's intermediate top-level `let`s (e.g.
3936                // `(let @v (bigint 3))`) are removed here so no globals
3937                // survive into execution.
3938                let desugared_typechecked = self.typecheck_program(&desugared)?;
3939                let desugared_typechecked =
3940                    remove_globals::remove_globals(desugared_typechecked, false);
3941
3942                new_typechecked.extend(desugared_typechecked);
3943            }
3944            Ok(ResolvedNCommands {
3945                desugared: new_typechecked,
3946                desugared_before_proofs: resolved_before_proofs,
3947            })
3948        }
3949    }

3951    /// Returns `Ok(())` iff every command in `input` passes the
3952    /// proof-encoding gate, matching exactly what the proof/term-encoding
3953    /// pipeline checks (see the gate in `resolve_command_before_proofs`).
3954    ///
3955    /// The gate must be evaluated on the typechecked program BEFORE global
3956    /// removal: global removal injects `:unsafe-seminaive` rules (for `Custom`
3957    /// globals referenced in a rule head), and the gate's `UnsafeSeminaive`
3958    /// arm would otherwise reject those internal rules — a false negative the
3959    /// real pipeline never hits, because it gates the user program first
3960    /// (`resolve_command_before_proofs`, proof branch) and only then removes
3961    /// globals. This is the supportability probe used by `file_supports_proofs`;
3962    /// it does not run term encoding (only the gate verdict is needed).
3963    pub(crate) fn check_program_proof_gate(
3964        &mut self,
3965        filename: Option<String>,
3966        input: &str,
3967    ) -> Result<(), Error> {
3968        let parsed = self.parser.get_program_from_string(filename, input)?;
3969        for before_expanded_command in parsed {
3970            let macro_expanded = self.command_macros.apply(
3971                before_expanded_command,
3972                &mut self.parser.symbol_gen,
3973                &self.type_info,
3974            )?;
3975            for command in macro_expanded {
3976                if let Command::Include(span, file) = &command {
3977                    let s = std::fs::read_to_string(file)
3978                        .map_err(|e| Error::IoError(file.clone().into(), e, span.clone()))?;
3979                    self.check_program_proof_gate(Some(file.clone()), &s)?;
3980                    continue;
3981                }
3982                let desugared =
3983                    desugar_command(command, &mut self.parser, self.proof_state.proof_testing)?;
3984                // Typecheck (building up type info for later commands), then run
3985                // the gate on the pre-global-removal form — identical to the
3986                // pipeline gate. `typecheck_program` mutates `self.type_info`,
3987                // so subsequent commands see earlier declarations.
3988                let typechecked = self.typecheck_program(&desugared)?;
3989                for command in &typechecked {
3990                    if let Err(reason) = command_supports_proof_encoding(
3991                        &command.to_command(),
3992                        &self.type_info,
3993                        // The probe answers "are proofs supported", so apply the
3994                        // full (proofs-enabled) set of checks.
3995                        true,
3996                    ) {
3997                        return Err(Error::UnsupportedProofCommand {
3998                            command: format!("{}", command.to_command()),
3999                            reason,
4000                        });
4001                    }
4002                }
4003            }
4004        }
4005        Ok(())
4006    }

4010    fn process_program_internal(
4011        &mut self,
4012        program: Vec<Command>,
4013        run_commands: bool,
4014    ) -> Result<ResolvedNCommandsWithOutput, Error> {
4015        let mut outputs = Vec::new();
4016        let mut desugared_before_proofs = Vec::new();
4017        let mut desugared = Vec::new();
4018
4019        for before_expanded_command in program {
4020            // First do user-provided macro expansion for this command,
4021            // which may rely on type information from previous commands.
4022            let macro_expanded = self.command_macros.apply(
4023                before_expanded_command,
4024                &mut self.parser.symbol_gen,
4025                &self.type_info,
4026            )?;

4028            for command in macro_expanded {

4030                if let Command::Include(span, file) = &command {

4032                        .map_err(|e| Error::IoError(file.clone().into(), e, span.clone()))?;

4034                        .parser

4036                    // run program internal on these include commands
4037                    let resolved = self.process_program_internal(included_program, run_commands)?;

4039                    desugared.extend(resolved.resolved);
4040                    desugared_before_proofs.extend(resolved.resolved_before_proofs);
4041                } else {
4042                    let resolved = self.resolve_command(command)?;
4043                    if run_commands && self.are_proofs_enabled() {
4044                        self.proof_check_program
4045                            .extend(resolved.desugared_before_proofs.clone());
4046                    }
4047
4048                    desugared_before_proofs.extend(resolved.desugared_before_proofs);
4049                    desugared.extend(resolved.desugared.clone());
4050
4051                    for processed in resolved.desugared {
4052                        // even in desugar mode we still run push and pop
4053                        if run_commands
4054                            || matches!(
4055                                processed,
4056                                ResolvedNCommand::Push(_) | ResolvedNCommand::Pop(_, _)
4057                            )
4058                        {
4059                            let result = self.run_command(processed)?;
4060                            outputs.extend(result);
4061                        }
4062                    }
4063                }
4064            }
4065        }

4067        Ok(ResolvedNCommandsWithOutput {
4068            outputs,
4069            resolved_before_proofs: desugared_before_proofs,

4308    type_info: &TypeInfo,
4309    name: &str,
4310    primitive: &core::SpecializedPrimitive,
4311    panic_id: ExternalFunctionId,
4312) -> ResolvedFunction {
4313    let Some(target_function) = type_info
4314        .get_sorts::<FunctionSort>()
4315        .into_iter()
4316        .find(|function| function.name() == primitive.output().name())
4317    else {
4318        panic!(
4319            "`unstable-fn` output sort `{}` is not a function sort",
4320            primitive.output().name()
4321        );
4322    };
4323
4324    let partial_arcsorts: Vec<_> = primitive.input().iter().skip(1).cloned().collect();
4325    let remaining_inputs = target_function.inputs();
4326    let output = target_function.output();
4327
4328    let id = if let Some(func) = functions.get(name) {
4329        let func_type = type_info
4330            .get_func_type(name)
4331            .unwrap_or_else(|| panic!("No resolution for {name:?}"));
4332        let expected_inputs = partial_arcsorts
4333            .iter()
4334            .chain(remaining_inputs)
4335            .collect::<Vec<_>>();
4336        let inputs_match = func_type.input.len() == expected_inputs.len()
4337            && func_type
4338                .input
4339                .iter()
4340                .zip(&expected_inputs)
4341                .all(|(actual, expected)| actual.name() == expected.name());
4342        if !inputs_match || func_type.output.name() != output.name() {
4343            let expected_input_names = expected_inputs
4344                .iter()
4345                .map(|sort| sort.name())
4346                .collect::<Vec<_>>()
4347                .join(", ");
4348            let actual_input_names = func_type
4349                .input
4350                .iter()
4351                .map(|sort| sort.name())
4352                .collect::<Vec<_>>()
4353                .join(", ");
4354            panic!(
4355                "function container lookup for `{name}` expected ({}) -> {}, found ({}) -> {}",
4356                expected_input_names,
4357                output.name(),
4358                actual_input_names,
4359                func_type.output.name(),
4360            );
4361        }
4362
4363        let action = egglog_bridge::TableAction::new(backend, func.backend_id);
4364        match func_type.subtype {
4365            ast::FunctionSubtype::Constructor => ResolvedFunctionId::Constructor(action),
4366            ast::FunctionSubtype::Custom => ResolvedFunctionId::Function(action),
4367        }
4368    } else if let Some(primitives) = type_info.get_prims(name) {
4369        let signature: Vec<_> = partial_arcsorts
4370            .iter()
4371            .chain(remaining_inputs)
4372            .chain(once(&output))
4373            .cloned()
4374            .collect();
4375        let candidates: Vec<_> = primitives
4376            .iter()
4377            .filter(|primitive| primitive.accept(&signature, type_info))
4378            .collect();
4379        let context_ids = enum_map::EnumMap::from_fn(|runtime_ctx| {
4380            let mut ids = candidates
4381                .iter()
4382                .filter_map(|primitive| primitive.context_ids[runtime_ctx]);
4383            match (ids.next(), ids.next()) {
4384                (None, _) => None,
4385                (Some(id), None) => Some(id),
4386                (Some(_), Some(_)) => panic!(
4387                    "Ambiguous primitive resolution for {name:?} in unstable-fn context {runtime_ctx:?}"
4388                ),
4389            }
4390        });
4391        if !context_ids.iter().any(|(_, id)| id.is_some()) {
4392            let (output_sort, input_sorts) = signature
4393                .split_last()
4394                .expect("primitive signature should include an output sort");
4395            let input_names = input_sorts
4396                .iter()
4397                .map(|sort| sort.name())
4398                .collect::<Vec<_>>()
4399                .join(", ");
4400            panic!(
4401                "no primitive overload matched expected signature for {name:?}: ({}) -> {}; \
4402                 context ids: {context_ids:?}",
4403                input_names,
4404                output_sort.name(),
4405            );
4406        }
4407        ResolvedFunctionId::Primitive { context_ids }
4408    } else {
4409        panic!("No resolution for {name:?}");
4410    };
4411
4412    ResolvedFunction {
4413        id,
4414        partial_arcsorts,
4415        name: name.to_owned(),
4416        panic_id,
4417    }
4418}

4501    fn prim(
4502        &mut self,
4503        prim: &core::SpecializedPrimitive,
4504        args: &[core::ResolvedAtomTerm],
4505        ctx: crate::Context,
4506    ) -> (ExternalFunctionId, Vec<QueryEntry>, ColumnTy) {
4507        // The typechecker has already checked that this primitive is
4508        // valid in `ctx`; pick the runtime id that stamps the same ctx
4509        // onto the state wrapper when invoked.
4510        //
4511        // The baked `SpecializedPrimitive` captured its external ids at
4512        // typecheck time. For a `:canon-prim` placeholder those ids are
4513        // later rebound to the real union-find canonicalizer (in
4514        // `declare_uf_function`) once the UF function is declared — but the
4515        // already-resolved rule still carries the stale placeholder id. Refresh
4516        // from the current `type_info` so the rebound id is used. This is a
4517        // no-op for every other primitive (its `type_info` id is unchanged
4518        // since typecheck).
4519        let resolved_id = self
4520            .type_info
4521            .current_primitive_external_id(prim.name(), ctx)
4522            .unwrap_or_else(|| prim.external_id(ctx));
4523
4524        let mut qe_args = self.args(args);
4525
4526        if prim.name() == "unstable-fn" {
4527            // `unstable-fn` is a bridge-only feature: it lives on the
4528            // ResolvedFunction container and requires a `TableAction`
4529            // that the trait surface doesn't expose. DuckDB programs
4530            // never reach this branch because `FunctionSort`
4531            // containers are gated out at upstream typechecking.
4532            let bridge = self
4533                .backend
4534                .as_any()
4535                .downcast_ref::<egglog_bridge::EGraph>()
4536                .expect("unstable-fn requires the bridge backend");
4537            let core::ResolvedAtomTerm::Literal(_, Literal::String(ref name)) = args[0] else {
4538                panic!("expected string literal after `unstable-fn`")
4539            };
4540            // Upstream's context-aware resolution (validates the
4541            // wrapped function's signature and bakes a panic id used by
4542            // `FunctionContainer::apply` to error on misapplication).
4543            // `self.rb` is the trait object; `new_panic` is on
4544            // `RuleBuilderOps`. We reuse the already-downcast `bridge`
4545            // for the (immutable) resolution + constant since the
4546            // function takes `&egglog_bridge::EGraph`.
4547            let panic_id = self.rb.new_panic(format!(
4548                "unstable-fn over `{name}` was applied in a context where its wrapped \
4549                 function is not valid for this call site, if in a rule, add :naive."
4550            ));
4551            let resolved = resolve_function_container_target_with_context(
4552                bridge,
4553                self.functions,
4554                self.type_info,
4555                name,
4556                prim,
4557                panic_id,
4558            );
4559            qe_args[0] = bridge.base_value_constant(resolved);
4560        }
4561
4562        // Type-disambiguate built-in primitive names for the duckdb
4563        // backend. Several egglog primitives share a name across
4564        // sorts (`^` is XOR for i64 / POWER for f64; `/` is integer-
4565        // div for i64 / float-div for f64; `+` is string concat for
4566        // String / numeric add for i64/f64). DuckDB's SQL operators
4567        // pick one meaning per symbol, and the duck rule-builder
4568        // dispatches by name, so without an override the wrong SQL
4569        // gets emitted. Each `SpecializedPrimitive` carries a unique
4570        // `external_id`; we rename per-id so the duck side maps each
4571        // id to the SQL form for its specific type.
4572        //
4573        // Hardcoded for now — see also `compile.rs::prim_sql`. A more
4574        // principled design (per-primitive SQL emitter registered via
4575        // the macro) is deferred.
4576        let pname = prim.name();
4577        let pout = prim.output().name();
4578        let in0 = prim.input().first().map(|s| s.name());
4579        // For sort-overloaded primitives, route the duckdb backend's
4580        // rule-builder to a sort-specific duck name. compile.rs's
4581        // `prim_sql` maps these to either native SQL ops or — for
4582        // BigRat operations that have no SQL equivalent — to per-op
4583        // UDFs registered on demand via
4584        // `EGraph::set_external_func_name` → `register_builtin_prim_udf`.
4585        if let Some(name) = duck_prim_rename(pname, pout, in0) {
4586            self.rb.rename_prim(prim.external_id(ctx), name.to_owned());
4587        }
4588
4589        (resolved_id, qe_args, prim.output().column_ty(self.backend))
4590    }

4600        for atom in &query.atoms {
4601            match &atom.head {
4602                ResolvedCall::Func(f) => {
4603                    let f = self.func(f);
4604                    let args = self.args(&atom.args);
4605                    let is_subsumed = match include_subsumed {
4606                        true => None,
4607                        false => Some(false),

4609                    self.rb.query_table(f, &args, is_subsumed).unwrap();

4611                ResolvedCall::Primitive(p) => {
4612                    let ctx = self.query_context();
4613                    let (p, args, ty) = self.prim(p, &atom.args, ctx);
4614                    self.rb.query_prim(p, &args, ty).unwrap()

4616                ResolvedCall::Values(_) => {
4617                    unreachable!("`values` is lowered to the underlying function atom before query")
4618                }

4623    fn actions(&mut self, actions: &core::ResolvedCoreActions) -> Result<(), Error> {
4624        for action in &actions.0 {
4625            match action {
4626                core::GenericCoreAction::Let(span, v, f, args) => {
4627                    let v = core::GenericAtomTerm::Var(span.clone(), v.clone());
4628                    let y = match f {
4629                        ResolvedCall::Func(f) => {
4630                            let name = f.name.clone();
4631                            let f = self.func(f);
4632                            let args = self.args(args);
4633                            let span = span.clone();
4634                            self.rb.lookup(
4635                                f,
4636                                &args,
4637                                Box::new(move || {
4638                                    format!("{span}: lookup of function {name} failed")
4639                                }),
4640                            )
4641                        }
4642                        ResolvedCall::Primitive(p) => {
4643                            let name = p.name().to_owned();
4644                            let ctx = self.action_context();
4645                            let (p, args, ty) = self.prim(p, args, ctx);
4646                            let span = span.clone();

4648                                p,
4649                                &args,
4650                                ty,

4652                                    format!("{span}: call of primitive {name} failed")
4653                                }),
4654                            )
4655                        }
4656                        ResolvedCall::Values(_) => {
4657                            panic!("`values` cannot be bound as a single value")
4658                        }
4659                    };
4660                    self.entries.insert(v, y);
4661                }
4662                core::GenericCoreAction::LetAtomTerm(span, v, x) => {
4663                    let v = core::GenericAtomTerm::Var(span.clone(), v.clone());
4664                    let x = self.entry(x);
4665                    self.entries.insert(v, x);
4666                }
4667                core::GenericCoreAction::Set(_, f, xs, ys) => match f {
4668                    ResolvedCall::Primitive(..) => panic!("runtime primitive set!"),
4669                    ResolvedCall::Values(..) => panic!("`values` is not a settable function"),
4670                    ResolvedCall::Func(f) => {
4671                        let f = self.func(f);
4672                        let args = self.args(xs.iter().chain(ys.iter()));
4673                        self.rb.set(f, &args)
4674                    }
4675                },
4676                core::GenericCoreAction::Change(span, change, f, args) => match f {
4677                    ResolvedCall::Primitive(..) => panic!("runtime primitive change!"),
4678                    ResolvedCall::Values(..) => panic!("`values` is not a changeable function"),
4679                    ResolvedCall::Func(f) => {
4680                        let name = f.name.clone();
4681                        let can_subsume = self.functions[&f.name].can_subsume;
4682                        let f = self.func(f);
4683                        let args = self.args(args);
4684                        match change {
4685                            Change::Delete => self.rb.remove(f, &args),
4686                            Change::Subsume if can_subsume => {
4687                                self.rb.subsume(f, &args).map_err(|e| {
4688                                    Error::BackendError(format!("subsume failed: {e}"))
4689                                })?;
4690                            }
4691                            Change::Subsume => {
4692                                return Err(Error::SubsumeMergeError(name, span.clone()));
4693                            }
4694                        }
4695                    }
4696                },
4697                core::GenericCoreAction::Union(_, x, y) => {
4698                    let x = self.entry(x);
4699                    let y = self.entry(y);
4700                    self.rb.union(x, y)
4701                }
4702                core::GenericCoreAction::Panic(_, message) => self.rb.panic(message.clone()),
4703            }
4704        }
4705        Ok(())
4706    }

4717/// Trait-routed literal-to-entry helper. Constructs a typed
4718/// `QueryEntry::Const` via the backend's `BaseValuePool` so the

4757/// Type-disambiguate a built-in primitive name for the DuckDB backend. Several
4758/// egglog primitives share a name across sorts (`^` is XOR for i64 / POWER for
4759/// f64; `/` is integer-div for i64 / float-div for f64; `+` is string-concat for
4760/// String / numeric add for i64/f64); BigInt/BigRat ops operate on pool handles
4761/// with no SQL equivalent. DuckDB's SQL operators / UDFs need one meaning per
4762/// symbol, so the duck side maps each `(name, output sort, first input sort)` to a
4763/// sort-specific name that `compile::prim_sql` translates. Returns `None` for a
4764/// name that needs no rewrite (the bare name already translates).
4765///
4766/// Shared by the rule-builder's `prim` lowering (which renames the primitive's
4767/// `external_id` for SQL emission inside a compiled rule) AND the native
4768/// VALUE-FOLD `:merge` MergeFn-tree builder (`translate_expr_to_mergefn`), so the
4769/// in-SQL fold resolves the SAME sort-specific UDFs the rule path would.
4770fn duck_prim_rename(pname: &str, pout: &str, in0: Option<&str>) -> Option<&'static str> {
4771    match (pname, pout, in0) {
4772        ("^", "i64", _) => Some("i64-xor"),
4773        ("/", "i64", _) => Some("int-div"),
4774        ("+", "String", _) => Some("string-concat"),
4775        // `to-string` overloads — duck routes each to a sort-specific UDF that
4776        // formats the input and interns the result string handle.
4777        ("to-string", "String", Some("i64")) => Some("i64-to-string"),
4778        ("to-string", "String", Some("f64")) => Some("f64-to-string"),
4779        ("to-string", "String", Some("BigInt")) => Some("bigint-to-string"),
4780        // BigInt arithmetic overloads: `+`, `*`, ... on BigInt have no SQL
4781        // equivalent (the operands are pool handles), so route each to a per-op
4782        // UDF that unwraps/operates/interns. Match on the BigInt *output* type to
4783        // disambiguate from the i64/f64 overloads of the same op name.
4784        ("+", "BigInt", _) => Some("bigint-add"),
4785        ("-", "BigInt", _) => Some("bigint-sub"),
4786        ("*", "BigInt", _) => Some("bigint-mul"),
4787        ("/", "BigInt", _) => Some("bigint-div"),
4788        ("%", "BigInt", _) => Some("bigint-rem"),
4789        ("&", "BigInt", _) => Some("bigint-and"),
4790        ("|", "BigInt", _) => Some("bigint-or"),
4791        ("^", "BigInt", _) => Some("bigint-xor"),
4792        ("min", "BigInt", _) => Some("bigint-min"),
4793        ("max", "BigInt", _) => Some("bigint-max"),
4794        // BigRat overloads: dispatch by op name and arg sort. Comparisons return
4795        // Unit; arithmetic returns BigRat; numer/denom return BigInt; to-f64
4796        // returns f64. We use the BigRat *argument* type to disambiguate from
4797        // i64/f64 overloads of the same op name.
4798        ("+", "BigRat", _) => Some("bigrat-add"),
4799        ("-", "BigRat", _) => Some("bigrat-sub"),
4800        ("*", "BigRat", _) => Some("bigrat-mul"),
4801        ("/", "BigRat", _) => Some("bigrat-div"),
4802        ("min", "BigRat", _) => Some("bigrat-min"),
4803        ("max", "BigRat", _) => Some("bigrat-max"),
4804        ("pow", "BigRat", _) => Some("bigrat-pow"),
4805        ("neg", "BigRat", _) => Some("bigrat-neg"),
4806        ("abs", "BigRat", _) => Some("bigrat-abs"),
4807        ("floor", "BigRat", _) => Some("bigrat-floor"),
4808        ("ceil", "BigRat", _) => Some("bigrat-ceil"),
4809        ("round", "BigRat", _) => Some("bigrat-round"),
4810        ("sqrt", "BigRat", _) => Some("bigrat-sqrt"),
4811        ("log", "BigRat", _) => Some("bigrat-log"),
4812        ("cbrt", "BigRat", _) => Some("bigrat-cbrt"),
4813        ("<", "Unit", Some("BigRat")) => Some("bigrat-lt"),
4814        (">", "Unit", Some("BigRat")) => Some("bigrat-gt"),
4815        ("<=", "Unit", Some("BigRat")) => Some("bigrat-le"),
4816        (">=", "Unit", Some("BigRat")) => Some("bigrat-ge"),
4817        ("numer", _, Some("BigRat")) => Some("bigrat-numer"),
4818        ("denom", _, Some("BigRat")) => Some("bigrat-denom"),
4819        ("to-f64", _, Some("BigRat")) => Some("bigrat-to-f64"),
4820        _ => None,
4821    }
4822}
