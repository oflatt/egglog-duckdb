1//! Host-side rule interpreter for the FlowLog backend (Milestone 3).
2//!
3//! This is the FlowLog analog of the Feldera backend's Milestone-3 host
4//! interpreter, and the **correctness oracle** for the engine path: it runs
5//! real `.egg` programs (general multi-atom bodies, primitives, head actions)
6//! to per-function tuple-count parity with the reference backend.
7//!
8//! One `run_rules` call = **one bounded egglog iteration**. The interpreter:
9//!
10//! 1. snapshots the relation mirror (the read view for this iteration — all
11//!    rules match against the same pre-iteration state, egglog's semi-naive
12//!    "match the old database, then apply" model for a single hop);
13//! 2. for each rule, runs a **nested-loop join** over the body's table atoms,
14//!    threading a variable→value binding environment, evaluating primitive body
15//!    atoms (`BodyOp::Prim`) against the embedded `Database` (guards like `!=`
16//!    prune; value-computing prims extend the env);
17//! 3. for every surviving binding, executes the head ops in order — `set` /
18//!    `remove` / `subsume` writes, RHS `lookup` (eq-sort constructor: create on
19//!    miss), RHS primitive `call`, `union`, `panic`;
20//! 4. applies all collected writes/removes to the mirror and resolves
21//!    functional-dependency conflicts per each touched function's merge mode.
22//!
23//! ## The engine split (Milestone 3 mandate)
24//!
25//! The relational table-atom join for the **DD-eligible** rule class (table
26//! atoms + `!=` guards, arity/width ≤ the engine row cap) runs on the
27//! Differential-Dataflow engine in a subprocess (see [`crate::dd_join`]); the
28//! primitive tail + head actions are applied HOST-side here, exactly mirroring
29//! Feldera M4's "table join on the engine, prim tail host-side" split. Rules
30//! outside the eligible class fall back entirely to this host nested-loop
31//! interpreter (the oracle).
32//!
33//! Primitives are invoked through `Database::with_execution_state`, so they see
34//! the same interned base `Value`s the frontend created — giving the FlowLog
35//! backend bit-for-bit value parity with the reference backend (no
36//! `eval_prim` trait change — flowlog keeps its zero-trait-change posture).
37

47/// Binding environment: variable id → bound `u32` value.
48pub(crate) type Env = HashMap<u32, u32>;

50/// A pending write to apply after all matches are computed.
51enum Write {
52    /// Insert/overwrite a full row.
53    Set(FunctionId, Row),
54    /// Retract by key (the slots address inputs for a function, whole row for a
55    /// relation).
56    Remove(FunctionId, Vec<u32>),
57}

59/// Run one bounded iteration of `rule_idxs` against the egraph. Returns whether
60/// the mirror changed.
61///
62/// **Seminaive** (mirrors Feldera M5): each rule fires only against bindings
63/// that touch at least one row NEW to that rule (`mirror[f] \ seen[r][f]`), so a
64/// rule that already matched a set of facts does NOT re-fire — which is what
65/// makes bounded `(saturate …)` loops converge instead of oscillating against
66/// rule-driven cleanup/retraction. The relational table-atom join still runs on
67/// the Differential-Dataflow engine for the DD-eligible class (now fed the
68/// delta per atom-occurrence); the primitive tail + head actions are host-side.
69pub fn run_iteration(eg: &mut EGraph, rule_idxs: &[usize]) -> Result<bool> {
70    // Snapshot the read view: rules match against the pre-iteration mirror.
71    //
72    // The snapshot shares each function's row set by `Rc` rather than
73    // deep-cloning every row: this is O(#functions), not O(state). Mutations to
74    // the mirror this call (head writes, hash-cons in `lookup_or_create`, merge
75    // resolution) go through `Rc::make_mut`, which copy-on-writes only the
76    // functions actually changed while this snapshot is alive — so `read` keeps
77    // the start-of-call contents and rules all match the pre-iteration state,
78    // exactly as before, but without paying for a full-mirror clone every call.
79    let read: HashMap<FunctionId, std::rc::Rc<HashSet<Row>>> = eg
80        .mirror
81        .iter()
82        .map(|(f, set)| (*f, std::rc::Rc::clone(set)))
83        .collect();
84
85    // Snapshot the fresh-id counter: any hash-cons (`lookup_or_create`) this
86    // call advances it, the O(1) signal that a new term row was created.
87    let next_id_at_start = eg.next_id;
88
89    let mut writes: Vec<Write> = Vec::new();
90    let mut touched: HashSet<FunctionId> = HashSet::new();
91    // Iteration-scoped `key -> output` index for `lookup_or_create` (eq-sort
92    // constructor hash-cons). Built lazily per function so repeated lookups in
93    // one iteration are O(1) instead of rescanning the growing mirror each time.
94    let mut lookup_index: HashMap<FunctionId, HashMap<Box<[u32]>, u32>> = HashMap::new();
95
96    // Collect each rule's index + IR up front (clone to avoid borrow conflicts
97    // while we also mutate the db / mirror via lookups). The index keys the
98    // per-rule seminaive `seen` snapshot.
99    let rules: Vec<(usize, RuleIr)> = rule_idxs
100        .iter()
101        .filter_map(|&i| eg.rules.get(i).and_then(|r| r.clone()).map(|r| (i, r)))
102        .collect();
103
104    // Shared start-of-iteration snapshots, built lazily once per function and
105    // reused (by refcount) for every rule's `seen` advance this call.
106    let mut shared_snapshot: HashMap<FunctionId, std::rc::Rc<HashSet<Row>>> = HashMap::new();
107    for (idx, rule) in &rules {
108        // The table relations this rule's body reads, and the seminaive delta of
109        // each (rows present now that this rule has not yet matched against).
110        let body_funcs: Vec<FunctionId> = rule
111            .body
112            .iter()
113            .filter_map(|op| match op {
114                BodyOp::Atom(a) => Some(a.func),
115                BodyOp::Prim { .. } => None,
116            })
117            .collect();
118        let delta: HashMap<FunctionId, HashSet<Row>> = body_funcs
119            .iter()
120            .map(|&f| {
121                let empty: HashSet<Row> = HashSet::new();
122                let cur = read.get(&f).map(|v| &**v).unwrap_or(&empty);
123                let seen = eg.seen.get(idx).and_then(|m| m.get(&f));
124                // Fast path: if this rule's `seen[f]` snapshot is the SAME `Rc`
125                // as the current read view, `f` is unchanged since the rule last
126                // ran ⇒ empty delta, no O(state) scan (the proven fed-diff trick).
127                if let (Some(cur_rc), Some(seen_rc)) = (read.get(&f), seen) {
128                    if std::rc::Rc::ptr_eq(cur_rc, seen_rc) {
129                        return (f, HashSet::new());
130                    }
131                }
132                let d: HashSet<Row> = cur
133                    .iter()
134                    .filter(|r| seen.map(|s| !s.contains(*r)).unwrap_or(true))
135                    .cloned()
136                    .collect();
137                (f, d)
138            })
139            .collect();
140
141        // The seminaive binding set: union over table-atom positions of the join
142        // with exactly that atom restricted to the delta. Each variant runs on
143        // the DD engine (if eligible) or the host nested-loop fallback.
144        let bindings = seminaive_bindings(eg, &read, &delta, rule)?;
145
146        for mut env in bindings {
147            apply_head(
148                eg,
149                &rule.head,
150                &mut env,
151                &mut writes,
152                &mut touched,
153                &mut lookup_index,
154            )?;
155        }
156
157        // Advance this rule's seen snapshot to the START-OF-ITERATION read view
158        // (NOT the post-write mirror): the rule has now matched everything
159        // currently present. A row deleted+readded later reappears in the delta.
160        // Build each touched function's snapshot once (shared by Rc across all
161        // rules in this call), then bump the refcount into this rule's seen.
162        for &f in &body_funcs {
163            if !shared_snapshot.contains_key(&f) {
164                // `read[f]` is already the start-of-call `Rc<HashSet<Row>>`;
165                // reuse its handle directly as this rule's seen snapshot (no
166                // rebuild). This also makes the next-iteration `Rc::ptr_eq`
167                // delta short-circuit fire when `f` is untouched.
168                let cur = read
169                    .get(&f)
170                    .map(std::rc::Rc::clone)
171                    .unwrap_or_else(|| std::rc::Rc::new(HashSet::new()));
172                shared_snapshot.insert(f, cur);
173            }
174        }
175        let entry = eg.seen.entry(*idx).or_default();
176        for &f in &body_funcs {
177            entry.insert(f, std::rc::Rc::clone(&shared_snapshot[&f]));
178        }
179    }
180
181    // Apply collected writes to the mirror.
182    //
183    // Removes are BATCHED per function: applying each `Write::Remove` with its
184    // own `set.retain` scan is O(|removes| · |state|) — quadratic, and the
185    // dominant per-iteration cost during rebuild (which retracts many rows at
186    // once). We collect all retraction keys per function into a hash set, then
187    // do a SINGLE `retain` pass per touched function: O(|state|) total. Removes
188    // are applied FIRST (batched), then Sets — preserving the term encoder's
189    // `(@uf)` "delete old leader, set new leader" delete-then-set ordering.
190    //
191    // `changed` is computed INCREMENTALLY as writes land (O(delta)), not via a
192    // full before/after content compare of every function (O(state)). A
193    // hash-cons in `lookup_or_create` always allocates a fresh id, so any term
194    // created this call advances `next_id` — that alone is a real mirror change.
195    let mut changed = eg.next_id != next_id_at_start;
196    let mut removes_by_func: HashMap<FunctionId, (usize, HashSet<Box<[u32]>>)> = HashMap::new();
197    let mut sets: Vec<(FunctionId, Row)> = Vec::new();
198    for w in writes {
199        match w {
200            Write::Set(f, row) => {
201                sets.push((f, row));
202            }
203            Write::Remove(f, key) => {
204                let entry = removes_by_func
205                    .entry(f)
206                    .or_insert_with(|| (key.len(), HashSet::new()));
207                entry.1.insert(key.into_boxed_slice());
208            }
209        }
210    }
211    for (f, (keylen, keys)) in removes_by_func {
212        if let Some(set) = eg.mirror.get_mut(&f) {
213            let before_len = set.len();
214            std::rc::Rc::make_mut(set).retain(|row| {
215                let k: Box<[u32]> = (0..keylen).map(|i| row_col(row, i)).collect();
216                !keys.contains(&k)
217            });
218            // A retraction that actually removed a row is a real change.
219            changed |= set.len() != before_len;
220        }
221    }
222    // Per-function set of input keys touched by a `set` this call — the only
223    // keys that can newly conflict and need merge resolution.
224    let mut touched_keys: HashMap<FunctionId, HashSet<Vec<u32>>> = HashMap::new();
225    for (f, row) in sets {
226        let inputs_len = eg.info(f).arity.saturating_sub(1);
227        let key: Vec<u32> = (0..inputs_len).map(|i| row_col(&row, i)).collect();
228        // `insert` returns true iff the row was genuinely new (set a row that
229        // already exists ⇒ no content change, so don't flag `changed`).
230        let inserted = std::rc::Rc::make_mut(eg.mirror.entry(f).or_default()).insert(row);
231        changed |= inserted;
232        touched_keys.entry(f).or_default().insert(key);
233    }
234
235    // Resolve FD conflicts on every function a head action wrote to (a `set`
236    // can introduce two rows sharing a key that must merge per the merge mode).
237    // Resolution is INCREMENTAL — only the keys whose rows were touched this
238    // call can newly conflict, and `resolve_merge` reports whether it actually
239    // collapsed/changed any row (another real change source for the loop).
240    let empty_keys: HashSet<Vec<u32>> = HashSet::new();
241    for &f in &touched {
242        let keys = touched_keys.get(&f).unwrap_or(&empty_keys);
243        changed |= eg.resolve_merge(f, keys);
244    }
245
246    // `change_detect` is now folded incrementally into apply/merge (O(delta));
247    // there is no separate full before/after compare.
248    Ok(changed)
249}

251/// Extend each binding env by matching `op` against the read view (table atom)
252/// or evaluating it (primitive). Returns the new list of envs.
253pub(crate) fn step_body(
254    eg: &mut EGraph,
255    read: &HashMap<FunctionId, std::rc::Rc<HashSet<Row>>>,
256    op: &BodyOp,
257    envs: Vec<Env>,
258) -> Result<Vec<Env>> {
259    match op {
260        BodyOp::Atom(atom) => {
261            let empty: HashSet<Row> = HashSet::new();
262            let rows = read.get(&atom.func).map(|v| &**v).unwrap_or(&empty);
263            Ok(step_atom(&atom.slots, rows, envs))
264        }
265        BodyOp::Prim { id, args, ret } => {
266            let mut out = Vec::new();
267            for env in envs {
268                let resolved: Option<Vec<Value>> = args
269                    .iter()
270                    .map(|s| slot_lookup(s, &|v| env.get(&v).copied()).map(Value::new))
271                    .collect();
272                let Some(argv) = resolved else { continue };
273                let result = eg.eval_prim_internal(*id, &argv);
274                let Some(result) = result else {
275                    // Primitive failed (e.g. `!=` of equal args) — prune.
276                    continue;
277                };
278                match ret {
279                    Slot::Var(v) => {
280                        let mut next = env.clone();
281                        match next.get(v) {
282                            Some(&existing) if existing != result.rep() => continue,
283                            _ => {
284                                next.insert(*v, result.rep());
285                            }
286                        }
287                        out.push(next);
288                    }
289                    Slot::Const(c) => {
290                        if *c == result.rep() {
291                            out.push(env);
292                        }
293                    }
294                }
295            }
296            Ok(out)
297        }
298    }
299}

301/// Match a single table atom (`slots`) against `rows` under each incoming env,
302/// returning the extended envs. Shared by the full-scan body step
303/// (`step_body`) and the seminaive delta match (`seminaive_bindings`): when
304/// `rows` is a relation's delta slice, this performs the delta-restricted join
305/// of that atom occurrence.
306pub(crate) fn step_atom(slots: &[Slot], rows: &HashSet<Row>, envs: Vec<Env>) -> Vec<Env> {
307    // JOIN KEY: columns whose slot is a variable already bound in the incoming
308    // envs (the same var must agree between env and row). If non-empty,
309    // hash-join on it instead of a full cartesian scan.
310    let bound_in_env = |v: u32| envs.first().map(|e| e.contains_key(&v)).unwrap_or(false);
311    let key_cols: Vec<usize> = slots
312        .iter()
313        .enumerate()
314        .filter_map(|(i, s)| match s {
315            Slot::Var(v) if bound_in_env(*v) => Some(i),
316            _ => None,
317        })
318        .collect();
319    let key_vars: Vec<u32> = key_cols
320        .iter()
321        .map(|&i| match &slots[i] {
322            Slot::Var(v) => *v,
323            _ => unreachable!(),
324        })
325        .collect();
326
327    let mut out = Vec::new();
328    if key_cols.is_empty() {
329        for env in &envs {
330            for row in rows {
331                if let Some(next) = match_atom(slots, row, env) {
332                    out.push(next);
333                }
334            }
335        }
336    } else {
337        let mut index: HashMap<Vec<u32>, Vec<&Row>> = HashMap::new();
338        for row in rows {
339            let key: Vec<u32> = key_cols.iter().map(|&i| row[i]).collect();
340            index.entry(key).or_default().push(row);
341        }
342        for env in &envs {
343            let key: Vec<u32> = key_vars.iter().map(|v| env[v]).collect();
344            if let Some(cands) = index.get(&key) {
345                for row in cands {
346                    if let Some(next) = match_atom(slots, row, env) {
347                        out.push(next);
348                    }
349                }
350            }
351        }
352    }
353    out
354}

356/// Compute the **seminaive** binding set for one rule: the union, over each
357/// table-atom position `j`, of the body join with atom `j` restricted to its
358/// relation's *delta* (rows new to this rule) and every other atom ranging over
359/// the full relation. This is exactly the set of bindings touching at least one
360/// newly-derived fact — egglog's seminaive semantics.
361///
362/// For the DD-eligible class the relational join runs on the Differential-
363/// Dataflow engine (`dd_join`); otherwise the host nested-loop fallback (the
364/// oracle) runs it. Body primitives (`!=` guards, value-computing prims) are
365/// applied host-side over the produced bindings. Bindings are deduplicated
366/// across the `j`-variants (a binding new in two positions appears twice).
367fn seminaive_bindings(
368    eg: &mut EGraph,
369    read: &HashMap<FunctionId, std::rc::Rc<HashSet<Row>>>,
370    delta: &HashMap<FunctionId, HashSet<Row>>,
371    rule: &RuleIr,
372) -> Result<Vec<Env>> {
373    // Positions of the table atoms within the body op list.
374    let atom_positions: Vec<usize> = rule
375        .body
376        .iter()
377        .enumerate()
378        .filter_map(|(i, op)| matches!(op, BodyOp::Atom(_)).then_some(i))
379        .collect();
380
381    if atom_positions.is_empty() {
382        // Prim-only / constant body: no table atoms to delta over. Evaluate it
383        // once (bounded + idempotent — rare, not on the saturation hot path).
384        eg.host_rule_runs += 1;
385        let mut envs: Vec<Env> = vec![Env::new()];
386        for op in &rule.body {
387            envs = step_body(eg, read, op, envs)?;
388            if envs.is_empty() {
389                break;
390            }
391        }
392        return Ok(envs);
393    }
394
395    // If NO body relation has any delta rows, the rule cannot produce a new
396    // binding this iteration — skip it entirely (the seminaive win that stops
397    // the oscillation).
398    let any_delta = atom_positions.iter().any(|&p| {
399        if let BodyOp::Atom(a) = &rule.body[p] {
400            delta.get(&a.func).map(|d| !d.is_empty()).unwrap_or(false)
401        } else {
402            false
403        }
404    });
405    if !any_delta {
406        return Ok(Vec::new());
407    }
408
409    // Plan the DD-eligible whole-body join once; reused per delta-atom variant.
410    let plan = if eg.dd_enabled {
411        dd_join::plan_join(eg, rule)
412    } else {
413        None
414    };
415
416    let mut seen_bindings: HashSet<Vec<(u32, u32)>> = HashSet::new();
417    let mut out: Vec<Env> = Vec::new();
418
419    for (atom_ord, &delta_pos) in atom_positions.iter().enumerate() {
420        // Skip a variant whose delta atom has no new rows (it would just rescan
421        // the full relation pointlessly).
422        let delta_func = match &rule.body[delta_pos] {
423            BodyOp::Atom(a) => a.func,
424            _ => unreachable!(),
425        };
426        if delta.get(&delta_func).map(|d| d.is_empty()).unwrap_or(true) {
427            continue;
428        }
429
430        let variant_envs = if let Some(plan) = &plan {
431            // DD path: run the relational join on the engine with this atom
432            // occurrence fed the delta rows and the others the full relation.
433            eg.dd_rule_runs += 1;
434            let bindings = dd_join::run_join_seminaive(eg, plan, atom_ord, read, delta)?;
435            let var_order = plan.var_order().to_vec();
436            let mut envs: Vec<Env> = Vec::new();
437            for bind in bindings {
438                let mut env: Env = Env::new();
439                for (i, &v) in var_order.iter().enumerate() {
440                    env.insert(v, bind[i]);
441                }
442                // Re-run body primitives host-side (value prims bind fresh vars;
443                // `!=` re-filters identically). Table atoms were handled on DD.
444                let mut es: Vec<Env> = vec![env];
445                for op in &rule.body {
446                    if let BodyOp::Prim { .. } = op {
447                        es = step_body(eg, read, op, es)?;
448                    }
449                }
450                envs.extend(es);
451            }
452            envs
453        } else {
454            // Host fallback: nested-loop body scan where the atom at `delta_pos`
455            // ranges over the delta rows, all others over the full read view.
456            eg.host_rule_runs += 1;
457            let empty: HashSet<Row> = HashSet::new();
458            let mut envs: Vec<Env> = vec![Env::new()];
459            for (pos, op) in rule.body.iter().enumerate() {
460                match op {
461                    BodyOp::Atom(atom) => {
462                        // Borrow the row set directly (no per-call clone): the
463                        // delta atom ranges over its delta set, the rest over
464                        // `read[f]`.
465                        let rows: &HashSet<Row> = if pos == delta_pos {
466                            delta.get(&atom.func).unwrap_or(&empty)
467                        } else {
468                            read.get(&atom.func).map(|v| &**v).unwrap_or(&empty)
469                        };
470                        envs = step_atom(&atom.slots, rows, envs);
471                    }
472                    BodyOp::Prim { .. } => {
473                        envs = step_body(eg, read, op, envs)?;
474                    }
475                }
476                if envs.is_empty() {
477                    break;
478                }
479            }
480            envs
481        };
482
483        // Deduplicate across the delta-position variants.
484        for env in variant_envs {
485            let mut key: Vec<(u32, u32)> = env.iter().map(|(&k, &v)| (k, v)).collect();
486            key.sort_unstable();
487            if seen_bindings.insert(key) {
488                out.push(env);
489            }
490        }
491    }
492
493    Ok(out)
494}

496/// Try to unify `slots` with `row` under `env`. Returns the extended env on
497/// success, `None` if a bound var / constant conflicts.
498fn match_atom(slots: &[Slot], row: &Row, env: &Env) -> Option<Env> {
499    let mut next = env.clone();
500    for (i, s) in slots.iter().enumerate() {
501        let col = row_col(row, i);
502        match s {
503            Slot::Const(c) => {
504                if *c != col {
505                    return None;
506                }
507            }
508            Slot::Var(v) => match next.get(v) {
509                Some(&bound) if bound != col => return None,
510                _ => {
511                    next.insert(*v, col);
512                }
513            },
514        }
515    }
516    Some(next)
517}

519/// Execute the head ops for one binding, accumulating writes.
520fn apply_head(
521    eg: &mut EGraph,
522    head: &[HeadOp],
523    env: &mut Env,
524    writes: &mut Vec<Write>,
525    touched: &mut HashSet<FunctionId>,
526    lookup_index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
527) -> Result<()> {
528    for op in head {
529        match op {
530            HeadOp::Set { func, slots } => {
531                let row = build_row(slots, env)?;
532                touched.insert(*func);
533                writes.push(Write::Set(*func, row));
534            }
535            HeadOp::Remove { func, slots } => {
536                let key: Vec<u32> = slots
537                    .iter()
538                    .map(|s| resolve(s, env))
539                    .collect::<Result<_>>()?;
540                touched.insert(*func);
541                writes.push(Write::Remove(*func, key));
542            }
543            HeadOp::Subsume { func, .. } => {
544                // Subsumption is not tracked; treat as a no-op (the row stays
545                // present). `supports_subsumption()` is false, so the frontend
546                // never relies on subsumed-row filtering on this backend.
547                touched.insert(*func);
548            }
549            HeadOp::Lookup { func, args, ret } => {
550                let key: Vec<Value> = args
551                    .iter()
552                    .map(|s| resolve(s, env).map(Value::new))
553                    .collect::<Result<_>>()?;
554                let val = lookup_or_create(eg, *func, &key, lookup_index);
555                env.insert(*ret, val.rep());
556            }
557            HeadOp::Call { id, args, ret } => {
558                let argv: Vec<Value> = args
559                    .iter()
560                    .map(|s| resolve(s, env).map(Value::new))
561                    .collect::<Result<_>>()?;
562                let result = eg.eval_prim_internal(*id, &argv);
563                if let Some(v) = result {
564                    env.insert(*ret, v.rep());
565                }
566                // A `None` result (primitive failure) in an action is a no-op;
567                // real failures surface as panics through `PanicFunc`.
568            }
569            HeadOp::Union { .. } => {
570                // Term-encoding mode emits unions as `(set (@uf …))` writes, not
571                // trait `union` calls (mirrors the DuckDB/Feldera backends). A
572                // direct `union` therefore never reaches a tractable program.
573            }
574            HeadOp::Panic(msg) => {
575                return Err(anyhow!("{msg}"));
576            }
577        }
578    }
579    Ok(())
580}

582/// Build a full row from head-action slots under `env`.
583fn build_row(slots: &[Slot], env: &Env) -> Result<Row> {
584    let vals: Vec<u32> = slots
585        .iter()
586        .map(|s| resolve(s, env))
587        .collect::<Result<_>>()?;
588    Ok(vals.into_boxed_slice())
589}
590
591/// Resolve a slot to a concrete value, erroring if it is an unbound variable.
592fn resolve(s: &Slot, env: &Env) -> Result<u32> {
593    slot_lookup(s, &|v| env.get(&v).copied())
594        .ok_or_else(|| anyhow!("unbound variable {s:?} in rule head"))
595}

597/// Look up the output of `func` for input `key`. If absent, create the row with
598/// a fresh id (eq-sort constructor semantics — mirrors `add_term`). The created
599/// row is written directly into the mirror so subsequent lookups in the same
600/// iteration see it (hash-cons).
601fn lookup_or_create(
602    eg: &mut EGraph,
603    func: FunctionId,
604    key: &[Value],
605    index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
606) -> Value {
607    let info = eg.info(func);
608    let inputs_len = info.arity.saturating_sub(1);
609    // Lazily build the key->output index for this function from the mirror so
610    // repeated lookups within one iteration are O(1) instead of O(state) scans.
611    let idx = index.entry(func).or_insert_with(|| {
612        let mut m: HashMap<Box<[u32]>, u32> = HashMap::new();
613        if let Some(set) = eg.mirror.get(&func) {
614            for row in set.iter() {
615                let k: Box<[u32]> = (0..inputs_len).map(|i| row_col(row, i)).collect();
616                m.insert(k, row_col(row, inputs_len));
617            }
618        }
619        m
620    });
621    let k: Box<[u32]> = key.iter().map(|v| v.rep()).collect();
622    if let Some(&out) = idx.get(&k) {
623        return Value::new(out);
624    }
625    let id = eg.fresh_id_internal();
626    idx.insert(k, id);
627    let mut full: Vec<u32> = key.iter().map(|v| v.rep()).collect();
628    full.push(id);
629    std::rc::Rc::make_mut(eg.mirror.entry(func).or_default()).insert(full.into_boxed_slice());
630    Value::new(id)
631}
