1//! Host-side iteration driver for the FlowLog backend.
2//!
3//! One `run_rules` call = **one bounded egglog iteration**. The body join runs
4//! on the in-process, build-once, epoch-driven raw differential-dataflow
5//! dataflow (`crate::dd_native`); this module owns the orchestration around it:
6//!
7//! 1. snapshot the relation mirror (the read view for this iteration — all rules
8//!    match against the same pre-iteration state, egglog's semi-naive "match the
9//!    old database, then apply" model for a single hop);
10//! 2. for each rule, drive its persistent DD join with the per-relation signed
11//!    delta vs. what that join was last fed, then re-run any body primitives
12//!    (`!=` guards, value-computing prims) host-side over the produced bindings;
13//! 3. for every surviving binding, execute the head ops in order — `set` /
14//!    `remove` / `subsume` writes, RHS `lookup` (eq-sort constructor: create on
15//!    miss), RHS primitive `call`, `union`, `panic`;
16//! 4. apply all collected writes/removes to the mirror and resolve
17//!    functional-dependency conflicts per each touched function's merge mode.
18//!
19//! ## The engine split (mirrors Feldera Stage C)
20//!
21//! The relational table-atom join is the ONLY thing on the engine, and it is the
22//! ONLY join path — there is no host nested-loop fallback. Any rule the DD plan
23//! cannot lower (a binding row exceeding the fixed width cap [`dd_native::W`], or
24//! any shape `plan_join` rejects) `panic!`s with a specific reason. The primitive
25//! tail + head actions are applied HOST-side here, exactly mirroring Feldera's
26//! "table join on the engine, prim tail host-side" split.
27//!
28//! Primitives are invoked through `Database::with_execution_state`, so they see
29//! the same interned base `Value`s the frontend created — giving the FlowLog
30//! backend bit-for-bit value parity with the reference backend (no `eval_prim`
31//! trait change — flowlog keeps its zero-trait-change posture).
32

41/// Binding environment: variable id → bound `u32` value.
42pub(crate) type Env = HashMap<u32, u32>;

44/// A pending write to apply after all matches are computed.
45enum Write {
46    /// Insert/overwrite a full row.
47    Set(FunctionId, Row),
48    /// Retract by key (the slots address inputs for a function, whole row for a
49    /// relation).
50    Remove(FunctionId, Vec<u32>),
51}

53/// One bounded egglog iteration with the body join running on the in-process,
54/// build-once, epoch-driven raw differential-dataflow dataflow
55/// (`crate::dd_native`). This is the FlowLog analog of the Feldera backend's
56/// `interpret::run_iteration` + `persistent_bindings`.
57///
58/// Per rule: compute the signed `+/-` delta of each body relation vs the rows
59/// last fed to that rule's persistent DD join, `step` the join (which feeds ONLY
60/// the delta into never-cleared InputSessions — genuinely incremental), turn the
61/// positive binding deltas into envs, re-run body prims host-side (value prims /
62/// guards the engine keeps off-circuit), then apply head actions. Writes +
63/// FD-merge are applied so results are bit-exact.
64pub fn run_iteration(eg: &mut EGraph, rule_idxs: &[usize]) -> Result<bool> {
65    // FlowLog runs the fast RELATIONAL term-encoding: EVERY rule — including the
66    // `@rebuild_rule*` canonicalization rules — lowers as an ordinary rule and
67    // joins on the fused DD worker (the sound `step_symmetric` path: `view ⋈ @UF_Sf`
68    // + full-recanon action). No native-UF host-pass, no synthetic `@DispΔ`
69    // relation, no rebuild-rule recognizer, no δUF-scoping special-casing.
70
71    // Snapshot the read view: rules match against the pre-iteration mirror.
72    //
73    // The snapshot shares each function's row set by `Rc` rather than
74    // deep-cloning every row: this is O(#functions), not O(state). Mutations to
75    // the mirror this call (head writes, hash-cons in `lookup_or_create`, merge
76    // resolution) go through `Rc::make_mut`, which copy-on-writes only the
77    // functions actually changed while this snapshot is alive — so `read` keeps
78    // the start-of-call contents and rules all match the pre-iteration state.
79    let read: HashMap<FunctionId, std::rc::Rc<HashSet<Row>>> = eg
80        .mirror
81        .iter()
82        .map(|(f, set)| (*f, std::rc::Rc::clone(set)))
83        .collect();
84
85    // Snapshot the fresh-id counter: any hash-cons (`lookup_or_create`) this
86    // call advances it, the O(1) signal that a new term row was created.
87    let next_id_at_start = eg.next_id;
88
89    let mut writes: Vec<Write> = Vec::new();
90    let mut touched: HashSet<FunctionId> = HashSet::new();
91    // Iteration-scoped `key -> output` index for `lookup_or_create` (eq-sort
92    // constructor hash-cons). Built lazily per function so repeated lookups in
93    // one iteration are O(1) instead of rescanning the growing mirror each time.
94    let mut lookup_index: HashMap<FunctionId, HashMap<Box<[u32]>, u32>> = HashMap::new();
95
96    // Collect each rule's index + IR up front (clone to avoid borrow conflicts
97    // while we also mutate the mirror via lookups). EVERY rule — including the
98    // `@rebuild_rule*` canonicalization rules — takes the ordinary atom-rule path
99    // (no recognizer, no rewrite, no drop): they are relational rules
100    // (`view ⋈ @UF_Sf` body + full-recanon head) and join on the fused DD worker
101    // like any other rule.
102    let rules: Vec<(usize, RuleIr)> = rule_idxs
103        .iter()
104        .filter_map(|&i| eg.rules.get(i).and_then(|r| r.clone()).map(|r| (i, r)))
105        .collect();
106
107    // Compute every rule's binding envs FIRST (so the whole atom-bearing ruleset
108    // runs on ONE fused DD worker in a single epoch — `fused_bindings`), THEN
109    // apply head actions in the original rule firing order. Atom-less rules
110    // (`(rule () …)`) have no input relation to drive the DD dataflow, so they
111    // stay host-side (fire once); they are computed inline below.
112    let envs_by_rule = fused_bindings(eg, &read, &rules)?;
113
114    for ((idx, rule), envs) in rules.iter().zip(envs_by_rule.into_iter()) {
115        let _ = idx;
116        for mut env in envs {
117            apply_head(
118                eg,
119                &rule.head,
120                &mut env,
121                &mut writes,
122                &mut touched,
123                &mut lookup_index,
124            )?;
125        }
126    }
127
128    // Apply collected writes to the mirror.
129    //
130    // Removes are BATCHED per function: applying each `Write::Remove` with its
131    // own `set.retain` scan is O(|removes| · |state|) — quadratic. We collect all
132    // retraction keys per function into a hash set, then do a SINGLE `retain`
133    // pass per touched function: O(|state|) total. Removes are applied FIRST
134    // (batched), then Sets — preserving the term encoder's `(@uf)` "delete old
135    // leader, set new leader" delete-then-set ordering.
136    //
137    // `changed` is computed INCREMENTALLY as writes land (O(delta)), not via a
138    // full before/after content compare. A hash-cons in `lookup_or_create`
139    // always allocates a fresh id, so any term created this call advances
140    // `next_id` — that alone is a real mirror change.
141    let mut changed = eg.next_id != next_id_at_start;
142    let mut removes_by_func: HashMap<FunctionId, (usize, HashSet<Box<[u32]>>)> = HashMap::new();
143    let mut sets: Vec<(FunctionId, Row)> = Vec::new();
144    for w in writes {
145        match w {
146            Write::Set(f, row) => sets.push((f, row)),
147            Write::Remove(f, key) => {
148                let entry = removes_by_func
149                    .entry(f)
150                    .or_insert_with(|| (key.len(), HashSet::new()));
151                entry.1.insert(key.into_boxed_slice());
152            }
153        }
154    }
155    for (f, (keylen, keys)) in removes_by_func {
156        if let Some(set) = eg.mirror.get_mut(&f) {
157            let before_len = set.len();
158            std::rc::Rc::make_mut(set).retain(|row| {
159                let k: Box<[u32]> = (0..keylen).map(|i| row_col(row, i)).collect();
160                !keys.contains(&k)
161            });
162            changed |= set.len() != before_len;
163        }
164    }
165    let mut touched_keys: HashMap<FunctionId, HashSet<Vec<u32>>> = HashMap::new();
166    for (f, row) in sets {
167        let inputs_len = eg.info(f).arity.saturating_sub(1);
168        let key: Vec<u32> = (0..inputs_len).map(|i| row_col(&row, i)).collect();
169        let inserted = std::rc::Rc::make_mut(eg.mirror.entry(f).or_default()).insert(row.clone());
170        changed |= inserted;
171        touched_keys.entry(f).or_default().insert(key);
172    }
173
174    // `--native-merge` TERM-BUILD scratch: cleared before the resolves so the
175    // view rows a term-build merge mints this iteration are recorded fresh.
176    eg.term_build_view_keys.clear();
177
178    // Resolve FD conflicts on every function a head action wrote to (a `set`
179    // can introduce two rows sharing a key that must merge per the merge mode).
180    let empty_keys: HashSet<Vec<u32>> = HashSet::new();
181    for &f in &touched {
182        let keys = touched_keys.get(&f).unwrap_or(&empty_keys);
183        changed |= eg.resolve_merge(f, keys);
184    }
185
186    // `--native-merge` TERM-BUILD: a term-build merge minted `@<C>View` rows
187    // (recorded in `term_build_view_keys`). Those views carry their OWN congruence
188    // merge, so reconcile their FD conflicts now — exactly the
189    // "TableInsert respects the target table's merge" semantics of the bridge.
190    // Drain to a fixpoint: resolving one view can enqueue unions but writes no new
191    // view rows, so this terminates in one pass in practice; the loop is defensive.
192    while !eg.term_build_view_keys.is_empty() {
193        let pending: Vec<(FunctionId, HashSet<Vec<u32>>)> =
194            eg.term_build_view_keys.drain().collect();
195        for (view, keys) in pending {
196            changed |= eg.resolve_merge(view, &keys);
197        }
198    }
199
200    Ok(changed)
201}

203/// Derive a stable ruleset LABEL for a `run_rules` call from the rules it runs,
204/// for `FLOWLOG_DD_RULESET_PROF`. The backend trait does not carry the ruleset
205/// name down to `run_rules`, but the term encoder gives each maintenance rule a
206/// `fresh()`-suffixed name with a stable PREFIX that identifies its ruleset
207/// (`uf_update*` / `singleparent*` / `uf_function_index*` / `congruence_rule*` /
208/// `rebuild_rule*` / `merge_rule*` / `merge_cleanup*` / `delete_rule*`); user
209/// rewrite rules are named by their full s-expression text. We map each rule
210/// name to its category, then label the call by the categories present (all
211/// rules in one `run_rules` call belong to one egglog ruleset).
212/// Map a single rule NAME to its bucket label. Maintenance rules emitted by the
213/// term encoder (proof_encoding.rs) carry a stable, `fresh()`-suffixed name;
214/// most are `@`-prefixed (`@uf_update`, `@congruence_rule`, …) but
215/// `singleparent@uf_update` is not. Match the identifying substring so the
216/// leading `@` and trailing digits don't matter. Order: most specific first
217/// (`singleparent` and the `_subsume` variant before their bare forms;
218/// `uf_function_index` before `uf_update`).
219pub(crate) fn rule_category(name: &str) -> &'static str {
220    const MAINT: &[(&str, &str)] = &[
221        ("singleparent", "single_parent"),
222        ("uf_function_index", "uf_function_index"),
223        // PR #782 native-UF drain (the `@uf_change_drain` ruleset). Must be
224        // matched BEFORE `uf_update` (substring overlap is none, but keep the
225        // most-specific UF rules grouped). Dropped under `--native-uf`.
226        ("uf_change_drain", "uf_change_drain"),
227        ("uf_update", "path_compress/uf_update"),
228        ("delete_rule_subsume", "delete_subsume"),
229        ("delete_rule", "delete_subsume"),
230        // `@congruence_rule` and `@rebuild_rule` are both in the term encoder's
231        // `rebuilding` ruleset and run in ONE fused `run_rules` call; split them
232        // into distinct buckets so the native-UF-addressable cost
233        // (canonicalization, which joins function rows against `@uf`) can be
234        // separated from congruence detection (which stays relational under a
235        // native UF).
236        ("congruence_rule", "congruence"),
237        ("rebuild_rule", "canonicalize"),
238        ("merge_cleanup", "rebuilding_cleanup"),
239        ("merge_rule", "merge_rule"),
240    ];
241    for (needle, label) in MAINT {
242        if name.contains(needle) {
243            return label;
244        }
245    }
246    if name.starts_with("eval_actions") {
247        return "eval_actions";
248    }
249    "<user>"
250}

252/// Compute every rule's binding envs in ONE fused pass: the whole atom-bearing
253/// ruleset's body joins run on a SINGLE shared timely worker
254/// ([`dd_native::FusedDdJoin`]) clocked once this iteration, then each rule's
255/// host-side prim tail is re-run over its own bindings. Atom-less rules
256/// (`(rule () …)`) have no input relation to drive the DD dataflow, so they are
257/// fired once host-side. Returns a `Vec<Vec<Env>>` parallel to `rules` (same
258/// order), ready for `apply_head`.
259fn fused_bindings(
260    eg: &mut EGraph,
261    read: &HashMap<FunctionId, std::rc::Rc<HashSet<Row>>>,
262    rules: &[(usize, RuleIr)],
263) -> Result<Vec<Vec<Env>>> {
264    use crate::dd_native;
265
266    let prof = dd_native::prof_enabled();
267    let rs_prof = dd_native::ruleset_prof_enabled();
268    // Per-ruleset attribution: the wall clock for the WHOLE atom-bearing path
269    // this call, plus the per-bucket nanos we accumulate below. The call's time
270    // is later apportioned across the rule CATEGORIES present (so `@rebuild_rule`
271    // and `@congruence_rule`, fused into one timely step, land in distinct
272    // `canonicalize`/`congruence` buckets).
273    let rs_t_total = std::time::Instant::now();
274    let mut rs_delta_ns: u64 = 0;
275    let mut rs_prim_ns: u64 = 0;
276    let mut rs_feed_ns: u64 = 0;
277    let mut rs_step_ns: u64 = 0;
278    let mut rs_delta_rows: u64 = 0;
279    let mut out: Vec<Vec<Env>> = vec![Vec::new(); rules.len()];
280
281    // Partition: atom-bearing rules drive the fused DD worker; atom-less rules
282    // fire once host-side. Record each atom-bearing rule's POSITION in `rules` so
283    // we can scatter the fused output back into `out` in the caller's order.
284    let mut atom_positions: Vec<usize> = Vec::new();
285    let mut atom_rule_idxs: Vec<usize> = Vec::new();
286    for (pos, (idx, rule)) in rules.iter().enumerate() {
287        let _ = idx;
288        let has_atoms = rule.body.iter().any(|op| matches!(op, BodyOp::Atom(_)));
289        if has_atoms {
290            atom_positions.push(pos);
291            atom_rule_idxs.push(*idx);
292        } else {
293            // Atom-less rule: fire once (presence in `seen` = already fired).
294            if eg.seen.contains_key(idx) {
295                continue;
296            }
297            eg.seen.insert(*idx, ());
298            let mut envs: Vec<Env> = vec![Env::new()];
299            for op in &rule.body {
300                envs = step_prim(eg, op, envs)?;
301                if envs.is_empty() {
302                    break;
303                }
304            }
305            out[pos] = envs;
306        }
307    }
308
309    if atom_positions.is_empty() {
310        return Ok(out);
311    }
312
313    // The fused join is keyed by the SORTED atom-bearing rule-index list (the
314    // ruleset identity), exactly like feldera's `FusedJoin`. Build it ONCE
315    // (lazily) per distinct ruleset, planning each rule. Any shape `plan_join`
316    // rejects PANICS (no host fallback; the DD dataflow is the only join path).
317    let mut key: Vec<usize> = atom_rule_idxs.clone();
318    key.sort_unstable();
319
320    if !eg.dd_fused.contains_key(&key) {
321        // Plan in the SAME order as `atom_positions` so the fused build order
322        // matches our scatter order (the fused join preserves plan order).
323        let mut plans: Vec<(usize, dd_native::JoinPlan)> = Vec::with_capacity(atom_positions.len());
324        for (&pos, &idx) in atom_positions.iter().zip(atom_rule_idxs.iter()) {
325            let rule = &rules[pos].1;
326            let plan = match dd_native::plan_join(rule) {
327                Ok(p) => p,
328                Err(reason) => panic!(
329                    "FlowLog DD join cannot lower rule {:?}: {reason} \
330                     (no host fallback; the DD dataflow is the only join path)",
331                    rule.name
332                ),
333            };
334            plans.push((idx, plan));
335        }
336        // `--wcoj`: enable the worst-case-optimal triangle delta query. The
337        // build detects the triangle shape per rule; non-triangle rules in the
338        // ruleset keep the binary `.join` chain (hybrid). Off ⇒ byte-identical
339        // to the pre-WCOJ build. EVERY rule — including the `@rebuild_rule*`
340        // canonicalization rules — runs through the same general fused join (the
341        // sound `step_symmetric` path); there is no δUF-scoping fast-rebuild
342        // variant anymore.
343        let wcoj = eg.wcoj_enabled;
344        let fused = dd_native::FusedDdJoin::build(&plans, wcoj, false)?;
345        eg.dd_fused.insert(key.clone(), fused);
346    }
347
348    // The fused join's internal rule order (its build order = `atom_positions`
349    // order). Map each fused output slot back to the caller `rules` position and
350    // capture each rule's canonical var order.
351    let (fused_rule_idxs, fused_body_funcs): (Vec<usize>, Vec<Vec<FunctionId>>) = {
352        let fused = eg.dd_fused.get(&key).expect("fused join present");
353        (
354            fused.rule_indices(),
355            (0..fused.rule_indices().len())
356                .map(|p| fused.rule_body_funcs(p).to_vec())
357                .collect(),
358        )
359    };
360    // The fused build order equals `atom_rule_idxs` (we built it that way), so
361    // map fused position -> caller `rules` position via `atom_positions`.
362    debug_assert_eq!(fused_rule_idxs, atom_rule_idxs, "fused build order");
363
364    // Each atom-bearing rule's canonical var order (for env reconstruction).
365    let var_orders: Vec<Vec<u32>> = atom_positions
366        .iter()
367        .map(|&pos| {
368            let rule = &rules[pos].1;
369            dd_native::plan_join(rule)
370                .expect("plan re-derivable")
371                .var_order()
372        })
373        .collect();
374
375    // Distinct relations across the whole ruleset → ONE combined signed delta map
376    // fed into the fused worker's SHARED inputs. The `fed` snapshot is per-ruleset
377    // (the fused join's identity), diffed against the live mirror like the per-rule
378    // `dd_native_fed`.
379    let t_delta = std::time::Instant::now();
380    let empty_set: std::rc::Rc<HashSet<Row>> = std::rc::Rc::new(HashSet::new());
381    let mut all_funcs: Vec<FunctionId> = Vec::new();
382    for bf in &fused_body_funcs {
383        for &f in bf {
384            if !all_funcs.contains(&f) {
385                all_funcs.push(f);
386            }
387        }
388    }
389    let mut delta: HashMap<FunctionId, Vec<(Vec<u32>, isize)>> = HashMap::new();
390    {
391        let fed = eg.dd_fused_fed.entry(key.clone()).or_default();
392        for &f in &all_funcs {
393            let cur = read.get(&f).cloned().unwrap_or_else(|| empty_set.clone());
394            let prev = fed.entry(f).or_insert_with(|| empty_set.clone());
395            if std::rc::Rc::ptr_eq(&cur, prev) {
396                *prev = cur;
397                continue;
398            }
399            let mut rows: Vec<(Vec<u32>, isize)> = Vec::new();
400            for r in cur.iter() {
401                if !prev.contains(r) {
402                    rows.push((r.to_vec(), 1));
403                }
404            }
405            for r in prev.iter() {
406                if !cur.contains(r) {
407                    rows.push((r.to_vec(), -1));
408                }
409            }
410            if !rows.is_empty() {
411                rs_delta_rows += rows.len() as u64;
412                delta.insert(f, rows);
413            }
414            *prev = cur;
415        }
416    }
417    let delta_elapsed = t_delta.elapsed().as_nanos() as u64;
418    if prof {
419        dd_native::PROF_DELTA_NS.fetch_add(delta_elapsed, std::sync::atomic::Ordering::Relaxed);
420    }
421    rs_delta_ns += delta_elapsed;
422
423    // ONE step of the shared worker for the WHOLE ruleset. `step` updates the
424    // global PROF_FEED_NS / PROF_STEP_NS counters when profiling is on (which it
425    // is whenever either env var is set), so snapshot them before/after to split
426    // this ruleset's feed vs worker_step time.
427    use std::sync::atomic::Ordering as ProfOrd;
428    let feed_before = dd_native::PROF_FEED_NS.load(ProfOrd::Relaxed);
429    let step_before = dd_native::PROF_STEP_NS.load(ProfOrd::Relaxed);
430    eg.dd_rule_runs += atom_positions.len() as u64;
431    let per_rule_bindings = {
432        let fused = eg.dd_fused.get_mut(&key).expect("fused join present");
433        fused.step(&delta)?
434    };
435    if rs_prof {
436        rs_feed_ns += dd_native::PROF_FEED_NS
437            .load(ProfOrd::Relaxed)
438            .wrapping_sub(feed_before);
439        rs_step_ns += dd_native::PROF_STEP_NS
440            .load(ProfOrd::Relaxed)
441            .wrapping_sub(step_before);
442    }
443
444    // Per-rule positive-binding count = the fused join's workload proxy for each
445    // rule (used below to apportion the single fused worker_step across the rule
446    // CATEGORIES present in this call). Length is parallel to `atom_positions`.
447    let mut rs_pos_bindings: Vec<u64> = vec![0; atom_positions.len()];
448
449    // Turn each rule's positive binding deltas into envs; re-run its body prims
450    // host-side. Negative weights are integral bookkeeping (a body row retracted)
451    // — egglog heads are monotone-fire, so we do NOT re-fire on disappearance.
452    let t_prim = std::time::Instant::now();
453    for (fpos, bindings) in per_rule_bindings.into_iter().enumerate() {
454        let caller_pos = atom_positions[fpos];
455        let rule = &rules[caller_pos].1;
456        let var_order = &var_orders[fpos];
457        let mut envs: Vec<Env> = Vec::new();
458        for (bind, w) in &bindings {
459            if *w <= 0 {
460                continue;
461            }
462            rs_pos_bindings[fpos] += 1;
463            let mut env: Env = Env::new();
464            for (i, &v) in var_order.iter().enumerate() {
465                env.insert(v, bind[i]);
466            }
467            let mut es: Vec<Env> = vec![env];
468            for op in &rule.body {
469                if let BodyOp::Prim { .. } = op {
470                    es = step_prim(eg, op, es)?;
471                }
472            }
473            envs.extend(es);
474        }
475        out[caller_pos] = envs;
476    }
477
478    let prim_elapsed = t_prim.elapsed().as_nanos() as u64;
479    if prof {
480        dd_native::PROF_PRIM_NS.fetch_add(prim_elapsed, std::sync::atomic::Ordering::Relaxed);
481    }
482    rs_prim_ns += prim_elapsed;
483
484    if rs_prof {
485        let rs_total_ns = rs_t_total.elapsed().as_nanos() as u64;
486        // Several rule CATEGORIES (e.g. `canonicalize` = `@rebuild_rule` and
487        // `congruence` = `@congruence_rule`) run in ONE fused timely step, so
488        // there is no per-category wall clock. Apportion this call's measured
489        // buckets across the categories present, weighted by each category's
490        // share of POSITIVE output bindings (the join workload it produced). If
491        // no rule produced output this call, fall back to an even split by rule
492        // count so the call's wall time is still attributed.
493        let mut cat_w: HashMap<&'static str, u64> = HashMap::new();
494        let mut cat_n: HashMap<&'static str, u64> = HashMap::new();
495        // Cross-check: nanos of (apportioned) worker_step attributable to rules
496        // whose BODY reads a `@uf` table (`UF_*` relation) — the native-UF-
497        // addressable fraction proxy.
498        let mut uf_body_w: u64 = 0;
499        for (fpos, &pos) in atom_positions.iter().enumerate() {
500            let rule = &rules[pos].1;
501            let cat = rule_category(&rule.name);
502            let w = rs_pos_bindings[fpos];
503            *cat_w.entry(cat).or_default() += w;
504            *cat_n.entry(cat).or_default() += 1;
505            let reads_uf = rule.body.iter().any(
506                |op| matches!(op, BodyOp::Atom(a) if eg.relation_name(a.func).contains("UF_")),
507            );
508            if reads_uf {
509                uf_body_w += w;
510            }
511        }
512        let total_w: u64 = cat_w.values().sum();
513        let total_n: u64 = cat_n.values().sum();
514        // Worker_step nanos apportioned to UF-body-reading rules.
515        let uf_step_ns = if total_w > 0 {
516            (rs_step_ns as u128 * uf_body_w as u128 / total_w as u128) as u64
517        } else {
518            0
519        };
520        dd_native::ruleset_uf_body_record(uf_step_ns, rs_step_ns);
521        for (cat, &w) in &cat_w {
522            let n = cat_n[cat];
523            // share by binding workload, else by rule count.
524            let (num, den): (u128, u128) = if total_w > 0 {
525                (w as u128, total_w as u128)
526            } else {
527                (n as u128, total_n.max(1) as u128)
528            };
529            let part = |v: u64| (v as u128 * num / den) as u64;
530            dd_native::ruleset_prof_record(
531                cat,
532                part(rs_total_ns),
533                part(rs_step_ns),
534                part(rs_feed_ns),
535                part(rs_prim_ns),
536                part(rs_delta_ns),
537                part(rs_delta_rows),
538            );
539        }
540    }
541
542    Ok(out)
543}

545/// Evaluate a primitive body op over each binding env, returning the new list of
546/// envs. A value-computing prim binds (or checks) its return var; a guard prim
547/// (`!=`) that fails prunes the env. Table atoms are NOT handled here — they run
548/// on the DD dataflow; this is only the host-side primitive tail.
549pub(crate) fn step_prim(eg: &mut EGraph, op: &BodyOp, envs: Vec<Env>) -> Result<Vec<Env>> {
550    let BodyOp::Prim { id, args, ret } = op else {
551        unreachable!("step_prim called on a non-primitive body op");
552    };
553    let mut out = Vec::new();
554    for env in envs {
555        let resolved: Option<Vec<Value>> = args
556            .iter()
557            .map(|s| slot_lookup(s, &|v| env.get(&v).copied()).map(Value::new))
558            .collect();
559        let Some(argv) = resolved else { continue };
560        let result = eg.eval_prim_internal(*id, &argv);
561        let Some(result) = result else {
562            // Primitive failed (e.g. `!=` of equal args) — prune.
563            continue;
564        };
565        match ret {
566            Slot::Var(v) => {
567                let mut next = env.clone();
568                match next.get(v) {
569                    Some(&existing) if existing != result.rep() => continue,
570                    _ => {
571                        next.insert(*v, result.rep());
572                    }
573                }
574                out.push(next);
575            }
576            Slot::Const(c) => {
577                if *c == result.rep() {
578                    out.push(env);
579                }
580            }
581        }
582    }
583    Ok(out)
584}

586/// Execute the head ops for one binding, accumulating writes.
587fn apply_head(
588    eg: &mut EGraph,
589    head: &[HeadOp],
590    env: &mut Env,
591    writes: &mut Vec<Write>,
592    touched: &mut HashSet<FunctionId>,
593    lookup_index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
594) -> Result<()> {
595    for op in head {
596        match op {
597            HeadOp::Set { func, slots } => {
598                let row = build_row(slots, env)?;
599                touched.insert(*func);
600                writes.push(Write::Set(*func, row));
601            }
602            HeadOp::Remove { func, slots } => {
603                let key: Vec<u32> = slots
604                    .iter()
605                    .map(|s| resolve(s, env))
606                    .collect::<Result<_>>()?;
607                touched.insert(*func);
608                writes.push(Write::Remove(*func, key));
609            }
610            HeadOp::Subsume { func, .. } => {
611                // Subsumption is not tracked; treat as a no-op (the row stays
612                // present). `supports_subsumption()` is false, so the frontend
613                // never relies on subsumed-row filtering on this backend.
614                touched.insert(*func);
615            }
616            HeadOp::Lookup { func, args, ret } => {
617                let key: Vec<Value> = args
618                    .iter()
619                    .map(|s| resolve(s, env).map(Value::new))
620                    .collect::<Result<_>>()?;
621                let val = lookup_or_create(eg, *func, &key, lookup_index);
622                env.insert(*ret, val.rep());
623            }
624            HeadOp::Call { id, args, ret } => {
625                let argv: Vec<Value> = args
626                    .iter()
627                    .map(|s| resolve(s, env).map(Value::new))
628                    .collect::<Result<_>>()?;
629                let result = eg.eval_prim_internal(*id, &argv);
630                if let Some(v) = result {
631                    env.insert(*ret, v.rep());
632                }
633                // A `None` result (primitive failure) in an action is a no-op;
634                // real failures surface as panics through `PanicFunc`.
635            }
636            HeadOp::Union { .. } => {
637                // Term-encoding mode emits unions as `(set (@uf …))` writes, not
638                // trait `union` calls (mirrors the DuckDB/Feldera backends). A
639                // direct `union` therefore never reaches a tractable program.
640            }
641            HeadOp::Panic(msg) => {
642                return Err(anyhow!("{msg}"));
643            }
644        }
645    }
646    Ok(())
647}

649/// Build a full row from head-action slots under `env`.
650fn build_row(slots: &[Slot], env: &Env) -> Result<Row> {
651    let vals: Vec<u32> = slots
652        .iter()
653        .map(|s| resolve(s, env))
654        .collect::<Result<_>>()?;
655    Ok(vals.into_boxed_slice())
656}

658/// Resolve a slot to a concrete value, erroring if it is an unbound variable.
659fn resolve(s: &Slot, env: &Env) -> Result<u32> {
660    slot_lookup(s, &|v| env.get(&v).copied())
661        .ok_or_else(|| anyhow!("unbound variable {s:?} in rule head"))
662}

664/// Look up the output of `func` for input `key`. If absent, create the row with
665/// a fresh id (eq-sort constructor semantics — mirrors `add_term`). The created
666/// row is written directly into the mirror so subsequent lookups in the same
667/// iteration see it (hash-cons).
668pub(crate) fn lookup_or_create(
669    eg: &mut EGraph,
670    func: FunctionId,
671    key: &[Value],
672    index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
673) -> Value {
674    let info = eg.info(func);
675    let inputs_len = info.arity.saturating_sub(1);
676    let identity_on_miss = info.identity_on_miss;
677    // Lazily build the key->output index for this function from the mirror so
678    // repeated lookups within one iteration are O(1) instead of O(state) scans.
679    let idx = index.entry(func).or_insert_with(|| {
680        let mut m: HashMap<Box<[u32]>, u32> = HashMap::new();
681        if let Some(set) = eg.mirror.get(&func) {
682            for row in set.iter() {
683                let k: Box<[u32]> = (0..inputs_len).map(|i| row_col(row, i)).collect();
684                m.insert(k, row_col(row, inputs_len));
685            }
686        }
687        m
688    });
689    let k: Box<[u32]> = key.iter().map(|v| v.rep()).collect();
690    if let Some(&out) = idx.get(&k) {
691        return Value::new(out);
692    }
693    // Identity-on-miss ("lookup-or-self"): a missing key resolves to the key
694    // itself and inserts no row. Used by the canonicalize-at-creation encoding
695    // for the flat UF-index `@UF_Sf`, where this is exactly `find_UFold(x)=x`
696    // for an id with no recorded leader.
697    if identity_on_miss {
698        debug_assert_eq!(key.len(), 1, "identity-on-miss expects a single key column");
699        return key[0];
700    }
701    let id = eg.fresh_id_internal();
702    idx.insert(k, id);
703    let mut full: Vec<u32> = key.iter().map(|v| v.rep()).collect();
704    full.push(id);
705    let row: Row = full.into_boxed_slice();
706    std::rc::Rc::make_mut(eg.mirror.entry(func).or_default()).insert(row);
707    Value::new(id)
708}
