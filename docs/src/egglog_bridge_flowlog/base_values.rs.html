1//! Base-value pool for the FlowLog backend.
2//!
3//! Stores base values exactly the way the reference bridge does: a real
4//! [`egglog_core_relations::BaseValues`] registry, owned by the embedded
5//! [`Database`] in `lib.rs`. This gives bit-for-bit `Value` parity with the
6//! reference backend (the M1 proof relies on `Value` reps matching).
7//!
8//! Provides the layout-preserving reference cast from `&BaseValues` /
9//! `&mut BaseValues` to `&dyn BaseValuePool` — the same `#[repr(transparent)]`
10//! trick the bridge uses with its `BaseValuesAsPool` newtype. Mirrors the
11//! Feldera backend's `base_values.rs` verbatim.
12//!
13//! [`Database`]: egglog_core_relations::Database
14
15use std::any::{Any, TypeId};
16
17use egglog_backend_trait::{BaseValueId, BaseValuePool, Value};
18use egglog_core_relations::{BaseValues, DynamicInternTable};
19
20/// `#[repr(transparent)]` view of [`BaseValues`] implementing [`BaseValuePool`].
21#[repr(transparent)]
22pub struct BaseValuesAsPool(BaseValues);
23
24/// View `&BaseValues` as `&dyn BaseValuePool`.
25pub fn base_values_as_pool(bv: &BaseValues) -> &dyn BaseValuePool {
26    // SAFETY: `BaseValuesAsPool` is `#[repr(transparent)]` over `BaseValues`,
27    // so the reference cast is sound.
28    let as_pool: &BaseValuesAsPool =
29        unsafe { &*(bv as *const BaseValues as *const BaseValuesAsPool) };
30    as_pool
31}
32
33/// View `&mut BaseValues` as `&mut dyn BaseValuePool`.
34pub fn base_values_as_pool_mut(bv: &mut BaseValues) -> &mut dyn BaseValuePool {
35    // SAFETY: as above.
36    let as_pool: &mut BaseValuesAsPool =
37        unsafe { &mut *(bv as *mut BaseValues as *mut BaseValuesAsPool) };
38    as_pool
39}
40
41impl BaseValuePool for BaseValuesAsPool {
42    fn register_type_dyn(
43        &mut self,
44        type_id: TypeId,
45        factory: Box<dyn FnOnce() -> Box<dyn DynamicInternTable>>,
46    ) -> BaseValueId {
47        self.0.register_type_dyn(type_id, factory)
48    }
49
50    fn get_ty_by_type_id(&self, type_id: TypeId) -> BaseValueId {
51        self.0.get_ty_by_id(type_id)
52    }
53
54    fn intern_dyn(&self, ty: BaseValueId, value: Box<dyn Any + Send + Sync>) -> Value {
55        let any_ref: &dyn Any = &*value;
56        self.0.intern_dyn_by_id(ty, any_ref)
57    }
58
59    fn unwrap_dyn(&self, ty: BaseValueId, val: Value) -> Box<dyn Any + Send + Sync> {
60        self.0.unwrap_dyn_by_id(ty, val)
61    }
62
63    fn has_ty(&self, type_id: TypeId) -> bool {
64        self.0.has_ty_by_id(type_id)
65    }
66}
