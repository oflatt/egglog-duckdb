1//! In-process, build-once, epoch-driven incremental body join on RAW
2//! `differential-dataflow` + `timely` — the FlowLog analog of the Feldera
3//! backend's `dbsp_join::PersistentJoin`.
4//!
5//! This is the ONLY join path for the FlowLog `Interpret` backend (driven by
6//! [`crate::interpret::run_iteration`]); there is no host nested-loop fallback.
7//! It panics (via the caller) on shapes it does not support — see `plan_join`.
8//!
9//! ## Two implementations: per-rule vs FUSED per-ruleset
10//!
11//! [`FusedDdJoin`] is the PERF-CRITICAL path the interpreter drives: ONE shared
12//! timely `Worker` hosts ONE dataflow for a whole RULESET, every distinct body
13//! relation is a single SHARED input `Collection`, and each rule is a join
14//! sub-stream reading those shared collections (the DD analog of feldera's
15//! `FusedJoin`). [`PersistentDdJoin`] is the original per-RULE worker; it is
16//! retained for the bridge-level incrementality unit test and documents the base
17//! architecture below. Both feed only signed DELTAS into never-cleared
18//! InputSessions, so the arrangements persist across epochs = incremental.
19//!
20//! ## The base architecture (mirrors feldera/DBSP)
21//!
22//! For each atom-bearing rule we build ONE differential-dataflow dataflow, ONCE,
23//! inside a single-threaded timely `Worker` we OWN (so we can `step` it across
24//! host calls). Each body atom occurrence is sourced from a
25//! `differential_dataflow::input::InputSession`; the rule's body join is a
26//! left-deep chain of DD `.join`s, with `!=` guards and value-prims inlined in
27//! `.flat_map`/`.filter`; the head binding rows flow out through `.inspect_batch`
28//! into a shared `Rc<RefCell<Vec<(Row, isize)>>>` capture buffer.
29//!
30//! Each egglog iteration (= one epoch) the host feeds ONLY the per-relation
31//! signed DELTA into the InputSessions (`+1` insert, `-1` retract), advances the
32//! timely timestamp, `step_while`s the worker to that epoch's fixpoint, and
33//! drains the capture buffer to get the OUTPUT binding deltas. The
34//! InputSessions are NEVER cleared — the DD arrangements persist across epochs,
35//! which is what makes the join genuinely incremental (epoch K does only
36//! delta·integral work, not a full recompute) — the whole point of the design.
37//!
38//! ## Fixpoint structure
39//!
40//! We use EXTERNAL epoch-drive (the host loop advances epochs and feeds head
41//! outputs back as the next epoch's inputs), NOT an in-dataflow `iterate()`
42//! scope. This matches egglog's bounded `(run N)` fire->rebuild->repeat model and
43//! sidesteps DD `iterate()`'s monotonicity constraints under retraction (a
44//! rebuild RETRACTS non-canonical rows, which `iterate()` cannot express
45//! cleanly). The dataflow itself is NON-recursive: one epoch = one bounded hop.

69/// Fixed binding-row width (DD `Data` needs a `Sized + Ord + Hash` type; an
70/// array gives us that without dbsp's `declare_tuples!`). Mirrors feldera's
71/// `JOIN_WIDTH = 32` but bumped to 48 to cover the widest rebuild rule the
72/// flowlog test corpus generates: `luminal-llama`'s `@rebuild_rule34` uses 35
73/// distinct body vars (a wide-arity congruence-closure rebuild). 48 covers
74/// every reachable program with headroom; a rule exceeding this is reported as
75/// a row-width-cap wall (raise `W` to extend coverage — it is purely a fixed
76/// array size, costing `W * 4` bytes per binding row).
77pub const W: usize = 48;

79/// A fixed-width binding / relation row flowing through the DD dataflow:
80/// `row[i]` is the value of canonical body variable `i` (0 if not yet bound).
81///
82/// A NEWTYPE over `[u32; W]` (rather than the bare array) because timely's
83/// `ExchangeData` bound — required by DD `.join`/`.distinct` — is
84/// `Serialize + Deserialize`, and `serde` only derives those for arrays up to
85/// length 32. The hand-written serde impl (serialize as a fixed-length seq of
86/// `W` `u32`s) lifts that cap so `W` can exceed 32 (the corpus needs 35). All
87/// other derives (`Ord`/`Hash`/`Clone`/`Copy`) are auto for any array size.
88#[derive(Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash, Debug)]
89pub struct Row([u32; W]);

110impl serde::Serialize for Row {
111    fn serialize<S: serde::Serializer>(&self, s: S) -> Result<S::Ok, S::Error> {
112        use serde::ser::SerializeTuple;
113        // Fixed-length tuple of W u32s — bincode-friendly, no length prefix
114        // needed (the deserializer knows W). Sidesteps serde's 32-array cap.
115        let mut t = s.serialize_tuple(W)?;
116        for v in &self.0 {
117            t.serialize_element(v)?;
118        }
119        t.end()
120    }
121}

123impl<'de> serde::Deserialize<'de> for Row {
124    fn deserialize<D: serde::Deserializer<'de>>(d: D) -> Result<Row, D::Error> {
125        struct RowVisitor;
126        impl<'de> serde::de::Visitor<'de> for RowVisitor {
127            type Value = Row;
128            fn expecting(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
129                write!(f, "a tuple of {W} u32s")
130            }
131            fn visit_seq<A: serde::de::SeqAccess<'de>>(self, mut seq: A) -> Result<Row, A::Error> {
132                let mut a = [0u32; W];
133                for (i, slot) in a.iter_mut().enumerate() {
134                    *slot = seq
135                        .next_element()?
136                        .ok_or_else(|| serde::de::Error::invalid_length(i, &self))?;
137                }
138                Ok(Row(a))
139            }
140        }
141        d.deserialize_tuple(W, RowVisitor)
142    }
143}

259/// Print the Step-0 profile to stderr if `FLOWLOG_DD_PROF` is set.
260pub fn dd_prof_dump() {
261    if !prof_enabled() {
262        return;
263    }
264    let workers = PROF_WORKERS.load(Ordering::Relaxed);
265    let sessions = PROF_INPUT_SESSIONS.load(Ordering::Relaxed);
266    let step_ns = PROF_STEP_NS.load(Ordering::Relaxed);
267    let feed_ns = PROF_FEED_NS.load(Ordering::Relaxed);
268    let calls = PROF_STEP_CALLS.load(Ordering::Relaxed);
269    let prim_ns = PROF_PRIM_NS.load(Ordering::Relaxed);
270    let delta_ns = PROF_DELTA_NS.load(Ordering::Relaxed);
271    #[allow(clippy::disallowed_macros)]
272    {
273        eprintln!(
274            "[FLOWLOG_DD_PROF] workers={workers} input_sessions={sessions} \
275             nonempty_step_calls={calls} worker_step={:.3}s feed={:.3}s \
276             host_prim={:.3}s delta_compute={:.3}s",
277            step_ns as f64 / 1e9,
278            feed_ns as f64 / 1e9,
279            prim_ns as f64 / 1e9,
280            delta_ns as f64 / 1e9,
281        );
282    }
283}

285/// Print the per-ruleset DD profile to stderr if `FLOWLOG_DD_RULESET_PROF` is
286/// set: one row per ruleset, sorted by total DD time descending, with each
287/// ruleset's share of the grand total.
288pub fn dd_ruleset_prof_dump() {
289    if !ruleset_prof_enabled() {
290        return;
291    }
292    let table = ruleset_prof_table().lock().expect("ruleset prof lock");
293    if table.is_empty() {
294        return;
295    }
296    let mut rows: Vec<(String, RulesetProf)> =
297        table.iter().map(|(k, v)| (k.clone(), v.clone())).collect();
298    rows.sort_by(|a, b| b.1.total_ns.cmp(&a.1.total_ns));
299    let grand_total: u64 = rows.iter().map(|(_, p)| p.total_ns).sum();
300    let s = |ns: u64| ns as f64 / 1e9;
301    #[allow(clippy::disallowed_macros)]
302    {
303        eprintln!(
304            "[FLOWLOG_DD_RULESET_PROF] grand_total_dd={:.3}s",
305            s(grand_total)
306        );
307        eprintln!(
308            "{:<28} {:>6} {:>9} {:>6} {:>11} {:>9} {:>9} {:>11}",
309            "ruleset", "calls", "total", "%dd", "worker_step", "feed", "host_prim", "delta_rows"
310        );
311        for (name, p) in &rows {
312            let pct = if grand_total > 0 {
313                100.0 * p.total_ns as f64 / grand_total as f64
314            } else {
315                0.0
316            };
317            eprintln!(
318                "{:<28} {:>6} {:>8.3}s {:>5.1}% {:>10.3}s {:>8.3}s {:>8.3}s {:>11}",
319                name,
320                p.calls,
321                s(p.total_ns),
322                pct,
323                s(p.worker_step_ns),
324                s(p.feed_ns),
325                s(p.host_prim_ns),
326                p.delta_rows,
327            );
328        }
329        // Cross-check: share of fused worker_step in rules whose body reads @uf.
330        let uf_step = UF_BODY_STEP_NS.load(Ordering::Relaxed);
331        let all_step = ALL_STEP_NS.load(Ordering::Relaxed);
332        let uf_pct = if all_step > 0 {
333            100.0 * uf_step as f64 / all_step as f64
334        } else {
335            0.0
336        };
337        eprintln!(
338            "[FLOWLOG_DD_RULESET_PROF] uf_body_worker_step={:.3}s of {:.3}s total worker_step \
339             = {:.1}% (rules whose body reads a UF_* table)",
340            s(uf_step),
341            s(all_step),
342            uf_pct,
343        );
344    }
345}

363/// A planned DD join: canonical body-variable order + the table atoms.
364pub struct JoinPlan {
365    /// `var_order[i]` is the variable id at binding-row column `i`.
366    var_order: Vec<u32>,
367    /// var id -> binding-row column index.
368    var_col: HashMap<u32, usize>,
369    /// Body table atoms in emission order.
370    atoms: Vec<PlanAtom>,
371    /// `Some` ⇒ row projection (`FLOWLOG_PROJECT`) found a column-reusing layout
372    /// that fits `W` though the static var count does not; the binary chain uses
373    /// the per-step columns instead of `var_col`. `None` ⇒ the historical static
374    /// layout (every distinct var gets a permanent column).
375    projection: Option<ProjectionPlan>,
376}
377
378struct PlanAtom {
379    func: FunctionId,
380    slots: Vec<Slot>,
381}
382
383impl JoinPlan {
384    /// The variable id at each CAPTURED binding-row column, in column order. With
385    /// projection this is the reduced surviving-var set (the build packs them into
386    /// columns `0..head_vars.len()`); without, it is the full static var order.
387    /// Either way the head scatter reads captured-row column `i` for var
388    /// `var_order()[i]`, so it stays correct.
389    pub fn var_order(&self) -> Vec<u32> {
390        match &self.projection {
391            Some(p) => p.head_vars.clone(),
392            None => self.var_order.clone(),
393        }
394    }
395}

397/// Build the join plan for `rule`, or `Err(reason)` if the DD dataflow cannot
398/// support its shape (the caller PANICS — there is no host fallback). Supported:
399/// one or more table atoms, at most [`W`] distinct body vars, atom arity at most
400/// [`W`]. Body prims (`!=` guards, value prims like `+`) are re-run host-side
401/// over the bindings by the caller (the table-join-on-engine /
402/// prim-tail-host-side split), so we accept them by leaving them to the host tail.
403pub fn plan_join(rule: &RuleIr) -> Result<JoinPlan, String> {
404    let mut var_order: Vec<u32> = Vec::new();
405    let mut var_col: HashMap<u32, usize> = HashMap::new();
406    let mut atoms: Vec<PlanAtom> = Vec::new();
407
408    let see = |v: u32, var_order: &mut Vec<u32>, var_col: &mut HashMap<u32, usize>| {
409        if !var_col.contains_key(&v) {
410            var_col.insert(v, var_order.len());
411            var_order.push(v);
412        }
413    };
414
415    for op in &rule.body {
416        match op {
417            BodyOp::Atom(atom) => {
418                if atom.slots.len() > W {
419                    return Err(format!("atom arity {} > W {}", atom.slots.len(), W));
420                }
421                for s in &atom.slots {
422                    if let Slot::Var(v) = s {
423                        see(*v, &mut var_order, &mut var_col);
424                    }
425                }
426                atoms.push(PlanAtom {
427                    func: atom.func,
428                    slots: atom.slots.clone(),
429                });
430            }
431            // Body prims (e.g. `!=` guards, value prims like `+`) are re-run
432            // host-side over the join bindings by the caller (the table-join-on-
433            // engine / prim-tail-host-side split). They do not affect join
434            // planning; a value prim may bind a fresh var the head reads.
435            BodyOp::Prim { .. } => {}
436        }
437    }
438
439    if atoms.is_empty() {
440        return Err("no body table atoms (atom-less rule)".to_string());
441    }
442    // ROW PROJECTION (gated `FLOWLOG_PROJECT`, default-off): when the static
443    // layout exceeds `W`, attempt a per-step register allocation that reuses
444    // binding-row columns for variables whose liveness intervals do not overlap
445    // (a body subterm var dies once the atom that consumes it has joined). If the
446    // reused-column frontier still fits `W`, the binary chain can run the rule on
447    // the existing fixed-width `Row` — this is what lets the giant Herbie seed
448    // rules (962 vars, frontier ~22) lower. Off ⇒ the historical bail below.
449    if project_enabled() {
450        if let Some(proj) = build_projection(&atoms, &rule.head, &rule.body) {
451            return Ok(JoinPlan {
452                var_order,
453                var_col,
454                atoms,
455                projection: Some(proj),
456            });
457        }
458        // Projection could not fit `W` (frontier too wide) ⇒ fall through to the
459        // historical reject.
460    }
461
462    if var_order.len() > W {
463        return Err(format!("too many body vars {} > W {}", var_order.len(), W));
464    }
465
466    Ok(JoinPlan {
467        var_order,
468        var_col,
469        atoms,
470        projection: None,
471    })
472}

483/// A per-step binding-row column layout produced by [`build_projection`]: column
484/// reuse (linear-scan register allocation over body-atom liveness) that keeps the
485/// frontier within [`W`] so a rule whose STATIC var count exceeds `W` can still
486/// run on the fixed-width `Row`. See the call site in [`plan_join`].
487#[derive(Clone, Debug)]
488pub(crate) struct ProjectionPlan {
489    /// `step_col[i]` maps each variable LIVE during step `i` (atom `i`'s join) to
490    /// its binding-row column at that step. A var's column is stable from its
491    /// birth step to its death step but may differ across non-overlapping vars.
492    pub step_col: Vec<HashMap<u32, usize>>,
493    /// The surviving (head/body-prim-relevant) variables and their FINAL columns,
494    /// in a deterministic order. Drives the reduced head-scatter `var_order`.
495    pub head_vars: Vec<u32>,
496    /// Final column of each surviving var (parallel to `head_vars`).
497    pub head_cols: Vec<usize>,
498}

500/// Build a column-reusing layout for the body atoms, or `None` if the reused
501/// frontier still exceeds `W`. Liveness is first-use..last-use over the EMITTED
502/// atom order, EXTENDED so any variable read by the head or a body prim stays
503/// live to the end (it must survive into the captured row). Linear-scan slot
504/// assignment: a column freed by a dead var is reused by a later var's birth.
505fn build_projection(
506    atoms: &[PlanAtom],
507    head: &[crate::compile::HeadOp],
508    body: &[BodyOp],
509) -> Option<ProjectionPlan> {
510    use hashbrown::HashSet;
511    let n = atoms.len();
512
513    // Variables the head / body prims read: these must survive to the end.
514    let mut survivor: HashSet<u32> = HashSet::new();
515    collect_head_vars(head, &mut survivor);
516    for op in body {
517        if let BodyOp::Prim { args, ret, .. } = op {
518            for s in args {
519                if let Slot::Var(v) = s {
520                    survivor.insert(*v);
521                }
522            }
523            if let Slot::Var(v) = ret {
524                survivor.insert(*v);
525            }
526        }
527    }
528
529    // first[v]/last[v] = first/last atom index where v appears. Survivors get
530    // last = n-1 (live to the final step) so they are never freed mid-chain.
531    let mut first: HashMap<u32, usize> = HashMap::new();
532    let mut last: HashMap<u32, usize> = HashMap::new();
533    for (i, a) in atoms.iter().enumerate() {
534        for s in &a.slots {
535            if let Slot::Var(v) = s {
536                first.entry(*v).or_insert(i);
537                last.insert(*v, i);
538            }
539        }
540    }
541    for v in &survivor {
542        if let Some(l) = last.get_mut(v) {
543            *l = n - 1;
544        }
545    }
546
547    // Vars born / dying at each step (in deterministic var-id order so the layout
548    // is reproducible).
549    let mut births: Vec<Vec<u32>> = vec![Vec::new(); n];
550    let mut deaths: Vec<Vec<u32>> = vec![Vec::new(); n];
551    let mut vars: Vec<u32> = first.keys().copied().collect();
552    vars.sort_unstable();
553    for v in vars {
554        births[first[&v]].push(v);
555        deaths[last[&v]].push(v);
556    }
557
558    // Linear-scan register allocation. `col_of` is the current column of each
559    // live var; `free` is a stack of reusable columns (lowest first, so the
560    // layout is deterministic); `next` is the high-water column allocator.
561    let mut col_of: HashMap<u32, usize> = HashMap::new();
562    let mut free: Vec<usize> = Vec::new();
563    let mut next: usize = 0;
564    let mut step_col: Vec<HashMap<u32, usize>> = Vec::with_capacity(n);
565    let mut peak = 0usize;
566    for i in 0..n {
567        // Births first (so this atom's fresh vars get columns before we snapshot).
568        for &v in &births[i] {
569            let c = if let Some(c) = free.pop() {
570                c
571            } else {
572                let c = next;
573                next += 1;
574                c
575            };
576            col_of.insert(v, c);
577        }
578        peak = peak.max(col_of.len());
579        if peak > W {
580            return None;
581        }
582        // Snapshot the layout AT this step (after births, before deaths) — every
583        // var the atom touches plus all still-live carried vars are present.
584        step_col.push(col_of.clone());
585        // Deaths free their columns for reuse by later births.
586        for &v in &deaths[i] {
587            if let Some(c) = col_of.remove(&v) {
588                free.push(c);
589            }
590            // freeing low-first keeps assignment deterministic
591            free.sort_unstable_by(|a, b| b.cmp(a));
592        }
593    }
594
595    // Surviving vars + their FINAL columns (the columns they hold at the last
596    // step, where they are guaranteed live). Deterministic order by var id.
597    let final_layout = &step_col[n - 1];
598    let mut head_vars: Vec<u32> = survivor
599        .into_iter()
600        .filter(|v| final_layout.contains_key(v))
601        .collect();
602    head_vars.sort_unstable();
603    let head_cols: Vec<usize> = head_vars.iter().map(|v| final_layout[v]).collect();
604
605    Some(ProjectionPlan {
606        step_col,
607        head_vars,
608        head_cols,
609    })
610}

612/// Collect every variable a head op references into `out`.
613fn collect_head_vars(head: &[crate::compile::HeadOp], out: &mut hashbrown::HashSet<u32>) {
614    use crate::compile::HeadOp;
615    let add = |s: &Slot, out: &mut hashbrown::HashSet<u32>| {
616        if let Slot::Var(v) = s {
617            out.insert(*v);
618        }
619    };
620    for op in head {
621        match op {
622            HeadOp::Set { slots, .. }
623            | HeadOp::Remove { slots, .. }
624            | HeadOp::Subsume { slots, .. } => {
625                for s in slots {
626                    add(s, out);
627                }
628            }
629            HeadOp::Lookup { args, ret, .. } => {
630                for s in args {
631                    add(s, out);
632                }
633                out.insert(*ret);
634            }
635            HeadOp::Call { args, ret, .. } => {
636                for s in args {
637                    add(s, out);
638                }
639                out.insert(*ret);
640            }
641            HeadOp::Union { l, r } => {
642                add(l, out);
643                add(r, out);
644            }
645            HeadOp::Panic(_) => {}
646        }
647    }

650// ===========================================================================
651// `--wcoj` triangle worst-case-optimal join
652// ===========================================================================
653//
654// Detects the reverse-distributivity triangle rule
655//   (rewrite (Add (Mul a b) (Mul a c)) (Mul a (Add b c)))
656// which lowers (term encoding) to three arity-4 atoms
657//   A0  Mul(a, b, m1, x1)     [a, b,  m1, x1]
658//   A1  Mul(a, c, m2, x2)     [a, c,  m2, x2]
659//   A2  Add(m1, m2, o, x3)    [m1, m2, o,  x3]
660// where A0,A1 are the SAME relation (a self-join sharing col-0 `a`), A2 is a
661// different relation joining A0.col2=m1 and A1.col2=m2, and every named
662// variable is distinct. The cyclic core is the triangle on (a, m1, m2): the
663// binary join Mul_a ⋈ Mul_b on `a` materializes a `Σ_a deg(a)²` intermediate
664// the WCOJ collapses to the output.
665
666/// The recognized triangle, with the BINDING-ROW column index of each of the 9
667/// distinct variables (a,b,m1,x1,c,m2,x2,o,x3) plus the two relation ids. All
668/// downstream `extend_using` key selectors and the final `Row` assembly index
669/// the binding row by these columns, so the WCOJ emits the SAME 9-column
670/// binding row as the binary join (bit-exact head firing).
671#[derive(Clone, Debug)]
672pub(crate) struct TriangleShape {
673    mul_func: FunctionId,
674    add_func: FunctionId,
675    // binding-row columns (positions in the n_vars-wide Row)
676    col_a: usize,
677    col_b: usize,
678    col_m1: usize,
679    col_x1: usize,
680    col_c: usize,
681    col_m2: usize,
682    col_x2: usize,
683    col_o: usize,
684    col_x3: usize,
685}
686
687/// Recognize the triangle shape in `plan`, or `None` if it does not match.
688/// Structural (no name/func-id hardcoding): three arity-4 atoms, atoms 0 and 1
689/// the same relation sharing exactly their col-0 var, atom 2 a different
690/// relation whose col-0 = atom0.col2 and col-1 = atom1.col2, and all 9 named
691/// variables distinct. Stage 1 recognizes exactly this 3-atom triangle;
692/// generalization is Stage 2.
693pub(crate) fn detect_triangle(plan: &JoinPlan) -> Option<TriangleShape> {
694    if plan.atoms.len() != 3 {
695        return None;
696    }
697    // Each atom must be arity 4 with all-distinct Var slots (no consts, no
698    // repeated vars within an atom).
699    let atom_var = |i: usize| -> Option<[u32; 4]> {
700        let s = &plan.atoms[i].slots;
701        if s.len() != 4 {
702            return None;
703        }
704        let mut out = [0u32; 4];
705        for (j, slot) in s.iter().enumerate() {
706            match slot {
707                Slot::Var(v) => out[j] = *v,
708                Slot::Const(_) => return None,
709            }
710        }
711        // distinct within the atom
712        for j in 0..4 {
713            for k in (j + 1)..4 {
714                if out[j] == out[k] {
715                    return None;
716                }
717            }
718        }
719        Some(out)
720    };
721    let a0 = atom_var(0)?;
722    let a1 = atom_var(1)?;
723    let a2 = atom_var(2)?;
724
725    let mul_func = plan.atoms[0].func;
726    // A0, A1 same relation; A2 a different relation.
727    if plan.atoms[1].func != mul_func {
728        return None;
729    }
730    let add_func = plan.atoms[2].func;
731    if add_func == mul_func {
732        return None;
733    }
734
735    // A0 = [a, b, m1, x1]; A1 = [a, c, m2, x2] — share col-0 (a), nothing else.
736    let (a, b, m1, x1) = (a0[0], a0[1], a0[2], a0[3]);
737    let (a_, c, m2, x2) = (a1[0], a1[1], a1[2], a1[3]);
738    if a != a_ {
739        return None;
740    }
741    // A2 = [m1, m2, o, x3] — col0 = A0.col2 (m1), col1 = A1.col2 (m2).
742    let (m1_, m2_, o, x3) = (a2[0], a2[1], a2[2], a2[3]);
743    if m1_ != m1 || m2_ != m2 {
744        return None;
745    }
746
747    // All 9 named variables distinct (rules out degenerate self-overlap the
748    // generic shape would not produce; the binary join would treat such a rule
749    // differently, so we leave it to the binary path).
750    let vars = [a, b, m1, x1, c, m2, x2, o, x3];
751    for i in 0..vars.len() {
752        for j in (i + 1)..vars.len() {
753            if vars[i] == vars[j] {
754                return None;
755            }
756        }
757    }
758
759    let col = |v: u32| -> usize { plan.var_col[&v] };
760    Some(TriangleShape {
761        mul_func,
762        add_func,
763        col_a: col(a),
764        col_b: col(b),
765        col_m1: col(m1),
766        col_x1: col(x1),
767        col_c: col(c),
768        col_m2: col(m2),
769        col_x2: col(x2),
770        col_o: col(o),
771        col_x3: col(x3),
772    })
773}

775// ===========================================================================
776// Stage 2: GENERAL incremental WCOJ for an arbitrary >=3-atom CYCLIC body
777// ===========================================================================
778//
779// Generalizes the hardcoded triangle (above) to any >=3-atom rule whose body
780// join is CYCLIC — the class where a binary-join chain materializes an
781// intermediate exceeding the output (the AGM blowup WCOJ removes). Acyclic and
782// <=2-atom rules keep the binary `.join` chain (hybrid, Free-Join style): WCOJ
783// buys nothing there and adds per-prefix index/intersection overhead.
784//
785// ## Detector (`detect_cyclic_cq`)
786//
787// Structural, no name/func-id dependence:
788//   1. >= 3 table atoms, all slots distinct `Var`s within each atom, no consts
789//      (a const/repeated-var atom is a SELECTION the binary path handles; we
790//      leave it on the binary chain to stay conservative + bit-exact).
791//   2. The body hypergraph is CYCLIC by GYO reduction (repeatedly drop an "ear"
792//      vertex appearing in <=1 remaining edge, and any edge subset of another;
793//      acyclic iff it reduces to empty). Cyclic ⇒ the binary chain blows up.
794//   3. A full WCOJ plan is constructible (see `build_cq_plan`): for EVERY
795//      driver atom there is a variable order extending the driver one variable
796//      at a time, each step binding a variable that is the LAST unbound variable
797//      of some atom (so that atom keys entirely on already-bound columns). If
798//      any driver cannot be covered this way, we BAIL (return None → binary
799//      chain). This is the "only fire where provably bit-exact" stance: the
800//      construction mirrors the Stage-1 triangle exactly (full k-stream delta
801//      decomposition, ALL reads inside the AltNeu scope), and we only emit it
802//      when every variable is bound by an in-scope extend/propose.
803//
804// ## Plan (`CqPlan`)
805//
806// `var_col` (carried) + a `CqDriver` per atom. Each driver records its initial
807// bound columns (the driver atom's vars, written straight from the delta row)
808// then a sequence of `CqStep`s, each extending the prefix with ONE variable
809// `bind_col` via a set of `CqExtender`s (`atom_idx` + the prefix columns the
810// atom keys on + the relation column it proposes + whether the atom reads ALT
811// (atom index < driver) or NEU (atom index > driver)). >=2 extenders for a step
812// ⇒ multiway `extend` (the WCOJ intersection); exactly 1 ⇒ `propose_using`.
813
814/// One extender in a WCOJ extension step: read relation `atom_idx`'s `propose_col`
815/// keyed by the prefix's `key_cols`, from the ALT (old) or NEU (new) trace.
816#[derive(Clone, Debug)]
817pub(crate) struct CqExtender {
818    /// Index into `CqPlan.atom_funcs` / the rule's atom list.
819    atom_idx: usize,
820    /// Binding-row columns (already bound in the prefix) this atom keys on.
821    /// Parallel to `key_atom_cols`.
822    key_cols: Vec<usize>,
823    /// The atom's relation columns (0-based slot positions) holding the key
824    /// vars, parallel to `key_cols` — used to build the atom's lookup index so
825    /// its key matches the prefix key_selector's columns.
826    key_atom_cols: Vec<usize>,
827    /// The relation column (0-based in the atom's row) that proposes the var.
828    propose_col: usize,
829    /// true ⇒ read the ALT (old) trace; false ⇒ the NEU (new) trace.
830    alt: bool,
831}
832
833/// One JOIN-variable extension step: bind binding-row column `bind_col` from
834/// `exts`. The intersected variable is a JOIN variable (appears in >=2 atoms).
835/// With >=2 extenders this is the multiway `extend` (the WCOJ intersection that
836/// collapses the intermediate); with a single extender it is a `propose_using`.
837#[derive(Clone, Debug)]
838pub(crate) struct CqStep {
839    /// The binding-row column this step writes.
840    bind_col: usize,
841    exts: Vec<CqExtender>,
842}
843
844/// A per-atom PAYLOAD recovery: once an atom's JOIN columns are all bound,
845/// recover its payload variables (those appearing only in this atom) in ONE
846/// `propose_using` keyed on the atom's join columns. `payload_cols[k]` =
847/// `(binding_col, relation_col)` for the k-th payload var.
848#[derive(Clone, Debug)]
849pub(crate) struct CqRecover {
850    atom_idx: usize,
851    /// Binding columns (already bound) the atom keys on. Parallel to
852    /// `key_atom_cols`.
853    key_cols: Vec<usize>,
854    /// The atom's relation columns holding the key (join) vars.
855    key_atom_cols: Vec<usize>,
856    /// `(binding_col, relation_col)` per payload var to write from the proposal.
857    payload: Vec<(usize, usize)>,
858    /// true ⇒ read the ALT (old) trace; false ⇒ the NEU (new) trace.
859    alt: bool,
860}

862/// One delta query `dQ/dA_driver`: seed the prefix from the driver atom's delta
863/// row, bind every JOIN variable via `steps` (the WCOJ core), then recover each
864/// non-driver atom's payload via `recovers`. ALL inside the AltNeu scope.
865#[derive(Clone, Debug)]
866pub(crate) struct CqDriver {
867    /// The driver atom's index.
868    driver_idx: usize,
869    /// `(binding_col, relation_col)` for each of the driver atom's variables —
870    /// written straight from the delta row into the initial prefix.
871    seed: Vec<(usize, usize)>,
872    steps: Vec<CqStep>,
873    recovers: Vec<CqRecover>,
874}

885/// The variables of an atom (its distinct `Var` slots), or `None` if the atom
886/// has any const or repeated var (we leave those selections on the binary path).
887fn atom_distinct_vars(slots: &[Slot]) -> Option<Vec<u32>> {
888    let mut out: Vec<u32> = Vec::with_capacity(slots.len());
889    for s in slots {
890        match s {
891            Slot::Var(v) => {
892                if out.contains(v) {
893                    return None;
894                }
895                out.push(*v);
896            }
897            Slot::Const(_) => return None,
898        }
899    }
900    Some(out)
901}

903/// GYO acyclicity test on the body hypergraph (atoms = hyperedges, vars =
904/// vertices). Returns `true` iff the query is CYCLIC (GYO does not reduce to
905/// empty). `atom_vars[i]` is atom i's vertex set.
906fn is_cyclic_cq(atom_vars: &[Vec<u32>]) -> bool {
907    // Work on owned mutable vertex sets; `removed[i]` drops an absorbed edge.
908    let mut edges: Vec<HashSet<u32>> = atom_vars
909        .iter()
910        .map(|vs| vs.iter().copied().collect())
911        .collect();
912    let mut removed = vec![false; edges.len()];
913
914    loop {
915        let mut progress = false;
916
917        // (a) Ear removal: drop a vertex that appears in <= 1 live edge.
918        let mut vert_count: HashMap<u32, usize> = HashMap::new();
919        for (i, e) in edges.iter().enumerate() {
920            if removed[i] {
921                continue;
922            }
923            for &v in e {
924                *vert_count.entry(v).or_insert(0) += 1;
925            }
926        }
927        let ears: Vec<u32> = vert_count
928            .iter()
929            .filter(|&(_, &c)| c <= 1)
930            .map(|(&v, _)| v)
931            .collect();
932        if !ears.is_empty() {
933            for e in edges.iter_mut() {
934                for v in &ears {
935                    e.remove(v);
936                }
937            }
938            progress = true;
939        }
940
941        // (b) Absorb an edge that is a subset of another live edge.
942        let live: Vec<usize> = (0..edges.len()).filter(|&i| !removed[i]).collect();
943        'outer: for &i in &live {
944            if edges[i].is_empty() {
945                removed[i] = true;
946                progress = true;
947                continue;
948            }
949            for &j in &live {
950                if i == j || removed[j] {
951                    continue;
952                }
953                if edges[i].is_subset(&edges[j]) {
954                    removed[i] = true;
955                    progress = true;
956                    continue 'outer;
957                }
958            }
959        }
960
961        if !progress {
962            break;
963        }
964    }
965
966    // Cyclic iff any non-empty edge remains (GYO got stuck).
967    edges
968        .iter()
969        .enumerate()
970        .any(|(i, e)| !removed[i] && !e.is_empty())
971}

973/// Build the general WCOJ plan for `plan`, or `None` if it is not constructible
974/// bit-exactly. Splits variables into JOIN vars (in >=2 atoms — the cyclic core)
975/// and PAYLOAD vars (in exactly 1 atom — eclass/extra columns). The WCOJ binds
976/// each JOIN var one at a time (each being the last-unbound JOIN var of >=1
977/// atom, so that atom keys on already-bound join columns), then recovers each
978/// atom's payload via one `propose`. If any driver's join vars cannot be ordered
979/// this way, bail (`None` ⇒ binary chain).
980fn build_cq_plan(plan: &JoinPlan, atom_vars: &[Vec<u32>]) -> Option<CqPlan> {
981    let n_atoms = plan.atoms.len();
982    let var_col = &plan.var_col;
983    let atom_funcs: Vec<FunctionId> = plan.atoms.iter().map(|a| a.func).collect();
984
985    // relation-col lookup: atom i, variable v -> the 0-based slot position.
986    let atom_var_col = |i: usize, v: u32| -> Option<usize> {
987        plan.atoms[i]
988            .slots
989            .iter()
990            .position(|s| matches!(s, Slot::Var(x) if *x == v))
991    };
992
993    // JOIN vars = vars appearing in >= 2 atoms. PAYLOAD vars = in exactly 1.
994    let mut var_atom_count: HashMap<u32, usize> = HashMap::new();
995    for vs in atom_vars {
996        for &v in vs {
997            *var_atom_count.entry(v).or_insert(0) += 1;
998        }
999    }
1000    let is_join = |v: u32| -> bool { var_atom_count.get(&v).copied().unwrap_or(0) >= 2 };
1001    let join_vars_of = |i: usize| -> Vec<u32> {
1002        atom_vars[i]
1003            .iter()
1004            .copied()
1005            .filter(|&v| is_join(v))
1006            .collect()
1007    };
1008    let all_join_vars: HashSet<u32> = atom_vars
1009        .iter()
1010        .flatten()
1011        .copied()
1012        .filter(|&v| is_join(v))
1013        .collect();
1014
1015    let mut drivers: Vec<CqDriver> = Vec::with_capacity(n_atoms);
1016    for driver_idx in 0..n_atoms {
1017        // Seed: the driver atom's own variables (join + payload), straight from
1018        // the delta row — they are all immediately bound.
1019        let mut bound: HashSet<u32> = atom_vars[driver_idx].iter().copied().collect();
1020        let seed: Vec<(usize, usize)> = atom_vars[driver_idx]
1021            .iter()
1022            .map(|&v| (var_col[&v], atom_var_col(driver_idx, v).unwrap()))
1023            .collect();
1024
1025        // Bind JOIN vars one at a time. The next join var must be the LAST
1026        // unbound JOIN var of >= 1 atom (payload vars never block — they are
1027        // recovered after, not used as keys). Prefer the var constrained by the
1028        // MOST atoms (most-constrained-variable = the deepest WCOJ
1029        // intersection); ties broken by var id for determinism.
1030        let mut steps: Vec<CqStep> = Vec::new();
1031        while bound.iter().filter(|&&v| is_join(v)).count() < all_join_vars.len() {
1032            let mut best: Option<(u32, Vec<usize>)> = None;
1033            for &v in &all_join_vars {
1034                if bound.contains(&v) {
1035                    continue;
1036                }
1037                let mut ready_atoms: Vec<usize> = Vec::new();
1038                for (i, vs) in atom_vars.iter().enumerate() {
1039                    if !vs.contains(&v) {
1040                        continue;
1041                    }
1042                    // v is bindable from atom i iff every OTHER JOIN var of atom
1043                    // i is already bound.
1044                    if vs
1045                        .iter()
1046                        .all(|&w| w == v || !is_join(w) || bound.contains(&w))
1047                    {
1048                        ready_atoms.push(i);
1049                    }
1050                }
1051                if ready_atoms.is_empty() {
1052                    continue;
1053                }
1054                let better = match &best {
1055                    None => true,
1056                    Some((bv, ba)) => {
1057                        (ready_atoms.len(), std::cmp::Reverse(v))
1058                            > (ba.len(), std::cmp::Reverse(*bv))
1059                    }
1060                };
1061                if better {
1062                    best = Some((v, ready_atoms));
1063                }
1064            }
1065
1066            let (v, ready_atoms) = best?; // no ready join var ⇒ not constructible.
1067            let exts: Vec<CqExtender> = ready_atoms
1068                .iter()
1069                .map(|&i| {
1070                    // Key on the atom's OTHER join vars (all bound by now).
1071                    let key_vars: Vec<u32> =
1072                        join_vars_of(i).into_iter().filter(|&w| w != v).collect();
1073                    CqExtender {
1074                        atom_idx: i,
1075                        key_cols: key_vars.iter().map(|w| var_col[w]).collect(),
1076                        key_atom_cols: key_vars
1077                            .iter()
1078                            .map(|&w| atom_var_col(i, w).unwrap())
1079                            .collect(),
1080                        propose_col: atom_var_col(i, v).unwrap(),
1081                        alt: i < driver_idx,
1082                    }
1083                })
1084                .collect();
1085            steps.push(CqStep {
1086                bind_col: var_col[&v],
1087                exts,
1088            });
1089            bound.insert(v);
1090        }
1091
1092        // Recover each NON-driver atom's payload vars (those appearing only in
1093        // it), keyed on the atom's join columns (all bound). The driver atom's
1094        // payload is already in the seed. An atom with NO payload var still must
1095        // be VALIDATED so its tuple constrains the result; we emit a recover
1096        // with an empty payload (a pure existence `propose`). That is sound (no
1097        // over-count): empty payload ⟺ every column is a join var ⟺ the key
1098        // covers the whole row, and rows are set-semantic, so the key matches at
1099        // most one row (weight 1). In the term encoding every atom carries an
1100        // eclass + extra column, so this empty-payload case does not arise.
1101        let mut recovers: Vec<CqRecover> = Vec::new();
1102        for (i, avs) in atom_vars.iter().enumerate() {
1103            if i == driver_idx {
1104                continue;
1105            }
1106            let payload_vars: Vec<u32> = avs.iter().copied().filter(|&v| !is_join(v)).collect();
1107            let key_vars = join_vars_of(i);
1108            recovers.push(CqRecover {
1109                atom_idx: i,
1110                key_cols: key_vars.iter().map(|w| var_col[w]).collect(),
1111                key_atom_cols: key_vars
1112                    .iter()
1113                    .map(|&w| atom_var_col(i, w).unwrap())
1114                    .collect(),
1115                payload: payload_vars
1116                    .iter()
1117                    .map(|&v| (var_col[&v], atom_var_col(i, v).unwrap()))
1118                    .collect(),
1119                alt: i < driver_idx,
1120            });
1121        }
1122
1123        drivers.push(CqDriver {
1124            driver_idx,
1125            seed,
1126            steps,
1127            recovers,
1128        });
1129    }
1130
1131    Some(CqPlan {
1132        atom_funcs,
1133        drivers,
1134    })
1135}

1137/// Detect a general WCOJ-worthy body: a >=3-atom CYCLIC query for which a
1138/// bit-exact WCOJ plan is constructible. Returns the plan, or `None` (stay on
1139/// the binary chain). See the module note above for the criteria.
1140pub(crate) fn detect_cyclic_cq(plan: &JoinPlan, allow_acyclic: bool) -> Option<CqPlan> {
1141    let trace = std::env::var_os("FLOWLOG_WCOJ_TRACE").is_some();
1142    if plan.atoms.len() < 3 {
1143        return None;
1144    }
1145    // All atoms must be pure distinct-Var atoms (no const/repeated-var
1146    // selections — those stay on the binary path).
1147    let mut atom_vars: Vec<Vec<u32>> = Vec::with_capacity(plan.atoms.len());
1148    for a in &plan.atoms {
1149        match atom_distinct_vars(&a.slots) {
1150            Some(vs) => atom_vars.push(vs),
1151            None => {
1152                #[allow(clippy::disallowed_macros)]
1153                if trace {
1154                    eprintln!(
1155                        "[WCOJ] detect_cyclic_cq: bail (atom with const/repeated var) atoms={}",
1156                        plan.atoms.len()
1157                    );
1158                }
1159                return None;
1160            }
1161        }
1162    }
1163    // Only fire where the binary chain provably blows up: a CYCLIC join.
1164    let cyclic = is_cyclic_cq(&atom_vars);
1165    #[allow(clippy::disallowed_macros)]
1166    if trace {
1167        eprintln!(
1168            "[WCOJ] detect_cyclic_cq: atoms={} cyclic={cyclic} atom_vars={atom_vars:?}",
1169            plan.atoms.len()
1170        );
1171    }
1172    // Broadening WCOJ to acyclic >=3-atom rules is off by default (callers
1173    // pass `allow_acyclic: false`). Tests that verify acyclic correctness
1174    // pass `true` explicitly.
1175    if !cyclic && !allow_acyclic {
1176        return None;
1177    }
1178    let plan_out = build_cq_plan(plan, &atom_vars);
1179    #[allow(clippy::disallowed_macros)]
1180    if trace && plan_out.is_none() {
1181        eprintln!("[WCOJ] detect_cyclic_cq: cyclic but NOT constructible -> binary chain");
1182    }
1183    plan_out
1184}

1186/// The persistent, in-process DD body join for one rule. Built once; driven
1187/// across epochs by [`step`]. Owns its timely `Worker`, so it can be stepped
1188/// between host iterations without re-spawning threads.
1189pub struct PersistentDdJoin {
1190    worker: Worker,
1191    /// One input session per atom occurrence (in `plan.atoms` order).
1192    inputs: Vec<InputSession<u32, Row, isize>>,
1193    /// Probe on the output, to `step_while` until the epoch is fully processed.
1194    probe: ProbeHandle<u32>,
1195    /// Shared capture buffer: the output binding deltas of the most-recent
1196    /// epoch, drained by [`step`]. `inspect_batch` appends `(row, weight)`.
1197    captured: Rc<RefCell<Vec<(Row, isize)>>>,
1198    /// `func` -> atom-occurrence indices reading it (self-join fan-out).
1199    occ_of_func: HashMap<FunctionId, Vec<usize>>,
1200    /// Number of canonical body variables (binding-row width in use).
1201    n_vars: usize,
1202    /// Current epoch (monotonic; advanced once per [`step`]).
1203    epoch: u32,
1204}
1205
1206impl PersistentDdJoin {
1207    /// Build the persistent DD dataflow for `plan` ONCE.
1208    pub fn build(plan: &JoinPlan) -> Result<PersistentDdJoin> {
1209        let alloc = Allocator::Thread(Thread::default());
1210        let mut worker = Worker::new(
1211            WorkerConfig::default(),
1212            alloc,
1213            Some(std::time::Instant::now()),
1214        );
1215        if prof_enabled() {
1216            PROF_WORKERS.fetch_add(1, Ordering::Relaxed);
1217            PROF_INPUT_SESSIONS.fetch_add(plan.atoms.len() as u64, Ordering::Relaxed);
1218        }
1219
1220        let n_atoms = plan.atoms.len();
1221        let atom_slots: Vec<Vec<Slot>> = plan.atoms.iter().map(|a| a.slots.clone()).collect();
1222        let var_col = plan.var_col.clone();
1223        let n_vars = plan.var_order.len();
1224        let captured: Rc<RefCell<Vec<(Row, isize)>>> = Rc::new(RefCell::new(Vec::new()));
1225        let captured_in = Rc::clone(&captured);
1226        let probe: ProbeHandle<u32> = ProbeHandle::new();
1227        let probe_in = probe.clone();
1228
1229        let inputs = worker.dataflow::<u32, _, _>(|scope| {
1230            let mut inputs: Vec<InputSession<u32, Row, isize>> = Vec::with_capacity(n_atoms);
1231            let mut collections = Vec::with_capacity(n_atoms);
1232            for _ in 0..n_atoms {
1233                let mut session: InputSession<u32, Row, isize> = InputSession::new();
1234                let coll = session.to_collection(scope);
1235                inputs.push(session);
1236                collections.push(coll);
1237            }
1238
1239            // Left-deep join. Start from atom 0's bindings (map each relation row
1240            // into a canonical binding row, dropping rows whose const / repeated-
1241            // var constraints fail). Then join successive atoms on shared bound
1242            // columns. `bound[c]` tracks which canonical columns are filled.
1243            let mut bound = vec![false; n_vars];
1244
1245            let slots0 = atom_slots[0].clone();
1246            let vc0 = var_col.clone();
1247            let mut cur = collections[0]
1248                .clone()
1249                .flat_map(move |r: Row| bind_atom(&r, &slots0, &vc0));
1250            mark_bound(&atom_slots[0], &var_col, &mut bound);
1251
1252            for i in 1..n_atoms {
1253                let slots = atom_slots[i].clone();
1254                // Shared variables = atom vars already bound by a previous atom.
1255                let shared: Vec<u32> = atom_vars(&slots)
1256                    .into_iter()
1257                    .filter(|v| var_col.get(v).map(|&c| bound[c]).unwrap_or(false))
1258                    .collect();
1259
1260                // Left side: key the current bindings by the shared canonical
1261                // columns. Right side: key the atom rows by the matching slot
1262                // positions. `.join` then yields `(key, (bind, relrow))` pairs.
1263                let shared_cols_left: Vec<usize> = shared.iter().map(|v| var_col[v]).collect();
1264                let shared_atom_cols: Vec<usize> = shared
1265                    .iter()
1266                    .map(|v| {
1267                        slots
1268                            .iter()
1269                            .position(|s| matches!(s, Slot::Var(x) if x == v))
1270                            .expect("shared var present in atom")
1271                    })
1272                    .collect();
1273
1274                let scl = shared_cols_left.clone();
1275                let left = cur.map(move |b: Row| (pack_key(&b, &scl), b));
1276                let sac = shared_atom_cols.clone();
1277                let right = collections[i]
1278                    .clone()
1279                    .map(move |r: Row| (pack_key(&r, &sac), r));
1280
1281                let slotsc = slots.clone();
1282                let vc = var_col.clone();
1283                let bound_now = bound.clone();
1284                cur = left
1285                    .join(right)
1286                    .flat_map(move |(_k, (b, r)): (Key, (Row, Row))| {
1287                        merge_atom_into(&b, &r, &slotsc, &vc, &bound_now)
1288                    });
1289
1290                mark_bound(&slots, &var_col, &mut bound);
1291            }
1292
1293            // `.distinct()` makes the binding set set-semantic (weights ±1),
1294            // matching egglog relations. `.consolidate()` collapses any
1295            // multiplicities before capture.
1296            let out = cur
1297                .map(|b: Row| (b, ()))
1298                .distinct()
1299                .map(|(b, ())| b)
1300                .consolidate();
1301
1302            // Capture the per-epoch output delta into the shared buffer via the
1303            // raw timely stream's batch inspector (DD `inspect_batch` gives us
1304            // `&[(Row, time, isize)]`). We DON'T integrate — each epoch we read
1305            // exactly the delta produced at that epoch's time.
1306            let cap = Rc::clone(&captured_in);
1307            out.inner
1308                .inspect_batch(move |_t, batch| {
1309                    let mut buf = cap.borrow_mut();
1310                    for (row, _time, w) in batch.iter() {
1311                        buf.push((*row, *w));
1312                    }
1313                })
1314                .probe_with(&probe_in);
1315
1316            inputs
1317        });
1318
1319        let mut occ_of_func: HashMap<FunctionId, Vec<usize>> = HashMap::new();
1320        for (ord, a) in plan.atoms.iter().enumerate() {
1321            occ_of_func.entry(a.func).or_default().push(ord);
1322        }
1323
1324        Ok(PersistentDdJoin {
1325            worker,
1326            inputs,
1327            probe,
1328            captured,
1329            occ_of_func,
1330            n_vars,
1331            epoch: 0,
1332        })
1333    }
1334
1335    /// Feed one epoch of signed relation deltas, advance the timestamp, run the
1336    /// worker to this epoch's fixpoint, and return the resulting binding delta as
1337    /// `(binding_row_as_var_order_vec, weight)`.
1338    ///
1339    /// CRUCIAL: the InputSessions are NEVER cleared — only the delta is pushed,
1340    /// so the DD arrangements persist and the join is genuinely incremental.
1341    pub fn step(
1342        &mut self,
1343        deltas: &HashMap<FunctionId, Vec<(Vec<u32>, isize)>>,
1344    ) -> Result<Vec<(Vec<u32>, isize)>> {
1345        let prof = prof_enabled();
1346        let t_feed = std::time::Instant::now();
1347        let mut pushed = false;
1348        for (func, rows) in deltas {
1349            if let Some(occs) = self.occ_of_func.get(func) {
1350                for &ord in occs {
1351                    for (row, w) in rows {
1352                        self.inputs[ord].update(pack_row(row), *w);
1353                        pushed = true;
1354                    }
1355                }
1356            }
1357        }
1358        // No input change for this rule's body ⇒ no new bindings: skip the epoch
1359        // entirely (still advance the logical clock so a later real delta is
1360        // ordered after). This short-circuits the many no-op rebuild re-runs.
1361        let next_epoch = self.epoch + 1;
1362        if !pushed {
1363            self.epoch = next_epoch;
1364            return Ok(Vec::new());
1365        }
1366
1367        self.captured.borrow_mut().clear();
1368
1369        // Advance every input to the next epoch and flush, then step the worker
1370        // until the probe frontier passes the epoch we just closed — i.e. all
1371        // output for this epoch has been produced.
1372        for inp in &mut self.inputs {
1373            inp.advance_to(next_epoch);
1374            inp.flush();
1375        }
1376        if prof {
1377            add_ns(&PROF_FEED_NS, t_feed.elapsed());
1378            PROF_STEP_CALLS.fetch_add(1, Ordering::Relaxed);
1379        }
1380        let t_step = std::time::Instant::now();
1381        let probe = self.probe.clone();
1382        self.worker.step_while(|| probe.less_than(&next_epoch));
1383        if prof {
1384            add_ns(&PROF_STEP_NS, t_step.elapsed());
1385        }
1386        self.epoch = next_epoch;
1387
1388        // Drain the capture buffer; consolidate duplicate rows that may have been
1389        // emitted across multiple internal steps within the epoch.
1390        let mut acc: HashMap<Vec<u32>, isize> = HashMap::new();
1391        for (row, w) in self.captured.borrow_mut().drain(..) {
1392            let key: Vec<u32> = (0..self.n_vars).map(|i| row[i]).collect();
1393            *acc.entry(key).or_insert(0) += w;
1394        }
1395        let out: Vec<(Vec<u32>, isize)> = acc.into_iter().filter(|(_, w)| *w != 0).collect();
1396        // SPIKE evidence (gated `FLOWLOG_DD_NATIVE_TRACE`): per-epoch INPUT delta
1397        // size vs OUTPUT binding-delta size. Incrementality shows up as: output
1398        // tracks the input delta, NOT the (growing) integral — epoch K does only
1399        // delta·integral work, never a full recompute. The `pos`/`neg` split shows
1400        // retraction (negative-weight outputs) flowing through DD's signed weights.
1401        if trace_enabled() {
1402            let in_n: usize = deltas.values().map(|v| v.len()).sum();
1403            let pos = out.iter().filter(|(_, w)| *w > 0).count();
1404            let neg = out.iter().filter(|(_, w)| *w < 0).count();
1405            #[allow(clippy::disallowed_macros)]
1406            {
1407                eprintln!(
1408                    "[dd_native] epoch {} : input_delta_rows={} -> output_binding_delta pos={} neg={}",
1409                    next_epoch, in_n, pos, neg
1410                );
1411            }
1412        }
1413        Ok(out)
1414    }
1415}

1417// ===========================================================================
1418// FusedDdJoin — ONE shared worker + ONE dataflow per RULESET
1419// ===========================================================================
1420//
1421// `PersistentDdJoin` builds one timely `Worker` PER RULE: a program with R
1422// atom-bearing rules spins up R independent workers, and a body relation read by
1423// K rules gets K separate `InputSession`s (each fed the same delta) + K separate
1424// arrangements, each stepped to fixpoint separately. Step-0 profiling
1425// (`FLOWLOG_DD_PROF`) on math-microbenchmark N=9 showed 79 workers, 173
1426// InputSessions, and 1.996s of 2.43s total spent inside `worker.step_while` — the
1427// per-rule-worker duplication is the dominant cost.
1428//
1429// `FusedDdJoin` collapses this to ONE worker hosting ONE `worker.dataflow(...)`
1430// scope for the whole ruleset (keyed by the sorted live rule-index list, exactly
1431// like feldera's `FusedJoin`). Within that scope:
1432//   - every DISTINCT body relation across all rules gets ONE `InputSession` →
1433//     ONE base `Collection`, `.distinct()`'d ONCE and SHARED by every atom
1434//     occurrence (in every rule) that reads it. Cloning a DD collection is a
1435//     handle copy, so the shared `.distinct()` arrangement is built once, not K
1436//     times — the dedup win.
1437//   - each rule is a left-deep join sub-stream reading those shared collections,
1438//     with its OWN `inspect_batch` capture into a per-rule `Rc<RefCell<Vec>>`.
1439// Per epoch: feed each relation's delta ONCE into its shared input, advance +
1440// step the SINGLE worker once, then drain each rule's capture buffer. The
1441// host-side prim re-run + `apply_head` are unchanged (the caller does them).
1442//
1443// The NEVER-CLEAR / fed-only-deltas invariant is preserved (the InputSessions
1444// persist across epochs = genuinely incremental), as is the external epoch-drive.
1445
1446/// A fused, delta-fed body join for a WHOLE ruleset on a single shared timely
1447/// `Worker`. Built once via [`FusedDdJoin::build`]; driven across epochs via
1448/// [`FusedDdJoin::step`] with a SINGLE `worker.step_while` per call.
1449pub struct FusedDdJoin {
1450    worker: Worker,
1451    /// One shared input session per DISTINCT body relation across all rules.
1452    inputs: HashMap<FunctionId, InputSession<u32, Row, isize>>,
1453    /// Single probe on all rule outputs (they share the dataflow scope, so one
1454    /// probe gates the whole epoch's fixpoint).
1455    probe: ProbeHandle<u32>,
1456    /// The fused rules, in build (= sorted rule-index) order. Each carries its
1457    /// own capture buffer + var width.
1458    rules: Vec<FusedRule>,
1459    /// Current epoch (monotonic; advanced once per [`step`]).
1460    epoch: u32,
1461}
1462
1463/// One rule's lowering inside a [`FusedDdJoin`]: its rule index (for routing
1464/// bindings to its head), its per-epoch output capture buffer, and its width.
1465struct FusedRule {
1466    idx: usize,
1467    /// This rule's per-epoch output binding-delta capture (`inspect_batch`
1468    /// appends `(row, weight)`; drained by [`FusedDdJoin::step`]).
1469    captured: Rc<RefCell<Vec<(Row, isize)>>>,
1470    /// Number of canonical body variables (binding-row width in use).
1471    n_vars: usize,
1472    /// `func` -> atom-occurrence indices reading it within THIS rule (self-join
1473    /// fan-out is handled at build time via the shared collection, so this is
1474    /// only used to know which relations this rule reads).
1475    body_funcs: Vec<FunctionId>,
1476}
1477
1478impl FusedDdJoin {
1479    /// Build ONE worker + ONE dataflow for the whole ruleset. `plans` pairs each
1480    /// rule's index with its [`JoinPlan`], in the order they should fire. EVERY
1481    /// rule — congruence, user, and `@rebuild_rule*` canonicalization — runs
1482    /// through the same general fused join (the sound `step_symmetric` path).
1483    pub fn build(
1484        plans: &[(usize, JoinPlan)],
1485        wcoj: bool,
1486        allow_acyclic: bool,
1487    ) -> Result<FusedDdJoin> {
1488        let alloc = Allocator::Thread(Thread::default());
1489        let mut worker = Worker::new(
1490            WorkerConfig::default(),
1491            alloc,
1492            Some(std::time::Instant::now()),
1493        );
1494        if prof_enabled() {
1495            PROF_WORKERS.fetch_add(1, Ordering::Relaxed);
1496        }
1497
1498        // Distinct body relations across all rules → one shared input each.
1499        let mut funcs: Vec<FunctionId> = Vec::new();
1500        for (_, plan) in plans {
1501            for a in &plan.atoms {
1502                if !funcs.contains(&a.func) {
1503                    funcs.push(a.func);
1504                }
1505            }
1506        }
1507        if prof_enabled() {
1508            PROF_INPUT_SESSIONS.fetch_add(funcs.len() as u64, Ordering::Relaxed);
1509        }
1510
1511        // Owned per-rule plan snapshots so the `move` dataflow closure is 'static.
1512        struct RulePlan {
1513            idx: usize,
1514            atoms: Vec<Vec<Slot>>,
1515            atom_funcs: Vec<FunctionId>,
1516            var_col: HashMap<u32, usize>,
1517            n_vars: usize,
1518            body_funcs: Vec<FunctionId>,
1519            /// `--wcoj` and this rule is the recognized triangle ⇒ build the WCOJ
1520            /// delta query instead of the binary `.join` chain. `None` ⇒ binary.
1521            triangle: Option<TriangleShape>,
1522            /// `--wcoj` and this rule is a general detected cyclic >=3-atom body
1523            /// (NOT already matched by `triangle`) ⇒ the general WCOJ delta
1524            /// query. `None` ⇒ triangle or binary.
1525            cq: Option<CqPlan>,
1526            /// `FLOWLOG_PROJECT`: per-step column-reuse layout ⇒ the binary chain
1527            /// runs with projecting merges and packs the captured row to the
1528            /// surviving vars. `None` ⇒ the historical static-column chain.
1529            projection: Option<ProjectionPlan>,
1530        }
1531        let rule_plans: Vec<RulePlan> = plans
1532            .iter()
1533            .map(|(idx, plan)| {
1534                let atom_funcs: Vec<FunctionId> = plan.atoms.iter().map(|a| a.func).collect();
1535                let mut body_funcs: Vec<FunctionId> = Vec::new();
1536                for &f in &atom_funcs {
1537                    if !body_funcs.contains(&f) {
1538                        body_funcs.push(f);
1539                    }
1540                }
1541                // Detect WCOJ shapes ONLY when `--wcoj` is set; off ⇒ both
1542                // `None`, so the build is byte-identical to the binary path. The
1543                // Stage-1 triangle is kept as its own hand-built path (the
1544                // regression guarantee); other detected cyclic >=3-atom bodies
1545                // route to the general construction. A rule matched by the
1546                // triangle never also takes the general path.
1547                let triangle = if wcoj { detect_triangle(plan) } else { None };
1548                let cq = if wcoj && triangle.is_none() {
1549                    detect_cyclic_cq(plan, allow_acyclic)
1550                } else {
1551                    None
1552                };
1553                // With projection the CAPTURED row is packed to the surviving
1554                // vars (columns `0..head_vars.len()`), so the capture/scatter
1555                // width is `head_vars.len()`, not the full static var count.
1556                let n_vars = match &plan.projection {
1557                    Some(p) => p.head_vars.len(),
1558                    None => plan.var_order.len(),
1559                };
1560                RulePlan {
1561                    idx: *idx,
1562                    atoms: plan.atoms.iter().map(|a| a.slots.clone()).collect(),
1563                    atom_funcs,
1564                    var_col: plan.var_col.clone(),
1565                    n_vars,
1566                    body_funcs,
1567                    triangle,
1568                    cq,
1569                    projection: plan.projection.clone(),
1570                }
1571            })
1572            .collect();
1573
1574        let probe: ProbeHandle<u32> = ProbeHandle::new();
1575        let probe_in = probe.clone();
1576        // Per-rule capture buffers, allocated outside the closure so we can keep a
1577        // clone here and route each rule's output to its head after `step`.
1578        let captures: Vec<Rc<RefCell<Vec<(Row, isize)>>>> = rule_plans
1579            .iter()
1580            .map(|_| Rc::new(RefCell::new(Vec::new())))
1581            .collect();
1582        let captures_in = captures.clone();
1583        let funcs_in = funcs.clone();
1584        // The per-rule metadata `FusedRule` needs (kept here; the closure consumes
1585        // `rule_plans` for the dataflow build).
1586        let rule_meta: Vec<(usize, usize, Vec<FunctionId>)> = rule_plans
1587            .iter()
1588            .map(|rp| (rp.idx, rp.n_vars, rp.body_funcs.clone()))
1589            .collect();
1590
1591        // PERF: the per-epoch input delta is already set-semantic — it is built
1592        // from a `HashSet` set-difference vs the fed view (`interpret::fused_bindings`),
1593        // so each row appears at most once with weight ±1. The input integral
1594        // therefore stays 0/1 per row WITHOUT `.distinct()`, making the input
1595        // distinct (a full integral + per-key consolidation every epoch, over the
1596        // LARGE relation integrals) pure overhead. Dropped by default; set
1597        // `FLOWLOG_DD_KEEP_INPUT_DISTINCT` to restore it. (Mirrors feldera's
1598        // `FELDERA_KEEP_INPUT_DISTINCT` finding.)
1599        let keep_input_distinct = std::env::var_os("FLOWLOG_DD_KEEP_INPUT_DISTINCT").is_some();
1600        let keep_output_distinct = std::env::var_os("FLOWLOG_DD_KEEP_OUTPUT_DISTINCT").is_some();
1601        let inputs = worker.dataflow::<u32, _, _>(move |scope| {
1602            // ONE shared input + base collection per distinct relation, shared by
1603            // every atom occurrence (in every rule) that reads it.
1604            let mut inputs: HashMap<FunctionId, InputSession<u32, Row, isize>> = HashMap::new();
1605            let mut rel_coll: HashMap<FunctionId, _> = HashMap::new();
1606            for &f in &funcs_in {
1607                let mut session: InputSession<u32, Row, isize> = InputSession::new();
1608                let base = session.to_collection(scope);
1609                let coll = if keep_input_distinct {
1610                    base.map(|r: Row| (r, ())).distinct().map(|(r, ())| r)
1611                } else {
1612                    base
1613                };
1614                inputs.insert(f, session);
1615                rel_coll.insert(f, coll);
1616            }
1617
1618            for (rp, cap) in rule_plans.iter().zip(captures_in.iter()) {
1619                // This rule's per-atom collection vector, from the SHARED relation
1620                // collections (cloning a DD collection is just a handle copy).
1621                let n_atoms = rp.atoms.len();
1622                let atom_slots = &rp.atoms;
1623                let var_col = &rp.var_col;
1624                let n_vars = rp.n_vars;
1625
1626                // `--wcoj`: the recognized triangle rule runs the worst-case-
1627                // optimal delta query (prefix-extension + AltNeu 3-stream
1628                // decomposition) over the SHARED Mul/Add input collections,
1629                // emitting the SAME n_vars-wide binding Row as the binary chain.
1630                let cur = if let Some(tri) = &rp.triangle {
1631                    // Diagnostic (gated `FLOWLOG_WCOJ_TRACE`): confirm which rules
1632                    // route to the WCOJ path. Mirrors the other `FLOWLOG_*` debug
1633                    // gates; zero cost when off.
1634                    #[allow(clippy::disallowed_macros)]
1635                    if std::env::var_os("FLOWLOG_WCOJ_TRACE").is_some() {
1636                        eprintln!(
1637                            "[WCOJ] triangle rule idx={} mul={:?} add={:?}",
1638                            rp.idx, tri.mul_func, tri.add_func
1639                        );
1640                    }
1641                    let mul = rel_coll[&tri.mul_func].clone();
1642                    let add = rel_coll[&tri.add_func].clone();
1643                    wcoj_triangle_collection(scope, mul, add, tri.clone())
1644                } else if let Some(cq) = &rp.cq {
1645                    // GENERAL WCOJ: a detected cyclic >=3-atom body. One shared
1646                    // input collection per atom (in atom order), fed to the
1647                    // k-stream delta decomposition. Emits the same n_vars-wide
1648                    // binding Row as the binary chain.
1649                    #[allow(clippy::disallowed_macros)]
1650                    if std::env::var_os("FLOWLOG_WCOJ_TRACE").is_some() {
1651                        eprintln!(
1652                            "[WCOJ] cyclic-cq rule idx={} atoms={} funcs={:?}",
1653                            rp.idx,
1654                            cq.atom_funcs.len(),
1655                            cq.atom_funcs
1656                        );
1657                    }
1658                    let colls: Vec<_> = cq.atom_funcs.iter().map(|f| rel_coll[f].clone()).collect();
1659                    wcoj_cq_collection(scope, &colls, cq.clone())
1660                } else if let Some(proj) = &rp.projection {
1661                    // PROJECTED binary chain (`FLOWLOG_PROJECT`): the binding row
1662                    // is laid out per-step with column REUSE (`proj.step_col`), so
1663                    // a rule whose static var count exceeds `W` still fits. The
1664                    // chain is otherwise identical to the static path below; the
1665                    // only differences are (a) every var's column is the per-step
1666                    // column instead of the static `var_col`, (b) each merge REMAPS
1667                    // carried-over vars whose column changed (reuse) and zeroes
1668                    // freed columns, and (c) a final `.map` packs the surviving
1669                    // vars into columns `0..head_vars.len()` for the capture.
1670                    let step_col = &proj.step_col;
1671                    let slots0 = atom_slots[0].clone();
1672                    let sc0 = step_col[0].clone();
1673                    let mut cur = rel_coll[&rp.atom_funcs[0]]
1674                        .clone()
1675                        .flat_map(move |r: Row| bind_atom(&r, &slots0, &sc0));
1676
1677                    for i in 1..n_atoms {
1678                        let slots = atom_slots[i].clone();
1679                        let prev = &step_col[i - 1];
1680                        let curl = &step_col[i];
1681                        // Shared = atom vars already live (present in `prev`).
1682                        let shared: Vec<u32> = atom_vars(&slots)
1683                            .into_iter()
1684                            .filter(|v| prev.contains_key(v))
1685                            .collect();
1686                        // Left key reads each shared var from its PREVIOUS column.
1687                        let shared_cols_left: Vec<usize> = shared.iter().map(|v| prev[v]).collect();
1688                        let shared_atom_cols: Vec<usize> = shared
1689                            .iter()
1690                            .map(|v| {
1691                                slots
1692                                    .iter()
1693                                    .position(|s| matches!(s, Slot::Var(x) if x == v))
1694                                    .expect("shared var present in atom")
1695                            })
1696                            .collect();
1697
1698                        let scl = shared_cols_left.clone();
1699                        let left = cur.map(move |b: Row| (pack_key(&b, &scl), b));
1700                        let sac = shared_atom_cols.clone();
1701                        let right = rel_coll[&rp.atom_funcs[i]]
1702                            .clone()
1703                            .map(move |r: Row| (pack_key(&r, &sac), r));
1704
1705                        // Carried vars: live at BOTH prev and cur but NOT produced
1706                        // by this atom — copy them from prev[v] to cur[v]. Atom
1707                        // vars: written/validated at cur[v]. (prev_col, cur_col,
1708                        // is_shared, atom_col) per relevant var, precomputed.
1709                        let slotsc = slots.clone();
1710                        let prevc = prev.clone();
1711                        let curc = curl.clone();
1712                        cur = left
1713                            .join(right)
1714                            .flat_map(move |(_k, (b, r)): (Key, (Row, Row))| {
1715                                remap_merge_atom_into(&b, &r, &slotsc, &prevc, &curc)
1716                            });
1717                    }
1718
1719                    // Final projection: pack the surviving vars (at their last-step
1720                    // columns `head_cols`) into the capture columns `0..k`, zeroing
1721                    // the rest, so the head scatter reads `bind[i]` for surviving
1722                    // var `i` exactly like the static path.
1723                    let head_cols = proj.head_cols.clone();
1724                    cur.map(move |b: Row| pack_key(&b, &head_cols))
1725                } else {
1726                    let mut bound = vec![false; n_vars];
1727                    let slots0 = atom_slots[0].clone();
1728                    let vc0 = var_col.clone();
1729                    let mut cur = rel_coll[&rp.atom_funcs[0]]
1730                        .clone()
1731                        .flat_map(move |r: Row| bind_atom(&r, &slots0, &vc0));
1732                    mark_bound(&atom_slots[0], var_col, &mut bound);
1733
1734                    for i in 1..n_atoms {
1735                        let slots = atom_slots[i].clone();
1736                        let shared: Vec<u32> = atom_vars(&slots)
1737                            .into_iter()
1738                            .filter(|v| var_col.get(v).map(|&c| bound[c]).unwrap_or(false))
1739                            .collect();
1740                        let shared_cols_left: Vec<usize> =
1741                            shared.iter().map(|v| var_col[v]).collect();
1742                        let shared_atom_cols: Vec<usize> = shared
1743                            .iter()
1744                            .map(|v| {
1745                                slots
1746                                    .iter()
1747                                    .position(|s| matches!(s, Slot::Var(x) if x == v))
1748                                    .expect("shared var present in atom")
1749                            })
1750                            .collect();
1751
1752                        let scl = shared_cols_left.clone();
1753                        let left = cur.map(move |b: Row| (pack_key(&b, &scl), b));
1754                        let sac = shared_atom_cols.clone();
1755                        let right = rel_coll[&rp.atom_funcs[i]]
1756                            .clone()
1757                            .map(move |r: Row| (pack_key(&r, &sac), r));
1758
1759                        let slotsc = slots.clone();
1760                        let vc = var_col.clone();
1761                        let bound_now = bound.clone();
1762                        cur = left
1763                            .join(right)
1764                            .flat_map(move |(_k, (b, r)): (Key, (Row, Row))| {
1765                                merge_atom_into(&b, &r, &slotsc, &vc, &bound_now)
1766                            });
1767                        mark_bound(&slots, var_col, &mut bound);
1768                    }
1769                    cur
1770                };
1771
1772                // PERF: the output `.distinct()` is redundant here. `step`
1773                // accumulates each rule's binding deltas into a per-key weight map
1774                // and `interpret::fused_bindings` inspects only the SIGN of the net
1775                // weight (>0 ⇒ one env; net-zero already filtered). distinct would
1776                // clamp the binding multiplicity to {0,1}, but since only the sign
1777                // is observed and net-zero rows are dropped, the clamp is
1778                // unobservable. `.consolidate()` still collapses per-key
1779                // multiplicities so the captured batch is one signed row per key.
1780                // Dropped by default; set `FLOWLOG_DD_KEEP_OUTPUT_DISTINCT` to
1781                // restore. (Mirrors feldera's `FELDERA_KEEP_OUTPUT_DISTINCT`.)
1782                let consolidated = if keep_output_distinct {
1783                    cur.map(|b: Row| (b, ()))
1784                        .distinct()
1785                        .map(|(b, ())| b)
1786                        .consolidate()
1787                } else {
1788                    cur.consolidate()
1789                };
1790                let out = consolidated;
1791
1792                let cap = Rc::clone(cap);
1793                out.inner
1794                    .inspect_batch(move |_t, batch| {
1795                        let mut buf = cap.borrow_mut();
1796                        for (row, _time, w) in batch.iter() {
1797                            buf.push((*row, *w));
1798                        }
1799                    })
1800                    .probe_with(&probe_in);
1801            }
1802
1803            inputs
1804        });
1805
1806        let rules: Vec<FusedRule> = rule_meta
1807            .into_iter()
1808            .zip(captures)
1809            .map(|((idx, n_vars, body_funcs), captured)| FusedRule {
1810                idx,
1811                captured,
1812                n_vars,
1813                body_funcs,
1814            })
1815            .collect();
1816
1817        Ok(FusedDdJoin {
1818            worker,
1819            inputs,
1820            probe,
1821            rules,
1822            epoch: 0,
1823        })
1824    }

1836    /// Feed one epoch of signed relation deltas into the SHARED inputs, advance
1837    /// the timestamp, run the SINGLE worker to this epoch's fixpoint, and return
1838    /// per-rule binding deltas. The outer `Vec` is in [`rule_indices`] order; each
1839    /// inner `Vec` is `(binding_row_as_var_order_vec, weight)`.
1840    ///
1841    /// CRUCIAL: the InputSessions are NEVER cleared — only the delta is pushed, so
1842    /// the DD arrangements persist and the join is genuinely incremental.
1843    pub fn step(
1844        &mut self,
1845        deltas: &HashMap<FunctionId, Vec<(Vec<u32>, isize)>>,
1846    ) -> Result<Vec<Vec<(Vec<u32>, isize)>>> {
1847        // EVERY rule — congruence, user, and `@rebuild_rule*` canonicalization —
1848        // runs through the SAME symmetric incremental join: feed ALL deltas, one
1849        // sub-step, capture every rule. FlowLog is on the fast relational
1850        // term-encoding, so the rebuild rules take no special path (the unsound
1851        // δUF-driven / transient-pulse rebuild variants have been deleted).
1852        self.step_symmetric(deltas)
1853    }
1854
1855    /// Step the SINGLE worker to `epoch`, accounting feed/step time.
1856    fn drive_to(&mut self, epoch: u32, t_feed: std::time::Instant, prof: bool) {
1857        for inp in self.inputs.values_mut() {
1858            inp.advance_to(epoch);
1859            inp.flush();
1860        }
1861        if prof {
1862            add_ns(&PROF_FEED_NS, t_feed.elapsed());
1863            PROF_STEP_CALLS.fetch_add(1, Ordering::Relaxed);
1864        }
1865        let t_step = std::time::Instant::now();
1866        let probe = self.probe.clone();
1867        self.worker.step_while(|| probe.less_than(&epoch));
1868        if prof {
1869            add_ns(&PROF_STEP_NS, t_step.elapsed());
1870        }
1871    }

1873    /// Drain each rule's capture buffer; keep a rule's output into `accs` only if
1874    /// `keep(rule)` (route a sub-step's output to the rules it is meant for).
1875    fn drain_into(&self, accs: &mut [HashMap<Vec<u32>, isize>], keep: impl Fn(&FusedRule) -> bool) {
1876        for (rule, acc) in self.rules.iter().zip(accs.iter_mut()) {
1877            let drained: Vec<(Row, isize)> = rule.captured.borrow_mut().drain(..).collect();
1878            if !keep(rule) {
1879                continue;
1880            }
1881            for (row, w) in drained {
1882                let key: Vec<u32> = (0..rule.n_vars).map(|i| row[i]).collect();
1883                *acc.entry(key).or_insert(0) += w;
1884            }
1885        }
1886    }

1888    /// The original symmetric incremental join: feed ALL deltas, one sub-step,
1889    /// capture every rule. Behavior-identical to the pre-Phase-2 `step`.
1890    fn step_symmetric(
1891        &mut self,
1892        deltas: &HashMap<FunctionId, Vec<(Vec<u32>, isize)>>,
1893    ) -> Result<Vec<Vec<(Vec<u32>, isize)>>> {
1894        let prof = prof_enabled();
1895        let t_feed = std::time::Instant::now();
1896        let mut pushed = false;
1897        for (func, rows) in deltas {
1898            if let Some(inp) = self.inputs.get_mut(func) {
1899                for (row, w) in rows {
1900                    inp.update(pack_row(row), *w);
1901                    pushed = true;
1902                }
1903            }
1904        }
1905        let next_epoch = self.epoch + 1;
1906        if !pushed {
1907            self.epoch = next_epoch;
1908            return Ok(vec![Vec::new(); self.rules.len()]);
1909        }
1910
1911        for rule in &self.rules {
1912            rule.captured.borrow_mut().clear();
1913        }
1914        self.drive_to(next_epoch, t_feed, prof);
1915        self.epoch = next_epoch;
1916
1917        if trace_enabled() {
1918            let total: usize = self.rules.iter().map(|r| r.captured.borrow().len()).sum();
1919            let n_rules = self.rules.len();
1920            use egglog_numeric_id::NumericId;
1921            let funcs: Vec<u32> = deltas.keys().map(|f| f.rep()).collect();
1922            let delta_rows: usize = deltas.values().map(|r| r.len()).sum();
1923            #[allow(clippy::disallowed_macros)]
1924            {
1925                eprintln!(
1926                    "[dd_symmetric] n_rules={n_rules} delta_funcs={funcs:?} delta_rows={delta_rows} total_out={total}"
1927                );
1928            }
1929        }
1930
1931        let mut accs: Vec<HashMap<Vec<u32>, isize>> =
1932            (0..self.rules.len()).map(|_| HashMap::new()).collect();
1933        self.drain_into(&mut accs, |_| true);
1934        Ok(accs
1935            .into_iter()
1936            .map(|acc| acc.into_iter().filter(|(_, w)| *w != 0).collect())
1937            .collect())
1938    }

1983/// Match the first atom's relation row against its slots, producing the initial
1984/// canonical binding row (or empty vec if a const / repeated-var constraint
1985/// fails). Returns a `Vec` for `flat_map`.
1986fn bind_atom(r: &Row, slots: &[Slot], var_col: &HashMap<u32, usize>) -> Vec<Row> {
1987    let mut out = empty_row();
1988    let mut local: HashMap<u32, u32> = HashMap::new();
1989    for (i, s) in slots.iter().enumerate() {
1990        let val = r[i];
1991        match s {
1992            Slot::Const(c) => {
1993                if *c != val {
1994                    return Vec::new();
1995                }
1996            }
1997            Slot::Var(v) => {
1998                if let Some(&prev) = local.get(v) {
1999                    if prev != val {
2000                        return Vec::new();
2001                    }
2002                } else {
2003                    local.insert(*v, val);
2004                    out[var_col[v]] = val;
2005                }
2006            }
2007        }
2008    }
2009    vec![out]
2010}

2012/// Merge atom row `r` into binding `b`: already-bound columns must agree;
2013/// previously-unbound atom vars are written. Empty vec on constraint failure.
2014fn merge_atom_into(
2015    b: &Row,
2016    r: &Row,
2017    slots: &[Slot],
2018    var_col: &HashMap<u32, usize>,
2019    bound: &[bool],
2020) -> Vec<Row> {
2021    let mut out = *b;
2022    let mut local: HashMap<u32, u32> = HashMap::new();
2023    for (i, s) in slots.iter().enumerate() {
2024        let val = r[i];
2025        match s {
2026            Slot::Const(c) => {
2027                if *c != val {
2028                    return Vec::new();
2029                }
2030            }
2031            Slot::Var(v) => {
2032                if let Some(&prev) = local.get(v) {
2033                    if prev != val {
2034                        return Vec::new();
2035                    }
2036                    continue;
2037                }
2038                local.insert(*v, val);
2039                let c = var_col[v];
2040                if bound[c] {
2041                    if out[c] != val {
2042                        return Vec::new();
2043                    }
2044                } else {
2045                    out[c] = val;
2046                }
2047            }
2048        }
2049    }
2050    vec![out]
2051}

2053/// Projecting merge for the `FLOWLOG_PROJECT` chain. Like [`merge_atom_into`] but
2054/// rebuilds the row from scratch under a NEW column layout: `prev` is the
2055/// left-row layout (step `i-1`), `cur` is the output layout (step `i`). Carried
2056/// vars (live in both layouts but not produced here) are copied `prev[v]→cur[v]`;
2057/// the atom's vars are validated (shared) or written (fresh) at `cur[v]`; every
2058/// other output column is left zeroed. This is what reuses freed columns and so
2059/// keeps the frontier within `W`. Empty vec on a const / repeated-var / shared-
2060/// var constraint failure (same semantics as the static merge).
2061fn remap_merge_atom_into(
2062    b: &Row,
2063    r: &Row,
2064    slots: &[Slot],
2065    prev: &HashMap<u32, usize>,
2066    cur: &HashMap<u32, usize>,
2067) -> Vec<Row> {
2068    let mut out = empty_row();
2069    // Carry over every still-live var (present in `cur`) that the left row
2070    // already holds (present in `prev`). Atom-fresh vars are absent from `prev`,
2071    // so they are NOT copied here — they are written from `r` below.
2072    for (&v, &pc) in prev {
2073        if let Some(&cc) = cur.get(&v) {
2074            out[cc] = b[pc];
2075        }
2076    }
2077    let mut local: HashMap<u32, u32> = HashMap::new();
2078    for (i, s) in slots.iter().enumerate() {
2079        let val = r[i];
2080        match s {
2081            Slot::Const(c) => {
2082                if *c != val {
2083                    return Vec::new();
2084                }
2085            }
2086            Slot::Var(v) => {
2087                if let Some(&prior) = local.get(v) {
2088                    if prior != val {
2089                        return Vec::new();
2090                    }
2091                    continue;
2092                }
2093                local.insert(*v, val);
2094                // Every atom var is live at this step ⇒ present in `cur`.
2095                let cc = cur[v];
2096                if prev.contains_key(v) {
2097                    // Shared (already bound): the carried value must agree. (The
2098                    // join key already enforces this for the key columns; this
2099                    // also covers shared vars not in the key, if any.)
2100                    if out[cc] != val {
2101                        return Vec::new();
2102                    }
2103                } else {
2104                    // Fresh var born at this atom: write it.
2105                    out[cc] = val;
2106                }
2107            }
2108        }
2109    }
2110    vec![out]
2111}

2137/// Build the GENERAL worst-case-optimal join for a detected cyclic body
2138/// (`CqPlan`) as a collection of full-width binding `Row`s — the SAME shape the
2139/// binary `.join` chain emits, so the downstream consolidate / capture / env
2140/// scatter is bit-identical.
2141///
2142/// Implements the full k-stream delta decomposition (one delta query per body
2143/// atom) inside ONE `AltNeu<u32>` nested scope, exactly mirroring the Stage-1
2144/// triangle: in delta query `dQ/dA_d`, atoms with index < d read the ALT (old)
2145/// trace and atoms with index > d read the NEU (new) trace, so simultaneous
2146/// cross-atom updates are counted exactly once. ALL reads (the multiway
2147/// intersections AND the per-atom payload recoveries) are inside the AltNeu
2148/// scope. Each JOIN step binds ONE join variable: a >=2-extender step is the
2149/// WCOJ multiway `extend` (the intersection that collapses the intermediate); a
2150/// 1-extender step is a `propose_using`. After the join core is pinned, each
2151/// non-driver atom's payload vars are recovered (and the atom validated) by one
2152/// `propose` keyed on its join columns.
2153fn wcoj_cq_collection<'scope>(
2154    scope: Scope<'scope, u32>,
2155    colls: &[VecCollection<'scope, u32, Row, isize>],
2156    plan: CqPlan,
2157) -> VecCollection<'scope, u32, Row, isize> {
2158    use differential_dataflow::AsCollection;

2317/// Build the worst-case-optimal triangle join as a differential-dataflow
2318/// collection of full `n_vars`-wide binding `Row`s — the SAME shape the binary
2319/// `.join` chain emits, so the downstream consolidate / capture / env scatter
2320/// is bit-identical.
2321///
2322/// Implements the FULL 3-stream delta decomposition (one delta query per body
2323/// atom) inside a single `AltNeu<u32>` nested scope, following dogsdogsdogs'
2324/// `examples/delta_query_wcoj.rs`. The total atom order is A0(Mul_a) <
2325/// A1(Mul_b) < A2(Add): in each delta query, atoms BEFORE the driving atom read
2326/// the `alt` (old) trace and atoms AFTER read the `neu` (new) trace, so
2327/// simultaneous cross-atom updates (incl. the Mul self-join) are not
2328/// double-counted. The ONLY multiway-intersected variable is the core `a`
2329/// (driven by dA2: intersect the `a`-sets of the two Mul eclasses m1,m2 — the
2330/// `Σ_a deg(a)²` collapse); every other variable is functionally recovered by a
2331/// single in-scope `propose_using`. ALL reads (core intersection AND payload
2332/// recovery) are inside the AltNeu scope — unlike the spike shortcut whose
2333/// out-of-scope recovery drifted at high iteration counts.
2334fn wcoj_triangle_collection<'scope>(
2335    scope: Scope<'scope, u32>,
2336    mul: VecCollection<'scope, u32, Row, isize>,
2337    add: VecCollection<'scope, u32, Row, isize>,
2338    tri: TriangleShape,
2339) -> VecCollection<'scope, u32, Row, isize> {
2340    use differential_dataflow::AsCollection;
2341    use timely::dataflow::operators::Concatenate;
2342
2343    // Raw-relation column layout (fixed by the term encoding's arity-4 schema):
2344    //   Mul row = [arg0=a/c, arg1=b/c-payload, eclass=m, rowextra=x]
2345    //   Add row = [m1, m2, eclass=o, rowextra=x]
2346    // (cols 0..3 are the RELATION columns; tri.col_* are the BINDING columns.)
2347    // Each stream assembles a full-width `empty_row()` and writes only the 9
2348    // triangle columns; columns >= n_vars stay 0 — identical to the binary
2349    // chain, which also leaves unused binding columns at 0. (n_vars itself is
2350    // not needed here; the downstream drain reads cols 0..rule.n_vars.)
2351    let (ca, cb, cm1, cx1) = (tri.col_a, tri.col_b, tri.col_m1, tri.col_x1);
2352    let (cc, cm2, cx2) = (tri.col_c, tri.col_m2, tri.col_x2);
2353    let (co, cx3) = (tri.col_o, tri.col_x3);
2354
2355    scope.scoped::<AltNeu<u32>, _, _>("WcojTriangle", move |inner| {
2356        let mul = mul.enter(inner);
2357        let add = add.enter(inner);
2358        let mul_neu = mul.clone().delay(|t| AltNeu::neu(t.time));
2359        let add_neu = add.clone().delay(|t| AltNeu::neu(t.time));
2360
2361        // --- indices over the entered collections ---------------------------
2362        // Mul indices: (K=eclass m, V=arg a), (K=arg a, V=eclass m),
2363        //              (K=(a,m), V=(b,x))  [payload recovery]
2364        let alt_mul_by_m = CollectionIndex::index(mul.clone().map(|r: Row| (r[2], r[0])));
2365        let alt_mul_by_a = CollectionIndex::index(mul.clone().map(|r: Row| (r[0], r[2])));
2366        let neu_mul_by_a = CollectionIndex::index(mul_neu.clone().map(|r: Row| (r[0], r[2])));
2367        let alt_mul_by_am =
2368            CollectionIndex::index(mul.clone().map(|r: Row| ((r[0], r[2]), (r[1], r[3]))));
2369        let neu_mul_by_am =
2370            CollectionIndex::index(mul_neu.clone().map(|r: Row| ((r[0], r[2]), (r[1], r[3]))));
2371        // Add indices: (K=m1, V=m2), (K=m2, V=m1), (K=(m1,m2), V=(o,x)).
2372        let neu_add_by_m1 = CollectionIndex::index(add_neu.clone().map(|r: Row| (r[0], r[1])));
2373        let neu_add_by_m2 = CollectionIndex::index(add_neu.clone().map(|r: Row| (r[1], r[0])));
2374        let neu_add_by_m1m2 =
2375            CollectionIndex::index(add_neu.clone().map(|r: Row| ((r[0], r[1]), (r[2], r[3]))));
2376
2377        // ===== dQ/dA0 : driven by dMul as Mul_a(a,b,m1,x1) ==================
2378        // bound a,b,m1,x1 ; A1(Mul_b) NEU, A2(Add) NEU.
2379        let changes0 = {
2380            // initial prefix: place the Mul delta row's cols into a,b,m1,x1.
2381            let prefix = mul.clone().map(move |r: Row| {
2382                let mut p = empty_row();
2383                p[ca] = r[0];
2384                p[cb] = r[1];
2385                p[cm1] = r[2];
2386                p[cx1] = r[3];
2387                p
2388            });
2389            // 1. intersect m2: Add(m1,·)=m2 (NEU) ∩ Mul(a,·)=m2 (NEU).
2390            let with_m2 = prefix
2391                .extend(&mut [
2392                    &mut neu_add_by_m1.extend_using(move |p: &Row| p[cm1]),
2393                    &mut neu_mul_by_a.extend_using(move |p: &Row| p[ca]),
2394                ])
2395                .map(move |(mut p, m2): (Row, u32)| {
2396                    p[cm2] = m2;
2397                    p
2398                });
2399            // 2. recover (c,x2) for Mul_b given (a,m2) (NEU).
2400            let with_cx2 = with_m2
2401                .propose_using(&mut neu_mul_by_am.extend_using(move |p: &Row| (p[ca], p[cm2])))
2402                .map(move |(mut p, (c, x2)): (Row, (u32, u32))| {
2403                    p[cc] = c;
2404                    p[cx2] = x2;
2405                    p
2406                });
2407            // 3. recover (o,x3) for Add given (m1,m2) (NEU).
2408            with_cx2
2409                .propose_using(&mut neu_add_by_m1m2.extend_using(move |p: &Row| (p[cm1], p[cm2])))
2410                .map(move |(mut p, (o, x3)): (Row, (u32, u32))| {
2411                    p[co] = o;
2412                    p[cx3] = x3;
2413                    p
2414                })
2415        };
2416
2417        // ===== dQ/dA1 : driven by dMul as Mul_b(a,c,m2,x2) ==================
2418        // bound a,c,m2,x2 ; A0(Mul_a) ALT, A2(Add) NEU.
2419        let changes1 = {
2420            let prefix = mul.clone().map(move |r: Row| {
2421                let mut p = empty_row();
2422                p[ca] = r[0];
2423                p[cc] = r[1];
2424                p[cm2] = r[2];
2425                p[cx2] = r[3];
2426                p
2427            });
2428            // 1. intersect m1: Add(·,m2)=m1 (NEU) ∩ Mul(a,·)=m1 (ALT).
2429            let with_m1 = prefix
2430                .extend(&mut [
2431                    &mut neu_add_by_m2.extend_using(move |p: &Row| p[cm2]),
2432                    &mut alt_mul_by_a.extend_using(move |p: &Row| p[ca]),
2433                ])
2434                .map(move |(mut p, m1): (Row, u32)| {
2435                    p[cm1] = m1;
2436                    p
2437                });
2438            // 2. recover (b,x1) for Mul_a given (a,m1) (ALT).
2439            let with_bx1 = with_m1
2440                .propose_using(&mut alt_mul_by_am.extend_using(move |p: &Row| (p[ca], p[cm1])))
2441                .map(move |(mut p, (b, x1)): (Row, (u32, u32))| {
2442                    p[cb] = b;
2443                    p[cx1] = x1;
2444                    p
2445                });
2446            // 3. recover (o,x3) for Add given (m1,m2) (NEU).
2447            with_bx1
2448                .propose_using(&mut neu_add_by_m1m2.extend_using(move |p: &Row| (p[cm1], p[cm2])))
2449                .map(move |(mut p, (o, x3)): (Row, (u32, u32))| {
2450                    p[co] = o;
2451                    p[cx3] = x3;
2452                    p
2453                })
2454        };
2455
2456        // ===== dQ/dA2 : driven by dAdd(m1,m2,o,x3) =========================
2457        // bound m1,m2,o,x3 ; A0(Mul_a) ALT, A1(Mul_b) ALT.
2458        let changes2 = {
2459            let prefix = add.clone().map(move |r: Row| {
2460                let mut p = empty_row();
2461                p[cm1] = r[0];
2462                p[cm2] = r[1];
2463                p[co] = r[2];
2464                p[cx3] = r[3];
2465                p
2466            });
2467            // 1. intersect a: Mul(·,·)=m1 → a (ALT) ∩ Mul(·,·)=m2 → a (ALT).
2468            //    THE worst-case-optimal collapse.
2469            let with_a = prefix
2470                .extend(&mut [
2471                    &mut alt_mul_by_m.extend_using(move |p: &Row| p[cm1]),
2472                    &mut alt_mul_by_m.extend_using(move |p: &Row| p[cm2]),
2473                ])
2474                .map(move |(mut p, a): (Row, u32)| {
2475                    p[ca] = a;
2476                    p
2477                });
2478            // 2. recover (b,x1) for Mul_a given (a,m1) (ALT).
2479            let with_bx1 = with_a
2480                .propose_using(&mut alt_mul_by_am.extend_using(move |p: &Row| (p[ca], p[cm1])))
2481                .map(move |(mut p, (b, x1)): (Row, (u32, u32))| {
2482                    p[cb] = b;
2483                    p[cx1] = x1;
2484                    p
2485                });
2486            // 3. recover (c,x2) for Mul_b given (a,m2) (ALT).
2487            with_bx1
2488                .propose_using(&mut alt_mul_by_am.extend_using(move |p: &Row| (p[ca], p[cm2])))
2489                .map(move |(mut p, (c, x2)): (Row, (u32, u32))| {
2490                    p[cc] = c;
2491                    p[cx2] = x2;
2492                    p
2493                })
2494        };
2495
2496        // Concat the three delta streams and leave the AltNeu scope.
2497        let streams = vec![changes0.inner, changes1.inner, changes2.inner];
2498        inner.concatenate(streams).as_collection().leave(scope)
2499    })
