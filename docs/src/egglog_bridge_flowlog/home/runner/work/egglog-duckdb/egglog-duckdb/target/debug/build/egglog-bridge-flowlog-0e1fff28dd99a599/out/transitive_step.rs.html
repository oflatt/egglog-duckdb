320        /// Apply all staged updates as one epoch. Panics if no
321        /// transaction is active — call `begin()` or any
322        /// `insert_*` / `remove_*` method first. Returns the
323        /// per-output deltas produced by this epoch.
324        pub fn commit(&mut self) -> IncrementalResults {

329            );

333                        bucket,
334                    );

337                    *self.path_slots[i].lock().expect("slot poisoned") = ::std::mem::take(
338                        bucket,

367        /// Stage tuples to insert at the next `commit()`. Auto-begins
368        /// a transaction if none is active; an empty `items` slice is
369        /// a no-op and does not auto-begin.
370        pub fn insert_edge(&mut self, items: Vec<rel::Edge>) {
371            if items.is_empty() {
372                return;
373            }
374            self.ensure_txn();
375            let total = items.len();
376            let workers = self.workers;
377            let chunk = total / workers;
378            let remainder = total % workers;
379            let mut iter = items.into_iter();
380            for i in 0..workers {
381                let take = chunk + if i < remainder { 1 } else { 0 };
382                if take == 0 {
383                    continue;
384                }
385                let bucket = &mut self.edge_staged[i];
386                bucket.reserve(take);
387                for item in iter.by_ref().take(take) {
388                    bucket.push((item, 1_i32));
389                }
390            }
391        }
