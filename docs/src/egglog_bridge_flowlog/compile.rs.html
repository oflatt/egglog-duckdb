1//! Rule IR and row representation for the FlowLog backend.
2//!
3//! ## The load-bearing design choice (per the milestone brief)
4//!
5//! egglog's `(run N)` applies a ruleset **N times with bounded extension per
6//! round** — a transitive-closure rule extends **N hops, NOT to full closure**.
7//! FlowLog's incremental engine (`commit()`) would, on a *recursive* `.dl`,
8//! saturate to a fixed point inside one `commit()`; that is *wrong* for
9//! egglog's bounded iteration (it is the same trap the Feldera Phase-0 spike's
10//! recursive DD scope fell into).
11//!
12//! So the M1 backend compiles a **non-recursive** flowlog program
13//! (`transitive_step.dl`): `hop(x,z) :- path(x,y), edge(y,z).` One `commit()`
14//! performs exactly **one round** of the join over the staged delta. The host
15//! (`lib.rs::run_rules`) drives `(run N)` by feeding the previous round's new
16//! `path` rows and calling `commit()` once per round — N calls = N hops.
17//!
18//! ## Row representation
19//!
20//! Every relation's rows are stored in the Rust-side mirror as a variable-width
21//! boxed slice of `u32` (egglog [`Value`] reps), exactly `arity` columns wide.
22//! `lookup_id` / `for_each` / `table_size` read the mirror.
23
24use egglog_backend_trait::{ExternalFunctionId, FunctionId, QueryEntry, Value};
25
26/// Upper bound on relation arity (sanity check; the mirror row is
27/// variable-width, so this is generous).
28pub const MAX_ARITY: usize = 64;
29
30/// Uniform mirror row type: a variable-width boxed slice of `u32`
31/// (egglog `Value` reps), exactly `arity` columns wide.
32pub type Row = Box<[u32]>;
33
34/// How a function resolves a functional-dependency conflict (two rows sharing
35/// the same key columns with different output columns). Recognized from the
36/// trait `MergeFn` (see `lib.rs::add_table`). Mirrors the Feldera backend.
37#[derive(Clone, Copy, Debug, PartialEq, Eq)]
38pub enum MergeMode {
39    /// Plain relation: the whole row is the key, no output column to resolve.
40    Relation,
41    /// `:merge old` / `AssertEq`: keep the existing value on conflict.
42    Old,
43    /// `:merge new`: keep the new value on conflict.
44    New,
45    /// `:merge (ordering-min old new)` / `UnionId`: keep the numerically
46    /// smallest value (the union-find leader). Load-bearing for M2 rebuild.
47    Min,
48}
49
50/// Pack a slice of `Value`s into a [`Row`] (exactly `vals.len()` columns).
51pub fn pack_row(vals: &[Value]) -> Row {
52    use egglog_numeric_id::NumericId;
53    assert!(
54        vals.len() <= MAX_ARITY,
55        "row arity {} exceeds {MAX_ARITY}",
56        vals.len()
57    );
58    vals.iter().map(|v| v.rep()).collect()
59}
60
61/// Read column `i` (0-based) out of a [`Row`].
62#[inline]
63pub fn row_col(r: &Row, i: usize) -> u32 {
64    r[i]
65}
66
67/// Unpack the first `arity` columns of a [`Row`] into a `Vec<Value>`.
68pub fn unpack_row(r: &Row, arity: usize) -> Vec<Value> {
69    use egglog_numeric_id::NumericId;
70    (0..arity).map(|i| Value::new(row_col(r, i))).collect()
71}

73/// One column reference in a rule body atom or head action: either a bound
74/// variable (identified by its [`egglog_backend_trait::VariableId`] rep) or a
75/// constant value.
76#[derive(Clone, Debug)]
77pub enum Slot {
78    Var(u32),
79    Const(u32),
80}
81
82/// Resolve a [`Slot`] to a concrete `u32` value: a constant resolves to itself,
83/// a variable resolves through `get` (the current binding env), or `None` if the
84/// variable is unbound.
85pub fn slot_lookup(s: &Slot, get: &dyn Fn(u32) -> Option<u32>) -> Option<u32> {
86    match s {
87        Slot::Var(v) => get(*v),
88        Slot::Const(c) => Some(*c),
89    }
90}
91
92impl Slot {
93    pub fn from_entry(e: &QueryEntry) -> Self {
94        use egglog_numeric_id::NumericId;
95        match e {
96            QueryEntry::Var(v) => Slot::Var(v.id.rep()),
97            QueryEntry::Const { val, .. } => Slot::Const(val.rep()),
98        }
99    }
100}
101
102/// A body atom: a function table reference with one [`Slot`] per column.
103#[derive(Clone, Debug)]
104pub struct BodyAtom {
105    pub func: FunctionId,
106    pub slots: Vec<Slot>,
107}
108
109impl BodyAtom {
110    pub fn from_entries(func: FunctionId, entries: &[QueryEntry]) -> Self {
111        BodyAtom {
112            func,
113            slots: entries.iter().map(Slot::from_entry).collect(),
114        }
115    }
116}

118/// One operation in a rule **body**, in emission order.
119#[derive(Clone, Debug)]
120pub enum BodyOp {
121    /// Match a table/relation atom.
122    Atom(BodyAtom),
123    /// Evaluate a primitive `func(args..)` (M2+; recorded but not executed in
124    /// the M1 flowlog-driven path).
125    #[allow(dead_code)]
126    Prim {
127        id: ExternalFunctionId,
128        args: Vec<Slot>,
129        ret: Slot,
130    },
131}
132
133/// One operation in a rule **head**, in emission order.
134#[derive(Clone, Debug)]
135pub enum HeadOp {
136    /// `(set func(key..) = val)` / relation `insert` — slots are the full row.
137    Set { func: FunctionId, slots: Vec<Slot> },
138    /// `(delete func(key..))` — retraction (M2+).
139    #[allow(dead_code)]
140    Remove { func: FunctionId, slots: Vec<Slot> },
141    /// `(subsume func(key..))` (M2+).
142    #[allow(dead_code)]
143    Subsume { func: FunctionId, slots: Vec<Slot> },
144    /// RHS function lookup binding `ret` (M2+).
145    #[allow(dead_code)]
146    Lookup {
147        func: FunctionId,
148        args: Vec<Slot>,
149        ret: u32,
150    },
151    /// RHS primitive call binding `ret` (M2+).
152    #[allow(dead_code)]
153    Call {
154        id: ExternalFunctionId,
155        args: Vec<Slot>,
156        ret: u32,
157    },
158    /// `(union l r)` (M2+).
159    #[allow(dead_code)]
160    Union { l: Slot, r: Slot },
161    /// `(panic msg)`.
162    #[allow(dead_code)]
163    Panic(String),
164}

166/// The compiled IR for one egglog rule.
167///
168/// `body` is an ordered list of table-atom matches (and, in later milestones,
169/// primitive evaluations); `head` is an ordered list of writes. In M1 the
170/// FlowLog-driven `run_rules` path recognizes the canonical
171/// transitive-closure-step shape (two table body atoms joined on a shared
172/// variable, one `set` head) and executes it through the bundled flowlog
173/// incremental engine's `commit()`. The full ordered IR is retained so M2 can
174/// add the host fallback / richer rule shapes the way Feldera M3 did.
175#[derive(Clone, Debug, Default)]
176pub struct RuleIr {
177    pub name: String,
178    pub body: Vec<BodyOp>,
179    pub head: Vec<HeadOp>,
180}
