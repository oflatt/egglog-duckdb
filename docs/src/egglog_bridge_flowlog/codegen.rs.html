1//! Runtime translation of an egglog rule (the trait-level `RuleBuilderOps` IR)
2//! into a FlowLog `.dl` program plus a thin **driver** crate, for the M2
3//! shell-out architecture.
4//!
5//! ## What is generated, at runtime
6//!
7//! M1 compiled a *build-time-fixed* `transitive_step.dl` and recognized the
8//! rule shape structurally. M2 instead takes the rule the frontend builds **at
9//! runtime** (`StepShape`, recognized from the live `RuleIr`) and emits:
10//!
11//! 1. a `.dl` for it (a non-recursive single-join step — one `commit()` = one
12//!    bounded hop, exactly the M1 per-iteration model, but the relation names,
13//!    join columns, and head come from the runtime rule, not a fixed file);
14//! 2. a driver `main.rs` that `include!`s the flowlog-build-generated engine and
15//!    speaks a line-based **stdin/stdout command protocol** (see `protocol`);
16//! 3. a `Cargo.toml` + `build.rs` so the temp crate compiles standalone.
17//!
18//! The caller (`subprocess.rs`) hashes the `.dl`, builds the crate **once** per
19//! rule-set (`cargo build`, ~tens of seconds cold), caches the binary by hash,
20//! then spawns it and drives it over the pipe for the whole program — so the
21//! flowlog incremental engine stays warm and incrementality is preserved.
22//!
23//! ## The non-recursive single-join shape
24//!
25//! The recognized step is `head(x, z) :- path(x, y), edge(y, z)`. We emit two
26//! command-staged input relations (`path`, `edge`) and one output `hop`:
27//!
28//! ```text
29//! .decl edge(src: int32, dst: int32)
30//! .input edge(IO="command", delimiter=",")
31//! .decl path(src: int32, dst: int32)
32//! .input path(IO="command", delimiter=",")
33//! .decl hop(src: int32, dst: int32)
34//! hop(x, z) :- path(x, y), edge(y, z).
35//! .output hop
36//! ```
37//!
38//! The host (`lib.rs::run_one_hop_shellout`) folds each commit's `hop` deltas
39//! into the Rust-side mirror and re-stages the new `path` rows next round, the
40//! same bounded host-feedback loop M1 used — only now driving a *subprocess*
41//! over a pipe instead of an in-process engine.
42
43/// The fixed relation names the generated `.dl` and driver use. The runtime
44/// rule's actual `FunctionId`s are mapped onto these three roles by the host
45/// (`StepShape`); the `.dl` itself only needs stable lowercase idents so the
46/// generated engine's API (`insert_path` / `insert_edge` / `IncrementalResults.
47/// hop`) and the driver's protocol dispatch are predictable.
48pub const REL_EDGE: &str = "edge";
49pub const REL_PATH: &str = "path";
50pub const REL_HOP: &str = "hop";
51
52/// Emit the runtime `.dl` for a recognized transitive-closure step.
53///
54/// The shape is fixed (non-recursive single join), but this is produced *at
55/// rule-install time from the runtime rule IR* — it is not a checked-in file.
56/// Future shapes (different arities / multiple joins) extend here.
57pub fn emit_dl() -> String {
58    let mut s = String::new();
59    s.push_str("// AUTO-GENERATED at runtime by egglog-bridge-flowlog (M2 shell-out).\n");
60    s.push_str("// One non-recursive join = one bounded egglog hop per commit().\n\n");
61    s.push_str(&format!(".decl {REL_EDGE}(src: int32, dst: int32)\n"));
62    s.push_str(&format!(
63        ".input {REL_EDGE}(IO=\"command\", delimiter=\",\")\n\n"
64    ));
65    s.push_str(&format!(".decl {REL_PATH}(src: int32, dst: int32)\n"));
66    s.push_str(&format!(
67        ".input {REL_PATH}(IO=\"command\", delimiter=\",\")\n\n"
68    ));
69    s.push_str(&format!(".decl {REL_HOP}(src: int32, dst: int32)\n"));
70    s.push_str(&format!(
71        "{REL_HOP}(x, z) :- {REL_PATH}(x, y), {REL_EDGE}(y, z).\n"
72    ));
73    s.push_str(&format!(".output {REL_HOP}\n"));
74    s
75}

77/// The driver crate's `Cargo.toml`, parameterized by the absolute path to the
78/// local flowlog-rs clone (so the temp crate finds `flowlog-runtime` /
79/// `flowlog-build`). The crate is intentionally NOT a workspace member
80/// (`[workspace]` empty table) so it builds standalone in its temp dir.
81pub fn emit_cargo_toml(crate_name: &str, flowlog_root: &str) -> String {
82    format!(
83        r#"[package]
84name = "{crate_name}"
85version = "0.0.0"
86edition = "2021"
87publish = false
88
89# Standalone: do NOT inherit the egglog workspace.
90[workspace]
91
92[[bin]]
93name = "{crate_name}"
94path = "src/main.rs"
95
96[dependencies]
97flowlog-runtime = {{ path = "{flowlog_root}/crates/flowlog-runtime" }}
98
99[build-dependencies]
100flowlog-build = {{ path = "{flowlog_root}/crates/flowlog-build" }}
101"#
102    )
103}

105/// The driver crate's `build.rs`: compile the runtime-emitted `program.dl` into
106/// `$OUT_DIR/program.rs` in incremental mode. Identical in spirit to the M1
107/// build.rs, but the `.dl` it compiles was written at runtime.
108pub fn emit_build_rs() -> String {
109    r#"//! AUTO-GENERATED driver build.rs (M2 shell-out).
110use flowlog_build::{Builder, ExecutionMode};
111
112fn main() {
113    println!("cargo:rerun-if-changed=program.dl");
114    Builder::default()
115        .string_intern(false)
116        .mode(ExecutionMode::DatalogInc)
117        .compile(&["program.dl"], &[] as &[&str])
118        .expect("flowlog-build failed to compile program.dl");
119}
120"#
121    .to_string()
122}

124/// The driver `main.rs`: include the generated engine, then run the line-based
125/// stdin/stdout command protocol. The host side that speaks this wire format is
126/// [`crate::subprocess::DriverHandle`].
127///
128/// Protocol (one command per line on stdin; responses on stdout):
129///   - `insert <rel> <c0> <c1>`  -> stage an insert delta (no response)
130///   - `remove <rel> <c0> <c1>`  -> stage a remove delta (no response)
131///   - `commit`  -> step ONE epoch; emit `delta hop <x> <z> <diff>` for every
132///     output delta, then a final `ok` line
133///   - `read <rel>`              -> (reserved; emits `ok`)
134///   - `quit`                    -> exit 0
135///
136/// The driver flushes stdout after every command so the host never deadlocks.
137pub fn emit_main_rs() -> String {
138    r#"//! AUTO-GENERATED driver (M2 shell-out): embeds a flowlog-rs
139//! `DatalogIncrementalEngine` for the runtime-emitted `program.dl` and drives
140//! it over a line-based stdin/stdout command protocol. One `commit` command =
141//! one engine `commit()` = one bounded egglog hop.
142
143#[allow(clippy::all)]
144#[allow(dead_code)]
145#[allow(unused)]
146mod gen {
147    include!(concat!(env!("OUT_DIR"), "/program.rs"));
148}
149
150use gen::DatalogIncrementalEngine;
151use std::io::{BufRead, Write};
152
153fn main() {
154    let mut engine = DatalogIncrementalEngine::new(1);
155
156    let stdin = std::io::stdin();
157    let stdout = std::io::stdout();
158    let mut out = stdout.lock();
159
160    for line in stdin.lock().lines() {
161        let line = match line {
162            Ok(l) => l,
163            Err(_) => break,
164        };
165        let line = line.trim();
166        if line.is_empty() {
167            continue;
168        }
169        let mut it = line.split_whitespace();
170        let cmd = it.next().unwrap_or("");
171        match cmd {
172            "insert" | "remove" => {
173                let rel = it.next().unwrap_or("");
174                let cols: Vec<i32> = it.filter_map(|t| t.parse::<i32>().ok()).collect();
175                if cols.len() != 2 {
176                    let _ = writeln!(out, "err expected 2 columns, got {}", cols.len());
177                    let _ = out.flush();
178                    continue;
179                }
180                let tup = (cols[0], cols[1]);
181                let items = vec![tup];
182                match (cmd, rel) {
183                    ("insert", "edge") => engine.insert_edge(items),
184                    ("insert", "path") => engine.insert_path(items),
185                    ("remove", "edge") => engine.remove_edge(items),
186                    ("remove", "path") => engine.remove_path(items),
187                    _ => {
188                        let _ = writeln!(out, "err unknown rel {rel}");
189                        let _ = out.flush();
190                    }
191                }
192            }
193            "commit" => {
194                let results = engine.commit();
195                for (t, d) in results.hop.into_iter() {
196                    let t: (i32, i32) = t;
197                    let _ = writeln!(out, "delta hop {} {} {}", t.0, t.1, d);
198                }
199                let _ = writeln!(out, "ok");
200                let _ = out.flush();
201            }
202            "read" => {
203                // Reserved: the host keeps its own mirror; nothing to stream.
204                let _ = writeln!(out, "ok");
205                let _ = out.flush();
206            }
207            "quit" => {
208                break;
209            }
210            other => {
211                let _ = writeln!(out, "err unknown command {other}");
212                let _ = out.flush();
213            }
214        }
215    }
216}
217"#
218    .to_string()
