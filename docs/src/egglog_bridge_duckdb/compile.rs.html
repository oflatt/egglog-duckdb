1//! Compile a high-level `Rule` into a `CompiledVariant` per
2//! seminaive variant.
3//!
4//! A rule with N function-table atoms produces N variants. Variant i
5//! adds `WHERE alias_i.ts >= ?1` on the focused atom and reads the
6//! current epoch as `?2`. Both come in as bind parameters at run
7//! time. Each variant runs all its actions in order on every match.
8//!
9//! When a rule has any `Action::LetCtor` (allocate a fresh EqSort
10//! ID and bind it for use in later actions), the compiler switches
11//! to the *materialized* path:
12//!   1. CREATE OR REPLACE TEMP TABLE __m_<rule>_<variant> AS
13//!      SELECT <body cols>, nextval('seq') AS v1, … FROM <body
14//!      atoms> WHERE focus.ts >= ?1.
15//!   2. Each subsequent action runs as INSERT/DELETE FROM __m,
16//!      referencing both body bindings and let-allocated ids by
17//!      column.
18//! The materialize SQL has `?1` baked in (the focus ts threshold);
19//! the action SQLs only need `?2` (the current epoch).
20

29/// Glue a list of WHERE atoms into a single SQL fragment that
30/// starts with a leading space + `WHERE`, or "" if empty.
31fn format_where(parts: &[String]) -> String {
32    if parts.is_empty() {
33        String::new()
34    } else {
35        format!(" WHERE {}", parts.join(" AND "))
36    }
37}

39/// Target table written by an action, if any. Used by the runner
40/// to bump the table's insert watermark after a successful execute.
41fn action_target(a: &Action, functions: &HashMap<String, FunctionInfo>) -> Option<String> {
42    match a {
43        Action::Insert { name, .. } | Action::Delete { name, .. } => Some(name.clone()),
44        // LetCtor that has no associated `@<name>View` table actually
45        // INSERTs into the constructor itself (term-encoding helpers
46        // `@to_delete_X`, `@to_subsume_X`, `mergecleanup-N`). For
47        // those, treat the constructor as the action's target so the
48        // body-table watermark gate sees the write and downstream
49        // rules don't get skipped.
50        Action::LetCtor { name, .. } => {
51            let view = format!("@{name}View");
52            if functions.contains_key(&view) {
53                None
54            } else {
55                Some(name.clone())
56            }
57        }
58        Action::LetExpr { .. } | Action::Panic { .. } => None,
59    }
60}

62/// Mark each body atom as demotable iff it is a function-table
63/// lookup whose function has a registered native union-find UDF
64/// (`info.native_uf_udf == Some(_)`), the inputs are bound by an
65/// earlier body atom, and the output is a fresh `Var`. The compiler
66/// will emit a scalar UDF call binding the output var rather than
67/// adding a join atom.
68///
69/// Walks the body greedily in source order. A demoted atom only
70/// contributes its output binding, so later atoms that depend on
71/// it can still resolve.
72fn compute_native_uf_demote(body: &[Atom], functions: &HashMap<String, FunctionInfo>) -> Vec<bool> {
73    let mut demote = vec![false; body.len()];
74    let mut bound: std::collections::HashSet<String> = std::collections::HashSet::new();
75    for (i, atom) in body.iter().enumerate() {
76        match atom {
77            Atom::Func { name, args } => {
78                let info = match functions.get(name) {
79                    Some(i) => i,
80                    None => continue,
81                };
82                let inputs_bound = args[..info.inputs_len].iter().all(|t| match t {
83                    Term::Var(v) => bound.contains(v),
84                    _ => true,
85                });
86                let output_fresh = info.inputs_len + 1 == args.len()
87                    && matches!(&args[info.inputs_len], Term::Var(v) if !bound.contains(v));
88                if info.native_uf_udf.is_some() && inputs_bound && output_fresh {
89                    demote[i] = true;
90                    if let Term::Var(v) = &args[info.inputs_len] {
91                        bound.insert(v.clone());
92                    }
93                } else {
94                    for arg in args {
95                        if let Term::Var(v) = arg {
96                            bound.insert(v.clone());
97                        }
98                    }
99                }
100            }
101            Atom::Bind { var, .. } => {
102                bound.insert(var.clone());
103            }
104            Atom::Filter(_) => {}
105        }
106    }
107    demote
108}

110/// Lift inline `Term::FuncCall`s that target an EqSort constructor
111/// (or its term-encoded view) into explicit `LetCtor` actions, so
112/// the existing hash-cons machinery in the materialized path can
113/// allocate-or-reuse an ID for them instead of doing a bare lookup.
114///
115/// Term encoding's proof mode is the only thing that currently emits
116/// nested calls here — e.g. the congruence rule's
117/// `(set (pname l s) (Trans p1 (Sym p2)))`. Without this lift the
118/// nested `(Trans …)` and `(Sym …)` compile to scalar subqueries
119/// against `@_trans` / `@_sym`, which return NULL when the proof
120/// term has never been constructed; the NULL then hits the
121/// `NOT NULL` constraint on `pname.c2`. After lifting, each nested
122/// call becomes a fresh `LetCtor` that the materialized path knows
123/// how to hash-cons-or-allocate.
124///
125/// Only Calls whose target is `eq_sort_ctor` get lifted: regular
126/// functions still resolve via a lookup, which is what the encoder
127/// expects for them.
128fn lift_nested_eq_sort_calls(rule: &Rule, functions: &HashMap<String, FunctionInfo>) -> Rule {
129    let mut counter: usize = 0;
130    let mut new_actions: Vec<Action> = Vec::with_capacity(rule.actions.len());
131    for action in &rule.actions {
132        let mut lifted = action.clone();
133        match &mut lifted {
134            Action::Insert { args, .. } | Action::Delete { key_args: args, .. } => {
135                let new_args = args
136                    .iter()
137                    .cloned()
138                    .map(|t| lift_term(t, &rule.name, &mut counter, &mut new_actions, functions))
139                    .collect();
140                *args = new_args;
141            }
142            Action::LetCtor { args, .. } => {
143                let new_args = args
144                    .iter()
145                    .cloned()
146                    .map(|t| lift_term(t, &rule.name, &mut counter, &mut new_actions, functions))
147                    .collect();
148                *args = new_args;
149            }
150            Action::LetExpr { expr, .. } => {
151                *expr = lift_term(
152                    expr.clone(),
153                    &rule.name,
154                    &mut counter,
155                    &mut new_actions,
156                    functions,
157                );
158            }
159            Action::Panic { .. } => {}
160        }
161        new_actions.push(lifted);
162    }
163    Rule {
164        name: rule.name.clone(),
165        ruleset: rule.ruleset.clone(),
166        body: rule.body.clone(),
167        actions: new_actions,
168    }
169}

171/// Recursively lift FuncCalls in `term`. Lifted calls become
172/// `LetCtor` actions pushed to `out` in dependency order (innermost
173/// first, outermost last), and the call site is replaced with a
174/// `Var` reference to the lifted name. Calls whose target is *not*
175/// an `eq_sort_ctor` are left in place — they read existing
176/// function rows.
177fn lift_term(
178    term: Term,
179    rule_name: &str,
180    counter: &mut usize,
181    out: &mut Vec<Action>,
182    functions: &HashMap<String, FunctionInfo>,
183) -> Term {
184    match term {
185        Term::FuncCall { name, args } => {
186            let lifted_args: Vec<Term> = args
187                .into_iter()
188                .map(|a| lift_term(a, rule_name, counter, out, functions))
189                .collect();
190            let is_eq_sort_ctor = functions
191                .get(&name)
192                .map(|i| i.eq_sort_ctor)
193                .unwrap_or(false);
194            if is_eq_sort_ctor {
195                let var = format!("__lifted_{rule_name}_{}", *counter);
196                *counter += 1;
197                out.push(Action::LetCtor {
198                    var: var.clone(),
199                    name,
200                    args: lifted_args,
201                });
202                Term::Var(var)
203            } else {
204                Term::FuncCall {
205                    name,
206                    args: lifted_args,
207                }
208            }
209        }
210        Term::Prim(op, args) => {
211            let lifted_args: Vec<Term> = args
212                .into_iter()
213                .map(|a| lift_term(a, rule_name, counter, out, functions))
214                .collect();
215            Term::Prim(op, lifted_args)
216        }
217        other => other,
218    }
219}

221pub(crate) fn compile_rule(
222    rule: &Rule,
223    functions: &HashMap<String, FunctionInfo>,
224    proofs_enabled: bool,
225) -> Result<CompiledRule> {
226    // Rewrite the rule before validation/compilation so any nested
227    // eq-sort constructor calls in action positions become explicit
228    // `LetCtor`s (hash-cons-or-allocate). See
229    // `lift_nested_eq_sort_calls` for why.
230    //
231    // NB: there used to be a `dedupe_body_atoms` pass here that merged
232    // body atoms with identical inputs. It existed only to make the
233    // inlined Herbie dumps' deeply-repeated patterns tractable, but it
234    // was unsound for self-joins (it collapsed the nullary-`:merge`
235    // reconciliation and per-sort congruence rules into always-false
236    // `x <> x` filters) and caused multiple bugs. It's been removed
237    // along with the Herbie `inline-bigrat-lets` workaround that
238    // created the duplicate atoms; non-inlined dumps don't blow up.
239    let rule_owned = lift_nested_eq_sort_calls(rule, functions);
240    let rule = &rule_owned;
241    // Validate function-table atoms.
242    for atom in &rule.body {
243        if let Atom::Func { name, args } = atom {
244            let info = functions
245                .get(name)
246                .ok_or_else(|| anyhow!("rule {}: unknown function {name}", rule.name))?;
247            if args.len() != info.arity() {
248                return Err(anyhow!(
249                    "rule {}: atom {name} has {} args, expected {}",
250                    rule.name,
251                    args.len(),
252                    info.arity()
253                ));
254            }
255        }
256    }
257    // Validate actions.
258    for action in &rule.actions {
259        match action {
260            Action::Insert { name, args } => {
261                let info = functions
262                    .get(name)
263                    .ok_or_else(|| anyhow!("rule {}: unknown action target {name}", rule.name))?;
264                if args.len() != info.arity() {
265                    return Err(anyhow!(
266                        "rule {}: action target {name} has {} args, expected {}",
267                        rule.name,
268                        args.len(),
269                        info.arity()
270                    ));
271                }
272            }
273            Action::Delete { name, key_args } => {
274                let info = functions
275                    .get(name)
276                    .ok_or_else(|| anyhow!("rule {}: unknown action target {name}", rule.name))?;
277                if key_args.len() != info.inputs_len {
278                    return Err(anyhow!(
279                        "rule {}: delete on {name} has {} args, expected {} (input cols)",
280                        rule.name,
281                        key_args.len(),
282                        info.inputs_len
283                    ));
284                }
285            }
286            Action::LetCtor { name, args, .. } => {
287                let info = functions
288                    .get(name)
289                    .ok_or_else(|| anyhow!("rule {}: unknown LetCtor target {name}", rule.name))?;
290                if !info.eq_sort_ctor {
291                    return Err(anyhow!(
292                        "rule {}: LetCtor target {name} is not an EqSort constructor",
293                        rule.name
294                    ));
295                }
296                if args.len() != info.inputs_len {
297                    return Err(anyhow!(
298                        "rule {}: LetCtor {name} has {} args, expected {}",
299                        rule.name,
300                        args.len(),
301                        info.inputs_len
302                    ));
303                }
304            }
305            Action::LetExpr { .. } => { /* no validation needed */ }
306            Action::Panic { .. } => { /* no validation needed */ }
307        }
308    }
309
310    // Native-UF demote plan: body atoms against `:merge ordering-min`
311    // functions get rewritten as scalar UDF calls
312    // (`duck_uf_<sort>_find(input)`) rather than join atoms. They
313    // don't contribute to the seminaive triangulation either —
314    // dropping them from `func_atom_indices` removes the variants
315    // that would have focused on them. Empty `Vec<bool>` (all
316    // false) when `--duck-native-uf` is off.
317    let demote = compute_native_uf_demote(&rule.body, functions);
318
319    let func_atom_indices: Vec<usize> = rule
320        .body
321        .iter()
322        .enumerate()
323        .filter_map(|(i, a)| (matches!(a, Atom::Func { .. }) && !demote[i]).then_some(i))
324        .collect();
325
326    if rule.actions.is_empty() {
327        return Err(anyhow!("rule {}: at least one action required", rule.name));
328    }
329
330    // Fuse all K seminaive variants for this rule into a single
331    // `CompiledVariant`. The materialize SELECT becomes a UNION ALL
332    // of K branches differing only in their ts predicates; the
333    // partitioning is disjoint by construction (each match has
334    // exactly one "first new" atom), so no dedup is needed. Cuts
335    // the per-iter statement count by ~K× without changing the
336    // total join work.
337    let mut variants = vec![compile_fused_variant(
338        rule,
339        &func_atom_indices,
340        functions,
341        &demote,
342        None,
343        proofs_enabled,
344    )?];
345    // For each distinct table referenced by a demoted atom, emit
346    // one gated "UF-changed" recovery variant: scan the body with
347    // no `>= ?1` lower bound, look up canonical roots via UDF,
348    // canonicalize stale rows. The runner runs this only when the
349    // gated table's watermark has advanced since this rule last
350    // ran — saves work when nothing has unified.
351    let mut demote_tables: Vec<String> = Vec::new();
352    for (i, atom) in rule.body.iter().enumerate() {
353        if !demote[i] {
354            continue;
355        }
356        if let Atom::Func { name, .. } = atom
357            && !demote_tables.iter().any(|n| n == name)
358        {
359            demote_tables.push(name.clone());
360        }
361    }
362    for gated in &demote_tables {
363        // Native-UF gating: writes that affect the demoted atom's
364        // canonicalization land in the corresponding pname (raw UF)
365        // table, not the function-form table itself. Derive pname
366        // by the term-encoding naming convention so the runner can
367        // skip the gated variant when nothing new has unified this
368        // iteration.
369        let gate_pname = gated
370            .strip_suffix('f')
371            .unwrap_or(gated.as_str())
372            .to_string();
373        variants.push(compile_fused_variant(
374            rule,
375            &func_atom_indices,
376            functions,
377            &demote,
378            Some(&gate_pname),
379            proofs_enabled,
380        )?);
381    }
382    Ok(CompiledRule {
383        name: rule.name.clone(),
384        ruleset: rule.ruleset.clone(),
385        variants,
386        // `add_rule` re-derives this from `rule.body` so it stays in
387        // sync with the IR even if compilation rewrites the variants.
388        body_tables: Vec::new(),
389    })
390}

392/// Compile all seminaive variants of `rule` into a single
393/// `CompiledVariant` by UNION-ALL-ing the per-focus branches.
394///
395/// Each branch shares the same FROM clause, hash-cons LEFT JOINs,
396/// and SELECT projection — the branches differ only in their `ts`
397/// predicates. Seminaive triangulation guarantees the partitions
398/// are disjoint, so UNION ALL preserves exactly the union-of-
399/// per-focus matches with no duplicates.
400fn compile_fused_variant(
401    rule: &Rule,
402    func_atom_indices: &[usize],
403    functions: &HashMap<String, FunctionInfo>,
404    demote: &[bool],
405    gate_table: Option<&str>,
406    proofs_enabled: bool,
407) -> Result<CompiledVariant> {
408    // 1) Build the body's FROM/WHERE and a binding for body vars.
409    let (from, base_where_parts, mut binding) =
410        walk_body(&rule.body, &rule.name, functions, demote)?;
411    // No Func atoms → use a 1-row dummy as FROM source.
412    let from = if from.is_empty() {
413        "(SELECT 1) __empty__".to_string()
414    } else {
415        from
416    };
417    let where_clause = format_where(&base_where_parts);
418
419    // 2) Decide whether we need the materialized path.
420    //
421    // We materialize the body when:
422    //  (a) Any action allocates a fresh ID (LetCtor) or computes a
423    //      derived value (LetExpr) that subsequent actions reference.
424    //  (b) The rule has multiple actions — even pure INSERT/DELETE
425    //      pairs need to see the SAME body snapshot. Without
426    //      materialization, an action like `(delete X)` would remove
427    //      rows from the body before a subsequent `(set X')` action
428    //      could read them, dropping the corresponding insertions.
429    //      This is the @uf_update/@singleparent pattern: delete an
430    //      old row, insert the canonicalized successor.
431    let has_let = rule
432        .actions
433        .iter()
434        .any(|a| matches!(a, Action::LetCtor { .. } | Action::LetExpr { .. }));
435    let needs_snapshot = has_let || rule.actions.len() > 1;
436
437    // The K seminaive partitions, expressed as per-focus WHERE
438    // clauses (built from the shared body WHERE + per-focus ts
439    // predicates). For gated variants, no `>= ?1` lower bound: the
440    // variant is triggered by an external watermark and wants a
441    // full scan. Empty body → one branch with no ts predicates.
442    let branch_wheres: Vec<String> = if gate_table.is_some() {
443        // Gated: full scan of non-demoted Func atoms with only the
444        // upper-bound `< ?2` filter (avoid seeing rows inserted by
445        // this same iteration).
446        let mut parts = base_where_parts.clone();
447        for (i, atom) in rule.body.iter().enumerate() {
448            if !matches!(atom, Atom::Func { .. }) {
449                continue;
450            }
451            if demote.get(i).copied().unwrap_or(false) {
452                continue;
453            }
454            parts.push(format!("t{i}.ts < ?2"));
455        }
456        vec![format_where(&parts)]
457    } else if func_atom_indices.is_empty() {
458        vec![where_clause.clone()]
459    } else {
460        func_atom_indices
461            .iter()
462            .map(|&focus| {
463                let mut parts = base_where_parts.clone();
464                parts.extend(ts_predicates_for_focus(&rule.body, focus, demote));
465                format_where(&parts)
466            })
467            .collect()
468    };
469    let _ = where_clause;
470
471    if !needs_snapshot {
472        // Single-action rule with no Let: simple INSERT/DELETE FROM
473        // <body> is correct, and avoids the temp-table overhead.
474        // For K-way fusion we UNION ALL the per-focus branches as
475        // the source of the action; the partitioning is disjoint so
476        // no rows are duplicated.
477        let actions = rule
478            .actions
479            .iter()
480            .map(|a| {
481                let sql = compile_simple_action_fused(
482                    a,
483                    &binding,
484                    &from,
485                    &branch_wheres,
486                    &rule.name,
487                    functions,
488                )?;
489                Ok(CompiledAction {
490                    sql,
491                    target: action_target(a, functions),
492                })
493            })
494            .collect::<Result<_>>()?;
495        return Ok(CompiledVariant {
496            materialize: None,
497            temp_table: None,
498            actions,
499            gate_table: gate_table.map(str::to_string),
500        });
501    }
502
503    // 3) Materialized path. Determine the columns of the temp table:
504    // - one column per body-bound variable (so subsequent actions can
505    //   reference body vars).
506    // - one nextval column per LetCtor.
507    let temp_table = format!("__m_{}", sanitize(&rule.name));
508
509    let mut select_cols: Vec<String> = Vec::new();
510    // body vars: take their stable binding location and assign an
511    // alias matching the var name. Sort to make codegen
512    // deterministic (binding is a HashMap).
513    let mut body_vars: Vec<(String, String)> = binding
514        .iter()
515        .map(|(v, expr)| (v.clone(), expr.clone()))
516        .collect();
517    body_vars.sort_by(|a, b| a.0.cmp(&b.0));
518    for (v, expr) in &body_vars {
519        select_cols.push(format!("{expr} AS {}", q(v)));
520    }
521    // After materialization, body vars are accessed via their alias.
522    let mut mat_binding: HashMap<String, String> = HashMap::new();
523    for (v, _) in &body_vars {
524        mat_binding.insert(v.clone(), q(v));
525    }
526    // Let allocations: each adds a column to the temp table.
527    // LetCtor: when all args resolve to body-bound vars (i.e., they
528    // don't reference an earlier LetCtor's var), hash-cons against
529    // the existing view via a scalar subquery — only burn a fresh
530    // sequence value when no row exists for these args. Avoids the
531    // duplicate-ID-then-congruence-collapse churn that drives
532    // @UF_<sort> growth. When args reference an earlier LetCtor's
533    // value, we can't read sibling SELECT-list aliases in the same
534    // SELECT, so fall back to plain nextval.
535    // LetExpr: just an alias for an expression evaluated against
536    // the body bindings.
537    // For LetCtors whose args are body-resolvable, we add a LEFT
538    // JOIN against a derived table that aggregates the constructor
539    // view by (input cols), giving at most one existing-id row per
540    // (args). The temp table's column for that LetCtor is then
541    // `COALESCE(hc<i>.id, nextval(seq))` — reusing an existing id
542    // when the term has been seen before, else allocating fresh.
543    // This is the in-band hash-cons that avoids the duplicate-id-
544    // then-congruence-collapse pattern.
545    let mut hc_joins: Vec<String> = Vec::new();
546    for action in &rule.actions {
547        match action {
548            Action::LetCtor { var, name, args } => {
549                let body_resolvable = args.iter().all(|t| match t {
550                    Term::Var(v) => binding.contains_key(v),
551                    _ => true,
552                });
553                // Which table to hash-cons against. In non-proof mode
554                // we use the canonicalized `@<name>View`: its id column
555                // is rewritten to the e-class leader by rebuild, so
556                // congruent terms fold onto one id (faster, fewer
557                // rows). In proof mode that is incorrect — reusing the
558                // leader gives the constructed term a *different*
559                // constructor's id, so proof extraction reconstructs
560                // the wrong (equal) term and the checker rejects it.
561                // The bare term table (`name`) keeps each application's
562                // own id, like the bridge backend, so proofs stay
563                // faithful. See `EGraph::proofs_enabled`.
564                let view = format!("@{name}View");
565                // Only redirect to the bare term table for genuine
566                // eq-sort *user* constructors — those that have a
567                // `@<name>View` (whose id column the rebuild rewrites
568                // to the leader). Proof-machinery constructors (AstC,
569                // Rule, Trans, …) have no view and keep their existing
570                // hash-cons/nextval behavior.
571                let view_exists = functions.contains_key(&view);
572                let hc_table = if proofs_enabled && view_exists {
573                    name.clone()
574                } else {
575                    view.clone()
576                };
577                // Hash-cons is only meaningful when the hash-cons table
578                // actually stores the output ID (i.e., it has
579                // args.len() + 1 columns: inputs + id). For
580                // constructors whose original return type was a
581                // primitive (e.g., a `(function foo () i64 :merge …)`
582                // gets rewritten by term encoding into a 1-arg
583                // constructor backed by a 1-column @fooView relation
584                // that only tracks existence, no id), there's no id
585                // to look up — fall back to plain nextval.
586                let view_has_id = functions
587                    .get(&hc_table)
588                    .map(|i| i.cols.len() > args.len())
589                    .unwrap_or(false);
590                // Hash-cons: look up an existing canonical id for
591                // (args) before allocating, falling back to a fresh
592                // sequence value when nothing matches. Saves
593                // substantial work on rule-heavy workloads
594                // (math-microbenchmark run-10: 34.6s/10.27 GB →
595                // 17.4s/668 MB) without affecting tuple counts —
596                // verified through run-11 against the reference
597                // backend, and the full snapshot suite.
598                let col_expr = if body_resolvable && view_has_id {
599                    let arg_sqls: Vec<String> = args
600                        .iter()
601                        .map(|t| term_sql(t, &binding, &rule.name))
602                        .collect::<Result<_>>()?;
603                    let hc_alias = format!("hc{}", hc_joins.len());
604                    let id_col_idx = arg_sqls.len();
605                    let in_cols: Vec<String> = (0..id_col_idx).map(|i| format!("c{i}")).collect();
606                    if arg_sqls.is_empty() {
607                        // 0-input: there's only ever one such row.
608                        // Cross join with a 1-row aggregate.
609                        hc_joins.push(format!(
610                            "LEFT JOIN (SELECT MIN(c{id_col_idx}) AS id FROM {}) {hc_alias} ON TRUE",
611                            q(&hc_table)
612                        ));
613                    } else {
614                        let on_conds: Vec<String> = arg_sqls
615                            .iter()
616                            .enumerate()
617                            .map(|(i, s)| format!("{hc_alias}.c{i} = {s}"))
618                            .collect();
619                        hc_joins.push(format!(
620                            "LEFT JOIN (SELECT {}, MIN(c{id_col_idx}) AS id FROM {} GROUP BY {}) {hc_alias} ON {}",
621                            in_cols.join(", "),
622                            q(&hc_table),
623                            in_cols.join(", "),
624                            on_conds.join(" AND ")
625                        ));
626                    }
627                    format!("COALESCE({hc_alias}.id, nextval('__egglog_eqsort_seq'))")
628                } else {
629                    "nextval('__egglog_eqsort_seq')".to_string()
630                };
631                select_cols.push(format!("{col_expr} AS {}", q(var)));
632                mat_binding.insert(var.clone(), q(var));
633            }
634            Action::LetExpr { var, expr } => {
635                // Pure aliases `(let $a @v6)` where the RHS is a
636                // var already bound by an earlier LetCtor live in
637                // `mat_binding`, not `binding`. SELECT-list aliases
638                // can't reference each other within the same SELECT,
639                // so just re-point the alias instead of emitting a
640                // column.
641                if let Term::Var(other) = expr
642                    && let Some(src) = mat_binding.get(other)
643                {
644                    let src = src.clone();
645                    mat_binding.insert(var.clone(), src);
646                    continue;
647                }
648                let expr_sql = term_sql(expr, &binding, &rule.name)?;
649                select_cols.push(format!("{expr_sql} AS {}", q(var)));
650                mat_binding.insert(var.clone(), q(var));
651            }
652            _ => {}
653        }
654    }
655    let from_with_hc = if hc_joins.is_empty() {
656        from.clone()
657    } else {
658        // SQL grammar: `t0, t1 LEFT JOIN d ON …` parses the LEFT
659        // JOIN as binding to `t1` alone, so the ON clause can't see
660        // `t0`. Convert comma-joins to explicit CROSS JOIN so the
661        // LEFT JOIN binds to the whole left side.
662        let xj = from.replace(", ", " CROSS JOIN ");
663        format!("{xj} {}", hc_joins.join(" "))
664    };
665    // Build the materialize SQL as a UNION ALL of one branch per
666    // focus. Each branch has the same SELECT projection and FROM
667    // clause (including hash-cons LEFT JOINs); only the WHERE
668    // changes. Seminaive partitions are disjoint by construction,
669    // so UNION ALL is exact — no need for UNION/DISTINCT.
670    let select_cols_str = select_cols.join(", ");
671    let branches: Vec<String> = branch_wheres
672        .iter()
673        .map(|w| format!("SELECT {select_cols_str} FROM {from_with_hc}{w}"))
674        .collect();
675    let materialize = format!(
676        "CREATE OR REPLACE TEMP TABLE {} AS {}",
677        q(&temp_table),
678        branches.join(" UNION ALL "),
679    );
680
681    // 4) Build per-action SQLs that select FROM the temp table.
682    let mut action_sqls: Vec<CompiledAction> = Vec::new();
683    for action in &rule.actions {
684        let sql =
685            compile_materialized_action(action, &mat_binding, &temp_table, &rule.name, functions)?;
686        action_sqls.push(CompiledAction {
687            sql,
688            target: action_target(action, functions),
689        });
690    }
691
692    // Suppress `binding` unused warnings; it's only relevant on the
693    // simple path above.
694    let _ = &mut binding;
695
696    // No explicit DROP — the next iteration's CREATE OR REPLACE
697    // recycles the temp table in place. Issuing a redundant DROP
698    // cost ~25% of our SQL-statement count on math-microbenchmark.
699    Ok(CompiledVariant {
700        materialize: Some(materialize),
701        temp_table: Some(temp_table),
702        actions: action_sqls,
703        gate_table: gate_table.map(str::to_string),
704    })
705}

707/// Compile a list of body atoms (as in a rule body or a `(check ...)`)
708/// into a SQL `SELECT 1 FROM <body> [WHERE …] LIMIT 1` returning a
709/// row iff the conjunction has any match. No seminaive focus
710/// predicate; that's only for rules.
711pub(crate) fn compile_body_select(
712    atoms: &[Atom],
713    functions: &std::collections::HashMap<String, crate::FunctionInfo>,
714) -> Result<String> {
715    // Validate.
716    for atom in atoms {
717        if let Atom::Func { name, args } = atom {
718            let info = functions
719                .get(name)
720                .ok_or_else(|| anyhow!("body atom: unknown function {name}"))?;
721            if args.len() != info.arity() {
722                return Err(anyhow!(
723                    "body atom {name} has {} args, expected {}",
724                    args.len(),
725                    info.arity()
726                ));
727            }
728        }
729    }
730    // `(check ...)` runs against the current state; no native-UF
731    // demote applies (we want the literal join, not a UDF lookup
732    // bypass) so all-false `demote`.
733    let no_demote = vec![false; atoms.len()];
734    let (from, where_parts, _binding) = walk_body(atoms, "<check>", functions, &no_demote)?;
735    let where_clause = format_where(&where_parts);
736    if from.is_empty() {
737        // No function atoms: a pure-primitive `(check (= 1 1))`-style
738        // assertion. Evaluate the WHERE clause as a constant.
739        return Ok(format!(
740            "SELECT COUNT(*) FROM (SELECT 1{}) LIMIT 1",
741            where_clause
742        ));
743    }
744    Ok(format!(
745        "SELECT COUNT(*) FROM (SELECT 1 FROM {from}{where_clause} LIMIT 1)"
746    ))
747}

749/// Shared body walker used by both rule compilation (via
750/// `build_query`) and check compilation (via `compile_body_select`).
751/// Try to process a single body atom. Returns `Ok(true)` if the
752/// atom was fully processed and contributed to `binding` /
753/// `from_parts` / `where_parts`; `Ok(false)` if it depends on a var
754/// that hasn't been bound yet and should be retried later. The Err
755/// path is reserved for genuine compilation errors (e.g., unbound
756/// var inside a Prim subterm).
757fn try_walk_atom(
758    atom: &Atom,
759    i: usize,
760    binding: &mut HashMap<String, String>,
761    from_parts: &mut Vec<String>,
762    where_parts: &mut Vec<String>,
763    rule_name: &str,
764    functions: &HashMap<String, FunctionInfo>,
765    demoted: bool,
766) -> Result<bool> {
767    match atom {
768        Atom::Func { name, args } => {
769            if demoted {
770                // Native-UF demote: emit a scalar UDF call binding
771                // the output var instead of adding a join atom. The
772                // compute_native_uf_demote pre-pass already verified
773                // shape (inputs bound, single fresh output Var), but
774                // the worklist may not have processed the dependency
775                // yet on this pass — defer if any input var is still
776                // unbound.
777                let info = &functions[name];
778                if !args[..info.inputs_len].iter().all(|t| match t {
779                    Term::Var(v) => binding.contains_key(v),
780                    _ => true,
781                }) {
782                    return Ok(false);
783                }
784                let udf = info
785                    .native_uf_udf
786                    .as_ref()
787                    .expect("demoted atom must have native_uf_udf set");
788                let arg_sqls: Vec<String> = args[..info.inputs_len]
789                    .iter()
790                    .map(|t| term_sql(t, binding, rule_name))
791                    .collect::<Result<_>>()?;
792                let call = format!("{}({})", udf, arg_sqls.join(", "));
793                if let Term::Var(v) = &args[info.inputs_len] {
794                    binding.insert(v.clone(), call);
795                }
796                return Ok(true);
797            }
798            let alias = format!("t{i}");
799            from_parts.push(format!("{} {alias}", q(name)));
800            for (col, term) in args.iter().enumerate() {
801                let lhs = format!("{alias}.c{col}");
802                match term {
803                    Term::Var(v) => match binding.get(v) {
804                        None => {
805                            binding.insert(v.clone(), lhs);
806                        }
807                        Some(prev) => {
808                            where_parts.push(format!("{prev} = {lhs}"));
809                        }
810                    },
811                    Term::Lit(_) | Term::Prim(_, _) | Term::FuncCall { .. } => {
812                        let rhs = term_sql(term, binding, rule_name)?;
813                        where_parts.push(format!("{lhs} = {rhs}"));
814                    }
815                }
816            }
817            Ok(true)
818        }
819        Atom::Filter(t) => {
820            if !term_vars_all_bound(t, binding) {
821                return Ok(false);
822            }
823            where_parts.push(term_sql(t, binding, rule_name)?);
824            Ok(true)
825        }
826        Atom::Bind { var, expr } => {
827            // Symmetric handling for Var-Var binds: if one side is
828            // already bound and the other isn't, propagate the
829            // binding to the unbound side. If neither is bound,
830            // defer (a later Func atom probably binds one of them).
831            if let Term::Var(other) = expr {
832                let var_bound = binding.contains_key(var);
833                let other_bound = binding.contains_key(other);
834                match (var_bound, other_bound) {
835                    (true, true) => {
836                        let l = binding[var].clone();
837                        let r = binding[other].clone();
838                        where_parts.push(format!("{l} = {r}"));
839                        Ok(true)
840                    }
841                    (true, false) => {
842                        let l = binding[var].clone();
843                        binding.insert(other.clone(), l);
844                        Ok(true)
845                    }
846                    (false, true) => {
847                        let r = binding[other].clone();
848                        binding.insert(var.clone(), r);
849                        Ok(true)
850                    }
851                    (false, false) => Ok(false),
852                }
853            } else {
854                if !term_vars_all_bound(expr, binding) {
855                    return Ok(false);
856                }
857                let s = term_sql(expr, binding, rule_name)?;
858                match binding.get(var) {
859                    None => {
860                        binding.insert(var.clone(), s);
861                    }
862                    Some(prev) => {
863                        where_parts.push(format!("{prev} = {s}"));
864                    }
865                }
866                Ok(true)
867            }
868        }
869    }
870}

872/// Check whether every `Term::Var` referenced inside `t` (directly
873/// or transitively through Prim/FuncCall children) is already
874/// bound. Used to decide whether a Filter/Bind atom is ready to
875/// process or should be deferred.
876fn term_vars_all_bound(t: &Term, binding: &HashMap<String, String>) -> bool {
877    match t {
878        Term::Var(v) => binding.contains_key(v),
879        Term::Lit(_) => true,
880        Term::Prim(_, args) | Term::FuncCall { args, .. } => {
881            args.iter().all(|a| term_vars_all_bound(a, binding))
882        }
883    }
884}

886/// Compute the seminaive ts predicates for a single focus choice.
887/// Returned strings reference table aliases `t{i}` for each Func
888/// atom; they're meant to be AND-combined with the body's own WHERE
889/// parts (which `walk_body` builds and returns unbounded).
890/// Demoted atoms (those that become scalar UDF calls) don't have a
891/// `tN` alias and are skipped — caller passes `demote` matching the
892/// per-atom plan used by `walk_body`.
893fn ts_predicates_for_focus(atoms: &[Atom], focus_idx: usize, demote: &[bool]) -> Vec<String> {
894    let mut parts = Vec::new();
895    for (i, atom) in atoms.iter().enumerate() {
896        if !matches!(atom, Atom::Func { .. }) {
897            continue;
898        }
899        if demote.get(i).copied().unwrap_or(false) {
900            continue;
901        }
902        match i.cmp(&focus_idx) {
903            std::cmp::Ordering::Equal => {
904                parts.push(format!("t{i}.ts >= ?1"));
905                parts.push(format!("t{i}.ts < ?2"));
906            }
907            std::cmp::Ordering::Less => {
908                parts.push(format!("t{i}.ts < ?1"));
909            }
910            std::cmp::Ordering::Greater => {
911                parts.push(format!("t{i}.ts < ?2"));
912            }
913        }
914    }
915    parts
916}

918fn walk_body(
919    atoms: &[Atom],
920    rule_name: &str,
921    functions: &HashMap<String, FunctionInfo>,
922    demote: &[bool],
923) -> Result<(String, Vec<String>, HashMap<String, String>)> {
924    let mut binding: HashMap<String, String> = HashMap::new();
925    let mut from_parts: Vec<String> = Vec::new();
926    let mut where_parts: Vec<String> = Vec::new();
927
928    // Term encoding can emit Bind atoms whose vars are introduced
929    // by *later* Func atoms (e.g. `(rewrite p $True :when …)` →
930    // body atom 0 is `(= @rewrite_var__ p)` where `p` is first
931    // bound by atom 3). Defer any atom we can't process yet and
932    // retry until we converge. Func atoms always process first
933    // pass since they introduce their own bindings unconditionally.
934    let mut pending: Vec<usize> = (0..atoms.len()).collect();
935    loop {
936        let mut made_progress = false;
937        let mut still_pending: Vec<usize> = Vec::new();
938        for i in pending {
939            let atom = &atoms[i];
940            let ok = try_walk_atom(
941                atom,
942                i,
943                &mut binding,
944                &mut from_parts,
945                &mut where_parts,
946                rule_name,
947                functions,
948                demote.get(i).copied().unwrap_or(false),
949            )?;
950            if ok {
951                made_progress = true;
952            } else {
953                still_pending.push(i);
954            }
955        }
956        if still_pending.is_empty() {
957            break;
958        }
959        if !made_progress {
960            // Pick the first unsatisfied atom and error with a
961            // descriptive message rather than looping forever.
962            let i = still_pending[0];
963            match &atoms[i] {
964                Atom::Bind { var, expr } => {
965                    return Err(anyhow!(
966                        "rule {rule_name}: cannot resolve Bind {var} = {expr:?}: dependent vars never get bound"
967                    ));
968                }
969                Atom::Filter(t) => {
970                    return Err(anyhow!(
971                        "rule {rule_name}: cannot resolve Filter {t:?}: dependent vars never get bound"
972                    ));
973                }
974                Atom::Func { name, .. } => {
975                    return Err(anyhow!(
976                        "rule {rule_name}: cannot resolve Func {name}: internal error"
977                    ));
978                }
979            }
980        }
981        pending = still_pending;
982    }
983    let from = from_parts.join(", ");
984    Ok((from, where_parts, binding))
985}

987/// Compile one rule action under the *simple* path, fused across
988/// all K seminaive partitions. Each `branch_wheres` entry is the
989/// fully-formed WHERE clause (including the per-focus ts
990/// predicates) for one partition; the action's source SELECT is
991/// the UNION ALL of one branch per entry.
992fn compile_simple_action_fused(
993    action: &Action,
994    binding: &HashMap<String, String>,
995    from: &str,
996    branch_wheres: &[String],
997    rule_name: &str,
998    functions: &HashMap<String, FunctionInfo>,
999) -> Result<String> {
1000    let union_branches = |body: &str| -> String {
1001        branch_wheres
1002            .iter()
1003            .map(|w| format!("{body}{w}"))
1004            .collect::<Vec<_>>()
1005            .join(" UNION ALL ")
1006    };
1007    match action {
1008        Action::Insert {
1009            name: target,
1010            args: targs,
1011        } => {
1012            let info = &functions[target];
1013            let select_cols: Vec<String> = targs
1014                .iter()
1015                .map(|t| term_sql(t, binding, rule_name))
1016                .collect::<Result<_>>()?;
1017            let target_cols: Vec<String> = (0..targs.len()).map(|i| format!("c{i}")).collect();
1018            let select_list = format!("{}?2", prefix_with_comma(&select_cols));
1019            let insert_cols = format!("{}ts", prefix_with_comma(&target_cols));
1020            let conflict = conflict_clause(info);
1021            let body = format!("SELECT {select_list} FROM {from}");
1022            Ok(format!(
1023                "INSERT INTO {} ({insert_cols}) {} {conflict}",
1024                q(target),
1025                union_branches(&body),
1026            ))
1027        }
1028        Action::Delete {
1029            name: target,
1030            key_args,
1031        } => {
1032            // 0-key deletes: SQL `()` and an empty SELECT are
1033            // invalid. Use `EXISTS (SELECT 1 FROM body)` instead —
1034            // delete every row of the target iff the body matches.
1035            if key_args.is_empty() {
1036                let body = format!("SELECT 1 FROM {from}");
1037                return Ok(format!(
1038                    "DELETE FROM {} WHERE EXISTS ({})",
1039                    q(target),
1040                    union_branches(&body),
1041                ));
1042            }
1043            let key_cols: Vec<String> = (0..key_args.len()).map(|i| format!("c{i}")).collect();
1044            let key_select: Vec<String> = key_args
1045                .iter()
1046                .map(|t| term_sql(t, binding, rule_name))
1047                .collect::<Result<_>>()?;
1048            let body = format!("SELECT {} FROM {from}", key_select.join(", "));
1049            Ok(format!(
1050                "DELETE FROM {} WHERE ({}) IN ({})",
1051                q(target),
1052                key_cols.join(", "),
1053                union_branches(&body),
1054            ))
1055        }
1056        Action::LetCtor { .. } | Action::LetExpr { .. } => {
1057            // Unreachable: presence of any Let switches us to the
1058            // materialized path before we reach this point.
1059            Err(anyhow!(
1060                "rule {rule_name}: internal: Let reached simple-action codegen"
1061            ))
1062        }
1063        Action::Panic { msg } => {
1064            // `error()` is only evaluated on returned rows, so 0 rows
1065            // = no error, ≥1 row = SQL exception.
1066            let safe = msg.replace('\'', "''");
1067            let body = format!("SELECT 1 FROM {from}");
1068            Ok(format!(
1069                "SELECT error('panic: {safe}') FROM ({}) LIMIT 1",
1070                union_branches(&body),
1071            ))
1072        }
1073    }
1074}

1076/// Compile a single action under the *materialized* path: INSERT/
1077/// DELETE FROM the temp table.
1078fn compile_materialized_action(
1079    action: &Action,
1080    mat_binding: &HashMap<String, String>,
1081    temp_table: &str,
1082    rule_name: &str,
1083    functions: &HashMap<String, FunctionInfo>,
1084) -> Result<String> {
1085    let from = q(temp_table);
1086    match action {
1087        Action::Insert {
1088            name: target,
1089            args: targs,
1090        } => {
1091            let info = &functions[target];
1092            let select_cols: Vec<String> = targs
1093                .iter()
1094                .map(|t| term_sql(t, mat_binding, rule_name))
1095                .collect::<Result<_>>()?;
1096            let target_cols: Vec<String> = (0..targs.len()).map(|i| format!("c{i}")).collect();
1097            let select_list = format!("{}?2", prefix_with_comma(&select_cols));
1098            let insert_cols = format!("{}ts", prefix_with_comma(&target_cols));
1099            let conflict = conflict_clause(info);
1100            Ok(format!(
1101                "INSERT INTO {} ({insert_cols}) SELECT {select_list} FROM {from} {conflict}",
1102                q(target),
1103            ))
1104        }
1105        Action::Delete {
1106            name: target,
1107            key_args,
1108        } => {
1109            if key_args.is_empty() {
1110                return Ok(format!(
1111                    "DELETE FROM {} WHERE EXISTS (SELECT 1 FROM {from})",
1112                    q(target)
1113                ));
1114            }
1115            let key_select: Vec<String> = key_args
1116                .iter()
1117                .map(|t| term_sql(t, mat_binding, rule_name))
1118                .collect::<Result<_>>()?;
1119            // Reshape `DELETE WHERE (cols) IN (SELECT … FROM temp)`
1120            // into `DELETE FROM target USING (SELECT …) __d WHERE
1121            // target.cN = __d.dN`. DuckDB's tuple-IN subquery
1122            // sometimes falls back to a full scan; the USING-join
1123            // form gives the optimizer a clear hash-join shape and
1124            // lets it pick the PK index on `target` for lookup.
1125            let proj_cols: Vec<String> = key_select
1126                .iter()
1127                .enumerate()
1128                .map(|(i, expr)| format!("{expr} AS d{i}"))
1129                .collect();
1130            let join_preds: Vec<String> = (0..key_select.len())
1131                .map(|i| format!("{}.c{i} = __d.d{i}", q(target)))
1132                .collect();
1133            Ok(format!(
1134                "DELETE FROM {} USING (SELECT {} FROM {from}) __d WHERE {}",
1135                q(target),
1136                proj_cols.join(", "),
1137                join_preds.join(" AND "),
1138            ))
1139        }
1140        Action::LetExpr { .. } => {
1141            // The expression's value is already in the temp table
1142            // column corresponding to `var`; nothing to insert.
1143            // Return a no-op SELECT so we have *something* in the
1144            // action list (the runner expects a SQL string).
1145            Ok("SELECT 1 WHERE FALSE".to_string())
1146        }
1147        Action::LetCtor { var, name, args } => {
1148            // Insert the (args…, fresh_id, ts) row into the
1149            // constructor's own table so it mirrors the bridge
1150            // backend's `TableAction::lookup` with `FreshId`
1151            // default. The fresh ID is already in the
1152            // materialized temp table's `var` column (via
1153            // `nextval`/hash-cons COALESCE); just project it
1154            // through.
1155            //
1156            // Previously this branch skipped the INSERT whenever
1157            // `@<name>View` existed (relying on the accompanying
1158            // `Insert @<name>View` to emit the canonical row).
1159            // That made the raw `<name>` table go unwritten, which
1160            // broke bridge-compatible callers that walk the
1161            // constructor table by name (`prove_exists`, the
1162            // extractor, etc.). Now that strings are stored as
1163            // interned `Value` handles (BIGINT) — matching the
1164            // bridge — `for_each_while` round-trips correctly, so
1165            // it's safe to populate both tables.
1166            let info = functions
1167                .get(name)
1168                .ok_or_else(|| anyhow!("rule {rule_name}: unknown LetCtor target {name}"))?;
1169            let arg_sqls: Vec<String> = args
1170                .iter()
1171                .map(|t| term_sql(t, mat_binding, rule_name))
1172                .collect::<Result<_>>()?;
1173            // Defensive: caller arity must equal `inputs_len`.
1174            if arg_sqls.len() != info.inputs_len {
1175                return Err(anyhow!(
1176                    "rule {rule_name}: LetCtor {name} arity {} doesn't match inputs_len {}",
1177                    arg_sqls.len(),
1178                    info.inputs_len
1179                ));
1180            }
1181            let target_cols: Vec<String> = (0..info.cols.len()).map(|i| format!("c{i}")).collect();
1182            let select_list = format!("{}{}, ?2", prefix_with_comma(&arg_sqls), q(var));
1183            let conflict = conflict_clause(info);
1184            Ok(format!(
1185                "INSERT INTO {} ({}, ts) SELECT {select_list} FROM {from} {conflict}",
1186                q(name),
1187                target_cols.join(", "),
1188            ))
1189        }
1190        Action::Panic { msg } => {
1191            let safe = msg.replace('\'', "''");
1192            Ok(format!("SELECT error('panic: {safe}') FROM {from} LIMIT 1"))
1193        }
1194    }
1195}

1197/// Compile a `Term` to SQL with an empty variable binding.
1198pub(crate) fn term_sql_no_binding(t: &Term, ctx: &str) -> Result<String> {
1199    let empty = HashMap::new();
1200    term_sql(t, &empty, ctx)
1201}

1203fn term_sql(t: &Term, binding: &HashMap<String, String>, rule_name: &str) -> Result<String> {
1204    match t {
1205        Term::Var(v) => binding
1206            .get(v)
1207            .cloned()
1208            .ok_or_else(|| anyhow!("rule {rule_name}: unbound variable {v}")),
1209        Term::Lit(l) => Ok(lit_sql(l)),
1210        Term::Prim(op, args) => {
1211            let arg_sqls: Vec<String> = args
1212                .iter()
1213                .map(|a| term_sql(a, binding, rule_name))
1214                .collect::<Result<_>>()?;
1215            prim_sql(op, &arg_sqls, rule_name)
1216        }
1217        Term::FuncCall { name, args } => {
1218            let arg_sqls: Vec<String> = args
1219                .iter()
1220                .map(|a| term_sql(a, binding, rule_name))
1221                .collect::<Result<_>>()?;
1222            let out_col = args.len();
1223            let where_clause = if arg_sqls.is_empty() {
1224                String::new()
1225            } else {
1226                let conjs: Vec<String> = arg_sqls
1227                    .iter()
1228                    .enumerate()
1229                    .map(|(i, s)| format!("c{i} = {s}"))
1230                    .collect();
1231                format!(" WHERE {}", conjs.join(" AND "))
1232            };
1233            Ok(format!(
1234                "(SELECT c{out_col} FROM {}{where_clause} LIMIT 1)",
1235                q(name)
1236            ))
1237        }
1238    }
1239}

1241fn prim_sql(op: &str, args: &[String], rule_name: &str) -> Result<String> {
1242    let binop = |sql_op: &str| -> Result<String> {
1243        if args.len() != 2 {
1244            return Err(anyhow!(
1245                "rule {rule_name}: primitive `{op}` expects 2 args, got {}",
1246                args.len()
1247            ));
1248        }
1249        Ok(format!("({} {sql_op} {})", args[0], args[1]))
1250    };
1251    let unop = |sql_op: &str| -> Result<String> {
1252        if args.len() != 1 {
1253            return Err(anyhow!(
1254                "rule {rule_name}: primitive `{op}` expects 1 arg, got {}",
1255                args.len()
1256            ));
1257        }
1258        Ok(format!("({sql_op} {})", args[0]))
1259    };
1260    let func =
1261        |sql_func: &str| -> Result<String> { Ok(format!("{sql_func}({})", args.join(", "))) };
1262    let variadic_join = |sql_op: &str, identity: &str| -> Result<String> {
1263        if args.is_empty() {
1264            return Ok(format!("({identity})"));
1265        }
1266        Ok(format!("({})", args.join(&format!(" {sql_op} "))))
1267    };
1268    match op {
1269        "+" | "-" | "*" | "/" => binop(op),
1270        "int-div" => binop("//"),
1271        // Bitwise integer ops. DuckDB uses & | for AND/OR but `^` is
1272        // exponentiation — for f64 `^` (power) we let it through; for
1273        // i64 `^` (XOR) the frontend rewrites to `i64-xor`.
1274        "&" | "|" => binop(op),
1275        "^" => binop(op),
1276        "i64-xor" => {
1277            if args.len() != 2 {
1278                return Err(anyhow!(
1279                    "rule {rule_name}: primitive `i64-xor` expects 2 args, got {}",
1280                    args.len()
1281                ));
1282            }
1283            Ok(format!("xor({}, {})", args[0], args[1]))
1284        }
1285        "<<" | ">>" => binop(op),
1286        "<" | "<=" | ">" | ">=" => binop(op),
1287        "=" => binop("="),
1288        // i64/bigint comparisons that return a native bool (see
1289        // `src/sort/i64.rs` and `src/sort/bigint.rs`). They map to the
1290        // same SQL comparison operators as the bare forms above.
1291        "bool-<" => binop("<"),
1292        "bool-<=" => binop("<="),
1293        "bool->" => binop(">"),
1294        "bool->=" => binop(">="),
1295        "bool-=" => binop("="),
1296        "!=" | "<>" | "bool-!=" => binop("<>"),
1297        "and" => binop("AND"),
1298        "or" => variadic_join("OR", "FALSE"),
1299        "not" => unop("NOT"),
1300        "ordering-max" | "max" => func("GREATEST"),
1301        "ordering-min" | "min" => func("LEAST"),
1302        "abs" => func("ABS"),
1303        "%" => binop("%"),
1304        "neg" => unop("-"),
1305        "not-i64" => unop("~"),
1306        "to-f64" => {
1307            if args.len() != 1 {
1308                return Err(anyhow!(
1309                    "rule {rule_name}: primitive `to-f64` expects 1 arg, got {}",
1310                    args.len()
1311                ));
1312            }
1313            Ok(format!("CAST({} AS DOUBLE)", args[0]))
1314        }
1315        "to-i64" => {
1316            if args.len() != 1 {
1317                return Err(anyhow!(
1318                    "rule {rule_name}: primitive `to-i64` expects 1 arg, got {}",
1319                    args.len()
1320                ));
1321            }
1322            Ok(format!("CAST({} AS BIGINT)", args[0]))
1323        }
1324        // The frontend (`rename_prim` in `lib.rs`) routes overloads
1325        // of `to-string` to sort-specific names so the SQL emitter
1326        // can pick the right UDF:
1327        //   i64-to-string    → __egglog_i64_to_string
1328        //   f64-to-string    → __egglog_f64_to_string
1329        //   bigint-to-string → __egglog_bigint_to_string
1330        // Strings now live as interned `Value` handles in BIGINT
1331        // columns, so we route to UDFs that take the input, format
1332        // it, and intern the result string. The original
1333        // `CAST(... AS VARCHAR)` path no longer works because
1334        // downstream consumers expect a BIGINT handle, not a raw
1335        // VARCHAR.
1336        "i64-to-string" => {
1337            if args.len() != 1 {
1338                return Err(anyhow!(
1339                    "rule {rule_name}: primitive `i64-to-string` expects 1 arg, got {}",
1340                    args.len()
1341                ));
1342            }
1343            Ok(format!("__egglog_i64_to_string({})", args[0]))
1344        }
1345        "f64-to-string" => {
1346            if args.len() != 1 {
1347                return Err(anyhow!(
1348                    "rule {rule_name}: primitive `f64-to-string` expects 1 arg, got {}",
1349                    args.len()
1350                ));
1351            }
1352            Ok(format!("__egglog_f64_to_string({})", args[0]))
1353        }
1354        "bigint-to-string" => {
1355            if args.len() != 1 {
1356                return Err(anyhow!(
1357                    "rule {rule_name}: primitive `bigint-to-string` expects 1 arg, got {}",
1358                    args.len()
1359                ));
1360            }
1361            Ok(format!("__egglog_bigint_to_string({})", args[0]))
1362        }
1363        // String-concat is variadic on the egglog side; the duck
1364        // UDF mirrors the signature with a variadic BIGINT arg list.
1365        // Empty concat is well-defined: emit a literal for the
1366        // empty string (interned at compile time would be cleaner,
1367        // but the 0-arg form is rare enough that we leave the UDF
1368        // to handle it).
1369        "string-concat" => {
1370            if args.is_empty() {
1371                return Err(anyhow!(
1372                    "rule {rule_name}: empty `string-concat` not supported on duckdb"
1373                ));
1374            }
1375            Ok(format!("__egglog_string_concat({})", args.join(", ")))
1376        }
1377        "replace" => {
1378            if args.len() != 3 {
1379                return Err(anyhow!(
1380                    "rule {rule_name}: primitive `replace` expects 3 args, got {}",
1381                    args.len()
1382                ));
1383            }
1384            Ok(format!(
1385                "__egglog_replace({}, {}, {})",
1386                args[0], args[1], args[2]
1387            ))
1388        }
1389        "count-matches" => {
1390            if args.len() != 2 {
1391                return Err(anyhow!(
1392                    "rule {rule_name}: primitive `count-matches` expects 2 args, got {}",
1393                    args.len()
1394                ));
1395            }
1396            Ok(format!("__egglog_count_matches({}, {})", args[0], args[1]))
1397        }
1398        "guard" => unop(""),
1399        // Pair container primitives. The term encoder uses these in
1400        // proof mode: `uf_function_<sort>` returns a
1401        // `(Pair sort proof)`, and rebuild rules read out the
1402        // leader with `(pair-first p)`. The bridge represents Pair
1403        // as a DuckDB STRUCT with `first` / `second` BIGINT fields.
1404        "pair" => {
1405            if args.len() != 2 {
1406                return Err(anyhow!(
1407                    "rule {rule_name}: primitive `pair` expects 2 args, got {}",
1408                    args.len()
1409                ));
1410            }
1411            // `struct_pack` keeps the field names attached so later
1412            // `.first` / `.second` access on the result is just a
1413            // typed field reference and DuckDB's planner can fold it.
1414            Ok(format!(
1415                "struct_pack(first := {}, second := {})",
1416                args[0], args[1]
1417            ))
1418        }
1419        "pair-first" => {
1420            if args.len() != 1 {
1421                return Err(anyhow!(
1422                    "rule {rule_name}: primitive `pair-first` expects 1 arg, got {}",
1423                    args.len()
1424                ));
1425            }
1426            Ok(format!("({}).first", args[0]))
1427        }
1428        "pair-second" => {
1429            if args.len() != 1 {
1430                return Err(anyhow!(
1431                    "rule {rule_name}: primitive `pair-second` expects 1 arg, got {}",
1432                    args.len()
1433                ));
1434            }
1435            Ok(format!("({}).second", args[0]))
1436        }
1437        // BigInt / BigRat constructors are routed to DuckDB scalar
1438        // UDFs registered by `EGraph::register_builtin_prim_udf` (see
1439        // `lib.rs`). The UDFs call into the shared base-value pool to
1440        // intern the parsed/constructed value and return its `Value`
1441        // rep as a BIGINT handle.
1442        "from-string" => {
1443            if args.len() != 1 {
1444                return Err(anyhow!(
1445                    "rule {rule_name}: primitive `from-string` expects 1 arg, got {}",
1446                    args.len()
1447                ));
1448            }
1449            Ok(format!("__egglog_from_string({})", args[0]))
1450        }
1451        "bigrat" => {
1452            if args.len() != 2 {
1453                return Err(anyhow!(
1454                    "rule {rule_name}: primitive `bigrat` expects 2 args, got {}",
1455                    args.len()
1456                ));
1457            }
1458            Ok(format!("__egglog_bigrat({}, {})", args[0], args[1]))
1459        }
1460        // egglog-experimental's `get-size!` zero-arg form. The
1461        // variadic table-filter form is not yet supported on duckdb;
1462        // `EGraph::refresh_table_sizes` must be called before each
1463        // query that may reference this UDF so it sees a fresh
1464        // snapshot.
1465        "get-size!" => {
1466            if !args.is_empty() {
1467                return Err(anyhow!(
1468                    "rule {rule_name}: variadic `get-size!` arg form is unsupported on duckdb \
1469                     (only the zero-arg total form works)"
1470                ));
1471            }
1472            Ok("__egglog_get_size()".to_string())
1473        }
1474        // BigRat-specific aliases — the frontend renames the
1475        // BigRat-overloaded `+`, `<`, `round`, ... call sites to
1476        // `bigrat-add`, `bigrat-lt`, `bigrat-round`, ... so we route
1477        // them to the per-op UDFs registered by
1478        // `register_builtin_prim_udf` (see `lib.rs`). The duck UDF
1479        // name uses `_` instead of `-` to keep it as a valid SQL
1480        // identifier — same convention as `sanitize_for_udf`.
1481        s if s.starts_with("bigrat-") => {
1482            let udf = format!("__egglog_{}", s.replace('-', "_"));
1483            Ok(format!("{udf}({})", args.join(", ")))
1484        }
1485        _ => Err(anyhow!("rule {rule_name}: unknown primitive `{op}`")),
1486    }
1487}

1489/// Public version of lit_sql for use by the bridge's seed inserts.
1490pub(crate) fn lit_sql_pub(l: &crate::Literal) -> String {
1491    lit_sql(l)
1492}

1494fn lit_sql(l: &crate::Literal) -> String {
1495    match l {
1496        crate::Literal::I64(i) => i.to_string(),
1497        crate::Literal::Bool(b) => if *b { "TRUE" } else { "FALSE" }.to_string(),
1498        crate::Literal::F64(f) => format!("CAST({} AS DOUBLE)", f),
1499        crate::Literal::Str(s) => format!("'{}'", s.replace('\'', "''")),
1500    }
1501}

1503/// Make a string safe to embed in a SQL identifier (temp table name).
1504fn sanitize(s: &str) -> String {
1505    s.chars()
1506        .map(|c| {
1507            if c.is_ascii_alphanumeric() || c == '_' {
1508                c
1509            } else {
1510                '_'
1511            }
1512        })
1513        .collect()
