1//! `impl egglog_backend_trait::RuleBuilderOps for DuckRuleBuilderOps` —
2//! Phase 2 Commit 10.
3//!
4//! This module bridges the trait-level rule-building API
5//! (`RuleBuilderOps`, defined in `egglog-backend-trait`) to the existing
6//! DuckDB-side data IR (`crate::Rule`, `crate::Atom`, `crate::Action`,
7//! `crate::Term`) and the existing `compile_rule` pipeline. There are
8//! **zero changes** to that internal IR — this is strictly a translation
9//! layer.
10//!
11//! ## How it works
12//!
13//! [`DuckRuleBuilderOps`] holds:
14//! - A mutable reference to the duckdb-backend [`crate::EGraph`] for the
15//!   builder's lifetime.
16//! - An in-progress [`crate::Rule`] (`name`, `ruleset = ""`, `body`,
17//!   `actions`) that each `RuleBuilderOps` method appends to.
18//! - A monotonically-increasing counter for fresh variable ids
19//!   (separate from `Variable::id` so that variables synthesized inside
20//!   the builder, e.g. for `lookup`'s output, never collide with
21//!   caller-provided ones).
22//!
23//! On [`RuleBuilderOps::build`], the accumulated rule is handed to the
24//! existing `EGraph::add_rule`. A fresh `RuleId` is allocated as the
25//! index into `EGraph::backend_rule_indices`.
26//!
27//! ## What is supported
28//!
29//! Variable bindings and constants (only for `Value`s that can be
30//! decoded inline; oversize integers and string base values require the
31//! `BaseValuePool` wiring from Commit 11). Function-table body atoms.
32//! Function lookups (`Action::LetExpr` with `Term::FuncCall`). Inserts,
33//! deletes, panics.
34//!
35//! ## What returns errors
36//!
37//! - `subsume`: DuckDB does not support subsumption in v1 (see
38//!   `Backend::supports_subsumption`).
39//! - `query_table` with `is_subsumed = Some(_)`: same.
40//! - `query_prim`: not yet wired through the trait (requires Commit 12's
41//!   primitive registry). Returns an error until then.
42//! - `call_external_func`: same — requires Commit 12. The trait
43//!   signature is infallible, so the error is deferred to `build()`.
44//! - `base_value_constant` (decoding from `Value` to a duck `Literal`):
45//!   requires the `BaseValuePool` wiring from Commit 11. Until then,
46//!   `QueryEntry::Const` with `ColumnTy::Base(_)` triggers a deferred
47//!   error. Pure-variable rules work today.
48//! - `union`: direct union actions are not supported in v1; programs
49//!   that need union go through encoded `(set (pname ...) ())` rules
50//!   emitted by term encoding.
51//!
52//! ## What we don't change in DuckDB's IR
53//!
54//! All translation lives in this file. The accumulator emits the
55//! exact `Atom`/`Action`/`Term` shapes the existing `compile_rule`
56//! pipeline already understands. If a method needs to reach into the
57//! egraph (e.g. to map a `FunctionId` to a duckdb-side table name), it
58//! consults `self.egraph.backend_function_names` directly at builder
59//! time — no new state is introduced.
60

71/// Decode a `Value` of base type `ty` into a duck-side [`Literal`] by
72/// consulting the [`BaseValuePool`]. Supported types: `i64`, `bool`,
73/// `f64`, `String`, `()`, `BigInt`, `BigRat`. Unsupported types yield
74/// an error.
75fn decode_base_const(pool: &DuckdbBaseValuePool, val: Value, ty: BaseValueId) -> Result<Term> {
76    use ordered_float::OrderedFloat;
77    use std::any::TypeId;
78    let pool_dyn: &dyn BaseValuePool = pool;
79    // Try common types. Order chosen for frequency in test programs.
80    if pool_dyn.has_ty(TypeId::of::<i64>()) && ty == pool_dyn.get_ty_by_type_id(TypeId::of::<i64>())
81    {
82        // Store the raw i64 value (not the handle). SQL operations
83        // (`+`, `-`, `~`, `&`, …) on BIGINT columns operate on the
84        // i64 itself; storing handles would break arithmetic
85        // comparison (e.g. `~ 0 = -1` would compare raw `-1` against
86        // the handle `2147483648`). The read path
87        // (`duck_value_to_trait_value`) re-derives the handle by
88        // re-interning when the i64 doesn't fit `try_box`'s 31-bit
89        // fast path.
90        let v = egglog_backend_trait::pool_unwrap::<i64>(pool_dyn, val);
91        return Ok(Term::Lit(Literal::I64(v)));
92    }
93    if pool_dyn.has_ty(TypeId::of::<bool>())
94        && ty == pool_dyn.get_ty_by_type_id(TypeId::of::<bool>())
95    {
96        let v = egglog_backend_trait::pool_unwrap::<bool>(pool_dyn, val);
97        return Ok(Term::Lit(Literal::Bool(v)));
98    }
99    if pool_dyn.has_ty(TypeId::of::<()>()) && ty == pool_dyn.get_ty_by_type_id(TypeId::of::<()>()) {
100        // Unit values: emit a sentinel i64(0) per the duck IR's
101        // existing handling of unit constants.
102        return Ok(Term::Lit(Literal::I64(0)));
103    }
104    type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
105    if pool_dyn.has_ty(TypeId::of::<FBoxed>())
106        && ty == pool_dyn.get_ty_by_type_id(TypeId::of::<FBoxed>())
107    {
108        let v = egglog_backend_trait::pool_unwrap::<FBoxed>(pool_dyn, val);
109        return Ok(Term::Lit(Literal::F64((*v).into_inner())));
110    }
111    if pool_dyn.has_ty(TypeId::of::<egglog_core_relations::Boxed<String>>())
112        && ty == pool_dyn.get_ty_by_type_id(TypeId::of::<egglog_core_relations::Boxed<String>>())
113    {
114        // Match the bridge's uniform encoding: strings live in the
115        // base-value pool and tables store the interned `Value`
116        // handle (i64), not the raw VARCHAR. The frontend already
117        // interned this string when building the QueryEntry::Const
118        // (see `literal_to_query_entry` in egglog/lib.rs), so the
119        // `Value` we got is the handle — just emit it as `I64`.
120        // SQL `=`/`!=` on handles is correct for string equality
121        // (interning is stable, so same string ⟺ same handle).
122        use egglog_numeric_id::NumericId;
123        return Ok(Term::Lit(Literal::I64(val.rep() as i64)));
124    }
125    // BigInt/BigRat etc. fall through to an error path the caller
126    // can route into deferred_err.
127    Err(anyhow!(
128        "DuckRuleBuilderOps: decoding a base-value constant of BaseValueId({}) into a \
129         duck Literal is not yet supported (i64/bool/Unit/f64/String are wired)",
130        ty.rep()
131    ))
132}

138/// In-progress rule built through the trait. Accumulates each
139/// `RuleBuilderOps` call into a fresh [`crate::Rule`], then commits to
140/// the egraph on [`RuleBuilderOps::build`].
141pub(crate) struct DuckRuleBuilderOps<'a> {
142    egraph: &'a mut EGraph,
143    rule: Rule,
144    /// Counter for synthetic variables (allocated by `new_var` /
145    /// `new_var_named` / `lookup` / `call_external_func`). Starts at a
146    /// value above any plausible caller-provided `VariableId` to avoid
147    /// collisions; in practice the caller mints variables through us
148    /// anyway via `new_var`, so the only risk is when `query_table` is
149    /// passed a `Variable` minted on the bridge side. We sidestep this
150    /// risk by deriving the duck::Term::Var name from the
151    /// `VariableId::rep()` directly — see [`Self::var_name`].
152    fresh_counter: u32,
153    /// Errors collected during the builder. The trait's
154    /// `query_table` / `query_prim` / `subsume` return `Result<()>`,
155    /// but `set` / `remove` / `union` / `panic` are infallible. To
156    /// keep DuckDB's "unsupported on duckdb" errors visible at
157    /// `build()` time, we accumulate any deferred error here and
158    /// surface it from `build`.
159    deferred_err: Option<anyhow::Error>,
160    /// Duck-side variable names that have been bound (by a body atom's
161    /// output, an `Atom::Bind`, or an `Action::LetExpr` / `LetCtor`).
162    /// Used by `query_prim` to decide between a `Bind` (fresh var) and
163    /// a `Filter` (assert-eq) translation, mirroring the bridge's
164    /// `inner.grounded` semantics.
165    bound_vars: std::collections::HashSet<String>,
166    /// Inline expansions for variables that were minted by `lookup` /
167    /// `call_external_func` but whose result we never bind through an
168    /// `Action::LetExpr`. Reason: DuckDB's `compile.rs` materializes
169    /// rule actions into a single SELECT, and SQL forbids referencing
170    /// sibling SELECT-list aliases (so `SELECT f(t.c0) AS v, g(v) AS w
171    /// FROM t` would fail). Instead we *inline* the lookup's
172    /// `Term::FuncCall { … }` directly into the args of any later
173    /// action that references it. Each entry maps `VariableId.rep()`
174    /// to the `Term` to substitute. When `entries_to_terms` translates
175    /// a `QueryEntry::Var(v)` whose id is in this map, it returns the
176    /// stored term instead of `Term::Var(name)`. Recursion handles
177    /// chains: a stored `Term::FuncCall` may itself contain
178    /// `Term::Var(other)` references, which the substitution walks
179    /// transitively.
180    inline_terms: std::collections::HashMap<u32, Term>,
181    /// Tracks which `inline_terms` entries have been substituted into
182    /// some other atom or action. Set in `entry_to_term` whenever a
183    /// `QueryEntry::Var(v)` resolves through the inline map. At
184    /// `build()` time, entries NOT in this set correspond to
185    /// primitives whose results were never consumed — typically
186    /// body-level filter primitives like `(!= a b)` whose fresh
187    /// result var is unused. Those get emitted as body `Filter` atoms
188    /// so the predicate actually constrains the rule.
189    consumed_inline: std::cell::RefCell<std::collections::HashSet<u32>>,
190}
191
192impl<'a> DuckRuleBuilderOps<'a> {
193    pub(crate) fn new(egraph: &'a mut EGraph, name: &str, _seminaive: bool) -> Self {
194        // `seminaive` is honored implicitly: DuckDB's `add_rule`
195        // compiles every rule into a seminaive variant per body atom,
196        // so the bridge-side flag has no DuckDB analog. We accept and
197        // ignore it.
198        //
199        // De-collide the rule name with an instance suffix: the
200        // frontend hands us non-unique names (e.g. "eval_actions" for
201        // every top-level command, "check_facts" for every check),
202        // and per-rule bookkeeping (`last_run_at`, `rule_perf_ns`) is
203        // keyed by name. Dispatch goes through `RuleId` → index, so
204        // the suffix only affects internal accounting.
205        let unique_name = format!("{name}#{}", egraph.backend_rule_indices.len());
206        let rule = Rule {
207            name: unique_name,
208            ruleset: String::new(),
209            body: Vec::new(),
210            actions: Vec::new(),
211        };
212        Self {
213            egraph,
214            rule,
215            fresh_counter: 1_000_000_000, // above any expected VariableId
216            deferred_err: None,
217            bound_vars: std::collections::HashSet::new(),
218            inline_terms: std::collections::HashMap::new(),
219            consumed_inline: std::cell::RefCell::new(std::collections::HashSet::new()),
220        }
221    }
222
223    /// Recursively expand any `Term::Var` references through
224    /// `inline_terms`. Used by `entries_to_terms` so that a lookup
225    /// whose result is then used as an arg to another action gets
226    /// inlined as a `Term::FuncCall` rather than left as a dangling
227    /// `Term::Var`. Bounded by the number of stored inlines (each
228    /// lookup adds one); cycles cannot occur because we only insert
229    /// into `inline_terms` with fresh ids.
230    fn inline_term(&self, t: Term) -> Term {
231        match t {
232            Term::Var(_) => t, // names; inlining is keyed on Variable ids in entry_to_term
233            Term::Lit(_) => t,
234            Term::Prim(op, args) => {
235                Term::Prim(op, args.into_iter().map(|a| self.inline_term(a)).collect())
236            }
237            Term::FuncCall {
238                name,
239                args,
240                identity_on_miss,
241            } => Term::FuncCall {
242                name,
243                args: args.into_iter().map(|a| self.inline_term(a)).collect(),
244                identity_on_miss,
245            },
246        }
247    }
248
249    /// Derive a stable, unique duckdb-side variable name from a
250    /// trait-side [`Variable`]. Variables minted by `new_var` /
251    /// `new_var_named` / `lookup` / `call_external_func` already have
252    /// unique ids; variables that the caller minted upstream and is
253    /// passing back to us also carry a unique id from their backend.
254    /// So `v{id}` is enough.
255    fn var_name(v: &Variable) -> String {
256        match &v.name {
257            Some(n) => format!("v{}_{}", v.id.rep(), sanitize(n)),
258            None => format!("v{}", v.id.rep()),
259        }
260    }
261
262    /// Translate a trait-level [`QueryEntry`] to a duck-level [`Term`].
263    ///
264    /// `QueryEntry::Var` → `Term::Var(name)`.
265    /// `QueryEntry::Const` with `ColumnTy::Id` → `Term::Lit(I64(...))`
266    /// using the value's `u32` representation cast to `i64`.
267    /// `QueryEntry::Const` with `ColumnTy::Base(_)` → decode the
268    /// concrete typed primitive via the base value pool and emit the
269    /// corresponding duck `Literal`. Supported types: `i64`, `bool`,
270    /// `f64`, `String`, `()`. Unsupported types yield an error.
271    fn entry_to_term(&self, entry: &QueryEntry) -> Result<Term> {
272        match entry {
273            QueryEntry::Var(v) => {
274                // If this variable was minted by a `lookup` /
275                // `call_external_func` whose result we stored inline,
276                // substitute its `Term::FuncCall` here. Mark it
277                // consumed so `build()` knows this entry was used
278                // (and shouldn't be promoted to a Filter atom).
279                // Otherwise emit a plain `Term::Var`.
280                if let Some(inline) = self.inline_terms.get(&v.id.rep()) {
281                    self.consumed_inline.borrow_mut().insert(v.id.rep());
282                    Ok(self.inline_term(inline.clone()))
283                } else {
284                    Ok(Term::Var(Self::var_name(v)))
285                }
286            }
287            QueryEntry::Const { val, ty } => match ty {
288                ColumnTy::Id => Ok(Term::Lit(Literal::I64(val.rep() as i64))),
289                ColumnTy::Base(bv) => {
290                    decode_base_const(&self.egraph.backend_base_value_pool, *val, *bv)
291                }
292            },
293        }
294    }
295
296    fn entries_to_terms(&self, entries: &[QueryEntry]) -> Result<Vec<Term>> {
297        entries.iter().map(|e| self.entry_to_term(e)).collect()
298    }
299
300    /// Allocate a fresh trait-level [`Variable`] with `name`.
301    fn alloc_var(&mut self, name: Option<&str>) -> Variable {
302        let id = self.fresh_counter;
303        self.fresh_counter = self
304            .fresh_counter
305            .checked_add(1)
306            .expect("DuckRuleBuilderOps: variable counter overflow");
307        Variable {
308            id: VariableId::new(id),
309            name: name.map(|s| s.into()),
310        }
311    }
312
313    fn lookup_function_name(&self, func: FunctionId) -> &str {
314        let idx = func.rep() as usize;
315        self.egraph
316            .backend_function_names
317            .get(idx)
318            .map(|s| s.as_str())
319            .unwrap_or_else(|| {
320                panic!(
321                    "DuckRuleBuilderOps: FunctionId({idx}) not registered via Backend::add_table"
322                )
323            })
324    }
325
326    /// `true` iff `name` is registered as a relation in the duck
327    /// `EGraph` (no output column on the SQL side). The bridge's
328    /// `RuleBuilder` API is relation-unaware — every table atom
329    /// includes a trailing "output" entry — so the duck adapter
330    /// strips that entry when emitting `Atom::Func` /
331    /// `Action::Insert` / `Action::Delete` against a relation.
332    fn is_relation(&self, name: &str) -> bool {
333        self.egraph
334            .functions
335            .get(name)
336            .map(|info| !info.has_output() && !info.eq_sort_ctor)
337            .unwrap_or(false)
338    }
339
340    /// `true` iff `name` is registered as an EqSort constructor. The
341    /// bridge's `RuleBuilder::lookup` semantics is "look up; allocate
342    /// fresh on miss if `DefaultVal::FreshId`". For eq-sort
343    /// constructors that means allocate-on-miss. In the duck IR this
344    /// is exactly what `Action::LetCtor` (with its hash-cons
345    /// LEFT JOIN + `COALESCE(... nextval(seq))`) does. So when
346    /// `lookup` targets an eq-sort constructor we emit a `LetCtor`
347    /// rather than inlining a `Term::FuncCall` (which would compile
348    /// to a pure SELECT, returning NULL on miss — wrong semantics).
349    fn is_eq_sort_ctor(&self, name: &str) -> bool {
350        self.egraph
351            .functions
352            .get(name)
353            .map(|info| info.eq_sort_ctor)
354            .unwrap_or(false)
355    }
356
357    /// `true` iff `name` is a function declared with `DefaultVal::Identity`
358    /// (the canonicalize-at-creation flat UF-index `@UF_Sf`). A `lookup`
359    /// against it must resolve a missing key to the key itself; the inlined
360    /// `Term::FuncCall` carries this so `compile::term_sql` emits a
361    /// `COALESCE((SELECT c1 …), <key>)` instead of a bare SELECT (NULL on miss).
362    fn is_identity_on_miss(&self, name: &str) -> bool {
363        self.egraph
364            .functions
365            .get(name)
366            .map(|info| info.identity_on_miss)
367            .unwrap_or(false)
368    }
369}

371/// Replace any non-alphanumeric character in a variable name with `_`.
372/// Variable display names can contain `@`/`$`/`-`/etc. which would
373/// break duck::Term::Var → SQL identifier substitution.
374fn sanitize(s: &str) -> String {
375    let mut out = String::with_capacity(s.len());
376    for c in s.chars() {
377        if c.is_ascii_alphanumeric() || c == '_' {
378            out.push(c);
379        } else {
380            out.push('_');
381        }
382    }
383    out
384}

390impl<'a> RuleBuilderOps for DuckRuleBuilderOps<'a> {
391    fn new_var(&mut self, _ty: ColumnTy) -> QueryEntry {
392        // ColumnTy is not stored on the duck side — variables carry
393        // their type via context (body atom signatures).
394        let var = self.alloc_var(None);
395        QueryEntry::Var(var)
396    }
397
398    fn new_var_named(&mut self, _ty: ColumnTy, name: &str) -> QueryEntry {
399        let var = self.alloc_var(Some(name));
400        QueryEntry::Var(var)
401    }
402
403    fn query_table(
404        &mut self,
405        func: FunctionId,
406        entries: &[QueryEntry],
407        is_subsumed: Option<bool>,
408    ) -> Result<()> {
409        // DuckDB does not model subsumption: every row is implicitly
410        // non-subsumed. `Some(false)` (filter to non-subsumed rows)
411        // therefore needs no filter — the SELECT already covers it.
412        // `Some(true)` (filter to subsumed rows) cannot be satisfied
413        // by any row and is rejected as an unsupported feature.
414        if is_subsumed == Some(true) {
415            return Err(anyhow!(
416                "DuckRuleBuilderOps::query_table: filtering to subsumed rows is not supported \
417                 on the DuckDB backend (Backend::supports_subsumption returns false)"
418            ));
419        }
420        let name = self.lookup_function_name(func).to_string();
421        let mut args = self.entries_to_terms(entries)?;
422        // Bug 2 fix: for relation atoms the bridge's RuleBuilder
423        // still adds a trailing wildcard "output" arg (the API is
424        // relation-unaware). Strip it so the duck-side
425        // `Atom::Func` matches the relation's actual column count.
426        if self.is_relation(&name) && !args.is_empty() {
427            args.pop();
428        }
429        // Body Func atoms bind every variable used as an argument.
430        for arg in &args {
431            if let Term::Var(v) = arg {
432                self.bound_vars.insert(v.clone());
433            }
434        }
435        self.rule.body.push(Atom::Func { name, args });
436        Ok(())
437    }
438
439    fn query_prim(
440        &mut self,
441        func: ExternalFunctionId,
442        entries: &[QueryEntry],
443        ret_ty: ColumnTy,
444    ) -> Result<()> {
445        // The bridge's `query_prim` semantics: the last entry is the
446        // expected return value. If it's a fresh variable, bind it to
447        // the call's result; otherwise, assert the call's result equals
448        // the entry. We translate accordingly into duck IR.
449        let name = self
450            .egraph
451            .external_func_name(func)
452            .ok_or_else(|| {
453                anyhow!(
454                    "DuckRuleBuilderOps::query_prim: ExternalFunctionId({}) has no associated \
455                     primitive name (was set_external_func_name called?)",
456                    func.rep()
457                )
458            })?
459            .to_string();
460        let translated = entries
461            .iter()
462            .map(|e| self.entry_to_term(e))
463            .collect::<Result<Vec<_>>>()?;
464        // Split into args and expected return value.
465        let mut translated = translated;
466        let expected = translated.pop().ok_or_else(|| {
467            anyhow!("DuckRuleBuilderOps::query_prim: must specify a return value")
468        })?;
469        // Constant-fold builtin primitives (`from-string`, `bigrat`,
470        // `bigrat-*`) when every input arg is a `Literal`. Avoids the
471        // SQL-expression-depth blowup on Herbie's `(bigrat (from-string
472        // …) (from-string …))` constants and ensures shared interning
473        // — same value, same handle, same SQL i64 across every rule
474        // that mentions it. See `EGraph::fold_builtin_prim`.
475        let call = if translated.iter().all(|t| matches!(t, Term::Lit(_)))
476            && let Some(folded) = self.egraph.fold_builtin_prim(&name, &translated)
477        {
478            folded
479        } else {
480            Term::prim(name.clone(), translated)
481        };
482        // Bug 5 fix: ALL `query_prim` results go through
483        // `inline_terms` so a downstream atom or action that uses
484        // the result gets the call inlined into its args. This is
485        // necessary because:
486        //   (a) `compile.rs` materializes a rule's actions into a
487        //       single SELECT, so chained `Atom::Bind { … }` body
488        //       atoms cannot reference each other's vars in SQL
489        //       (sibling SELECT-list aliases aren't visible);
490        //   (b) the bridge's model uniformly threads every prim
491        //       through `query_prim`'s "fresh return var" idiom,
492        //       so a `(guard (or (bool-!= a b)))` rule body comes
493        //       in as three separate `query_prim` calls — each
494        //       returning a fresh var that the next call's args
495        //       reference. Inlining flattens them into one nested
496        //       `Term::Prim` chain that compile.rs handles as a
497        //       single expression.
498        //
499        // For the `guard` primitive specifically — which is the
500        // user-visible "this expression must be true" marker —
501        // additionally push a `Filter(Prim("guard", [operand]))`
502        // atom so the rule actually has a body-level WHERE clause.
503        // Without this the rule would fire unconditionally.
504        match (&expected, entries.last().unwrap()) {
505            (Term::Var(varname), QueryEntry::Var(v)) if !self.bound_vars.contains(varname) => {
506                // Fresh ungrounded var: bind it inline.
507                let _ = v;
508                self.inline_terms.insert(v.id.rep(), call.clone());
509            }
510            _ => {
511                // The expected return is already bound or is a
512                // literal: emit an equality filter `call = expected`.
513                self.rule
514                    .body
515                    .push(Atom::Filter(Term::prim("=", vec![call.clone(), expected])));
516            }
517        }
518        // Standalone `guard` calls turn into a body-level Filter.
519        // The encoder emits these to mark a Boolean predicate
520        // assertion; compile.rs translates `Filter(Prim("guard",
521        // [x]))` to `WHERE (x)` directly.
522        if name == "guard" {
523            // The operand is the (now-inlined) first arg of the
524            // call; rebuild a Filter around it.
525            if let Term::Prim(_, args) = &call
526                && let Some(arg) = args.first()
527            {
528                self.rule
529                    .body
530                    .push(Atom::Filter(Term::prim("guard", vec![arg.clone()])));
531            }
532        }
533        // Primitives whose return type is `()` (Unit) act as body
534        // filters — they succeed if their predicate holds, fail
535        // otherwise (e.g. `(!= x y)`, `(< x y)`, `(value-eq x y)`).
536        // The bridge's `query_prim` idiom threads a fresh return var
537        // through every primitive call, so the inline-the-result path
538        // above captures the value-propagation half of the contract.
539        // But Unit-returning primitives don't propagate anything
540        // meaningful — their semantics is purely the filter — so the
541        // inline alone leaves the body unconstrained. Emit a Filter
542        // atom on top so the SQL WHERE clause actually checks the
543        // predicate. (The bridge backend gets this for free by
544        // halting rule eval when invoke returns `None`; duck needs
545        // it spelled out in SQL.) `guard` already handled above.
546        let unit_return = {
547            use egglog_backend_trait::{BaseValuePool, ColumnTy};
548            use std::any::TypeId;
549            let pool: &dyn BaseValuePool = &self.egraph.backend_base_value_pool;
550            matches!(ret_ty, ColumnTy::Base(bv)
551                if pool.has_ty(TypeId::of::<()>())
552                    && bv == pool.get_ty_by_type_id(TypeId::of::<()>()))
553        };
554        if unit_return && name != "guard" {
555            self.rule.body.push(Atom::Filter(call.clone()));
556        }
557        Ok(())
558    }
559
560    fn call_external_func(
561        &mut self,
562        func: ExternalFunctionId,
563        args: &[QueryEntry],
564        _ret_ty: ColumnTy,
565        // DuckDB compiles primitive calls to SQL and never panics through
566        // this path, so the lazy message is dropped unevaluated.
567        _panic_msg: PanicMsg,
568    ) -> QueryEntry {
569        let name = match self.egraph.external_func_name(func) {
570            Some(n) => n.to_string(),
571            None => {
572                if self.deferred_err.is_none() {
573                    self.deferred_err = Some(anyhow!(
574                        "DuckRuleBuilderOps::call_external_func: ExternalFunctionId({}) has \
575                         no associated primitive name (was set_external_func_name called?)",
576                        func.rep()
577                    ));
578                }
579                let var = self.alloc_var(Some("__ext_no_name"));
580                return QueryEntry::Var(var);
581            }
582        };
583        let translated_args: Vec<Term> = match args
584            .iter()
585            .map(|e| self.entry_to_term(e))
586            .collect::<Result<_>>()
587        {
588            Ok(v) => v,
589            Err(e) => {
590                if self.deferred_err.is_none() {
591                    self.deferred_err = Some(e);
592                }
593                Vec::new()
594            }
595        };
596        // Same constant-fold as `query_prim` — see that comment for
597        // why this is load-bearing on Herbie.
598        let inline_term = if translated_args.iter().all(|t| matches!(t, Term::Lit(_)))
599            && let Some(folded) = self.egraph.fold_builtin_prim(&name, &translated_args)
600        {
601            folded
602        } else {
603            Term::prim(name, translated_args)
604        };
605        let result = self.alloc_var(Some("call_prim"));
606        // Inline the primitive call into `inline_terms` for the same
607        // reason `lookup` does — see `lookup`'s comment. Subsequent
608        // actions get the call substituted into their args directly,
609        // sidestepping the sibling-SELECT-list-alias restriction.
610        self.inline_terms.insert(result.id.rep(), inline_term);
611        QueryEntry::Var(result)
612    }
613
614    fn lookup(
615        &mut self,
616        func: FunctionId,
617        entries: &[QueryEntry],
618        _panic_msg: PanicMsg,
619    ) -> QueryEntry {
620        // The bridge's `lookup` is "look up the function's output for
621        // these inputs; if missing, use the function's `DefaultVal`".
622        // Two cases on the duck side:
623        //
624        // 1. Eq-sort constructor (`DefaultVal::FreshId`). Looking up
625        //    should **allocate-on-miss**, returning the constructor's
626        //    fresh id. The duck IR represents this via
627        //    `Action::LetCtor`, whose hash-cons LEFT JOIN +
628        //    `COALESCE(... nextval('__egglog_eqsort_seq'))` lowering
629        //    in `compile.rs` does exactly that.
630        //
631        // 2. Plain function (with an output column). Looking up
632        //    returns the output's stored value; on miss we get NULL.
633        //    The duck IR for this is a `Term::FuncCall { name,
634        //    args }` *inlined* into the consuming action's args (see
635        //    `inline_terms` below). We can't emit `Action::LetExpr`
636        //    here because compile.rs materializes a rule's actions
637        //    into one SELECT and SQL forbids sibling-alias refs.
638        let name = self.lookup_function_name(func).to_string();
639        let args = match self.entries_to_terms(entries) {
640            Ok(a) => a,
641            Err(e) => {
642                if self.deferred_err.is_none() {
643                    self.deferred_err = Some(e);
644                }
645                Vec::new()
646            }
647        };
648        let var = self.alloc_var(Some("lookup"));
649        if self.is_eq_sort_ctor(&name) {
650            let var_name = Self::var_name(&var);
651            self.bound_vars.insert(var_name.clone());
652            self.rule.actions.push(Action::LetCtor {
653                var: var_name,
654                name,
655                args,
656            });
657        } else {
658            let identity_on_miss = self.is_identity_on_miss(&name);
659            self.inline_terms.insert(
660                var.id.rep(),
661                Term::FuncCall {
662                    name,
663                    args,
664                    identity_on_miss,
665                },
666            );
667        }
668        QueryEntry::Var(var)
669    }
670
671    fn subsume(&mut self, _func: FunctionId, _entries: &[QueryEntry]) -> Result<()> {
672        // DuckDB does not model subsumption (`Backend::supports_subsumption
673        // == false`). Term-encoded rules may still emit `subsume` for
674        // hash-cons / canonicalization bookkeeping; treat such requests
675        // as no-ops on duckdb. Behaviorally this is a known degradation
676        // — subsumed rows continue to match — but it matches the
677        // legacy duckdb pipeline, which also silently drops subsume.
678        // Programs that rely on subsumption for correctness are
679        // upstream-gated by `program_supports_proofs` and never reach
680        // the duckdb backend.
681        Ok(())
682    }
683
684    fn set(&mut self, func: FunctionId, entries: &[QueryEntry]) {
685        let name = self.lookup_function_name(func).to_string();
686        let mut args = match self.entries_to_terms(entries) {
687            Ok(a) => a,
688            Err(e) => {
689                if self.deferred_err.is_none() {
690                    self.deferred_err = Some(e);
691                }
692                return;
693            }
694        };
695        // Bug 3 fix (insert side): strip the trailing "output" entry
696        // for relation inserts. Same reasoning as in `query_table`.
697        if self.is_relation(&name) && !args.is_empty() {
698            args.pop();
699        }
700        self.rule.actions.push(Action::Insert { name, args });
701    }
702
703    fn remove(&mut self, func: FunctionId, entries: &[QueryEntry]) {
704        let name = self.lookup_function_name(func).to_string();
705        let args = match self.entries_to_terms(entries) {
706            Ok(a) => a,
707            Err(e) => {
708                if self.deferred_err.is_none() {
709                    self.deferred_err = Some(e);
710                }
711                return;
712            }
713        };
714        // `remove` already takes only input columns (no output), so
715        // no relation-arity adjustment is needed here.
716        self.rule.actions.push(Action::Delete {
717            name,
718            key_args: args,
719        });
720    }
721
722    fn union(&mut self, _l: QueryEntry, _r: QueryEntry) {
723        // Unions in the existing DuckDB pipeline are emitted by term
724        // encoding as `(set (pname (ordering-max l r) (ordering-min l
725        // r)) ())`. Reaching the same shape from a trait-level
726        // `union(l, r)` call requires knowing the eq-sort's pname,
727        // which is per-sort and not directly addressable through the
728        // FunctionId/Value surface. Most rule actions go through
729        // encoded `set` calls on the pname function directly, so this
730        // path is rarely exercised in practice.
731        //
732        // Defer an error to build() so the rule body can be inspected
733        // / diagnosed.
734        if self.deferred_err.is_none() {
735            self.deferred_err = Some(anyhow!(
736                "DuckRuleBuilderOps::union: direct union actions are not supported on the \
737                 DuckDB backend in v1; programs that need union should go through encoded \
738                 `(set (pname ...) ())` rules emitted by term encoding"
739            ));
740        }
741    }

743    fn panic(&mut self, message: String) {
744        self.rule.actions.push(Action::Panic { msg: message });
745    }

747    fn rename_prim(&mut self, id: ExternalFunctionId, name: String) {
748        self.egraph.set_external_func_name(id, name);

751    fn build(self: Box<Self>) -> Result<RuleId> {
752        // Bring the struct out of the Box so we can move fields.
753        let Self {
754            egraph,
755            mut rule,
756            deferred_err,
757            inline_terms,
758            consumed_inline,
759            ..
760        } = *self;
761
762        if let Some(e) = deferred_err {
763            return Err(e);
764        }
765
766        // Any `inline_terms` entry whose var was never substituted
767        // (consumed) into a subsequent atom or action represents a
768        // primitive call whose result is unused — typically a
769        // body-level filter primitive like `(!= a b)` whose fresh
770        // result var nothing reads. Emit each such entry as a body
771        // `Filter(Prim(...))` so the predicate actually constrains
772        // the rule. Without this the rule would fire unconditionally.
773        let consumed = consumed_inline.into_inner();
774        let mut unused: Vec<(u32, Term)> = inline_terms
775            .into_iter()
776            .filter(|(id, _)| !consumed.contains(id))
777            .collect();
778        unused.sort_by_key(|(id, _)| *id);
779        for (_id, term) in unused {
780            if matches!(term, Term::Prim(_, _) | Term::FuncCall { .. }) {
781                rule.body.push(Atom::Filter(term));
782            }
783        }
784
785        // If every action got silently dropped (e.g. the rule consisted
786        // entirely of subsume calls, which are no-ops on duckdb), skip
787        // the `add_rule` call entirely — `add_rule` insists on at least
788        // one action, but a no-op rule is just an empty rule. Allocate
789        // a "freed"-style slot so the returned RuleId is valid but
790        // running it does nothing.
791        let name = rule.name.clone();
792        if rule.actions.is_empty() {
793            let idx = egraph.backend_rule_indices.len() as u32;
794            // Allocate a slot pointing at no compiled rule. `run_rules`
795            // skips ids with `None` here so an empty-action rule
796            // becomes a true no-op.
797            egraph.backend_rule_indices.push(None);
798            return Ok(RuleId::new(idx));
799        }
800
801        // Register the rule with the existing pipeline. Keep the
802        // original ruleset name — the trait API filters by RuleId,
803        // not by ruleset, so there's no need for a synthetic one.
804        let _ = name;
805        let rule_idx = egraph.rules.len();
806        egraph
807            .add_rule(rule)
808            .map_err(|e| anyhow!("DuckRuleBuilderOps::build: add_rule failed: {e}"))?;
809
810        // `add_rule` can DROP a rule without compiling it (returns
811        // `Ok(())` but doesn't push to `egraph.rules`): under
812        // `--native-uf` the UF-maintenance rulesets and the
813        // `@uf_change_drain_rule*` rules are skipped (see
814        // `EGraph::add_rule`). In that case `rule_idx` (captured
815        // *before* `add_rule`) does NOT name this rule — it names the
816        // slot the *next* compiled rule will occupy. Pointing this
817        // RuleId at `rule_idx` would alias the next real rule, so
818        // running the dropped rule's ruleset would spuriously fire
819        // that real rule (e.g. a user rule firing during a rebuild
820        // pass). Detect the drop by checking whether `rules` grew and
821        // allocate a `None` slot (a true no-op, like the empty-action
822        // case above) instead.
823        let id = egraph.backend_rule_indices.len() as u32;
824        let slot = if egraph.rules.len() > rule_idx {
825            Some(rule_idx)
826        } else {
827            None
828        };
829        egraph.backend_rule_indices.push(slot);
830        Ok(RuleId::new(id))
831    }
832
833    /// Consume the builder, finalize the accumulated body atoms, and
834    /// run them as an existential check query against the egraph.
835    /// Used by the frontend's `(check …)` implementation. Actions
836    /// accumulated through `call_external_func` (the bridge's
837    /// side-channel sentinel pattern) are ignored — the duckdb path
838    /// answers `matches?` directly via SQL.
839    fn build_check(self: Box<Self>) -> Result<bool> {
840        let Self {
841            egraph,
842            mut rule,
843            deferred_err,
844            inline_terms,
845            consumed_inline,
846            ..
847        } = *self;
848
849        if let Some(e) = deferred_err {
850            return Err(e);
851        }
852
853        // Same unused-inline-term-to-Filter dance as `build`: any
854        // inline term whose result var was never consumed represents a
855        // body-level predicate (e.g. `(!= a b)`) that must constrain
856        // the check. Without lifting these into the body the check
857        // would incorrectly answer "matches" on every row.
858        let consumed = consumed_inline.into_inner();
859        let mut unused: Vec<(u32, Term)> = inline_terms
860            .into_iter()
861            .filter(|(id, _)| !consumed.contains(id))
862            .collect();
863        unused.sort_by_key(|(id, _)| *id);
864        for (_id, term) in unused {
865            // Skip the check-facts sentinel call: it has no
866            // semantic content on duckdb (it's the bridge's
867            // side-channel hook, which we ignore here).
868            if let Term::Prim(name, _) = &term
869                && name == "__check_facts_sentinel"
870            {
871                continue;
872            }
873            if matches!(term, Term::Prim(_, _) | Term::FuncCall { .. }) {
874                rule.body.push(Atom::Filter(term));
875            }
876        }
877
878        egraph.body_exists(&rule.body)
879    }
880}

893    /// Build a tiny "copy" rule via the trait surface:
894    /// `(rule ((R x)) ((R x)))`. Then run it. The trait flow must
895    /// not blow up end-to-end: `Backend::new_rule` →
896    /// `RuleBuilderOps::query_table` / `set` / `build` →

899    fn trait_rule_roundtrip_copy_rule_smoke() {
900        let mut backend: Box<dyn Backend> =
901            Box::new(EGraph::new().expect("DuckDB EGraph::new failed"));
902
903        // Register a 1-column relation `R(_)`. Use ColumnTy::Id since
904        // the trait dispatch maps everything to I64 in this commit.
905        let r = backend.add_table(FunctionConfig {
906            schema: vec![ColumnTy::Id],
907            // For our current dispatch: AssertEq + Fail picks the
908            // function-with-output path (the last column treated as
909            // output). That's fine for the smoke test — the rule
910            // accumulates and round-trips through compile_rule.
911            default: DefaultVal::Fail,
912            merge: MergeFn::AssertEq,
913            name: "R_trait_smoke".to_string(),
914            can_subsume: false,
915        });
916
917        // Empty table is fine.
918        assert_eq!(backend.table_size(r), 0);
919
920        // Build the rule via the trait surface. The actual SQL
921        // compilation runs inside `add_rule` on `build()`.
922        let rule_id = {
923            let mut rb = backend.new_rule("copy_rule_smoke", true);
924            let x = TraitVar {
925                id: VariableId::new(0),
926                name: Some("x".into()),
927            };
928            let entries = vec![QueryEntry::Var(x.clone())];
929            rb.query_table(r, &entries, None)
930                .expect("query_table failed");
931            // The function expects [input, output]; for a 1-column
932            // "relation"-looking schema treated as a function, the
933            // body atom is 1 entry (the output column) and `set`
934            // expects an empty key + the output. Match that by
935            // calling set with the same single entry; either it
936            // compiles or `add_rule` errors visibly.
937            rb.set(r, &entries);
938            rb.build().expect("build failed")
939        };
940
941        // Sanity: the rule id maps back to a registered name. Use
942        // the as_any escape hatch to peek at the registry.
943        {
944            let eg = backend.as_any().downcast_ref::<EGraph>().expect("downcast");
945            let stored_idx = eg
946                .backend_rule_indices
947                .get(rule_id.rep() as usize)
948                .and_then(|x| *x);
949            assert!(stored_idx.is_some(), "rule index should be stored");
950            assert_eq!(
951                stored_idx
952                    .and_then(|i| eg.rules.get(i))
953                    .map(|r| &r.name[..]),
954                Some("copy_rule_smoke#0")
955            );
956        }
957
958        // run_rules with no seeded data: the table remains empty,
959        // and run_rules must not error.
960        let report = backend.run_rules(&[rule_id]).expect("run_rules");
961        // Accept any report shape (duckdb returns a minimal default
962        // for now).
963        let _ = report.search_and_apply_time();
964
965        // free_rule: clears the slot, doesn't panic.
966        backend.free_rule(rule_id);
967        let eg = backend.as_any().downcast_ref::<EGraph>().expect("downcast");
968        assert!(
969            eg.backend_rule_indices
970                .get(rule_id.rep() as usize)
971                .map(|x| x.is_none())
972                .unwrap_or(false),
973            "free_rule should clear the registry slot"
974        );
975    }
