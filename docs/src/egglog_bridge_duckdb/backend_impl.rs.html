90/// Walk a VALUE-FOLD [`MergeFn`] tree into a DuckDB SQL scalar expression over the
91/// existing row (`c{out_col}`) and the incoming row (`EXCLUDED.c{out_col}`).
92///
93/// A value-fold body is a pure fold of primitives / literals over `old`/`new`
94/// (e.g. `(from-string (to-string (* new old)))`); the frontend lowers it to a
95/// tree of `Primitive` / `Const` / `Old` / `New` / `OldCol` / `NewCol`. Every
96/// column is a single output, so `Old`/`OldCol(0)` -> `c{out_col}` and
97/// `New`/`NewCol(0)` -> `EXCLUDED.c{out_col}`. Each [`MergeFn::Primitive`] is
98/// translated to its SQL scalar via [`crate::compile::prim_sql`] (the SAME
99/// translation the rule path uses), so anything `prim_sql` knows (arithmetic,
100/// `LEAST`/`GREATEST`, `to-string`/`from-string`, the BigInt/BigRat/string UDFs)
101/// folds directly.
102///
103/// Returns `None` (so the caller falls back to the prior `MergeMode`, never
104/// panicking) for any shape that is NOT a pure value fold: a non-zero
105/// `OldCol`/`NewCol` (a tuple-output reference — single-output DuckDB cannot
106/// express it), a `Function` call (a term build, not a value fold), any
107/// multi-action variant, OR a `Primitive` whose op `prim_sql` cannot translate.
108fn mergefn_to_sql(
109    merge: &MergeFn,
110    table: &str,
111    out_col: usize,
112    funcs: &external_func::DuckdbExternalFuncRegistry,
113) -> Option<String> {
114    match merge {
115        // Single output column: `Old`/`OldCol(0)` is the EXISTING row's value,
116        // `New`/`NewCol(0)` is the INCOMING value. A non-zero column index is a
117        // tuple-output reference DuckDB's single-output `ON CONFLICT` cannot model.
118        //
119        // The existing-column reference MUST be TABLE-QUALIFIED (`"<table>".c{out}`),
120        // not a bare `c{out}`. DuckDB evaluates an `ON CONFLICT DO UPDATE SET` over
121        // the WHOLE inserted chunk — including rows that do NOT conflict, whose
122        // existing-row column is uninitialized — and discards the result for
123        // non-conflicting rows. A bare `c{out}` reads that uninitialized memory
124        // (garbage), which our pool-handle UDFs then treat as a handle and crash on
125        // (out-of-bounds intern-table lookup). Qualifying the column makes DuckDB
126        // yield NULL for the absent existing row instead of garbage, so the UDFs see
127        // a defined value on the discarded rows (built-in SQL ops tolerate the
128        // garbage, which is why only chained pool-handle UDFs hit this).
129        MergeFn::Old | MergeFn::OldCol(0) => Some(format!("{}.c{out_col}", q(table))),
130        MergeFn::New | MergeFn::NewCol(0) => Some(format!("EXCLUDED.c{out_col}")),
131        MergeFn::OldCol(_) | MergeFn::NewCol(_) => None,
132        // A constant is the raw BIGINT handle (the `Value`'s u32 rep widened),
133        // directly comparable to the BIGINT-typed columns.
134        MergeFn::Const(v) => Some((v.rep() as i64).to_string()),
135        MergeFn::Primitive(ext_id, args) => {
136            let op = funcs.name(*ext_id)?;
137            let arg_sqls: Option<Vec<String>> = args
138                .iter()
139                .map(|a| mergefn_to_sql(a, table, out_col, funcs))
140                .collect();
141            // `prim_sql` errors on an op it cannot translate -> fall back to the
142            // prior `MergeMode` (the frontend's `@merge_rule` path) rather than
143            // emitting broken SQL.
144            crate::compile::prim_sql(op, &arg_sqls?, "@merge").ok()
145        }
146        // A `Function` call builds terms (term-build merge, not yet supported on
147        // DuckDB); every other variant is a non-value-fold shape.
148        _ => None,
149    }
150}

152/// Map a *scalar* (single value column) [`MergeFn`] to the DuckDB [`MergeMode`] that resolves one
153/// output column's FD conflict (the `ON CONFLICT` clause). Recognizes:
154///   - `Old` / `AssertEq`         => keep the existing value
155///   - `New`                      => keep the new value
156///   - `UnionId` / `UnionIntoUf`  => lattice-min (the union-find leader). Reachable under
157///     `--native-merge --duckdb` only via the constructor congruence view, whose union is routed
158///     through the in-core UF; mapped like `UnionId` for completeness.
159///   - `Primitive(ordering-min)`  => `Min` (the term encoder's `@UF_<Sort>f` index).
160///   - a VALUE-FOLD `Primitive` tree (`--native-merge`) => `Fold(sql)`, the fold run NATIVELY
161///     in-SQL (see [`mergefn_to_sql`]). Falls back to `Old` if the fold cannot be translated.
162///   - `OldCol(_)`                => keep the existing value
163///   - `NewCol(_)`                => keep the new value
164///   - everything else            => `Old` (complex merges are gated out by
165///     `supports_complex_merge` = false).
166///
167/// `table` is the function's table name and `out_col` is its output column index
168/// (`inputs_len`); together they reference the existing (`"<table>".c{out_col}`) /
169/// incoming (`EXCLUDED.c{out_col}`) value in a `Fold` (see [`mergefn_to_sql`] for why
170/// the existing reference must be table-qualified).
171fn duck_merge_mode_scalar(
172    merge: &MergeFn,
173    table: &str,
174    out_col: usize,
175    funcs: &external_func::DuckdbExternalFuncRegistry,
176) -> MergeMode {
177    match merge {
178        MergeFn::Old | MergeFn::AssertEq => MergeMode::Old,
179        MergeFn::New => MergeMode::New,
180        // `UnionIntoParentTable` is the bridge-only RELATIONAL native-merge merge
181        // (`--native-merge` without `--native-uf`); DuckDB always pairs
182        // `--native-merge` with `--native-uf`, so this never reaches it, but the
183        // surviving value is the min.
184        MergeFn::UnionId | MergeFn::UnionIntoUf(_) | MergeFn::UnionIntoParentTable { .. } => {
185            MergeMode::Min
186        }
187        MergeFn::Primitive(ext_id, _) => match funcs.name(*ext_id) {
188            // `ordering-min` is the term encoder's `@UF_<Sort>f` index: keep the
189            // dedicated `Min` mode (byte-identical UF path), not a `Fold(LEAST(...))`.
190            Some("ordering-min") => MergeMode::Min,
191            // Any other top-level primitive is a VALUE-FOLD custom `:merge`. Walk it
192            // into SQL; if every node translates, run it NATIVELY in-SQL via `Fold`.
193            // Otherwise fall back to `Old` (the frontend keeps the rule encoding
194            // when `supports_value_fold_merge` would have produced this only under
195            // `--native-merge`; an untranslatable fold is rare and `Old` is safe).
196            _ => match mergefn_to_sql(merge, table, out_col, funcs) {
197                Some(sql) => MergeMode::Fold(sql),
198                None => MergeMode::Old,
199            },
200        },
201        MergeFn::OldCol(_) => MergeMode::Old,
202        MergeFn::NewCol(_) => MergeMode::New,
203        MergeFn::Const(_) | MergeFn::Function(_, _) => MergeMode::Old,
204        // Proof-mode native-merge is not wired on DuckDB (it keeps congruence
205        // rule-encoded under `--native-merge` — `supports_native_congruence_merge`
206        // is `false`), so the 2-table `UnionIntoParentTableWithProof`, the bridge's
207        // tuple-output proof merges, and the proof-mode term-build `KeyCol` never
208        // reach it.
209        MergeFn::UnionIntoParentTableWithProof { .. }
210        | MergeFn::UnionIntoUfWithProof { .. }
211        | MergeFn::EclassMinProof { .. }
212        | MergeFn::KeyCol(_) => {
213            panic!("DuckDB backend does not support proof-mode native-merge merges")
214        }
215        MergeFn::Columns(_) => unreachable!("duck_merge_mode_scalar called on a Columns merge"),
216        // `Seq` / `TableInsert` / `Construct` / `IfEq` are the bridge-only
217        // `:merge`-multiple-actions variants (PR #933 + the term-build guard);
218        // DuckDB has no multi-action merge support (term-build `:merge` still
219        // rule-encodes via `supports_term_build_merge` = false), so they never
220        // reach this backend.
221        // `IfEq` / `Seq` are the top-level shapes of a TERM-BUILD custom `:merge`
222        // (`translate_term_build_merge_to_seq`). DuckDB runs these NATIVELY as a
223        // between-statement set-based SQL pass (`emit_term_build_merges`), NOT in
224        // the `ON CONFLICT` clause (which cannot mint constructors). The view is
225        // registered ALL-COLUMNS keyed with `ON CONFLICT DO NOTHING` (so the two
226        // conflicting writes coexist for the self-join to read), and the tree
227        // itself is retained in `FunctionInfo::merge_tree` by `add_table`. The
228        // mode here is therefore `Old` (the view's `ON CONFLICT DO NOTHING`).
229        MergeFn::Seq(_) | MergeFn::IfEq { .. } => MergeMode::Old,
230        // A bare `TableInsert` / `Construct` cannot appear at the TOP of a view
231        // merge (they are always nested inside a `Seq`); reaching here is a
232        // frontend bug.
233        MergeFn::TableInsert(..) | MergeFn::Construct(..) => {
234            panic!(
235                "DuckDB backend: a top-level TableInsert/Construct merge is not a \
236                 term-build tree (expected IfEq/Seq)"
237            )
238        }
239    }
240}

242/// Resolve the [`MergeMode`] for a function table with a single output column. A top-level
243/// `Columns` of length 1 is the multi-value spelling of a single-output merge and collapses to its
244/// one inner column's mode. DuckDB's `ON CONFLICT` model resolves a single output column, so a
245/// genuine multi-value `Columns` (len > 1) — a tuple-output function — cannot be expressed and is
246/// a clean, documented error. (Tuple-output views are a later step; DuckDB never receives them.)
247///
248/// `table` is the function's table name and `out_col` its output column index
249/// (`inputs_len`), threaded so a value-fold merge can reference the existing
250/// (`"<table>".c{out_col}`) / incoming (`EXCLUDED.c{out_col}`) value.
251fn duck_merge_mode(
252    merge: &MergeFn,
253    table: &str,
254    out_col: usize,
255    funcs: &external_func::DuckdbExternalFuncRegistry,
256) -> MergeMode {
257    match merge {
258        MergeFn::Columns(cols) => match cols.as_slice() {
259            [single] => duck_merge_mode_scalar(single, table, out_col, funcs),
260            _ => panic!(
261                "DuckDB backend does not support tuple-output (multi-value) merges \
262                 (a {}-column `Columns` merge)",
263                cols.len()
264            ),
265        },
266        other => duck_merge_mode_scalar(other, table, out_col, funcs),
267    }
268}

270/// True iff `merge` is a TERM-BUILD custom `:merge` tree — the top-level shape
271/// `translate_term_build_merge_to_seq` produces in TERM (non-proof) mode: an
272/// `IfEq` guard around a `Seq`, or a bare `Seq`. A single-element `Columns`
273/// (the multi-value spelling of a single-output merge) is unwrapped. DuckDB runs
274/// such a tree natively as a between-statement set-based pass
275/// (`emit_term_build_merges`); `add_table` retains the tree on the view's
276/// [`FunctionInfo::merge_tree`] and registers the view all-columns keyed.
277///
278/// Returns `false` for value-fold / congruence / scalar merges (which DuckDB
279/// resolves in-SQL via `ON CONFLICT`), so those keep their existing paths.
280/// (The PROOF-mode term-build merge is a `Columns([eclass, proof])` tuple — not
281/// reachable on DuckDB, which rejects proof-mode native-merge.)
282fn is_term_build_tree(merge: &MergeFn) -> bool {
283    match merge {
284        MergeFn::IfEq { .. } | MergeFn::Seq(_) => true,
285        MergeFn::Columns(cols) => matches!(cols.as_slice(), [single] if is_term_build_tree(single)),
286        _ => false,
287    }
288}

339/// Translate a [`ColumnTy`] from the backend trait crate to the
340/// duckdb-bridge's own [`crate::ColumnTy`].
341///
342/// Currently every column maps to `I64`. The trait surface uses a
343/// single `Value` (`u32`) for all column kinds, and DuckDB stores
344/// every `Value` as the underlying integer bits in a `BIGINT`
345/// column. This is lossy with respect to the *display* of values
346/// (e.g. a `f64` column rendered as `BIGINT` shows raw bits in a
347/// `SELECT *`), but it is fully round-trippable through the
348/// trait — `value_to_literal` / `duck_value_to_trait_value` agree
349/// on this encoding.
350///
351/// A more precise mapping (Bool / F64 / Str columns) would require
352/// reaching into the egglog-frontend's typed `Sort` system rather
353/// than `BaseValuePool` alone: the pool stores typed intern tables
354/// keyed by Rust `TypeId`, but the frontend's actual primitive types
355/// (`Boxed<OrderedFloat<f64>>`, `Arc<str>`, custom `BaseValue` impls)
356/// rarely match the natural DuckDB column kinds 1:1. Since this code
357/// path is dead until Commit 14 routes a `Box<dyn Backend>` through
358/// `add_table`, the lossy mapping is fine.
359///
360/// **TODO (Commit 14):** revisit when a real `Box<dyn Backend>`
361/// caller exercises `add_table` and we see what column kinds the
362/// frontend actually asks for.
363fn trait_col_ty_to_duck(ty: ColumnTy, pool: &dyn BaseValuePool) -> DuckColumnTy {
364    use crate::DuckPairMarker;
365    use ordered_float::OrderedFloat;
366    use std::any::TypeId;
367    match ty {
368        ColumnTy::Id => DuckColumnTy::I64,
369        ColumnTy::Base(bv) => {
370            // Probe the pool's registered TypeIds in the same order
371            // as `decode_base_const` in rule_builder.rs. Each branch
372            // checks whether `bv` is the registered id for that
373            // concrete `BaseValue` type and, if so, returns the
374            // matching DuckDB column kind. Falls back to `I64` for
375            // unknown / unregistered base types (which is also what
376            // egglog-bridge does for them — see `BaseValue::sql_ty`).
377            if pool.has_ty(TypeId::of::<DuckPairMarker>())
378                && bv == pool.get_ty_by_type_id(TypeId::of::<DuckPairMarker>())
379            {
380                // Pair marker — see `EGraph::pair_column_ty`.
381                return DuckColumnTy::PairI64;
382            }
383            if pool.has_ty(TypeId::of::<i64>()) && bv == pool.get_ty_by_type_id(TypeId::of::<i64>())
384            {
385                return DuckColumnTy::I64;
386            }
387            if pool.has_ty(TypeId::of::<bool>())
388                && bv == pool.get_ty_by_type_id(TypeId::of::<bool>())
389            {
390                return DuckColumnTy::Bool;
391            }
392            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
393            if pool.has_ty(TypeId::of::<FBoxed>())
394                && bv == pool.get_ty_by_type_id(TypeId::of::<FBoxed>())
395            {
396                return DuckColumnTy::F64;
397            }
398            // Strings used to map to `DuckColumnTy::Str` (VARCHAR),
399            // but that broke bridge-API parity: the trait surface
400            // (`for_each_while`, etc.) returns `Value` handles, and
401            // `duck_value_to_trait_value` had no way to round-trip
402            // a VARCHAR back to a `Value` without re-interning
403            // against the pool. The bridge stores strings as
404            // interned handles in BIGINT columns; we now match.
405            //
406            // SQL semantics: equality / inequality on the BIGINT
407            // handle is equivalent to equality on the original
408            // string because interning is stable. Ordering ops on
409            // handles aren't meaningful, but egglog doesn't expose
410            // `<` / `>` on strings either, so that's a non-issue.
411            // String-producing primitives (`to-string`, `string-
412            // concat`, `replace`) need UDFs that intern their
413            // output — see compile.rs's prim_sql.
414            //
415            // Unit and any other registered base types: store as
416            // i64. This includes BigInt / BigRat / Rational64
417            // which the bridge interns; their `Value` is an
418            // intern-table index that fits in i64.
419            DuckColumnTy::I64
420        }
421    }
422}

424/// Decode an egglog [`Value`] into a duckdb-side [`Literal`] for
425/// emission into a `BIGINT`-style SQL column.
426///
427/// Pairs with [`trait_col_ty_to_duck`]: that helper maps every trait
428/// column kind to `DuckColumnTy::I64`, so the corresponding `Literal`
429/// is `Literal::I64(value.rep() as i64)`. The `pool` parameter is
430/// reserved for the future refined mapping (see the TODO on
431/// `trait_col_ty_to_duck`).
432/// Decode a `Value` for a duck-side column of [`DuckColumnTy`] into
433/// the appropriate [`Literal`]. Used by `insert_row_inner` so a
434/// row whose column was registered as `STR`/`BOOL`/`F64` gets the
435/// matching SQL literal (not `Literal::I64(value.rep() as i64)`,
436/// which would produce a SQL type-conversion error).
437fn duck_value_to_literal(val: Value, col: DuckColumnTy, pool: &DuckdbBaseValuePool) -> Literal {
438    use ordered_float::OrderedFloat;
439    use std::any::TypeId;
440    let pool_dyn: &dyn BaseValuePool = pool;
441    match col {
442        DuckColumnTy::I64 => {
443            // Could be an eq-sort id, a primitive i64 value, or a
444            // handle to a non-i64 intern table (BigInt, Boxed<String>,
445            // etc. — all of which use the BIGINT representation on
446            // the duck side). For i64 base values, unwrap to the raw
447            // i64 so SQL arithmetic on the column operates on the
448            // value, not the handle. The read path
449            // (`duck_value_to_trait_value`) re-interns the raw value
450            // back to a handle so callers that expect `Value`s
451            // (the trait API contract) get a usable rep.
452            if pool_dyn.has_ty(TypeId::of::<i64>()) && val.rep() & (1 << 31) != 0 {
453                let i = egglog_backend_trait::pool_unwrap::<i64>(pool_dyn, val);
454                return Literal::I64(i);
455            }
456            Literal::I64(val.rep() as i64)
457        }
458        DuckColumnTy::Bool => {
459            if pool_dyn.has_ty(TypeId::of::<bool>()) {
460                let b = egglog_backend_trait::pool_unwrap::<bool>(pool_dyn, val);
461                return Literal::Bool(b);
462            }
463            Literal::Bool(val.rep() != 0)
464        }
465        DuckColumnTy::F64 => {
466            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
467            if pool_dyn.has_ty(TypeId::of::<FBoxed>()) {
468                let v = egglog_backend_trait::pool_unwrap::<FBoxed>(pool_dyn, val);
469                return Literal::F64((*v).into_inner());
470            }
471            Literal::F64(val.rep() as f64)
472        }
473        DuckColumnTy::Str => {
474            type SBoxed = egglog_core_relations::Boxed<String>;
475            if pool_dyn.has_ty(TypeId::of::<SBoxed>()) {
476                let v = egglog_backend_trait::pool_unwrap::<SBoxed>(pool_dyn, val);
477                return Literal::Str((*v).clone());
478            }
479            // Fallback: stringify the bits. Almost certainly wrong
480            // but at least doesn't trigger a SQL type cast error.
481            Literal::Str(format!("v{}", val.rep()))
482        }
483        DuckColumnTy::PairI64 => {
484            // Pair columns are not emitted via trait `add_table`
485            // (only via duck-internal registration), so this branch
486            // shouldn't be reachable from `insert_row_inner`.
487            panic!("duck_value_to_literal: PairI64 unexpected in trait-driven insert path")
488        }
489    }
490}

492#[allow(dead_code)]
493fn value_to_literal(val: Value, ty: ColumnTy, pool: &dyn BaseValuePool) -> Literal {
494    use ordered_float::OrderedFloat;
495    use std::any::TypeId;
496    match ty {
497        ColumnTy::Id => Literal::I64(val.rep() as i64),
498        ColumnTy::Base(bv) => {
499            if pool.has_ty(TypeId::of::<i64>()) && bv == pool.get_ty_by_type_id(TypeId::of::<i64>())
500            {
501                let v = egglog_backend_trait::pool_unwrap::<i64>(pool, val);
502                return Literal::I64(v);
503            }
504            if pool.has_ty(TypeId::of::<bool>())
505                && bv == pool.get_ty_by_type_id(TypeId::of::<bool>())
506            {
507                let v = egglog_backend_trait::pool_unwrap::<bool>(pool, val);
508                return Literal::Bool(v);
509            }
510            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
511            if pool.has_ty(TypeId::of::<FBoxed>())
512                && bv == pool.get_ty_by_type_id(TypeId::of::<FBoxed>())
513            {
514                let v = egglog_backend_trait::pool_unwrap::<FBoxed>(pool, val);
515                return Literal::F64((*v).into_inner());
516            }
517            type SBoxed = egglog_core_relations::Boxed<String>;
518            if pool.has_ty(TypeId::of::<SBoxed>())
519                && bv == pool.get_ty_by_type_id(TypeId::of::<SBoxed>())
520            {
521                let v = egglog_backend_trait::pool_unwrap::<SBoxed>(pool, val);
522                return Literal::Str((*v).clone());
523            }
524            // Fallback: treat as i64 (intern-table index).
525            Literal::I64(val.rep() as i64)
526        }
527    }

530/// Convert a DuckDB row value into a [`Value`] for surfacing through
531/// [`FunctionRow`].
532///
533/// The current DuckDB pipeline stores all egglog values as `BIGINT`
534/// (eq-sort ids) or other concrete SQL types. The Backend trait surface
535/// uses a single `Value` (a `u32` newtype) for every column. The
536/// mapping is:
537///
538/// - `BIGINT` -> low 32 bits cast to `Value`. For the eq-sort id case
539///   this is the canonical mapping; for primitive `BIGINT`s storing
540///   `i64` literals it loses the high bits, but only the trait-side
541///   pool wired in Commit 11 will know how to interpret the bits, and
542///   `for_each` callers reaching this path will be added in Commit
543///   14+. For now correctness is moot since no caller exercises it.
544/// Convert a DuckDB `ValueRef` to a trait-level `Value` handle.
545///
546/// For BIGINT columns this gets tricky because i64 base values
547/// are stored RAW in DuckDB (so SQL arithmetic works) but the trait
548/// API expects `Value` handles. The bridge unifies these: a "small"
549/// i64 (top bit clear, fits in u32) IS its own handle; an "interned"
550/// i64 (out of the 31-bit fast path — negative or > 2^31) has a
551/// handle of `intern_table_idx + VAL_OFFSET` (top bit set).
552///
553/// So when reading a BIGINT value `i`:
554///   - `i >= 0 && i < 2^31`: the raw i64 fits the fast-path, so
555///     `Value(i as u32)` is already a valid handle. Same for eq-sort
556///     ids (which always fit), `Boxed<String>` handles (intern
557///     indices fit easily in practice), and `Boxed<BigInt>` /
558///     `Boxed<BigRat>` handles (same).
559///   - `i < 0 || i >= 2^31`: must be a raw i64 base value that was
560///     stored without going through the intern table at insertion
561///     time. Intern it now to produce a stable handle for the trait
562///     consumer; pool_unwrap on the resulting handle round-trips.
563///
564/// The pool is `&dyn` to keep this callable from contexts that only
565/// have the abstract trait; in for_each_while we already need the
566/// pool to register an i64 type if it isn't yet (rare; defensively).
567fn duck_value_to_trait_value(v: ValueRef<'_>, pool: &dyn BaseValuePool) -> Value {
568    use duckdb::types::ValueRef as V;
569    use std::any::TypeId;
570
571    let raw_i64: Option<i64> = match v {
572        V::Null => return Value::new(u32::MAX),
573        V::Boolean(b) => return Value::new(b as u32),
574        V::TinyInt(i) => Some(i as i64),
575        V::SmallInt(i) => Some(i as i64),
576        V::Int(i) => Some(i as i64),
577        V::BigInt(i) => Some(i),
578        V::HugeInt(i) => Some(i as i64),
579        V::UTinyInt(i) => return Value::new(i as u32),
580        V::USmallInt(i) => return Value::new(i as u32),
581        V::UInt(i) => return Value::new(i),
582        V::UBigInt(i) => return Value::new(i as u32),
583        // DOUBLE column: f64 base values are stored RAW (so SQL
584        // arithmetic works). Intern via the pool to lift to a
585        // handle the trait consumer can use with `pool_unwrap::<F>`.
586        V::Float(f) => {
587            use ordered_float::OrderedFloat;
588            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
589            if pool.has_ty(TypeId::of::<FBoxed>()) {
590                let ty = pool.get_ty_by_type_id(TypeId::of::<FBoxed>());
591                return pool.intern_dyn(ty, Box::new(FBoxed::new(OrderedFloat(f as f64))));
592            }
593            return Value::new(u32::MAX);
594        }
595        V::Double(f) => {
596            use ordered_float::OrderedFloat;
597            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
598            if pool.has_ty(TypeId::of::<FBoxed>()) {
599                let ty = pool.get_ty_by_type_id(TypeId::of::<FBoxed>());
600                return pool.intern_dyn(ty, Box::new(FBoxed::new(OrderedFloat(f))));
601            }
602            return Value::new(u32::MAX);
603        }
604        // Other types: unreached for the BIGINT-heavy trait paths,
605        // but emit u32::MAX so a panic downstream is more obvious
606        // than a silently-wrong value.
607        _ => return Value::new(u32::MAX),
608    };
609    let Some(i) = raw_i64 else {
610        return Value::new(u32::MAX);
611    };
612
613    // The fast-path range covers all eq-sort ids, MAY_UNBOX-compatible
614    // i64 bases, and intern table indices for any reasonable program.
615    // Values outside it (large positive or any negative) can only be
616    // raw i64 base values: intern via the pool to lift to a handle.
617    if (0..(1i64 << 31)).contains(&i) {
618        return Value::new(i as u32);
619    }
620    if pool.has_ty(TypeId::of::<i64>()) {
621        let ty = pool.get_ty_by_type_id(TypeId::of::<i64>());
622        return pool.intern_dyn(ty, Box::new(i));
623    }
624    // No i64 type registered — can't intern. Fall back to the raw
625    // bits; callers that try to `pool_unwrap::<i64>` will panic
626    // cleanly with an unknown-table message.
627    Value::new(i as u32)
628}

630impl EGraph {
631    /// Look up the duckdb-bridge table name for a trait [`FunctionId`].
632    fn name_for_function_id(&self, id: FunctionId) -> &str {
633        let idx = id.rep() as usize;
634        self.backend_function_names
635            .get(idx)
636            .map(|s| s.as_str())
637            .unwrap_or_else(|| panic!("FunctionId({idx}) is not registered"))
638    }
639
640    /// Insert a single row's values into the function's table. The
641    /// row is decoded against the function's stored schema (duck-side
642    /// `ColumnTy` per column) and emitted via the existing
643    /// `EGraph::insert` seed path.
644    ///
645    /// Encoding strategy:
646    /// - Bool/I64/F64/Str duck columns → encode via the corresponding
647    ///   `Literal` variant. For `Bool`/`F64`/`Str` we need the value
648    ///   as the corresponding Rust type, which means using the base
649    ///   value pool for non-trivial unboxing. For `I64` we just cast
650    ///   `Value::rep() as i64`.
651    /// - `PairI64` columns: not supported through the trait yet;
652    ///   panics.
653    ///
654    /// Panics on schema mismatches; this method is called by
655    /// `add_values` / `insert_rows` / `add_term`, all of which are
656    /// unreachable until Commit 14 flips the test harness.
657    pub(crate) fn insert_row_inner(&mut self, func: FunctionId, row: &[Value]) {
658        let name = self.name_for_function_id(func).to_string();
659        let info = self
660            .functions
661            .get(&name)
662            .unwrap_or_else(|| panic!("insert_row_inner: function `{name}` not registered"))
663            .clone();
664        if row.len() != info.arity() {
665            panic!(
666                "insert_row_inner: arity mismatch for `{name}`: got {}, expected {}",
667                row.len(),
668                info.arity()
669            );
670        }
671
672        // Decode each value through the schema's `DuckColumnTy` so
673        // String/Bool/F64 columns receive the right `Literal` kind.
674        // We don't have the trait `ColumnTy` here (the duck-side
675        // `FunctionInfo.cols` already lost it), so re-derive the
676        // duck literal kind from `info.cols[i]`.
677        let lits: Vec<Literal> = row
678            .iter()
679            .enumerate()
680            .map(|(i, v)| {
681                let col_ty = info.cols[i];
682                duck_value_to_literal(*v, col_ty, &self.backend_base_value_pool)
683            })
684            .collect();
685
686        self.insert(&name, &lits)
687            .unwrap_or_else(|e| panic!("insert_row_inner: insert(`{name}`) failed: {e}"));
688    }
689}

698    fn add_table(&mut self, config: FunctionConfig) -> FunctionId {
699        // Allocate a numeric id and register the name. The duckdb
700        // backend itself uses string names; the trait callers use
701        // numeric ids, so we maintain a Vec<String> mapping.
702        let idx = self.backend_function_names.len() as u32;
703        let id = FunctionId::new(idx);
704
705        // Split the schema into input columns + output column (if any).
706        // For relations (no output) the entire schema is "inputs".
707        // For eq-sort constructors and functions, the last entry is the
708        // output column.
709        //
710        // Heuristic for dispatch (per the prompt):
711        //   - DefaultVal::FreshId => add_eq_sort_constructor (constructor)
712        //   - DefaultVal::Fail / Const + MergeFn::AssertEq / UnionId
713        //     => add_function with a merge mode derived from MergeFn
714        //   - If the schema has no output column treat as relation:
715        //     add_relation_with_pname.
716        //
717        // The dispatch is intentionally simple: this code path is dead
718        // until Commit 14, so the exact semantics of less-common
719        // combinations aren't load-bearing yet.
720
721        let name = config.name.clone();
722        let schema = &config.schema;
723
724        // Convert all schema columns. The trait's `FunctionConfig`
725        // always includes the output (if any) as the last entry of
726        // `schema`; the duckdb-bridge `add_function` and
727        // `add_eq_sort_constructor` take inputs + output separately
728        // while `add_relation_with_pname` takes the full input list.
729        let duck_cols: Vec<DuckColumnTy> = {
730            // Borrow the pool through the concrete field to dodge a
731            // re-borrow conflict with `self`: we need `&self.backend_base_value_pool`
732            // for the column-type translation but `&mut self` later
733            // for `add_function`/`add_relation_with_pname`. Translating
734            // up front gives us a fully-owned `Vec<DuckColumnTy>` to
735            // hand the mutating methods.
736            let pool: &dyn BaseValuePool = &self.backend_base_value_pool;
737            schema
738                .iter()
739                .copied()
740                .map(|t| trait_col_ty_to_duck(t, pool))
741                .collect()
742        };
743
744        let result = match config.default {
745            DefaultVal::FreshId => {
746                // EqSort constructor: schema = [inputs..., Id]. The
747                // duckdb-bridge appends the ID column itself, so we pass
748                // only the inputs (= schema without the trailing id).
749                let inputs = if duck_cols.is_empty() {
750                    &[][..]
751                } else {
752                    &duck_cols[..duck_cols.len() - 1]
753                };
754                self.add_eq_sort_constructor(&name, inputs, None)
755            }
756            // `DefaultVal::Identity` (canonicalize-at-creation): the flat
757            // UF-index `@UF_Sf` — a single-key key->output function whose lookup
758            // must return the key itself on miss (`find_UFold(x)=x` for an id
759            // with no recorded leader). Register it as an ordinary function
760            // (so inserts/merges of recorded leaders behave normally), then set
761            // `identity_on_miss` so `compile::term_sql` wraps reads in
762            // `COALESCE((SELECT c1 …), <key>)`. Validated as exactly a 2-column
763            // (single key -> output) function.
764            DefaultVal::Identity => {
765                assert_eq!(
766                    duck_cols.len(),
767                    2,
768                    "DefaultVal::Identity (`{name}`) expects a single key column \
769                     (2-column key->output function), got {} columns",
770                    duck_cols.len()
771                );
772                // Map the merge mode the same way the Fail/Const arm does; the
773                // term encoder declares `@UF_Sf` with `:merge (ordering-min …)`
774                // (a `Primitive` referencing `ordering-min`) => `MergeMode::Min`.
775                let merge_mode = duck_merge_mode(
776                    &config.merge,
777                    &name,
778                    duck_cols.len() - 1,
779                    &self.backend_external_funcs,
780                );
781                let inputs = &duck_cols[..duck_cols.len() - 1];
782                let output = duck_cols[duck_cols.len() - 1];
783                let r = self.add_function(&name, inputs, output, merge_mode);
784                if r.is_ok()
785                    && let Some(info) = self.functions.get_mut(&name)
786                {
787                    info.identity_on_miss = true;
788                }
789                r
790            }
791            DefaultVal::Fail | DefaultVal::Const(_) => {
792                // Either a function with an explicit output, or a
793                // relation if the output is Unit (term encoding's
794                // pattern for view tables — `(function @XView (...)
795                // Unit :merge old)`). The parallel pipeline's
796                // `add_function` in backend_duckdb.rs detected this
797                // by string-matching `output_sort == "Unit"`; on the
798                // trait side we don't have the sort name, only the
799                // `BaseValueId`. Use the pool to compare against the
800                // Unit type id.
801                // The term encoder emits `:merge (ordering-min old new)` for the
802                // function-form union-find table (`@UF_<Sort>f`), lowered to a
803                // `Primitive` referencing the `ordering-min` external => `Min`.
804                // Without recognizing it the duckdb conflict clause would fall back
805                // to `DO NOTHING`, silently dropping UFf upserts. The output column
806                // index is `duck_cols.len() - 1`; `saturating_sub` guards the
807                // 0-column (nullary relation) case below, where the mode is unused.
808                let merge_mode = Some(duck_merge_mode(
809                    &config.merge,
810                    &name,
811                    duck_cols.len().saturating_sub(1),
812                    &self.backend_external_funcs,
813                ));
814
815                // Detect Unit output: schema's last entry is
816                // `ColumnTy::Base(id)` where `id` matches the pool's
817                // registration for the `()` type.
818                let output_is_unit = schema.last().is_some_and(|t| match t {
819                    ColumnTy::Base(bv) => {
820                        let pool: &dyn BaseValuePool = &self.backend_base_value_pool;
821                        pool.has_ty(TypeId::of::<()>())
822                            && *bv == pool.get_ty_by_type_id(TypeId::of::<()>())
823                    }
824                    _ => false,
825                });
826
827                // TERM-BUILD custom `:merge` view (`--native-merge`): the merge
828                // body builds e-nodes (a top-level `IfEq`/`Seq` of
829                // `Construct`/`TableInsert`). Register the view ALL-COLUMNS keyed
830                // (conflicting writes coexist) and RETAIN the tree so
831                // `emit_term_build_merges` can compile + run it at the iteration
832                // boundary. Checked before `output_is_unit` (a term-build output
833                // is an eq-sort id, never Unit) and before the scalar
834                // `merge_mode` arm (a term-build tree is not a scalar `ON CONFLICT`
835                // merge). Requires a real output column (the eclass).
836                if !duck_cols.is_empty() && !output_is_unit && is_term_build_tree(&config.merge) {
837                    let inputs = &duck_cols[..duck_cols.len() - 1];
838                    let output = duck_cols[duck_cols.len() - 1];
839                    // Unwrap a single-element `Columns` to its inner tree (the
840                    // multi-value spelling of a single-output merge).
841                    let tree = match &config.merge {
842                        MergeFn::Columns(cols) if cols.len() == 1 => cols[0].clone(),
843                        other => other.clone(),
844                    };
845                    self.add_term_build_view_function(&name, inputs, output, tree)
846                } else if duck_cols.is_empty() {
847                    // No columns at all: treat as a nullary relation.
848                    self.add_relation_with_pname(&name, &[], None)
849                } else if output_is_unit {
850                    // Function with Unit output: matches term
851                    // encoding's relation pattern (e.g.
852                    // `(function @UF_@pathSort (Math Math) Unit
853                    // :merge old)`). The parallel pipeline drops the
854                    // Unit output and registers the function as a
855                    // relation whose key spans the INPUTS only — so
856                    // the body atoms reference inputs without a
857                    // trailing wildcard. Strip the trailing Unit
858                    // column to match.
859                    let inputs = &duck_cols[..duck_cols.len() - 1];
860                    self.add_relation_with_pname(&name, inputs, None)
861                } else if let Some(mode) = merge_mode {
862                    let inputs = &duck_cols[..duck_cols.len() - 1];
863                    let output = duck_cols[duck_cols.len() - 1];
864                    self.add_function(&name, inputs, output, mode)
865                } else {
866                    self.add_relation_with_pname(&name, &duck_cols, None)
867                }
868            }
869        };
870
871        if let Err(e) = result {
872            // The trait method does not return Result; if the duckdb
873            // backend rejects the table registration, we have to
874            // panic. Until Commit 14 flips the test harness this
875            // panic is unreachable.
876            panic!("DuckDB add_table({}): {}", name, e);
877        }
878
879        self.backend_function_names.push(name);
880        id
881    }

883    /// `--native-merge` CONSTRUCTOR-CONGRUENCE: associate an `@<C>View` constructor
884    /// view (`view_func`) with the union-find function (`uf_func`, the `@UF_Sf`) that
885    /// owns its eclass column. The frontend calls this (via `register_native_merge_views`)
886    /// once per native-merge view before the first run.
887    ///
888    /// NOTE: term-mode native-merge on DuckDB is now REJECTED at the CLI (its
889    /// congruence host-pass required the removed native-UF machinery), so this
890    /// registration is only exercised by the PROOF-mode native-merge path, which
891    /// resolves congruence via `emit_native_congruence_proof` /
892    /// [`FunctionInfo::native_congruence_proof`]. `native_congruence_uf` (the
893    /// term-mode association) is retained but inert on the fast-relational default.
894    fn register_native_merge_view(&mut self, uf_func: FunctionId, view_func: FunctionId) {
895        let uf_table = self.name_for_function_id(uf_func).to_string();
896        let view_name = self.name_for_function_id(view_func).to_string();
897        if let Some(info) = self.functions.get_mut(&view_name) {
898            info.native_congruence_uf = Some(uf_table);
899        }

902    /// `--proofs --native-merge` CONSTRUCTOR-CONGRUENCE: associate a constructor view
903    /// (`view_func`, the all-columns COEXIST shape `(children..., eclass) -> Proof`)
904    /// with its proof-congruence config (relational proof-UF `@UF_S` =
905    /// `parent_table`, the `@<F>ViewProof` side-table, and the `Trans`/`Sym` proof
906    /// constructors). [`EGraph::emit_native_congruence_proof`] then resolves each FD
907    /// conflict on the view via a self-join: it reads both colliding rows' proofs
908    /// from the side-table, MINTS `Trans(larger_pf, Sym(smaller_pf))` in SQL, and
909    /// INSERTs the proof-carrying union edge `(larger, smaller, edge_proof)` into
910    /// `@UF_S` — exactly the row the dropped `@congruence_rule*` wrote via
911    /// `union(new, old, Trans(prf1, Sym(prf2)))`. The relational singleparent /
912    /// path_compress rules then canonicalize the proof-UF.
913    fn register_native_merge_proof_view(
914        &mut self,
915        view_func: FunctionId,
916        parent_table: FunctionId,
917        proof_side_table: FunctionId,
918        trans: FunctionId,
919        sym: FunctionId,
920    ) {
921        let view_name = self.name_for_function_id(view_func).to_string();
922        let parent = self.name_for_function_id(parent_table).to_string();
923        let side = self.name_for_function_id(proof_side_table).to_string();
924        let trans_table = self.name_for_function_id(trans).to_string();
925        let sym_table = self.name_for_function_id(sym).to_string();
926        if let Some(info) = self.functions.get_mut(&view_name) {
927            info.native_congruence_proof = Some(crate::NativeCongruenceProof {
928                parent_table: parent,
929                side_table: side,
930                trans_table,
931                sym_table,
932            });
933        }
934    }

968    fn for_each_while(
969        &self,
970        table: FunctionId,
971        f: &mut dyn for<'r> FnMut(FunctionRow<'r>) -> bool,
972    ) {
973        let name = self.name_for_function_id(table);
974        let info = self
975            .functions
976            .get(name)
977            .unwrap_or_else(|| panic!("for_each_while: function `{name}` not registered"));
978        let arity = info.arity();
979
980        // Select all columns (c0..c{arity-1}); skip the `ts` column.
981        let cols: Vec<String> = (0..arity).map(|i| format!("c{i}")).collect();
982        let sql = if cols.is_empty() {
983            format!("SELECT 1 FROM {} LIMIT 1", q(name))
984        } else {
985            format!("SELECT {} FROM {}", cols.join(", "), q(name))
986        };
987
988        let mut stmt = self
989            .conn
990            .prepare(&sql)
991            .unwrap_or_else(|e| panic!("for_each_while: prepare failed: {e}"));
992        let mut rows = stmt
993            .query([])
994            .unwrap_or_else(|e| panic!("for_each_while: query failed: {e}"));
995
996        let mut buf: Vec<Value> = Vec::with_capacity(arity);
997        while let Some(row) = rows
998            .next()
999            .unwrap_or_else(|e| panic!("for_each_while: row fetch failed: {e}"))
1000        {
1001            buf.clear();
1002            for i in 0..arity {
1003                let v = row
1004                    .get_ref(i)
1005                    .unwrap_or_else(|e| panic!("for_each_while: get_ref({i}) failed: {e}"));
1006                buf.push(duck_value_to_trait_value(v, &self.backend_base_value_pool));
1007            }
1008            let frow = FunctionRow {
1009                vals: &buf,
1010                subsumed: false,
1011            };
1012            if !f(frow) {
1013                break;
1014            }
1015        }
1016    }

1030    fn lookup_id(&self, func: FunctionId, key: &[Value]) -> Option<Value> {
1031        let name = self.name_for_function_id(func);
1032        let info = self
1033            .functions
1034            .get(name)
1035            .unwrap_or_else(|| panic!("lookup_id: function `{name}` not registered"));
1036        if !info.has_output() {
1037            // Relations don't have an output column. The trait method's
1038            // contract is "key -> output"; for relations there is no
1039            // output. Return None.
1040            return None;
1041        }
1042        let inputs_len = info.inputs_len;
1043        if key.len() != inputs_len {
1044            return None;
1045        }
1046        let where_parts: Vec<String> = (0..inputs_len)
1047            .map(|i| format!("c{i} = {}", key[i].rep() as i64))
1048            .collect();
1049        let where_clause = if where_parts.is_empty() {
1050            String::new()
1051        } else {
1052            format!(" WHERE {}", where_parts.join(" AND "))
1053        };
1054        let sql = format!(
1055            "SELECT c{inputs_len} FROM {}{} LIMIT 1",
1056            q(name),
1057            where_clause
1058        );
1059        let mut stmt = self.conn.prepare(&sql).ok()?;
1060        let mut rows = stmt.query([]).ok()?;
1061        let row = rows.next().ok()??;
1062        let v = row.get_ref(0).ok()?;
1063        Some(duck_value_to_trait_value(v, &self.backend_base_value_pool))
1064    }

1066    fn add_values(&mut self, values: Box<dyn Iterator<Item = (FunctionId, Vec<Value>)> + '_>) {
1067        // Bulk insert across one or many functions, one row at a
1068        // time. Mirrors the bridge's `add_values` semantics. Each
1069        // row's `Vec<Value>` is decoded against the destination
1070        // function's schema, transformed to a `Vec<Literal>`, and
1071        // forwarded to the existing `EGraph::insert`.
1072        //
1073        // The bridge's `add_values` is followed by `flush_updates`;
1074        // on DuckDB seed inserts are immediate (no staging), so the

1081    fn add_term(&mut self, func: FunctionId, inputs: &[Value]) -> Value {
1082        // Allocate a fresh id, insert `(inputs..., fresh_id)` into the
1083        // function table, and return the id. Mirrors the bridge's
1084        // `add_term`. Uses the existing `allocate_and_insert` for
1085        // EqSort constructors and `insert_row_inner` for regular
1086        // functions.
1087        let name = self.name_for_function_id(func).to_string();
1088        let info = self
1089            .functions
1090            .get(&name)
1091            .unwrap_or_else(|| panic!("add_term: function `{name}` not registered"));
1092
1093        if info.eq_sort_ctor {
1094            // EqSort constructor — allocate fresh id via the
1095            // sequence; the body row is written by a subsequent
1096            // `(set @<name>View args fresh_id) ()`. To mirror the
1097            // bridge's "insert the row" semantics we have to also
1098            // write to the view here. For now we just allocate the
1099            // id (matching `allocate_and_insert`'s docs); writing the
1100            // view row is the caller's responsibility — when this
1101            // method is reached through Commit 14's flipped pipeline,
1102            // the frontend's term-encoding pass inserts the view row
1103            // separately.
1104            let pool: &dyn BaseValuePool = &self.backend_base_value_pool;
1105            let lits: Vec<Literal> = inputs
1106                .iter()
1107                .zip(info.cols.iter().take(info.inputs_len))
1108                .map(|(v, _duck_ty)| {
1109                    // We don't yet have the trait-side ColumnTy for
1110                    // the input column (the schema is stored as
1111                    // duck::ColumnTy on FunctionInfo). Fall back to
1112                    // the I64 encoding — for eq-sort constructors
1113                    // inputs are themselves ids, so this is correct.
1114                    let _ = pool;
1115                    Literal::I64(v.rep() as i64)
1116                })
1117                .collect();
1118            let id = self
1119                .allocate_and_insert(&name, &lits)
1120                .unwrap_or_else(|e| panic!("add_term: allocate_and_insert failed: {e}"));
1121            Value::new(id as u32)
1122        } else {
1123            // Plain function: allocate a fresh id for the output and
1124            // insert `(inputs..., fresh_id)` directly.
1125            let id_i64: i64 = self
1126                .conn
1127                .query_row("SELECT nextval('__egglog_eqsort_seq')", [], |r| r.get(0))
1128                .unwrap_or_else(|e| panic!("add_term: nextval failed: {e}"));
1129            let mut full_row: Vec<Value> = inputs.to_vec();
1130            full_row.push(Value::new(id_i64 as u32));
1131            self.insert_row_inner(func, &full_row);
1132            Value::new(id_i64 as u32)
1133        }
1134    }

1146    fn lookup_constructor_rows(&mut self, table: FunctionId, rows: &[Vec<Value>]) {
1147        // The bridge's `lookup_constructor_rows` is "look-up-or-allocate"
1148        // for an EqSort constructor: for each input key, return the
1149        // existing output id if any, or allocate a fresh one and
1150        // insert. The trait does not expose the resulting ids — the
1151        // caller uses the side-effect (rows are present in the table
1152        // afterward).
1153        //
1154        // We translate this to: for each key row, call `lookup_id`
1155        // first; if absent, call `add_term`. The behavior mirrors
1156        // `EGraph::allocate_and_insert` plus the view-row write the
1157        // bridge's `TableAction::lookup` does internally.
1158        let name = self.name_for_function_id(table).to_string();
1159        let info = self
1160            .functions
1161            .get(&name)
1162            .unwrap_or_else(|| panic!("lookup_constructor_rows: function `{name}` not registered"));
1163        let inputs_len = info.inputs_len;
1164
1165        for row in rows {
1166            if row.len() != inputs_len {
1167                panic!(
1168                    "lookup_constructor_rows: row arity mismatch for `{name}`: got {}, expected {}",
1169                    row.len(),
1170                    inputs_len
1171                );
1172            }
1173            // Try a lookup first.
1174            if self.lookup_id(table, row).is_some() {
1175                continue;
1176            }
1177            // Miss — allocate.
1178            let _ = self.add_term(table, row);
1179        }
1180    }

1250        // Mark the slot as freed. We intentionally keep the slot in

1262    fn run_rules(&mut self, rules: &[RuleId]) -> Result<IterationReport> {
1263        // The trait's `run_rules(&[ids])` is "run exactly these rules
1264        // once." Resolve each id to its `self.rules` index and call
1265        // `run_iteration_for_indices`. Empty input — and input where
1266        // every id resolves to a freed slot — is a true no-op (no
1267        // ruleset-name detour that could fall through to "run all
1268        // rules"). The duckdb backend doesn't surface per-rule
1269        // timing in a `RuleSetReport` shape (it accounts via
1270        // `rule_perf_ns`), so we return a minimal `IterationReport`
1271        // with `changed` reflecting whether any rows were affected.
1272        let indices: Vec<usize> = rules
1273            .iter()
1274            .filter_map(|id| {
1275                self.backend_rule_indices
1276                    .get(id.rep() as usize)
1277                    .and_then(|opt| *opt)
1278            })
1279            .collect();
1280        if indices.is_empty() {
1281            return Ok(IterationReport::default());
1282        }
1283
1284        let before = self.rules_affected_total();
1285        let _ = EGraph::run_iteration_for_indices(self, &indices)?;
1286        let after = self.rules_affected_total();
1287        if std::env::var("DUCK_TRACE_RUN_RULES").is_ok() {
1288            let names: Vec<&str> = indices
1289                .iter()
1290                .filter_map(|i| self.rules.get(*i).map(|r| r.name.as_str()))
1291                .collect();
1292            eprintln!(
1293                "[duck/run_rules] names={:?} delta={}",
1294                names,
1295                after - before,
1296            );
1297        }
1298
1299        let mut report = IterationReport::default();
1300        report.rule_set_report.changed = after != before;
1301        Ok(report)
1302    }

1411    /// (a fold of primitives / literals over `old`/`new` that yields a plain value,
1412    /// e.g. `(from-string (to-string (* new old)))`) NATIVELY in-SQL: the frontend
1413    /// lowers the body to a `MergeFn` tree of `Primitive`/`Const`/`Old`/`New`,

1415    /// table's `ON CONFLICT DO UPDATE SET c{out} = <expr>` resolves the FD conflict
1416    /// in-SQL. The output is NOT an eq-sort (a plain value — no congruence cascade),
1417    /// so this is pure SQL. `supports_complex_merge` stays `false` (it gates the
1418    /// bridge's value-fold-via-callback path that DuckDB does not use) — this split
1419    /// capability enables the native value-fold path. Mirrors FlowLog/Feldera (but

1425    /// `--native-merge` CONSTRUCTOR-CONGRUENCE. Returning `true` makes the encoder
1426    /// DROP the `@congruence_rule*` self-join (proof_encoding
1427    /// `handle_merge_or_congruence`) and record the view -> output-sort-UF
1428    /// association (via `register_native_merge_view`).
1429    ///
1430    /// TERM MODE: DuckDB's term-mode native-merge (the old inline `emit_native_
1431    /// congruence` host pass into `@UF_Sf`) has been REMOVED with the native-UF
1432    /// migration — `--native-merge --duckdb` in non-proof mode is now rejected at
1433    /// the CLI, and plain `--duckdb` keeps the rule-encoded `@congruence_rule*`
1434    /// (this method's DROP effect only fires when the encoder's `native_merge()`
1435    /// is set, which no longer happens for the non-proof duckdb default).
1436    ///
1437    /// PROOF-mode native-merge IS still supported, via the 2-table proof-congruence
1438    /// encoding rendered in the all-columns COEXIST shape: the constructor's
1439    /// `@<C>View` keeps the BASELINE all-columns shape `(children..., eclass) ->
1440    /// Proof` and a paired `@<F>ViewProof` SIDE-TABLE holds the per-row view proof.
1441    /// On an FD conflict the host pass [`EGraph::emit_native_congruence_proof`]
1442    /// reads both colliding rows' proofs from the side-table, MINTS the congruence
1443    /// edge proof `Trans(larger_pf, Sym(smaller_pf))` in SQL, and writes the
1444    /// proof-carrying union edge `(larger, smaller, edge_proof)` into the relational
1445    /// proof-UF `@UF_S` — byte-for-byte the row the dropped `@congruence_rule*`
1446    /// wrote via `union(new, old, Trans(prf1, Sym(prf2)))`. So this returns `true`
1447    /// so PROOF-mode native-merge congruence keeps working (the proof-carrying

1458    /// eclass)` row — a children-only delete on coexisting rows would self-wipe the
1459    /// canonical row, and a full-key delete can't be expressed on an FD function.

1461    /// `emit_native_congruence_proof` (proof mode) host pass. VALUE-FOLD / TERM-BUILD
1462    /// views are unaffected (they keep the FD shape — see `uses_fd_native_merge_view`).

1480    /// `--native-merge` TERM-BUILD custom `:merge`: DuckDB runs a term-building
1481    /// custom `:merge` (a merge body that mints e-nodes via constructor calls,
1482    /// e.g. `(C2 (C1 old new) (C2 old new))`) NATIVELY as INLINE set-based SQL.
1483    ///
1484    /// The frontend lowers the merge body to a `MergeFn::Seq` term-build tree
1485    /// (`IfEq` guard around a `Seq` of `Construct` / `TableInsert` / canon
1486    /// `Function` nodes) and installs it as the function's `@<F>View` merge.
1487    /// `add_table` retains that tree in [`FunctionInfo::merge_tree`] and registers

1522    fn dump_debug_info(&self) {
1523        eprintln!("== DuckDB tables (dump_debug_info) ==");
1524        let names: Vec<String> = self.functions.keys().cloned().collect();
1525        for name in &names {
1526            // Two-phase: first COUNT to skip empty tables, then
1527            // re-query for rows. The `prepare → query → next` flow
1528            // produced "statement not executed" panics on some
1529            // duckdb-rs versions; this two-phase shape sidesteps it.
1530            let n_rows: i64 = self
1531                .conn
1532                .query_row(&format!("SELECT COUNT(*) FROM {}", q(name)), [], |r| {
1533                    r.get(0)
1534                })
1535                .unwrap_or(0);
1536            if n_rows == 0 {
1537                continue;
1538            }
1539            eprintln!("== DuckDB table `{name}` ({n_rows} rows) ==");
1540            // Use CAST(... AS VARCHAR) so we never have to inspect
1541            // the column type (avoids duckdb-rs's "statement not
1542            // executed" panic on some types). Compose the SELECT
1543            // dynamically based on the function's known column count.
1544            let arity = self.functions.get(name).map(|f| f.cols.len()).unwrap_or(0);
1545            // +1 for the ts column.
1546            let cols_str: Vec<String> = (0..arity + 1)
1547                .map(|i| {
1548                    let cn = if i == arity {
1549                        "ts".to_string()
1550                    } else {
1551                        format!("c{i}")
1552                    };
1553                    format!("CAST({cn} AS VARCHAR)")
1554                })
1555                .collect();
1556            let sql = format!("SELECT {} FROM {}", cols_str.join(", "), q(name));
1557            let collected: Vec<String> = match self.conn.prepare(&sql) {
1558                Ok(mut stmt) => stmt
1559                    .query_map([], |row| {
1560                        let n_cols = arity + 1;
1561                        let mut parts: Vec<String> = Vec::with_capacity(n_cols);
1562                        for i in 0..n_cols {
1563                            let s: String = row.get(i).unwrap_or_default();
1564                            parts.push(s);
1565                        }
1566                        Ok(parts.join(", "))
1567                    })
1568                    .map(|iter| iter.filter_map(|r| r.ok()).collect())
1569                    .unwrap_or_default(),
1570                Err(e) => {
1571                    eprintln!("  prepare failed: {e}");
1572                    continue;
1573                }
1574            };
1575            for line in collected {
1576                eprintln!("  {line}");
1577            }
1578        }
1579    }

1606// ---------------------------------------------------------------------------
1607// `unsafe impl Send + Sync` for EGraph
1608// ---------------------------------------------------------------------------
1609//
1610// The `Backend` trait requires `Send + Sync`. `EGraph` holds a
1611// `duckdb::Connection`, which is `Send` but not `Sync` (DuckDB
1612// connections are thread-local). Within the egglog backend the
1613// connection is always accessed from a single thread (DuckDB itself
1614// is configured with `SET threads = 1` for determinism), so for the
1615// purposes of the trait-object wrapper we assert both. If a future
1616// caller drives the backend from multiple threads concurrently this
1617// promise will need re-evaluation.
1618unsafe impl Send for EGraph {}
1619unsafe impl Sync for EGraph {}
