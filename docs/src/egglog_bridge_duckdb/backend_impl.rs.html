3//! This file provides a **stub** implementation of the `Backend` trait
4//! for the DuckDB-backed [`crate::EGraph`]. The frontend `EGraph` in
5//! `src/lib.rs` does **not yet** reach this impl: the duckdb test combos

10//! ## What is implemented
11//!
12//! These methods have real bodies; they call directly into the existing
13//! duckdb-bridge primitives:
14//!
15//! - [`Backend::add_table`]: dispatches on [`FunctionConfig::default`]
16//!   and [`FunctionConfig::merge`] to `add_function` /
17//!   `add_relation_with_pname` / `add_eq_sort_constructor`. Returns a
18//!   numeric [`FunctionId`] tracked in `EGraph::backend_function_names`.
19//! - [`Backend::table_size`]: `db.count(name)`.

21//!   `SELECT * FROM <table>` cursor; rows are packed into a per-row
22//!   `Vec<Value>` buffer and surfaced as `FunctionRow`. The subsumption
23//!   bit is always `false` (DuckDB does not model subsumption; see
24//!   [`Backend::supports_subsumption`]).
25//! - [`Backend::lookup_id`]: `SELECT c<n> FROM t WHERE c0=? AND … LIMIT 1`.
26//! - [`Backend::set_report_level`]: stores the level on the egraph.
27//! - [`Backend::dump_debug_info`]: iterates registered tables and logs
28//!   `SELECT * FROM <table>` via `log::info!`.

30//!   [`Backend::supports_inline_table_lookups`],
31//!   [`Backend::supports_subsumption`],
32//!   [`Backend::supports_complex_merge`],
33//!   [`Backend::supports_containers`].
34//! - [`Backend::as_any`] / [`Backend::as_any_mut`]: trivial.
35//! - [`Backend::container_pool`] / [`Backend::container_pool_mut`]:
36//!   return `&self.backend_container_pool` /
37//!   `&mut self.backend_container_pool` — the zero-sized stub defined
38//!   below.
39//! - [`Backend::flush_updates`]: returns `false` (no-op until Commit 10
40//!   wires rule running).

134/// Translate a [`ColumnTy`] from the backend trait crate to the
135/// duckdb-bridge's own [`crate::ColumnTy`].
136///
137/// Currently every column maps to `I64`. The trait surface uses a
138/// single `Value` (`u32`) for all column kinds, and DuckDB stores
139/// every `Value` as the underlying integer bits in a `BIGINT`
140/// column. This is lossy with respect to the *display* of values
141/// (e.g. a `f64` column rendered as `BIGINT` shows raw bits in a
142/// `SELECT *`), but it is fully round-trippable through the
143/// trait — `value_to_literal` / `duck_value_to_trait_value` agree
144/// on this encoding.
145///
146/// A more precise mapping (Bool / F64 / Str columns) would require
147/// reaching into the egglog-frontend's typed `Sort` system rather
148/// than `BaseValuePool` alone: the pool stores typed intern tables
149/// keyed by Rust `TypeId`, but the frontend's actual primitive types
150/// (`Boxed<OrderedFloat<f64>>`, `Arc<str>`, custom `BaseValue` impls)
151/// rarely match the natural DuckDB column kinds 1:1. Since this code
152/// path is dead until Commit 14 routes a `Box<dyn Backend>` through
153/// `add_table`, the lossy mapping is fine.
154///
155/// **TODO (Commit 14):** revisit when a real `Box<dyn Backend>`
156/// caller exercises `add_table` and we see what column kinds the
157/// frontend actually asks for.
158fn trait_col_ty_to_duck(ty: ColumnTy, pool: &dyn BaseValuePool) -> DuckColumnTy {
159    use crate::DuckPairMarker;
160    use ordered_float::OrderedFloat;
161    use std::any::TypeId;
162    match ty {
163        ColumnTy::Id => DuckColumnTy::I64,
164        ColumnTy::Base(bv) => {
165            // Probe the pool's registered TypeIds in the same order
166            // as `decode_base_const` in rule_builder.rs. Each branch
167            // checks whether `bv` is the registered id for that
168            // concrete `BaseValue` type and, if so, returns the
169            // matching DuckDB column kind. Falls back to `I64` for
170            // unknown / unregistered base types (which is also what
171            // egglog-bridge does for them — see `BaseValue::sql_ty`).
172            if pool.has_ty(TypeId::of::<DuckPairMarker>())
173                && bv == pool.get_ty_by_type_id(TypeId::of::<DuckPairMarker>())
174            {
175                // Pair marker — see `EGraph::pair_column_ty`.
176                return DuckColumnTy::PairI64;
177            }
178            if pool.has_ty(TypeId::of::<i64>()) && bv == pool.get_ty_by_type_id(TypeId::of::<i64>())
179            {
180                return DuckColumnTy::I64;
181            }
182            if pool.has_ty(TypeId::of::<bool>())
183                && bv == pool.get_ty_by_type_id(TypeId::of::<bool>())
184            {
185                return DuckColumnTy::Bool;
186            }
187            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
188            if pool.has_ty(TypeId::of::<FBoxed>())
189                && bv == pool.get_ty_by_type_id(TypeId::of::<FBoxed>())
190            {
191                return DuckColumnTy::F64;
192            }
193            // Strings used to map to `DuckColumnTy::Str` (VARCHAR),
194            // but that broke bridge-API parity: the trait surface
195            // (`for_each_while`, etc.) returns `Value` handles, and
196            // `duck_value_to_trait_value` had no way to round-trip
197            // a VARCHAR back to a `Value` without re-interning
198            // against the pool. The bridge stores strings as
199            // interned handles in BIGINT columns; we now match.
200            //
201            // SQL semantics: equality / inequality on the BIGINT
202            // handle is equivalent to equality on the original
203            // string because interning is stable. Ordering ops on
204            // handles aren't meaningful, but egglog doesn't expose
205            // `<` / `>` on strings either, so that's a non-issue.
206            // String-producing primitives (`to-string`, `string-
207            // concat`, `replace`) need UDFs that intern their
208            // output — see compile.rs's prim_sql.
209            //
210            // Unit and any other registered base types: store as
211            // i64. This includes BigInt / BigRat / Rational64
212            // which the bridge interns; their `Value` is an
213            // intern-table index that fits in i64.
214            DuckColumnTy::I64
215        }
216    }
217}

219/// Decode an egglog [`Value`] into a duckdb-side [`Literal`] for
220/// emission into a `BIGINT`-style SQL column.
221///
222/// Pairs with [`trait_col_ty_to_duck`]: that helper maps every trait
223/// column kind to `DuckColumnTy::I64`, so the corresponding `Literal`
224/// is `Literal::I64(value.rep() as i64)`. The `pool` parameter is
225/// reserved for the future refined mapping (see the TODO on
226/// `trait_col_ty_to_duck`).
227/// Decode a `Value` for a duck-side column of [`DuckColumnTy`] into
228/// the appropriate [`Literal`]. Used by `insert_row_inner` so a
229/// row whose column was registered as `STR`/`BOOL`/`F64` gets the
230/// matching SQL literal (not `Literal::I64(value.rep() as i64)`,
231/// which would produce a SQL type-conversion error).
232fn duck_value_to_literal(val: Value, col: DuckColumnTy, pool: &DuckdbBaseValuePool) -> Literal {
233    use ordered_float::OrderedFloat;
234    use std::any::TypeId;
235    let pool_dyn: &dyn BaseValuePool = pool;
236    match col {
237        DuckColumnTy::I64 => {
238            // Could be an eq-sort id, a primitive i64 value, or a
239            // handle to a non-i64 intern table (BigInt, Boxed<String>,
240            // etc. — all of which use the BIGINT representation on
241            // the duck side). For i64 base values, unwrap to the raw
242            // i64 so SQL arithmetic on the column operates on the
243            // value, not the handle. The read path
244            // (`duck_value_to_trait_value`) re-interns the raw value
245            // back to a handle so callers that expect `Value`s
246            // (the trait API contract) get a usable rep.
247            if pool_dyn.has_ty(TypeId::of::<i64>()) && val.rep() & (1 << 31) != 0 {
248                let i = egglog_backend_trait::pool_unwrap::<i64>(pool_dyn, val);
249                return Literal::I64(i);
250            }
251            Literal::I64(val.rep() as i64)
252        }
253        DuckColumnTy::Bool => {
254            if pool_dyn.has_ty(TypeId::of::<bool>()) {
255                let b = egglog_backend_trait::pool_unwrap::<bool>(pool_dyn, val);
256                return Literal::Bool(b);
257            }
258            Literal::Bool(val.rep() != 0)
259        }
260        DuckColumnTy::F64 => {
261            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
262            if pool_dyn.has_ty(TypeId::of::<FBoxed>()) {
263                let v = egglog_backend_trait::pool_unwrap::<FBoxed>(pool_dyn, val);
264                return Literal::F64((*v).into_inner());
265            }
266            Literal::F64(val.rep() as f64)
267        }
268        DuckColumnTy::Str => {
269            type SBoxed = egglog_core_relations::Boxed<String>;
270            if pool_dyn.has_ty(TypeId::of::<SBoxed>()) {
271                let v = egglog_backend_trait::pool_unwrap::<SBoxed>(pool_dyn, val);
272                return Literal::Str((*v).clone());
273            }
274            // Fallback: stringify the bits. Almost certainly wrong
275            // but at least doesn't trigger a SQL type cast error.
276            Literal::Str(format!("v{}", val.rep()))
277        }
278        DuckColumnTy::PairI64 => {
279            // Pair columns are not emitted via trait `add_table`
280            // (only via duck-internal registration), so this branch
281            // shouldn't be reachable from `insert_row_inner`.
282            panic!("duck_value_to_literal: PairI64 unexpected in trait-driven insert path")
283        }
284    }
285}

288fn value_to_literal(val: Value, ty: ColumnTy, pool: &dyn BaseValuePool) -> Literal {
289    use ordered_float::OrderedFloat;
290    use std::any::TypeId;
291    match ty {
292        ColumnTy::Id => Literal::I64(val.rep() as i64),
293        ColumnTy::Base(bv) => {
294            if pool.has_ty(TypeId::of::<i64>()) && bv == pool.get_ty_by_type_id(TypeId::of::<i64>())
295            {
296                let v = egglog_backend_trait::pool_unwrap::<i64>(pool, val);
297                return Literal::I64(v);
298            }
299            if pool.has_ty(TypeId::of::<bool>())
300                && bv == pool.get_ty_by_type_id(TypeId::of::<bool>())
301            {
302                let v = egglog_backend_trait::pool_unwrap::<bool>(pool, val);
303                return Literal::Bool(v);
304            }
305            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
306            if pool.has_ty(TypeId::of::<FBoxed>())
307                && bv == pool.get_ty_by_type_id(TypeId::of::<FBoxed>())
308            {
309                let v = egglog_backend_trait::pool_unwrap::<FBoxed>(pool, val);
310                return Literal::F64((*v).into_inner());
311            }
312            type SBoxed = egglog_core_relations::Boxed<String>;
313            if pool.has_ty(TypeId::of::<SBoxed>())
314                && bv == pool.get_ty_by_type_id(TypeId::of::<SBoxed>())
315            {
316                let v = egglog_backend_trait::pool_unwrap::<SBoxed>(pool, val);
317                return Literal::Str((*v).clone());
318            }
319            // Fallback: treat as i64 (intern-table index).
320            Literal::I64(val.rep() as i64)
321        }
322    }

325/// Convert a DuckDB row value into a [`Value`] for surfacing through
326/// [`FunctionRow`].
327///
328/// The current DuckDB pipeline stores all egglog values as `BIGINT`
329/// (eq-sort ids) or other concrete SQL types. The Backend trait surface
330/// uses a single `Value` (a `u32` newtype) for every column. The
331/// mapping is:
332///
333/// - `BIGINT` -> low 32 bits cast to `Value`. For the eq-sort id case
334///   this is the canonical mapping; for primitive `BIGINT`s storing
335///   `i64` literals it loses the high bits, but only the trait-side
336///   pool wired in Commit 11 will know how to interpret the bits, and
337///   `for_each` callers reaching this path will be added in Commit
338///   14+. For now correctness is moot since no caller exercises it.
339/// Convert a DuckDB `ValueRef` to a trait-level `Value` handle.
340///
341/// For BIGINT columns this gets tricky because i64 base values
342/// are stored RAW in DuckDB (so SQL arithmetic works) but the trait
343/// API expects `Value` handles. The bridge unifies these: a "small"
344/// i64 (top bit clear, fits in u32) IS its own handle; an "interned"
345/// i64 (out of the 31-bit fast path — negative or > 2^31) has a
346/// handle of `intern_table_idx + VAL_OFFSET` (top bit set).
347///
348/// So when reading a BIGINT value `i`:
349///   - `i >= 0 && i < 2^31`: the raw i64 fits the fast-path, so
350///     `Value(i as u32)` is already a valid handle. Same for eq-sort
351///     ids (which always fit), `Boxed<String>` handles (intern
352///     indices fit easily in practice), and `Boxed<BigInt>` /
353///     `Boxed<BigRat>` handles (same).
354///   - `i < 0 || i >= 2^31`: must be a raw i64 base value that was
355///     stored without going through the intern table at insertion
356///     time. Intern it now to produce a stable handle for the trait
357///     consumer; pool_unwrap on the resulting handle round-trips.
358///
359/// The pool is `&dyn` to keep this callable from contexts that only
360/// have the abstract trait; in for_each_while we already need the
361/// pool to register an i64 type if it isn't yet (rare; defensively).
362fn duck_value_to_trait_value(v: ValueRef<'_>, pool: &dyn BaseValuePool) -> Value {
363    use duckdb::types::ValueRef as V;
364    use std::any::TypeId;
365
366    let raw_i64: Option<i64> = match v {
367        V::Null => return Value::new(u32::MAX),
368        V::Boolean(b) => return Value::new(b as u32),
369        V::TinyInt(i) => Some(i as i64),
370        V::SmallInt(i) => Some(i as i64),
371        V::Int(i) => Some(i as i64),
372        V::BigInt(i) => Some(i),
373        V::HugeInt(i) => Some(i as i64),
374        V::UTinyInt(i) => return Value::new(i as u32),
375        V::USmallInt(i) => return Value::new(i as u32),
376        V::UInt(i) => return Value::new(i),
377        V::UBigInt(i) => return Value::new(i as u32),
378        // DOUBLE column: f64 base values are stored RAW (so SQL
379        // arithmetic works). Intern via the pool to lift to a
380        // handle the trait consumer can use with `pool_unwrap::<F>`.
381        V::Float(f) => {
382            use ordered_float::OrderedFloat;
383            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
384            if pool.has_ty(TypeId::of::<FBoxed>()) {
385                let ty = pool.get_ty_by_type_id(TypeId::of::<FBoxed>());
386                return pool.intern_dyn(ty, Box::new(FBoxed::new(OrderedFloat(f as f64))));
387            }
388            return Value::new(u32::MAX);
389        }
390        V::Double(f) => {
391            use ordered_float::OrderedFloat;
392            type FBoxed = egglog_core_relations::Boxed<OrderedFloat<f64>>;
393            if pool.has_ty(TypeId::of::<FBoxed>()) {
394                let ty = pool.get_ty_by_type_id(TypeId::of::<FBoxed>());
395                return pool.intern_dyn(ty, Box::new(FBoxed::new(OrderedFloat(f))));
396            }
397            return Value::new(u32::MAX);
398        }
399        // Other types: unreached for the BIGINT-heavy trait paths,
400        // but emit u32::MAX so a panic downstream is more obvious
401        // than a silently-wrong value.
402        _ => return Value::new(u32::MAX),
403    };
404    let Some(i) = raw_i64 else {
405        return Value::new(u32::MAX);
406    };
407
408    // The fast-path range covers all eq-sort ids, MAY_UNBOX-compatible
409    // i64 bases, and intern table indices for any reasonable program.
410    // Values outside it (large positive or any negative) can only be
411    // raw i64 base values: intern via the pool to lift to a handle.
412    if (0..(1i64 << 31)).contains(&i) {
413        return Value::new(i as u32);
414    }
415    if pool.has_ty(TypeId::of::<i64>()) {
416        let ty = pool.get_ty_by_type_id(TypeId::of::<i64>());
417        return pool.intern_dyn(ty, Box::new(i));
418    }
419    // No i64 type registered — can't intern. Fall back to the raw
420    // bits; callers that try to `pool_unwrap::<i64>` will panic
421    // cleanly with an unknown-table message.
422    Value::new(i as u32)
423}

425impl EGraph {
426    /// Look up the duckdb-bridge table name for a trait [`FunctionId`].
427    fn name_for_function_id(&self, id: FunctionId) -> &str {
428        let idx = id.rep() as usize;
429        self.backend_function_names
430            .get(idx)
431            .map(|s| s.as_str())
432            .unwrap_or_else(|| panic!("FunctionId({idx}) is not registered"))
433    }
434
435    /// Insert a single row's values into the function's table. The
436    /// row is decoded against the function's stored schema (duck-side
437    /// `ColumnTy` per column) and emitted via the existing
438    /// `EGraph::insert` seed path.
439    ///
440    /// Encoding strategy:
441    /// - Bool/I64/F64/Str duck columns → encode via the corresponding
442    ///   `Literal` variant. For `Bool`/`F64`/`Str` we need the value
443    ///   as the corresponding Rust type, which means using the base
444    ///   value pool for non-trivial unboxing. For `I64` we just cast
445    ///   `Value::rep() as i64`.
446    /// - `PairI64` columns: not supported through the trait yet;
447    ///   panics.
448    ///
449    /// Panics on schema mismatches; this method is called by
450    /// `add_values` / `insert_rows` / `add_term`, all of which are
451    /// unreachable until Commit 14 flips the test harness.
452    pub(crate) fn insert_row_inner(&mut self, func: FunctionId, row: &[Value]) {
453        let name = self.name_for_function_id(func).to_string();
454        let info = self
455            .functions
456            .get(&name)
457            .unwrap_or_else(|| panic!("insert_row_inner: function `{name}` not registered"))
458            .clone();
459        if row.len() != info.arity() {
460            panic!(
461                "insert_row_inner: arity mismatch for `{name}`: got {}, expected {}",
462                row.len(),
463                info.arity()
464            );
465        }
466
467        // Decode each value through the schema's `DuckColumnTy` so
468        // String/Bool/F64 columns receive the right `Literal` kind.
469        // We don't have the trait `ColumnTy` here (the duck-side
470        // `FunctionInfo.cols` already lost it), so re-derive the
471        // duck literal kind from `info.cols[i]`.
472        let lits: Vec<Literal> = row
473            .iter()
474            .enumerate()
475            .map(|(i, v)| {
476                let col_ty = info.cols[i];
477                duck_value_to_literal(*v, col_ty, &self.backend_base_value_pool)
478            })
479            .collect();
480
481        self.insert(&name, &lits)
482            .unwrap_or_else(|e| panic!("insert_row_inner: insert(`{name}`) failed: {e}"));
483    }
484}

493    fn add_table(&mut self, config: FunctionConfig) -> FunctionId {
494        // Allocate a numeric id and register the name. The duckdb
495        // backend itself uses string names; the trait callers use
496        // numeric ids, so we maintain a Vec<String> mapping.
497        let idx = self.backend_function_names.len() as u32;
498        let id = FunctionId::new(idx);
499
500        // Split the schema into input columns + output column (if any).
501        // For relations (no output) the entire schema is "inputs".
502        // For eq-sort constructors and functions, the last entry is the
503        // output column.
504        //
505        // Heuristic for dispatch (per the prompt):
506        //   - DefaultVal::FreshId => add_eq_sort_constructor (constructor)
507        //   - DefaultVal::Fail / Const + MergeFn::AssertEq / UnionId
508        //     => add_function with a merge mode derived from MergeFn
509        //   - If the schema has no output column treat as relation:
510        //     add_relation_with_pname.
511        //
512        // The dispatch is intentionally simple: this code path is dead
513        // until Commit 14, so the exact semantics of less-common
514        // combinations aren't load-bearing yet.
515
516        let name = config.name.clone();
517        let schema = &config.schema;
518
519        // Convert all schema columns. The trait's `FunctionConfig`
520        // always includes the output (if any) as the last entry of
521        // `schema`; the duckdb-bridge `add_function` and
522        // `add_eq_sort_constructor` take inputs + output separately
523        // while `add_relation_with_pname` takes the full input list.
524        let duck_cols: Vec<DuckColumnTy> = {
525            // Borrow the pool through the concrete field to dodge a
526            // re-borrow conflict with `self`: we need `&self.backend_base_value_pool`
527            // for the column-type translation but `&mut self` later
528            // for `add_function`/`add_relation_with_pname`. Translating
529            // up front gives us a fully-owned `Vec<DuckColumnTy>` to
530            // hand the mutating methods.
531            let pool: &dyn BaseValuePool = &self.backend_base_value_pool;
532            schema
533                .iter()
534                .copied()
535                .map(|t| trait_col_ty_to_duck(t, pool))
536                .collect()
537        };
538
539        let result = match config.default {
540            DefaultVal::FreshId => {
541                // EqSort constructor: schema = [inputs..., Id]. The
542                // duckdb-bridge appends the ID column itself, so we pass
543                // only the inputs (= schema without the trailing id).
544                let inputs = if duck_cols.is_empty() {
545                    &[][..]
546                } else {
547                    &duck_cols[..duck_cols.len() - 1]
548                };
549                self.add_eq_sort_constructor(&name, inputs, None)
550            }
551            DefaultVal::Fail | DefaultVal::Const(_) => {
552                // Either a function with an explicit output, or a
553                // relation if the output is Unit (term encoding's
554                // pattern for view tables — `(function @XView (...)
555                // Unit :merge old)`). The parallel pipeline's
556                // `add_function` in backend_duckdb.rs detected this
557                // by string-matching `output_sort == "Unit"`; on the
558                // trait side we don't have the sort name, only the
559                // `BaseValueId`. Use the pool to compare against the
560                // Unit type id.
561                let merge_mode = match &config.merge {
562                    MergeFn::Old | MergeFn::AssertEq => Some(MergeMode::Old),
563                    MergeFn::New => Some(MergeMode::New),
564                    MergeFn::UnionId => Some(MergeMode::Min),
565                    MergeFn::Primitive(ext_id, _) => {
566                        // The term encoder emits `:merge (ordering-min
567                        // old new)` for the function-form union-find
568                        // table (`@UF_<Sort>f`). The frontend lowers
569                        // that to a `Primitive` referencing the
570                        // `ordering-min` external. Without recognizing
571                        // it, the duckdb conflict clause falls back to
572                        // `DO NOTHING`, which silently drops UFf
573                        // upserts and leaves singleparent unable to
574                        // canonicalize — observed as an infinite
575                        // `(8,13) ↔ (13,8)` toggle in @UF_Math while
576                        // @UF_Mathf stays stale at (8,8)/(13,13).
577                        match self.backend_external_funcs.name(*ext_id) {
578                            Some("ordering-min") => Some(MergeMode::Min),
579                            _ => Some(MergeMode::Old),
580                        }
581                    }
582                    MergeFn::Const(_) | MergeFn::Function(_, _) => {
583                        // Per Phase 1 design: complex merges are gated
584                        // out by `supports_complex_merge` (`false`).
585                        // Fall back to Old for unreachable code paths.
586                        Some(MergeMode::Old)
587                    }
588                };
589
590                // Detect Unit output: schema's last entry is
591                // `ColumnTy::Base(id)` where `id` matches the pool's
592                // registration for the `()` type.
593                let output_is_unit = schema.last().is_some_and(|t| match t {
594                    ColumnTy::Base(bv) => {
595                        let pool: &dyn BaseValuePool = &self.backend_base_value_pool;
596                        pool.has_ty(TypeId::of::<()>())
597                            && *bv == pool.get_ty_by_type_id(TypeId::of::<()>())
598                    }
599                    _ => false,
600                });
601
602                if duck_cols.is_empty() {
603                    // No columns at all: treat as a nullary relation.
604                    self.add_relation_with_pname(&name, &[], None)
605                } else if output_is_unit {
606                    // Function with Unit output: matches term
607                    // encoding's relation pattern (e.g.
608                    // `(function @UF_@pathSort (Math Math) Unit
609                    // :merge old)`). The parallel pipeline drops the
610                    // Unit output and registers the function as a
611                    // relation whose key spans the INPUTS only — so
612                    // the body atoms reference inputs without a
613                    // trailing wildcard. Strip the trailing Unit
614                    // column to match.
615                    let inputs = &duck_cols[..duck_cols.len() - 1];
616                    self.add_relation_with_pname(&name, inputs, None)
617                } else if let Some(mode) = merge_mode {
618                    let inputs = &duck_cols[..duck_cols.len() - 1];
619                    let output = duck_cols[duck_cols.len() - 1];
620                    self.add_function(&name, inputs, output, mode)
621                } else {
622                    self.add_relation_with_pname(&name, &duck_cols, None)
623                }
624            }
625        };
626
627        if let Err(e) = result {
628            // The trait method does not return Result; if the duckdb
629            // backend rejects the table registration, we have to
630            // panic. Until Commit 14 flips the test harness this
631            // panic is unreachable.
632            panic!("DuckDB add_table({}): {}", name, e);
633        }
634
635        self.backend_function_names.push(name);
636        id
637    }

651    fn for_each_while(
652        &self,
653        table: FunctionId,
654        f: &mut dyn for<'r> FnMut(FunctionRow<'r>) -> bool,
655    ) {
656        let name = self.name_for_function_id(table);
657        let info = self
658            .functions
659            .get(name)
660            .unwrap_or_else(|| panic!("for_each_while: function `{name}` not registered"));
661        let arity = info.arity();
662
663        // Select all columns (c0..c{arity-1}); skip the `ts` column.
664        let cols: Vec<String> = (0..arity).map(|i| format!("c{i}")).collect();
665        let sql = if cols.is_empty() {
666            format!("SELECT 1 FROM {} LIMIT 1", q(name))
667        } else {
668            format!("SELECT {} FROM {}", cols.join(", "), q(name))
669        };
670
671        let mut stmt = self
672            .conn
673            .prepare(&sql)
674            .unwrap_or_else(|e| panic!("for_each_while: prepare failed: {e}"));
675        let mut rows = stmt
676            .query([])
677            .unwrap_or_else(|e| panic!("for_each_while: query failed: {e}"));
678
679        let mut buf: Vec<Value> = Vec::with_capacity(arity);
680        while let Some(row) = rows
681            .next()
682            .unwrap_or_else(|e| panic!("for_each_while: row fetch failed: {e}"))
683        {
684            buf.clear();
685            for i in 0..arity {
686                let v = row
687                    .get_ref(i)
688                    .unwrap_or_else(|e| panic!("for_each_while: get_ref({i}) failed: {e}"));
689                buf.push(duck_value_to_trait_value(v, &self.backend_base_value_pool));
690            }
691            let frow = FunctionRow {
692                vals: &buf,
693                subsumed: false,
694            };
695            if !f(frow) {
696                break;
697            }
698        }
699    }

701    fn for_each(&self, table: FunctionId, f: &mut dyn for<'r> FnMut(FunctionRow<'r>)) {
702        // Default impl in the trait is `unimplemented!()` — provide a
703        // concrete one by threading `for_each_while` with an always-
704        // true continuation.
705        self.for_each_while(table, &mut |row| {
706            f(row);
707            true
708        });

713    fn lookup_id(&self, func: FunctionId, key: &[Value]) -> Option<Value> {
714        let name = self.name_for_function_id(func);
715        let info = self
716            .functions
717            .get(name)
718            .unwrap_or_else(|| panic!("lookup_id: function `{name}` not registered"));
719        if !info.has_output() {
720            // Relations don't have an output column. The trait method's
721            // contract is "key -> output"; for relations there is no
722            // output. Return None.
723            return None;
724        }
725        let inputs_len = info.inputs_len;
726        if key.len() != inputs_len {
727            return None;
728        }
729        let where_parts: Vec<String> = (0..inputs_len)
730            .map(|i| format!("c{i} = {}", key[i].rep() as i64))
731            .collect();
732        let where_clause = if where_parts.is_empty() {
733            String::new()
734        } else {
735            format!(" WHERE {}", where_parts.join(" AND "))
736        };
737        let sql = format!(
738            "SELECT c{inputs_len} FROM {}{} LIMIT 1",
739            q(name),
740            where_clause
741        );
742        let mut stmt = self.conn.prepare(&sql).ok()?;
743        let mut rows = stmt.query([]).ok()?;
744        let row = rows.next().ok()??;
745        let v = row.get_ref(0).ok()?;
746        Some(duck_value_to_trait_value(v, &self.backend_base_value_pool))
747    }

750        // Bulk insert across one or many functions, one row at a
751        // time. Mirrors the bridge's `add_values` semantics. Each
752        // row's `Vec<Value>` is decoded against the destination

764    fn add_term(&mut self, func: FunctionId, inputs: &[Value]) -> Value {
765        // Allocate a fresh id, insert `(inputs..., fresh_id)` into the
766        // function table, and return the id. Mirrors the bridge's
767        // `add_term`. Uses the existing `allocate_and_insert` for
768        // EqSort constructors and `insert_row_inner` for regular
769        // functions.
770        let name = self.name_for_function_id(func).to_string();
771        let info = self
772            .functions
773            .get(&name)
774            .unwrap_or_else(|| panic!("add_term: function `{name}` not registered"));
775
776        if info.eq_sort_ctor {
777            // EqSort constructor — allocate fresh id via the
778            // sequence; the body row is written by a subsequent
779            // `(set @<name>View args fresh_id) ()`. To mirror the
780            // bridge's "insert the row" semantics we have to also
781            // write to the view here. For now we just allocate the
782            // id (matching `allocate_and_insert`'s docs); writing the
783            // view row is the caller's responsibility — when this
784            // method is reached through Commit 14's flipped pipeline,
785            // the frontend's term-encoding pass inserts the view row
786            // separately.
787            let pool: &dyn BaseValuePool = &self.backend_base_value_pool;
788            let lits: Vec<Literal> = inputs
789                .iter()
790                .zip(info.cols.iter().take(info.inputs_len))
791                .map(|(v, _duck_ty)| {
792                    // We don't yet have the trait-side ColumnTy for
793                    // the input column (the schema is stored as
794                    // duck::ColumnTy on FunctionInfo). Fall back to
795                    // the I64 encoding — for eq-sort constructors
796                    // inputs are themselves ids, so this is correct.
797                    let _ = pool;
798                    Literal::I64(v.rep() as i64)
799                })
800                .collect();
801            let id = self
802                .allocate_and_insert(&name, &lits)
803                .unwrap_or_else(|e| panic!("add_term: allocate_and_insert failed: {e}"));
804            Value::new(id as u32)
805        } else {
806            // Plain function: allocate a fresh id for the output and
807            // insert `(inputs..., fresh_id)` directly.
808            let id_i64: i64 = self
809                .conn
810                .query_row("SELECT nextval('__egglog_eqsort_seq')", [], |r| r.get(0))
811                .unwrap_or_else(|e| panic!("add_term: nextval failed: {e}"));
812            let mut full_row: Vec<Value> = inputs.to_vec();
813            full_row.push(Value::new(id_i64 as u32));
814            self.insert_row_inner(func, &full_row);
815            Value::new(id_i64 as u32)
816        }
817    }

829    fn lookup_constructor_rows(&mut self, table: FunctionId, rows: &[Vec<Value>]) {
830        // The bridge's `lookup_constructor_rows` is "look-up-or-allocate"
831        // for an EqSort constructor: for each input key, return the
832        // existing output id if any, or allocate a fresh one and
833        // insert. The trait does not expose the resulting ids — the
834        // caller uses the side-effect (rows are present in the table
835        // afterward).
836        //
837        // We translate this to: for each key row, call `lookup_id`
838        // first; if absent, call `add_term`. The behavior mirrors
839        // `EGraph::allocate_and_insert` plus the view-row write the
840        // bridge's `TableAction::lookup` does internally.
841        let name = self.name_for_function_id(table).to_string();
842        let info = self
843            .functions
844            .get(&name)
845            .unwrap_or_else(|| panic!("lookup_constructor_rows: function `{name}` not registered"));
846        let inputs_len = info.inputs_len;
847
848        for row in rows {
849            if row.len() != inputs_len {
850                panic!(
851                    "lookup_constructor_rows: row arity mismatch for `{name}`: got {}, expected {}",
852                    row.len(),
853                    inputs_len
854                );
855            }
856            // Try a lookup first.
857            if self.lookup_id(table, row).is_some() {
858                continue;
859            }
860            // Miss — allocate.
861            let _ = self.add_term(table, row);
862        }
863    }

865    fn get_canon_repr(&self, val: Value, ty: ColumnTy) -> Value {
866        // For base values, canonicalization is the identity.
867        if matches!(ty, ColumnTy::Base(_)) {
868            return val;
869        }
870        // For eq-sort ids under `--duck-native-uf`, we have an
871        // in-memory union-find indexed by the sort's pname. The
872        // trait's `get_canon_repr` doesn't surface a sort handle —
873        // only `ColumnTy::Id`. As a pragmatic interim, we consult
874        // every registered native UF and take the first one that
875        // contains the id; if none does, return the id unchanged.
876        //
877        // Under the non-native-UF path, canonicalization happens
878        // implicitly through the SQL pname tables; the trait
879        // surface's `get_canon_repr` is not consulted at all by the
880        // existing pipeline. Until Commit 14 routes a caller through
881        // here, this method is correctness-irrelevant — a NOP-style
882        // identity returns the right answer in every situation that
883        // matters today.
884        let needle = val.rep() as i64;
885        for uf in self.native_ufs.values() {
886            // The UDF reads the UF read-only; we mirror that here.
887            let canon = uf.lock().unwrap().find_ro(needle);
888            if canon != needle {
889                return Value::new(canon as u32);
890            }
891        }
892        val
893    }

955    fn free_rule(&mut self, id: RuleId) {
956        // Mark the slot as freed. We intentionally keep the slot in
957        // place rather than truncating the vector so subsequent
958        // `RuleId`s retain their numeric meaning. The compiled rule
959        // in `self.rules` is left intact: removing it from there
960        // would require recomputing every index. Clear the slot so
961        // subsequent `run_rules([id])` skips this rule.
962        let idx = id.rep() as usize;
963        if let Some(slot) = self.backend_rule_indices.get_mut(idx) {
964            *slot = None;
965        }
966    }

968    fn run_rules(&mut self, rules: &[RuleId]) -> Result<IterationReport> {
969        // The trait's `run_rules(&[ids])` is "run exactly these rules
970        // once." Resolve each id to its `self.rules` index and call
971        // `run_iteration_for_indices`. Empty input — and input where
972        // every id resolves to a freed slot — is a true no-op (no
973        // ruleset-name detour that could fall through to "run all
974        // rules"). The duckdb backend doesn't surface per-rule
975        // timing in a `RuleSetReport` shape (it accounts via
976        // `rule_perf_ns`), so we return a minimal `IterationReport`
977        // with `changed` reflecting whether any rows were affected.
978        let indices: Vec<usize> = rules
979            .iter()
980            .filter_map(|id| {
981                self.backend_rule_indices
982                    .get(id.rep() as usize)
983                    .and_then(|opt| *opt)
984            })
985            .collect();
986        if indices.is_empty() {
987            return Ok(IterationReport::default());
988        }
989
990        let before = self.rules_affected_total();
991        let _ = EGraph::run_iteration_for_indices(self, &indices)?;
992        let after = self.rules_affected_total();
993        if std::env::var("DUCK_TRACE_RUN_RULES").is_ok() {
994            let names: Vec<&str> = indices
995                .iter()
996                .filter_map(|i| self.rules.get(*i).map(|r| r.name.as_str()))
997                .collect();
998            eprintln!(
999                "[duck/run_rules] names={:?} delta={}",
1000                names,
1001                after - before,
1002            );
1003        }
1004
1005        let mut report = IterationReport::default();
1006        report.rule_set_report.changed = after != before;
1007        Ok(report)
1008    }

1010    fn flush_updates(&mut self) -> bool {
1011        // DuckDB has no rule-side staged-update concept: rule actions
1012        // commit through `INSERT` statements immediately. The native
1013        // UF queue is drained as part of `run_iteration_in_set`. As a
1014        // result `flush_updates` is a no-op; we return `false` to
1015        // indicate "no incremental change accrued outside a normal
1016        // `run_rules`". Callers that need to be sure the state is
1017        // settled should invoke `run_rules` instead.

1126    fn dump_debug_info(&self) {
1127        eprintln!("== DuckDB tables (dump_debug_info) ==");
1128        let names: Vec<String> = self.functions.keys().cloned().collect();
1129        for name in &names {
1130            // Two-phase: first COUNT to skip empty tables, then
1131            // re-query for rows. The `prepare → query → next` flow
1132            // produced "statement not executed" panics on some
1133            // duckdb-rs versions; this two-phase shape sidesteps it.
1134            let n_rows: i64 = self
1135                .conn
1136                .query_row(&format!("SELECT COUNT(*) FROM {}", q(name)), [], |r| {
1137                    r.get(0)
1138                })
1139                .unwrap_or(0);
1140            if n_rows == 0 {
1141                continue;
1142            }
1143            eprintln!("== DuckDB table `{name}` ({n_rows} rows) ==");
1144            // Use CAST(... AS VARCHAR) so we never have to inspect
1145            // the column type (avoids duckdb-rs's "statement not
1146            // executed" panic on some types). Compose the SELECT
1147            // dynamically based on the function's known column count.
1148            let arity = self.functions.get(name).map(|f| f.cols.len()).unwrap_or(0);
1149            // +1 for the ts column.
1150            let cols_str: Vec<String> = (0..arity + 1)
1151                .map(|i| {
1152                    let cn = if i == arity {
1153                        "ts".to_string()
1154                    } else {
1155                        format!("c{i}")
1156                    };
1157                    format!("CAST({cn} AS VARCHAR)")
1158                })
1159                .collect();
1160            let sql = format!("SELECT {} FROM {}", cols_str.join(", "), q(name));
1161            let collected: Vec<String> = match self.conn.prepare(&sql) {
1162                Ok(mut stmt) => stmt
1163                    .query_map([], |row| {
1164                        let n_cols = arity + 1;
1165                        let mut parts: Vec<String> = Vec::with_capacity(n_cols);
1166                        for i in 0..n_cols {
1167                            let s: String = row.get(i).unwrap_or_default();
1168                            parts.push(s);
1169                        }
1170                        Ok(parts.join(", "))
1171                    })
1172                    .map(|iter| iter.filter_map(|r| r.ok()).collect())
1173                    .unwrap_or_default(),
1174                Err(e) => {
1175                    eprintln!("  prepare failed: {e}");
1176                    continue;
1177                }
1178            };
1179            for line in collected {
1180                eprintln!("  {line}");
1181            }
1182        }
1183    }

1210// ---------------------------------------------------------------------------
1211// `unsafe impl Send + Sync` for EGraph
1212// ---------------------------------------------------------------------------
1213//
1214// The `Backend` trait requires `Send + Sync`. `EGraph` holds a
1215// `duckdb::Connection`, which is `Send` but not `Sync` (DuckDB
1216// connections are thread-local). Within the egglog backend the
1217// connection is always accessed from a single thread (DuckDB itself
1218// is configured with `SET threads = 1` for determinism), so for the
1219// purposes of the trait-object wrapper we assert both. If a future
1220// caller drives the backend from multiple threads concurrently this
1221// promise will need re-evaluation.
1222unsafe impl Send for EGraph {}
1223unsafe impl Sync for EGraph {}
