1//! `DuckdbExternalFuncRegistry` — Phase 2 Commit 12.
2//!
3//! Storage scaffolding for user-registered primitives (`ExternalFunction`)
4//! and deferred-panic sentinels on the DuckDB backend.
5//!
6//! ## What is implemented
7//!
8//! - **Storage**: a `Vec<Slot>` indexed by `ExternalFunctionId`. Each
9//!   slot is one of:
10//!   - `Func { func: Box<dyn ExternalFunction> }` — a user-registered
11//!     primitive. Stored opaquely; not yet wired to a DuckDB VScalar
12//!     UDF (see "Open issues" below).
13//!   - `Panic { message: String }` — a deferred-panic sentinel
14//!     registered via `Backend::new_panic`. The rule path (owned by the
15//!     Commit 10 agent) translates references to a `Panic` slot into
16//!     `Action::Panic(message)` in the duckdb IR.
17//!   - `Freed` — a slot whose entry has been dropped via
18//!     `Backend::free_external_func`. The slot index is not reused so
19//!     ids stay stable.
20//!
21//! - **`new_panic`** returns a fresh id pointing at a `Panic` slot.
22//!   The compiled rule pipeline (Commit 10) inspects the slot to detect
23//!   the panic case.
24//!
25//! - **`register_external_func`** returns a fresh id pointing at a
26//!   `Func` slot. `free_external_func` marks it `Freed`.
27//!
28//! ## What is NOT implemented (deferred)
29//!
30//! Wiring a registered `ExternalFunction` to a DuckDB VScalar UDF so
31//! that the function is callable from rule SQL is **deferred**. The
32//! scaffolding here only stores the function; it does not register a
33//! UDF with DuckDB. As a result, primitives registered through the
34//! trait are unreachable from compiled rules on DuckDB.
35//!
36//! ### Rationale for the deferral
37//!
38//! Wiring a general `ExternalFunction` (which expects an
39//! `&mut ExecutionState`) into a DuckDB VScalar UDF (which expects
40//! a vectorized `DataChunkHandle` of typed SQL values) is **complex**:
41//!
42//! 1. The UDF's signature has to be derived from the function's egglog
43//!    signature, which the trait does not surface; primitives currently
44//!    declare their schema through `prelude.rs`'s
45//!    `add_primitive_with_validator` machinery.
46//! 2. Each call has to decode raw DuckDB inputs (`i64`, `String`, etc.)
47//!    back to `Value` via the base value pool, package them, run the
48//!    primitive's `invoke`, and encode the result back to a DuckDB
49//!    value.
50//! 3. The primitive can't reenter the database synchronously (DuckDB
51//!    UDFs run inside the executor and `register_scalar_function` would
52//!    deadlock on a reentrant SQL call). This is exactly what
53//!    [`Backend::supports_inline_table_lookups`] = `false` is meant to
54//!    surface — primitives that try to call back into tables must be
55//!    gated out at registration time.
56//!
57//! Without a real `Box<dyn Backend>` caller exercising the path (the
58//! frontend doesn't route DuckDB through the trait until Commit 14),
59//! the storage-only stub here is sufficient. Commit 14 will surface any
60//! reachable callers; we revisit the VScalar wrapper then.
61//!
62//! ## Open issues for Commit 14
63//!
64//! - VScalar wrapper for `ExternalFunction` (this file's primary TODO).
65//! - Decode / encode primitives for `Value` ↔ DuckDB SQL value via the
66//!   `BaseValuePool`. The mapping is mostly trivial (identity for
67//!   inline-encodable types; intern-table lookup for the rest), but
68//!   needs to thread through the UDF entry point.
69
70use egglog_backend_trait::{ExternalFunction, ExternalFunctionId};
71use egglog_numeric_id::NumericId;
72
73/// Per-id storage slot. The `Vec<Slot>` index is the
74/// `ExternalFunctionId::rep()` value.
75#[allow(dead_code)]
76pub(crate) enum ExternalFuncSlot {
77    /// A user-registered primitive. Held opaquely; not yet wired to a
78    /// VScalar UDF. The function is read at this slot by the future
79    /// Commit 14 VScalar wrapper.
80    Func(Box<dyn ExternalFunction + 'static>),
81    /// A deferred-panic sentinel created by `Backend::new_panic`. The
82    /// rule path translates references into `Action::Panic(message)`.
83    Panic(String),
84    /// A slot whose contents were dropped by
85    /// `Backend::free_external_func`. The slot is retained (instead of
86    /// reusing the index) so existing ids remain stable.
87    Freed,
88}
89
90/// Storage for primitives + panic sentinels registered through the
91/// `Backend` trait.
92#[derive(Default)]
93pub(crate) struct DuckdbExternalFuncRegistry {
94    slots: Vec<ExternalFuncSlot>,
95    /// Per-id primitive name. `None` for slots that have no name
96    /// associated (anonymous panic sentinels, freed slots). The name
97    /// is the egglog-level operator (e.g. `"+"`, `"!="`, `"value-eq"`).
98    /// Phase 2 Commit 14 wires this up so the duckdb rule-builder can
99    /// translate `ExternalFunctionId` references to duck `Term::Prim`
100    /// calls.
101    names: Vec<Option<String>>,
102}
103
104impl DuckdbExternalFuncRegistry {
105    /// Add a user-registered primitive, returning its id.
106    pub(crate) fn add_func(
107        &mut self,
108        func: Box<dyn ExternalFunction + 'static>,
109    ) -> ExternalFunctionId {
110        let idx = self.slots.len();
111        self.slots.push(ExternalFuncSlot::Func(func));
112        self.names.push(None);
113        ExternalFunctionId::from_usize(idx)
114    }
115
116    /// Add a deferred-panic sentinel, returning its id.
117    pub(crate) fn add_panic(&mut self, message: String) -> ExternalFunctionId {
118        let idx = self.slots.len();
119        self.slots.push(ExternalFuncSlot::Panic(message));
120        self.names.push(None);
121        ExternalFunctionId::from_usize(idx)
122    }
123
124    /// Drop the entry at `id`. Future accesses see `Freed`.
125    ///
126    /// Out-of-range or already-freed ids are silently tolerated (mirrors
127    /// the bridge's `free_external_func` semantics).
128    pub(crate) fn free(&mut self, id: ExternalFunctionId) {
129        let idx = id.rep() as usize;
130        if let Some(slot) = self.slots.get_mut(idx) {
131            *slot = ExternalFuncSlot::Freed;
132        }
133        if let Some(slot_name) = self.names.get_mut(idx) {
134            *slot_name = None;
135        }
136    }
137
138    /// Associate a primitive name with the slot at `id`. Used by the
139    /// frontend's typechecker (post-`register_external_func`) so the
140    /// duckdb rule-builder can emit `Term::Prim(name, ...)` when the
141    /// id is referenced from a rule body.
142    pub(crate) fn set_name(&mut self, id: ExternalFunctionId, name: String) {
143        let idx = id.rep() as usize;
144        if idx >= self.names.len() {
145            self.names.resize(idx + 1, None);
146        }
147        self.names[idx] = Some(name);
148    }
149
150    /// Look up the primitive name stored at `id`. Returns `None` if no
151    /// name was associated (e.g. the slot is anonymous, has been
152    /// freed, or is a deferred-panic sentinel).
153    pub(crate) fn name(&self, id: ExternalFunctionId) -> Option<&str> {
154        self.names.get(id.rep() as usize)?.as_deref()
155    }
156
157    /// Borrow the slot at `id`. Returns `None` for out-of-range ids;
158    /// returns `Some(Freed)` for freed entries.
159    #[allow(dead_code)]
160    pub(crate) fn get(&self, id: ExternalFunctionId) -> Option<&ExternalFuncSlot> {
161        self.slots.get(id.rep() as usize)
162    }
163
164    /// Clone the registered primitive at `id`, if the slot holds a `Func`.
165    /// Used by `Backend::eval_prim` to invoke the primitive against an
166    /// ephemeral execution state (pure primitives — comparisons, arithmetic —
167    /// do not touch table state, so an empty database suffices). The clone is
168    /// cheap (most primitives are zero-sized closures).
169    pub(crate) fn clone_func(
170        &self,
171        id: ExternalFunctionId,
172    ) -> Option<Box<dyn ExternalFunction + 'static>> {
173        match self.slots.get(id.rep() as usize)? {
174            ExternalFuncSlot::Func(f) => Some(dyn_clone::clone_box(f.as_ref())),
175            _ => None,
176        }
177    }
178
179    /// Look up the panic message stored at `id`. Returns `None` if `id`
180    /// does not refer to a `Panic` slot (e.g. was freed, or holds a
181    /// regular `Func`).
182    ///
183    /// Used by the Commit 10 rule-builder to translate a
184    /// `call_external_func(panic_id, …)` reference into
185    /// `Action::Panic(message)` in the duckdb IR.
186    #[allow(dead_code)]
187    pub(crate) fn panic_message(&self, id: ExternalFunctionId) -> Option<&str> {
188        match self.get(id)? {
189            ExternalFuncSlot::Panic(msg) => Some(msg.as_str()),
190            _ => None,
191        }
192    }
193}

199#[cfg(test)]
200mod tests {
201    use super::*;
202    use egglog_backend_trait::{ExecutionState, Value};
203
204    #[derive(Clone)]
205    struct NoOpFunc;
206
207    impl ExternalFunction for NoOpFunc {
208        fn invoke(&self, _state: &mut ExecutionState, _args: &[Value]) -> Option<Value> {
209            None
210        }
211    }
212
213    /// Sanity-check the registry's id assignment, panic recall, and
214    /// free behavior.
215    #[test]
216    fn registry_assigns_distinct_ids_and_recalls() {
217        let mut reg = DuckdbExternalFuncRegistry::default();
218        let f_id = reg.add_func(Box::new(NoOpFunc));
219        let p_id = reg.add_panic("kaboom".into());
220        assert_ne!(f_id, p_id);
221
222        // Slots are populated.
223        assert!(matches!(reg.get(f_id), Some(ExternalFuncSlot::Func(_))));
224        assert!(matches!(reg.get(p_id), Some(ExternalFuncSlot::Panic(_))));
225
226        assert_eq!(reg.panic_message(p_id), Some("kaboom"));
227        assert_eq!(reg.panic_message(f_id), None);
228
229        // Free the function slot; the panic slot is unaffected.
230        reg.free(f_id);
231        assert!(matches!(reg.get(f_id), Some(ExternalFuncSlot::Freed)));
232        assert_eq!(reg.panic_message(p_id), Some("kaboom"));
233    }
234
235    /// Out-of-range and already-freed ids do not panic.
236    #[test]
237    fn registry_handles_invalid_ids() {
238        let mut reg = DuckdbExternalFuncRegistry::default();
239        // Free of an empty registry.
240        reg.free(ExternalFunctionId::from_usize(0));
241        assert!(reg.get(ExternalFunctionId::from_usize(0)).is_none());
242
243        let id = reg.add_panic("oops".into());
244        reg.free(id);
245        reg.free(id);
246        assert!(matches!(reg.get(id), Some(ExternalFuncSlot::Freed)));
247    }
248}
