1//! DuckDB-backed executor for a small subset of egglog's resolved IR.
2//!
3//! Phase 1.1 scope: relations + functions with outputs and merge
4//! modes (`:merge old`, `:merge new`). No term encoding, no UF, no
5//! primitives. See `../duckdb-backend-plan.md` for the full plan.
6//!
7//! Design notes:
8//! - One DuckDB table per registered relation/function.
9//! - Every table carries a `ts BIGINT NOT NULL` column for seminaive.
10//! - `next_ts` and `last_run_at_<rule>` live in Rust state; they are
11//!   bind parameters in generated SQL, never database rows.
12//! - Each rule with N function-table atoms compiles to N seminaive
13//!   variants (one per focused atom), emitted as separate prepared
14//!   `INSERT INTO target SELECT ...` statements with appropriate
15//!   `ON CONFLICT` clauses depending on merge mode.
16

28/// DuckDB scalar UDF that wraps `UfTable::find_ro`. Registered as
29/// `duck_uf_<sort>_find(x BIGINT) -> BIGINT` for each native-UF
30/// function. Uses the read-only walk (no path compression) so the
31/// UDF can run from inside a SELECT without taking a write lock
32/// on the table; we re-compress periodically off the hot path.
33struct UfFindScalar;
34
35impl VScalar for UfFindScalar {
36    type State = Arc<Mutex<UfTable>>;
37
38    unsafe fn invoke(
39        state: &Self::State,
40        input: &mut DataChunkHandle,
41        output: &mut dyn WritableVector,
42    ) -> std::result::Result<(), Box<dyn std::error::Error>> {
43        unsafe {
44            let n = input.len();
45            let input_vec = input.flat_vector(0);
46            let inputs = input_vec.as_slice_with_len::<i64>(n);
47            let mut output_vec = output.flat_vector();
48            let outputs = output_vec.as_mut_slice::<i64>();
49            let uf = state.lock().unwrap();
50            for i in 0..n {
51                outputs[i] = uf.find_ro(inputs[i]);
52            }
53            Ok(())
54        }
55    }
56
57    fn signatures() -> Vec<ScalarFunctionSignature> {
58        vec![ScalarFunctionSignature::exact(
59            vec![LogicalTypeHandle::from(LogicalTypeId::Bigint)],
60            LogicalTypeHandle::from(LogicalTypeId::Bigint),
61        )]
62    }
63}

65/// State shared by all bigint/bigrat-related UDFs: a snapshot of the
66/// base-value pool whose interior intern tables are Arc-shared with
67/// the EGraph's pool, plus the resolved `BaseValueId`s for `Z` and
68/// (when registered) `Q`. Cloning is cheap and preserves intern
69/// sharing — every clone agrees on Value handles, so a `Q` interned
70/// by the UDF at row-time appears in the EGraph's pool by the time a
71/// rule action consumes it.
72///
73/// `bigrat_ty` is `None` until `BigRatSort` has been registered. Since
74/// `BigIntSort` registers first (and Herbie programs need both to do
75/// anything), the bigrat UDFs delay registration to
76/// `set_external_func_name`-time when the `Q` id is known to exist.
77#[derive(Clone)]
78struct BigPoolState {
79    pool: base_values::DuckdbBaseValuePool,
80    bigint_ty: egglog_backend_trait::BaseValueId,
81    bigrat_ty: Option<egglog_backend_trait::BaseValueId>,
82}

84/// UDF for the `from-string` primitive (egglog's `S -> Z` constructor
85/// for `BigInt`). Takes a VARCHAR row, parses it as a `BigInt`, and
86/// interns the result in the shared base-value pool. Returns the
87/// `Value`'s `u32` rep widened to `BIGINT`.
88///
89/// On parse failure: the bridge backend's `from-string` is fallible
90/// and returns `None`; we emit `NULL` so a downstream rule action
91/// referencing the result drops out via standard SQL NULL propagation.
92struct FromStringScalar;
93
94impl VScalar for FromStringScalar {
95    type State = BigPoolState;
96
97    unsafe fn invoke(
98        state: &Self::State,
99        input: &mut DataChunkHandle,
100        output: &mut dyn WritableVector,
101    ) -> std::result::Result<(), Box<dyn std::error::Error>> {
102        unsafe {
103            use egglog_backend_trait::{BaseValuePool, Value};
104            use egglog_numeric_id::NumericId;
105            use num::BigInt;
106            use std::any::TypeId;
107            use std::str::FromStr;
108
109            // Strings on duck are stored as interned `Boxed<String>`
110            // handles in BIGINT columns (matches bridge encoding). So
111            // this UDF's input is a BIGINT handle, not a raw VARCHAR.
112            // Unwrap the handle to get the string, then parse as BigInt.
113            let string_ty = state
114                .pool
115                .get_ty_by_type_id(TypeId::of::<egglog_core_relations::Boxed<String>>());
116
117            let n = input.len();
118            let in_vec = input.flat_vector(0);
119            let in_slice = in_vec.as_slice_with_len::<i64>(n);
120
121            let mut results: Vec<Option<i64>> = Vec::with_capacity(n);
122            for i in 0..n {
123                let handle = Value::new(in_slice[i] as u32);
124                let boxed = state.pool.unwrap_dyn(string_ty, handle);
125                let s_boxed: &egglog_core_relations::Boxed<String> = boxed
126                    .downcast_ref()
127                    .ok_or_else(|| -> Box<dyn std::error::Error> {
128                        "FromStringScalar: handle did not refer to a Boxed<String>".into()
129                    })?;
130                // Mirror the bridge's `-?>` semantics: parse failure
131                // means the rule firing should not produce a value.
132                // Encode as NULL so downstream actions/filters drop
133                // rows that resolve to NULL.
134                results.push(BigInt::from_str(s_boxed.0.as_str()).ok().map(|bi| {
135                    let z = egglog_core_relations::Boxed::new(bi);
136                    let val = state.pool.intern_dyn(state.bigint_ty, Box::new(z));
137                    val.rep() as i64
138                }));
139            }
140
141            let mut out_vec = output.flat_vector();
142            {
143                let out_slice = out_vec.as_mut_slice::<i64>();
144                for (i, r) in results.iter().enumerate() {
145                    out_slice[i] = r.unwrap_or(0);
146                }
147            }
148            for (i, r) in results.iter().enumerate() {
149                if r.is_none() {
150                    out_vec.set_null(i);
151                }
152            }
153            Ok(())
154        }
155    }
156
157    fn signatures() -> Vec<ScalarFunctionSignature> {
158        vec![ScalarFunctionSignature::exact(
159            vec![LogicalTypeHandle::from(LogicalTypeId::Bigint)],
160            LogicalTypeHandle::from(LogicalTypeId::Bigint),
161        )]
162    }
163}

165/// UDF for the `bigrat` primitive (egglog's `Z × Z -> Q` constructor
166/// for `BigRat`). Inputs are two `BIGINT` columns holding `Z` handles
167/// produced by `__egglog_from_string` (or any other source); the UDF
168/// unwraps each to a `num::BigInt`, builds a `num::BigRational`, and
169/// interns it as `Q = Boxed<BigRational>` in the shared pool.
170///
171/// State requires `bigrat_ty` to be `Some(...)` — the EGraph's
172/// `set_external_func_name` path checks this before registering the
173/// UDF (BigRatSort always registers before any `bigrat` call site).
174struct BigratScalar;
175
176impl VScalar for BigratScalar {
177    type State = BigPoolState;

179    unsafe fn invoke(
180        state: &Self::State,
181        input: &mut DataChunkHandle,
182        output: &mut dyn WritableVector,
183    ) -> std::result::Result<(), Box<dyn std::error::Error>> {
184        unsafe {
185            use egglog_backend_trait::{BaseValuePool, Value};
186            use egglog_numeric_id::NumericId;
187            use num::{BigInt, BigRational};
188
189            let bigrat_ty = state
190                .bigrat_ty
191                .ok_or_else(|| -> Box<dyn std::error::Error> {
192                    "BigratScalar: bigrat type id missing (BigRatSort not registered before \
193                 __egglog_bigrat was called)"
194                        .into()
195                })?;
196
197            let n = input.len();
198            let num_col = input.flat_vector(0);
199            let den_col = input.flat_vector(1);
200            let num_slice = num_col.as_slice_with_len::<i64>(n);
201            let den_slice = den_col.as_slice_with_len::<i64>(n);
202
203            let mut out_vec = output.flat_vector();
204            let out_slice = out_vec.as_mut_slice::<i64>();
205
206            for i in 0..n {
207                let num_val = Value::new(num_slice[i] as u32);
208                let den_val = Value::new(den_slice[i] as u32);
209                let num_boxed = state.pool.unwrap_dyn(state.bigint_ty, num_val);
210                let den_boxed = state.pool.unwrap_dyn(state.bigint_ty, den_val);
211                let num_bigint: &egglog_core_relations::Boxed<BigInt> = num_boxed
212                    .downcast_ref()
213                    .ok_or_else(|| -> Box<dyn std::error::Error> {
214                        "BigratScalar: numerator was not a Boxed<BigInt>".into()
215                    })?;
216                let den_bigint: &egglog_core_relations::Boxed<BigInt> = den_boxed
217                    .downcast_ref()
218                    .ok_or_else(|| -> Box<dyn std::error::Error> {
219                        "BigratScalar: denominator was not a Boxed<BigInt>".into()
220                    })?;
221                let q = egglog_core_relations::Boxed::new(BigRational::new(
222                    num_bigint.0.clone(),
223                    den_bigint.0.clone(),
224                ));
225                let val = state.pool.intern_dyn(bigrat_ty, Box::new(q));
226                out_slice[i] = val.rep() as i64;
227            }
228            Ok(())
229        }
230    }
231
232    fn signatures() -> Vec<ScalarFunctionSignature> {
233        vec![ScalarFunctionSignature::exact(
234            vec![
235                LogicalTypeHandle::from(LogicalTypeId::Bigint),
236                LogicalTypeHandle::from(LogicalTypeId::Bigint),
237            ],
238            LogicalTypeHandle::from(LogicalTypeId::Bigint),
239        )]
240    }

243/// BigRat operations that need a UDF wrapper because their egglog
244/// name (`+`, `<`, `round`, ...) is overloaded across sorts and DuckDB
245/// SQL has no native concept of arbitrary-precision rationals. The
246/// frontend renames each call site to a sort-specific duck name (e.g.
247/// `bigrat-add`); each variant of this enum corresponds to one such
248/// name and selects the operation the UDF performs.
249///
250/// Unary ops (`Neg`/`Abs`/…) take one `Q` handle and return one. Binary
251/// ops (`Add`/…/`Pow`) take two. Comparison ops (`Lt`/`Gt`/…) take two
252/// `Q` handles and return BOOLEAN. `Numer`/`Denom` return a `Z` handle.
253/// `ToF64` returns a DOUBLE.
254#[derive(Copy, Clone, Debug)]
255enum BigratOp {
256    // Q × Q → Q (fallible for +, -, *, /, pow via checked_*)
257    Add,
258    Sub,
259    Mul,
260    Div,
261    Min,
262    Max,
263    Pow,
264    // Q → Q
265    Neg,
266    Abs,
267    Floor,
268    Ceil,
269    Round,
270    Sqrt,
271    Log,
272    Cbrt,
273    // Q × Q → BOOL
274    Lt,
275    Gt,
276    Le,
277    Ge,
278    // Q → Z
279    Numer,
280    Denom,
281    // Q → f64
282    ToF64,
283}
284
285impl BigratOp {
286    fn from_duck_name(name: &str) -> Option<Self> {
287        Some(match name {
288            "bigrat-add" => BigratOp::Add,
289            "bigrat-sub" => BigratOp::Sub,
290            "bigrat-mul" => BigratOp::Mul,
291            "bigrat-div" => BigratOp::Div,
292            "bigrat-min" => BigratOp::Min,
293            "bigrat-max" => BigratOp::Max,
294            "bigrat-pow" => BigratOp::Pow,
295            "bigrat-neg" => BigratOp::Neg,
296            "bigrat-abs" => BigratOp::Abs,
297            "bigrat-floor" => BigratOp::Floor,
298            "bigrat-ceil" => BigratOp::Ceil,
299            "bigrat-round" => BigratOp::Round,
300            "bigrat-sqrt" => BigratOp::Sqrt,
301            "bigrat-log" => BigratOp::Log,
302            "bigrat-cbrt" => BigratOp::Cbrt,
303            "bigrat-lt" => BigratOp::Lt,
304            "bigrat-gt" => BigratOp::Gt,
305            "bigrat-le" => BigratOp::Le,
306            "bigrat-ge" => BigratOp::Ge,
307            "bigrat-numer" => BigratOp::Numer,
308            "bigrat-denom" => BigratOp::Denom,
309            "bigrat-to-f64" => BigratOp::ToF64,
310            _ => return None,
311        })
312    }
313
314    fn is_unary(&self) -> bool {
315        use BigratOp::*;
316        matches!(
317            self,
318            Neg | Abs | Floor | Ceil | Round | Sqrt | Log | Cbrt | Numer | Denom | ToF64
319        )
320    }
321
322    fn is_comparison(&self) -> bool {
323        use BigratOp::*;
324        matches!(self, Lt | Gt | Le | Ge)
325    }
326
327    fn returns_z(&self) -> bool {
328        matches!(self, BigratOp::Numer | BigratOp::Denom)
329    }
330
331    fn returns_f64(&self) -> bool {
332        matches!(self, BigratOp::ToF64)
333    }
334}

347/// Run a unary or binary BigRat → BigRat operation. Returns
348/// `Some(Q)` on success, `None` for fallible ops on bad inputs
349/// (division by zero, `pow` with fractional exponent, etc.) so the
350/// UDF can emit SQL NULL and downstream rules drop out via NULL
351/// propagation. Mirrors the bridge-side closures in
352/// `egglog::sort::bigrat::register_primitives`.
353fn run_bigrat_q(op: BigratOp, args: &[num::BigRational]) -> Option<num::BigRational> {
354    use num::traits::{One, Signed, Zero};
355    use num::{BigInt, BigRational};
356    let one = || BigRational::one();
357    let zero = || BigRational::zero();
358    let _ = one;
359    let _ = zero;
360    match op {
361        BigratOp::Add => Some(&args[0] + &args[1]),
362        BigratOp::Sub => Some(&args[0] - &args[1]),
363        BigratOp::Mul => Some(&args[0] * &args[1]),
364        BigratOp::Div => {
365            if args[1].is_zero() {
366                None
367            } else {
368                Some(&args[0] / &args[1])
369            }
370        }
371        BigratOp::Min => Some(args[0].clone().min(args[1].clone())),
372        BigratOp::Max => Some(args[0].clone().max(args[1].clone())),
373        BigratOp::Pow => {
374            let a = &args[0];
375            let b = &args[1];
376            if !b.is_integer() {
377                return None;
378            }
379            if a.is_zero() {
380                if b.is_zero() {
381                    return Some(BigRational::one());
382                }
383                if b.numer() > &BigInt::from(0) {
384                    return Some(BigRational::zero());
385                }
386                return None;
387            }
388            let is_neg = b.numer() < &BigInt::from(0);
389            let adj_base = if is_neg { a.recip() } else { a.clone() };
390            let adj_exp = if is_neg {
391                BigRational::new(-b.numer().clone(), b.denom().clone())
392            } else {
393                b.clone()
394            };
395            let exp_i64 = adj_exp.numer().to_string().parse::<i64>().ok()?;
396            if exp_i64 < 0 {
397                return None;
398            }
399            let exp = exp_i64 as usize;
400            num::traits::checked_pow(adj_base, exp)
401        }
402        BigratOp::Neg => Some(-args[0].clone()),
403        BigratOp::Abs => Some(BigRational::new(
404            args[0].numer().abs().clone(),
405            args[0].denom().clone(),
406        )),
407        BigratOp::Floor => Some(args[0].floor()),
408        BigratOp::Ceil => Some(args[0].ceil()),
409        BigratOp::Round => Some(args[0].round()),
410        BigratOp::Sqrt => {
411            // Closed-form sqrt only if both numer and denom are
412            // perfect squares — matches the bridge's behavior in
413            // `bigrat.rs`.
414            if args[0].numer() < &BigInt::from(0) {
415                return None;
416            }
417            let n_sqrt = args[0].numer().sqrt();
418            let d_sqrt = args[0].denom().sqrt();
419            if &(n_sqrt.clone() * n_sqrt.clone()) == args[0].numer()
420                && &(d_sqrt.clone() * d_sqrt.clone()) == args[0].denom()
421            {
422                Some(BigRational::new(n_sqrt, d_sqrt))
423            } else {
424                None
425            }
426        }
427        BigratOp::Log => {
428            // Bridge only handles `log(1) = 0`; everything else
429            // panics. Match that conservatively (just return None).
430            if args[0].numer() == &BigInt::from(1) && args[0].denom() == &BigInt::from(1) {
431                Some(BigRational::zero())
432            } else {
433                None
434            }
435        }
436        BigratOp::Cbrt => {
437            // Bridge only handles `cbrt(1) = 1`.
438            if args[0].numer() == &BigInt::from(1) && args[0].denom() == &BigInt::from(1) {
439                Some(BigRational::one())
440            } else {
441                None
442            }
443        }
444        BigratOp::Lt
445        | BigratOp::Gt
446        | BigratOp::Le
447        | BigratOp::Ge
448        | BigratOp::Numer
449        | BigratOp::Denom
450        | BigratOp::ToF64 => unreachable!("not a Q-returning op: {op:?}"),
451    }
452}

454/// Unwrap a `Q`-handle (BIGINT) row into a `BigRational`.
455fn unwrap_bigrat(
456    pool: &base_values::DuckdbBaseValuePool,
457    bigrat_ty: egglog_backend_trait::BaseValueId,
458    raw: i64,
459) -> std::result::Result<num::BigRational, Box<dyn std::error::Error>> {
460    use egglog_backend_trait::{BaseValuePool, Value};
461    use egglog_numeric_id::NumericId;
462    let val = Value::new(raw as u32);
463    let boxed = pool.unwrap_dyn(bigrat_ty, val);
464    let q: &egglog_core_relations::Boxed<num::BigRational> =
465        boxed
466            .downcast_ref()
467            .ok_or_else(|| -> Box<dyn std::error::Error> {
468                "expected Boxed<BigRational> from pool".into()
469            })?;
470    Ok(q.0.clone())
471}

1162/// Quote a SQL identifier with double quotes, escaping any embedded
1163/// double quote. Necessary because egglog identifiers can contain
1164/// `@`, `$`, etc., which DuckDB rejects in unquoted form.
1165///
1166/// DuckDB compares quoted identifiers case-insensitively (so
1167/// `"AConst"` and `"aConst"` collide as table names). We encode the
1168/// case bits with a two-step escape so distinct case-shapes always
1169/// produce distinct lowered identifiers:
1170///   1. `_` → `__` (escape literal underscore so it can't be
1171///      confused with the uppercase marker).
1172///   2. ASCII uppercase `X` → `_<lowercase x>`.
1173pub fn q(name: &str) -> String {
1174    let mut out = String::with_capacity(name.len() + 4);
1175    for c in name.chars() {
1176        if c == '_' {
1177            out.push('_');
1178            out.push('_');
1179        } else if c.is_ascii_uppercase() {
1180            out.push('_');
1181            out.push(c.to_ascii_lowercase());
1182        } else {
1183            out.push(c);
1184        }
1185    }
1186    format!("\"{}\"", out.replace('"', "\"\""))
1187}

1191/// list is non-empty and an empty string otherwise. Used to safely
1192/// concatenate column lists with the always-present trailing `ts`
1193/// column without producing `(, ts)` when there are no other cols.

1199/// duckdb-rust 1.5 didn't actually keep the planned representation
1200/// across calls in our setup — overall wall time roughly doubled.
1201/// Plain `execute` with string substitution beats it.
1202pub(crate) fn exec_bound(conn: &Connection, sql: &str, last: i64, cur: i64) -> Result<usize> {
1203    let bound = sql
1204        .replace("?1", &last.to_string())
1205        .replace("?2", &cur.to_string());
1206    if std::env::var("DUCK_TRACE_BOUND").is_ok() {
1207        eprintln!("[duck/bound last={last} cur={cur}] {bound}");
1208    }
1209    Ok(conn.execute(&bound, [])?)
1210}

1212/// Execute all variants of a single rule for one seminaive
1213/// iteration. Pulled out of `run_iteration_in*` so both entry points
1214/// share the same shape and so per-rule timing accounting lives in
1215/// one place. Borrows `self.rules` (read-only) and the timing fields
1216/// disjointly — the caller passes those as `&mut` references.
1217#[allow(clippy::too_many_arguments)]
1218fn run_rule_variants(
1219    rule: &CompiledRule,
1220    last: i64,
1221    cur: i64,
1222    conn: &Connection,
1223    time_mat_ns: &mut u64,
1224    time_mat_act_ns: &mut u64,
1225    time_act_ns: &mut u64,
1226    rules_affected: &mut u64,
1227    rule_perf_ns: &mut HashMap<String, (u64, u64)>,
1228    table_watermarks: &mut HashMap<String, i64>,
1229    rules_skipped: &mut u64,
1230) -> Result<usize> {
1231    // Watermark gate: if no body table has had inserts since this
1232    // rule last ran, every variant's seminaive predicate is empty.
1233    // Skip the whole rule. Falls through for rules with no body
1234    // (shouldn't happen, but cheap to handle).
1235    if !rule.body_tables.is_empty() {
1236        let mut any_fresh = false;
1237        for t in &rule.body_tables {
1238            let wm = table_watermarks.get(t).copied().unwrap_or(0);
1239            if wm >= last {
1240                any_fresh = true;
1241                break;
1242            }
1243        }
1244        if !any_fresh {
1245            *rules_skipped = rules_skipped.wrapping_add(1);
1246            return Ok(0);
1247        }
1248    }
1249
1250    let mut total: usize = 0;
1251    let mut rule_mat_ns: u64 = 0;
1252    let mut rule_act_ns: u64 = 0;
1253    let trace = std::env::var("DUCK_TRACE_RULE_AFFECTED").is_ok();
1254    let trace_sql = std::env::var("DUCK_TRACE_SQL").is_ok();
1255    for variant in &rule.variants {
1256        // Gated variant: skip unless its trigger table has been
1257        // updated since this rule last ran. Used by the native-UF
1258        // recovery variants, which only need to fire when a new
1259        // union has landed.
1260        if let Some(gt) = &variant.gate_table {
1261            let wm = table_watermarks.get(gt).copied().unwrap_or(0);
1262            if wm < last {
1263                continue;
1264            }
1265        }
1266        let mut skip_actions = false;
1267        if let Some(mat_sql_template) = &variant.materialize {
1268            if trace_sql {
1269                eprintln!("[duck/mat] {mat_sql_template}");
1270            }
1271            let t0 = std::time::Instant::now();
1272            exec_bound(conn, mat_sql_template, last, cur)?;
1273            let dt = t0.elapsed().as_nanos() as u64;
1274            *time_mat_ns = time_mat_ns.wrapping_add(dt);
1275            rule_mat_ns = rule_mat_ns.wrapping_add(dt);
1276            // Skip the action set if the materialize produced
1277            // zero rows. CREATE TABLE AS SELECT is DDL so its
1278            // `execute` return is 0 regardless of SELECT row
1279            // count; we use a follow-up COUNT(*) on the temp
1280            // table to actually find out. The count is O(1)
1281            // against DuckDB's row-count metadata.
1282            if let Some(temp) = &variant.temp_table {
1283                let count_sql = format!("SELECT COUNT(*) FROM {}", q(temp));
1284                let n: i64 = conn.query_row(&count_sql, [], |r| r.get(0))?;
1285                if n == 0 {
1286                    skip_actions = true;
1287                }
1288            }
1289        }
1290        if skip_actions {
1291            continue;
1292        }
1293        for act in &variant.actions {
1294            if trace_sql {
1295                eprintln!(
1296                    "[duck/{}] {}",
1297                    if variant.materialize.is_some() {
1298                        "mat-act"
1299                    } else {
1300                        "act"
1301                    },
1302                    act.sql,
1303                );
1304            }
1305            let t0 = std::time::Instant::now();
1306            let n = exec_bound(conn, &act.sql, last, cur)?;
1307            let dt = t0.elapsed().as_nanos() as u64;
1308            if variant.materialize.is_some() {
1309                *time_mat_act_ns = time_mat_act_ns.wrapping_add(dt);
1310            } else {
1311                *time_act_ns = time_act_ns.wrapping_add(dt);
1312            }
1313            rule_act_ns = rule_act_ns.wrapping_add(dt);
1314            *rules_affected = rules_affected.wrapping_add(n as u64);
1315            total += n;
1316            if n > 0 {
1317                if let Some(target) = &act.target {
1318                    let e = table_watermarks.entry(target.clone()).or_insert(0);
1319                    if cur > *e {
1320                        *e = cur;
1321                    }
1322                }
1323                if trace {
1324                    eprintln!("[duck/rule_n] {} +{n}", rule.name);
1325                }
1326            }
1327        }
1328    }
1329    if rule_mat_ns != 0 || rule_act_ns != 0 {
1330        let e = rule_perf_ns.entry(rule.name.clone()).or_insert((0, 0));
1331        e.0 = e.0.wrapping_add(rule_mat_ns);
1332        e.1 = e.1.wrapping_add(rule_act_ns);
1333    }
1334    Ok(total)
1335}

1400/// A literal value usable in seed inserts and `check`/`lookup`.
1401///
1402/// `Hash` / `Eq` / `PartialEq` are derived so `compile::dedupe_body_atoms`
1403/// can key on `(name, input args)` to collapse duplicate body atoms.
1404/// `f64` doesn't implement `Eq`/`Hash` natively; we wrap it through
1405/// `to_bits()` in the impls below so the trait bounds are satisfied
1406/// for the dedup pass (NaN-equality semantics don't matter here:
1407/// rule bodies never contain NaN literals).
1408#[derive(Debug, Clone)]
1409pub enum Literal {
1410    I64(i64),
1411    Bool(bool),
1412    F64(f64),
1413    Str(String),
1414}
1415
1416impl PartialEq for Literal {
1417    fn eq(&self, other: &Self) -> bool {
1418        match (self, other) {
1419            (Literal::I64(a), Literal::I64(b)) => a == b,
1420            (Literal::Bool(a), Literal::Bool(b)) => a == b,
1421            (Literal::F64(a), Literal::F64(b)) => a.to_bits() == b.to_bits(),
1422            (Literal::Str(a), Literal::Str(b)) => a == b,
1423            _ => false,
1424        }
1425    }
1426}
1427impl Eq for Literal {}
1428impl std::hash::Hash for Literal {
1429    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
1430        match self {
1431            Literal::I64(v) => {
1432                0u8.hash(state);
1433                v.hash(state);
1434            }
1435            Literal::Bool(v) => {
1436                1u8.hash(state);
1437                v.hash(state);
1438            }
1439            Literal::F64(v) => {
1440                2u8.hash(state);
1441                v.to_bits().hash(state);
1442            }
1443            Literal::Str(v) => {
1444                3u8.hash(state);
1445                v.hash(state);
1446            }
1447        }
1448    }
1449}
1450
1451impl ToSql for Literal {
1452    fn to_sql(&self) -> duckdb::Result<duckdb::types::ToSqlOutput<'_>> {
1453        match self {
1454            Literal::I64(i) => i.to_sql(),
1455            Literal::Bool(b) => b.to_sql(),
1456            Literal::F64(f) => f.to_sql(),
1457            Literal::Str(s) => s.as_str().to_sql(),
1458        }
1459    }
1460}

1496/// A body atom of a rule.

1514/// A rule action.

1517    /// Insert a row. The trailing arg is the output value for

1540    /// `(panic msg)` action: raise a runtime error if the rule body

1804    pub fn new() -> Result<Self> {
1805        let conn = Connection::open_in_memory()?;
1806        // Tuning: DuckDB defaults to row-order preservation, which
1807        // uses extra memory to track insertion order through joins.
1808        // We don't care about row order — egglog is set-semantics.
1809        // Disabling drops a sizeable chunk of per-iteration overhead
1810        // on workloads that do many INSERT…SELECT.
1811        conn.execute("SET preserve_insertion_order = false", [])?;
1812        // Herbie egglog dumps inline deeply-nested `(bigrat
1813        // (from-string …) (from-string …))` constants into rule
1814        // bodies; the resulting WHERE clauses + SELECT lists can
1815        // easily blow past DuckDB's default expression-depth cap of
1816        // 1000. Bump it generously so taylor*.egg dumps with 100+
1817        // levels of nesting compile.
1818        conn.execute("SET max_expression_depth = 100000", [])?;
1819        // Determinism: parallel execution can produce rows from a
1820        // SELECT in any order. With hash-cons (or any rule action
1821        // that side-effects via nextval()), per-row evaluation order
1822        // determines which sequence values get burned for which
1823        // matches, which affects unification chains and thus tuple
1824        // counts at bounded iteration. Force single-threaded so our
1825        // output is reproducible run-to-run.

1828        // microbenchmark showed per-statement overhead at ~35-43us

1830        // math-microbenchmark; the actual SQL work is. Leaving the

1834        // rows with the same inputs are intentional — congruence

2251        match name {
2252            "from-string" => {
2253                let s = as_str(args.first()?)?;
2254                let bi = BigInt::from_str(&s).ok()?;
2255                Some(intern_z(bi))
2256            }
2257            "bigrat" => {
2258                let n_raw = as_i64(args.first()?)?;
2259                let d_raw = as_i64(args.get(1)?)?;
2260                let n_val = Value::new(n_raw as u32);
2261                let d_val = Value::new(d_raw as u32);
2262                let n_boxed = pool_dyn.unwrap_dyn(bigint_ty, n_val);
2263                let d_boxed = pool_dyn.unwrap_dyn(bigint_ty, d_val);
2264                let n: &egglog_core_relations::Boxed<BigInt> = n_boxed.downcast_ref()?;
2265                let d: &egglog_core_relations::Boxed<BigInt> = d_boxed.downcast_ref()?;
2266                let q = BigRational::new(n.0.clone(), d.0.clone());
2267                intern_q(q)
2268            }
2269            other => {
2270                let op = BigratOp::from_duck_name(other)?;
2271                // Comparison ops fold to a Bool literal — emit it
2272                // and the rule_builder's downstream Filter on this
2273                // term sees `Lit(Bool(true/false))`.
2274                if op.is_comparison() {
2275                    let a = unwrap_q(as_i64(args.first()?)?)?;
2276                    let b = unwrap_q(as_i64(args.get(1)?)?)?;
2277                    let r = match op {
2278                        BigratOp::Lt => a < b,
2279                        BigratOp::Gt => a > b,
2280                        BigratOp::Le => a <= b,
2281                        BigratOp::Ge => a >= b,
2282                        _ => unreachable!(),
2283                    };
2284                    return Some(Term::Lit(Literal::Bool(r)));
2285                }
2286                // numer/denom fold to Z handles.
2287                if op.returns_z() {
2288                    let q = unwrap_q(as_i64(args.first()?)?)?;
2289                    let bi = match op {
2290                        BigratOp::Numer => q.numer().clone(),
2291                        BigratOp::Denom => q.denom().clone(),
2292                        _ => unreachable!(),
2293                    };
2294                    return Some(intern_z(bi));
2295                }
2296                // to-f64 folds to F64 literal.
2297                if op.returns_f64() {
2298                    use num::ToPrimitive;
2299                    let q = unwrap_q(as_i64(args.first()?)?)?;
2300                    return Some(Term::Lit(Literal::F64(q.to_f64()?)));
2301                }
2302                // Unary Q→Q: round, sqrt, neg, abs, ceil, floor, log, cbrt
2303                if op.is_unary() {
2304                    let q = unwrap_q(as_i64(args.first()?)?)?;
2305                    let result = run_bigrat_q(op, &[q])?;
2306                    return intern_q(result);
2307                }
2308                // Binary Q×Q→Q: add, sub, mul, div, min, max, pow
2309                let a = unwrap_q(as_i64(args.first()?)?)?;
2310                let b = unwrap_q(as_i64(args.get(1)?)?)?;
2311                let result = run_bigrat_q(op, &[a, b])?;
2312                intern_q(result)
2313            }
2314        }
2315    }

2317    /// Refresh the row-count snapshot consulted by the `get-size!`
2318    /// UDF. Queries each function table once via `SELECT COUNT(*)`
2319    /// and writes the result into the shared `table_sizes` map.
2320    /// Internal-symbol-prefixed tables (term-encoding bookkeeping)
2321    /// are excluded so the result matches the bridge `get-size!`
2322    /// primitive, which applies the same filter.
2323    ///
2324    /// Callers should invoke this immediately before each
2325    /// existence-check query that may reference `get-size!`. The
2326    /// snapshot stays valid for the duration of that query.
2327    pub fn refresh_table_sizes(&self) -> Result<()> {
2328        const INTERNAL_PREFIX: &str = "@";
2329        // Filter to non-internal tables. Order is stable so the
2330        // returned columns line up with `names` for read-out.
2331        let names: Vec<&str> = self
2332            .functions
2333            .keys()
2334            .filter(|n| !n.starts_with(INTERNAL_PREFIX))
2335            .map(|s| s.as_str())
2336            .collect();
2337        if names.is_empty() {
2338            let mut guard = self
2339                .table_sizes
2340                .lock()
2341                .map_err(|e| anyhow!("table_sizes mutex poisoned: {e}"))?;
2342            guard.clear();
2343            return Ok(());
2344        }
2345        // Batch into a single SELECT with N scalar `(SELECT COUNT(*)
2346        // FROM t)` subqueries. Avoids N round-trips per :until check
2347        // (Herbie's `(repeat 50 …)` schedules call `get-size!` once
2348        // per iteration; the per-query overhead added up to seconds).
2349        let cols: Vec<String> = names
2350            .iter()
2351            .map(|n| format!("(SELECT COUNT(*) FROM {})", q(n)))
2352            .collect();
2353        let sql = format!("SELECT {}", cols.join(", "));
2354        let mut new_sizes: HashMap<String, i64> = HashMap::new();
2355        self.conn
2356            .query_row(&sql, [], |row| {
2357                for (i, n) in names.iter().enumerate() {
2358                    let count: i64 = row.get(i)?;
2359                    new_sizes.insert((*n).to_string(), count);
2360                }
2361                Ok(())
2362            })
2363            .map_err(|e| anyhow!("refresh_table_sizes: batched COUNT failed: {e}"))?;
2364        let mut guard = self
2365            .table_sizes
2366            .lock()
2367            .map_err(|e| anyhow!("table_sizes mutex poisoned: {e}"))?;
2368        *guard = new_sizes;
2369        Ok(())
2370    }

2386    /// Pull every union assertion newer than the last sync from
2387    /// each native UF's corresponding pname (raw UF) table, apply
2388    /// them via the queue + drain protocol, and advance the
2389    /// per-UF watermark.
2390    ///
2391    /// The find UDF reads from each `UfTable` via `find_ro`; this
2392    /// method is what keeps that read view consistent with whatever
2393    /// rule actions have written into the SQL pname tables. The
2394    /// runner calls it at the top of each iteration so any UDF
2395    /// invocations within that iteration's rules see fresh state.
2396    fn sync_native_ufs(&mut self) -> Result<usize> {
2397        if self.native_ufs.is_empty() {
2398            return Ok(0);
2399        }
2400        let mut total_displaced: usize = 0;
2401        // Snapshot names to avoid double-borrowing self.
2402        let names: Vec<String> = self.native_ufs.keys().cloned().collect();
2403        for uf_name in names {
2404            let pname = match self.native_uf_pname.get(&uf_name) {
2405                Some(p) => p.clone(),
2406                None => continue,
2407            };
2408            // Watermark check: if pname's max-ts watermark hasn't
2409            // advanced past what we synced, there's nothing new.
2410            let pname_wm = self.table_watermarks.get(&pname).copied().unwrap_or(0);
2411            let last_synced = self.last_uf_sync_ts.get(&uf_name).copied().unwrap_or(-1);
2412            if pname_wm <= last_synced {
2413                continue;
2414            }
2415            let sql = format!(
2416                "SELECT c0, c1, ts FROM {} WHERE ts > {}",
2417                q(&pname),
2418                last_synced,
2419            );
2420            let uf_arc = self.native_ufs[&uf_name].clone();
2421            let mut stmt = self.conn.prepare(&sql)?;
2422            let mut rows = stmt.query([])?;
2423            let mut max_ts_seen = last_synced;
2424            let mut count: u64 = 0;
2425            {
2426                let mut uf = uf_arc.lock().unwrap();
2427                while let Some(row) = rows.next()? {
2428                    let a: i64 = row.get(0)?;
2429                    let b: i64 = row.get(1)?;
2430                    let ts: i64 = row.get(2)?;
2431                    uf.enqueue_union(a, b);
2432                    if ts > max_ts_seen {
2433                        max_ts_seen = ts;
2434                    }
2435                    count += 1;
2436                }
2437                if count > 0 {
2438                    uf.drain_pending();
2439                    let drained = uf.drain_displaced();
2440                    total_displaced += drained.len();
2441                }
2442            }
2443            self.native_uf_unions_synced = self.native_uf_unions_synced.wrapping_add(count);
2444            self.last_uf_sync_ts.insert(uf_name, max_ts_seen);
2445        }
2446        Ok(total_displaced)
2447    }

2471    /// Run `EXPLAIN ANALYZE` against the materialize SELECT of each
2472    /// variant of the top-`top` rules by total time. Uses `?1 = 0`
2473    /// and `?2 = next_ts + 1` so the scan covers all rows currently
2474    /// in each body table — that's the "everything fits" plan
2475    /// DuckDB picks once the DB is fully populated, which is what
2476    /// we care about for steady-state behavior. Returns a
2477    /// pre-formatted multi-line report.
2478    pub fn explain_top_rules(&self, top: usize) -> anyhow::Result<String> {
2479        let mut by_total: Vec<(String, u64)> = self
2480            .rule_perf_ns
2481            .iter()
2482            .map(|(n, (m, a))| (n.clone(), m + a))
2483            .collect();
2484        by_total.sort_by(|x, y| y.1.cmp(&x.1));
2485        let last: i64 = 0;
2486        let cur: i64 = self.next_ts + 1;
2487        let mut out = String::new();
2488        for (name, total) in by_total.into_iter().take(top) {
2489            let Some(rule) = self.rules.iter().find(|r| r.name == name) else {
2490                continue;
2491            };
2492            out.push_str(&format!(
2493                "=== {name} (ruleset={}, total {:.3}s) ===\n",
2494                rule.ruleset,
2495                total as f64 / 1e9,
2496            ));
2497            for (vi, variant) in rule.variants.iter().enumerate() {
2498                let Some(mat) = &variant.materialize else {
2499                    out.push_str(&format!("  variant {vi}: no materialize\n"));
2500                    continue;
2501                };
2502                // The materialize SQL is
2503                // `CREATE OR REPLACE TEMP TABLE "..." AS SELECT ...`.
2504                // Strip the wrapper so EXPLAIN ANALYZE doesn't
2505                // (re)create a temp table as a side effect.
2506                let Some(idx) = mat.find(" AS SELECT ") else {
2507                    out.push_str(&format!("  variant {vi}: unexpected SQL shape\n"));
2508                    continue;
2509                };
2510                let inner = &mat[idx + " AS ".len()..];
2511                let bound = inner
2512                    .replace("?1", &last.to_string())
2513                    .replace("?2", &cur.to_string());
2514                let explain_sql = format!("EXPLAIN ANALYZE {bound}");
2515                out.push_str(&format!("\n  --- variant {vi} ---\n"));
2516                match self.conn.prepare(&explain_sql) {
2517                    Ok(mut stmt) => {
2518                        let mut rows = stmt.query([])?;
2519                        while let Some(row) = rows.next()? {
2520                            // EXPLAIN ANALYZE returns
2521                            // (explain_key, explain_value).
2522                            let val: Result<String, _> = row.get::<_, String>(1);
2523                            if let Ok(v) = val {
2524                                for line in v.lines() {
2525                                    out.push_str("    ");
2526                                    out.push_str(line);
2527                                    out.push('\n');
2528                                }
2529                            }
2530                        }
2531                    }
2532                    Err(e) => {
2533                        out.push_str(&format!("    error: {e}\n    sql: {explain_sql}\n"));
2534                    }
2535                }
2536            }
2537            out.push('\n');
2538        }
2539        Ok(out)
2540    }

2717    fn declare(&mut self, name: &str, info: FunctionInfo) -> Result<()> {
2718        if self.functions.contains_key(name) {
2719            return Err(anyhow!("function {name} already registered"));
2720        }
2721        let col_decls: Vec<String> = info
2722            .cols
2723            .iter()
2724            .enumerate()
2725            .map(|(i, ty)| format!("c{i} {} NOT NULL", ty.sql()))
2726            .collect();
2727        // PK width:
2728        // - relations and eq-sort constructors: cover ALL columns
2729        //   (eq-sort ctors expect duplicate input keys with distinct
2730        //   IDs; relations have no output to exclude).
2731        // - functions with merge mode: cover input columns only, so
2732        //   ON CONFLICT can update the output.
2733        let pk_width = match (&info.merge, info.eq_sort_ctor) {
2734            (Some(_), false) => info.inputs_len,
2735            _ => info.cols.len(),
2736        };
2737        let pk: Vec<String> = (0..pk_width).map(|i| format!("c{i}")).collect();
2738        let pk_clause = if pk.is_empty() {
2739            String::new()
2740        } else {
2741            format!(", PRIMARY KEY ({})", pk.join(", "))
2742        };
2743        // For 0-column tables (nullary relations / nullary
2744        // constructor helpers from term encoding), DuckDB rejects
2745        // an empty leading list — skip the leading comma.
2746        let col_list = if col_decls.is_empty() {
2747            "ts BIGINT NOT NULL".to_string()
2748        } else {
2749            format!("{}, ts BIGINT NOT NULL", col_decls.join(", "))
2750        };
2751        let sql = format!("CREATE TABLE {} ({col_list}{pk_clause})", q(name));
2752        self.debug_sql("create", &sql);
2753        self.conn.execute(&sql, [])?;
2754        // We tried two flavors of auxiliary indexes here and both
2755        // were a net loss on math-microbenchmark:
2756        //  - Secondary B-tree indexes on each non-leading input
2757        //    column (intended to accelerate rebuild variants that
2758        //    join `view.c_i = uf.c0` for i > 0). DuckDB's planner
2759        //    correctly chose hash joins over index seeks (the NEW
2760        //    side is small, the build cost is amortized), so the
2761        //    indexes went unused while costing per-insert
2762        //    maintenance. mm-microbenchmark slowed by ~26%.
2763        //  - Index on the `ts` column (intended to speed up the
2764        //    seminaive `ts >= ?1 AND ts < ?2` range filter). Lost
2765        //    to DuckDB's built-in zone maps for monotonically-
2766        //    inserted ts; insert maintenance dominated. Slowed by
2767        //    ~14%.
2768        // Insert-heavy analytical workloads on DuckDB don't want
2769        // OLTP-style auxiliary indexes — the columnar storage and
2770        // zone maps already cover the access patterns we use.
2771        // Native-UF registration. We attach a UfTable + scalar UDF
2772        // when the function has `:merge (ordering-min old new)` —
2773        // term encoding emits exactly one such function per eq-sort
2774        // (the function-form UF, `@_u_f___<sort>f`). The SQL table
2775        // is still created above; writes go through it so the
2776        // existing rulesets see them. The UDF is for fast reads.
2777        let mut info = info;
2778        if self.native_uf_enabled && info.merge == Some(MergeMode::Min) {
2779            // pname (raw UF) is what term encoding writes raw union
2780            // assertions into. It's declared *before* the function-
2781            // form table and follows the `@UF_<sort>` / `@UF_<sort>f`
2782            // naming convention: pname is the function name with a
2783            // trailing `f` stripped. Resolve and check up front so
2784            // we don't half-register if the convention is broken.
2785            let pname = name.strip_suffix('f').ok_or_else(|| {

2807    /// Compile and store a rule. Compilation produces one SQL
2808    /// statement per (variant × action).
2809    pub fn add_rule(&mut self, rule: Rule) -> Result<()> {
2810        if std::env::var("DUCK_DUMP_RULES").is_ok() {
2811            eprintln!(
2812                "[duck/add_rule] name={} ruleset={}",
2813                rule.name, rule.ruleset
2814            );
2815            for atom in &rule.body {
2816                eprintln!("  body: {atom:?}");
2817            }
2818            for action in &rule.actions {
2819                eprintln!("  action: {action:?}");
2820            }
2821        }
2822        // Under `--duck-native-uf` the UF-maintenance rulesets are
2823        // skipped at every iter (the native UF maintains the same
2824        // invariants in-memory). Compile-time, some of their rules
2825        // use primitives the bridge doesn't yet support (notably
2826        // `pair` / `pair-first` / `pair-second` that proof-mode's
2827        // UF function index emits). Skipping the compile for those
2828        // rules outright avoids a hard error on programs that would
2829        // never execute the rules anyway.
2830        if self.native_uf_enabled && is_uf_maintenance_ruleset(&rule.ruleset) {
2831            return Ok(());
2832        }
2833        let mut compiled = compile::compile_rule(&rule, &self.functions, self.proofs_enabled)?;
2834        // Body tables = distinct Atom::Func names. The watermark gate
2835        // reads this set to decide whether the rule has anything new
2836        // to look at on a given iteration.
2837        //
2838        // Native-UF substitution: when a body atom references a
2839        // function with a registered native UF (e.g.
2840        // `@_u_f___mathf`), the corresponding SQL table is no
2841        // longer written to — rule actions go through the union
2842        // UDF or land in the raw pname table. So its watermark
2843        // never advances, and gating on it would permanently
2844        // block the rule after its first run. We substitute pname
2845        // (the raw UF, where union assertions land) so the gate
2846        // tracks the table that actually receives writes. Caused
2847        // a silent miss before this fix: rebuild rules would skip
2848        // because the @UF_*f watermark stayed at 0 even as new
2849        // unions accumulated in the native UF.
2850        let mut bt: Vec<String> = Vec::new();
2851        for atom in &rule.body {
2852            if let Atom::Func { name, .. } = atom {
2853                let effective = if let Some(info) = self.functions.get(name) {
2854                    if info.native_uf_udf.is_some() {
2855                        // Map @UF_<sort>f -> @UF_<sort> (pname).
2856                        self.native_uf_pname
2857                            .get(name)
2858                            .cloned()
2859                            .unwrap_or_else(|| name.clone())
2860                    } else {
2861                        name.clone()
2862                    }
2863                } else {
2864                    name.clone()
2865                };
2866                if !bt.iter().any(|n| n == &effective) {
2867                    bt.push(effective);
2868                }
2869            }
2870        }
2871        compiled.body_tables = bt;
2872        self.last_run_at.insert(rule.name.clone(), 0);
2873        self.rule_to_ruleset
2874            .insert(rule.name.clone(), rule.ruleset.clone());
2875        self.rules.push(compiled);
2876        Ok(())
2877    }

3046    /// Run rules in `ruleset` once (or all rules when `ruleset` is
3047    /// `None`). Returns total rows inserted across rules and
3048    /// variants.
3049    pub fn run_iteration_in(&mut self, ruleset: Option<&str>) -> Result<usize> {
3050        let synced_displaced = self.sync_native_ufs()?;
3051        self.rules_affected = self.rules_affected.wrapping_add(synced_displaced as u64);
3052        self.next_ts += 1;

3055        let last_run_ats: HashMap<String, i64> = self.last_run_at.clone();
3056        let skip_uf_maintenance = self.native_uf_enabled;
3057        for rule in &self.rules {
3058            if let Some(rs) = ruleset
3059                && rule.ruleset != rs
3060            {
3061                continue;
3062            }
3063            if skip_uf_maintenance && is_uf_maintenance_ruleset(&rule.ruleset) {
3064                continue;
3065            }
3066            // Inline-congruence backend optimization: under native
3067            // UF, the runner emits the equivalent SQL after every
3068            // iter via `emit_inline_congruence`. The encoding's
3069            // `congruence_rule<N>` would do the same work, so skip
3070            // it to avoid duplicate scans.
3071            if skip_uf_maintenance && is_congruence_rule_name(&rule.name) {
3072                continue;
3073            }
3074            let last = *last_run_ats.get(&rule.name).unwrap_or(&0);
3075            total += run_rule_variants(
3076                rule,
3077                last,
3078                cur,
3079                &self.conn,
3080                &mut self.time_mat_ns,
3081                &mut self.time_mat_act_ns,
3082                &mut self.time_act_ns,
3083                &mut self.rules_affected,
3084                &mut self.rule_perf_ns,
3085                &mut self.table_watermarks,
3086                &mut self.rules_skipped,
3087            )?;
3088            self.last_run_at.insert(rule.name.clone(), cur);
3089        }
3090        // After rules fire, run the inline-congruence pass over
3091        // every EqSort constructor table that received new rows
3092        // this iteration. The encoding's `@congruence_rule*` rules
3093        // are skipped above; this is where their work lands instead.
3094        //
3095        // Only fire on rulesets that can produce new view rows: the
3096        // user's iter (`ruleset = None`) and `@rebuilding` (where
3097        // the rebuild rule canonicalizes view rows). Other rulesets
3098        // (cleanup, single_parent, parent, uf_function_index) only
3099        // delete or push to pname, never insert into views, so the
3100        // emit would scan empty `v1` sets repeatedly.
3101        let should_emit_cong = self.native_uf_enabled
3102            && (ruleset.is_none() || ruleset.is_some_and(is_rebuilding_ruleset));
3103        if should_emit_cong {
3104            total += self.emit_inline_congruence(cur)?;
3105        }
3106        let _ = synced_displaced;
3107        Ok(total)
3108    }

3110    /// For each EqSort constructor with a registered pname whose
3111    /// table watermark advanced this iter, INSERT into the pname any
3112    /// same-input/different-id pair as a `(max, min, cur)` union
3113    /// assertion. Returns the total rows inserted across all tables.
3114    ///
3115    /// Semantics match the term encoding's `@congruence_rule<N>`
3116    /// scanning new view rows against the current view, with
3117    /// `ON CONFLICT DO NOTHING` to dedupe pairs that two
3118    /// `v1`/`v2` orderings would otherwise emit twice.
3119    fn emit_inline_congruence(&mut self, cur: i64) -> Result<usize> {
3120        let mut total: usize = 0;
3121        // Snapshot the eligible (view, pname, inputs_len) triples so
3122        // we don't hold a borrow on `self.functions` during execute.
3123        // A function is eligible iff it has a pname registered and
3124        // can hold same-input/different-id rows (i.e. it has an ID
3125        // column). For both `eq_sort_ctor` and view relations the ID
3126        // is the last column.
3127        let entries: Vec<(String, String, usize)> = self
3128            .functions
3129            .iter()
3130            .filter_map(|(name, info)| {
3131                let pname = info.eq_sort_pname.clone()?;
3132                if info.cols.len() < 2 {
3133                    return None;
3134                }
3135                // For eq_sort_ctor (term tables) `inputs_len` is the
3136                // input arity; the ID is at index `inputs_len`. For
3137                // view relations there is no separate output column —
3138                // all `cols` are part of the row; the ID sits at the
3139                // last position by convention (term encoding emits
3140                // views shaped as `(inputs…, id)`).
3141                let inputs_len = if info.eq_sort_ctor {
3142                    info.inputs_len
3143                } else {
3144                    info.cols.len() - 1
3145                };
3146                if inputs_len == 0 {
3147                    return None;
3148                }
3149                // Skip if no inserts since the last inline-cong scan.
3150                // `wm` is the max-ever `ts` of any insert into this
3151                // view; `last_at` is the upper end of the previous
3152                // scan window. If `wm < last_at` the scan range is
3153                // guaranteed empty. (Note: NOT `wm < cur` — the
3154                // view's last write might be from a few iters ago and
3155                // we'd still want to pick up pairs that involve it
3156                // until they've been scanned.)
3157                let wm = self.table_watermarks.get(name).copied().unwrap_or(0);
3158                let last_at = self.last_inline_cong_at.get(name).copied().unwrap_or(0);
3159                if wm < last_at {
3160                    return None;
3161                }
3162                Some((name.clone(), pname, inputs_len))
3163            })
3164            .collect();
3165        for (view, pname, inputs_len) in entries {
3166            let id_col = inputs_len; // ID is at index `inputs_len`.
3167            let join_preds: Vec<String> = (0..inputs_len)
3168                .map(|i| format!("v1.c{i} = v2.c{i}"))
3169                .collect();
3170            // Seminaive triangulation: `v1` is the "new since last
3171            // scan" side (`ts >= last_inline_cong_at`); `v2` covers
3172            // everything earlier-or-equal (`ts < cur`, since the
3173            // current iter's rows are accounted for via `v1`). The
3174            // pair (v1=old, v2=new) is the same pair as (v1=new,
3175            // v2=old) with the two halves swapped, so we don't need
3176            // to emit a second branch — `ON CONFLICT DO NOTHING` on
3177            // pname dedupes if v1 and v2 are both new.
3178            let last_at = self.last_inline_cong_at.get(&view).copied().unwrap_or(0);
3179            let sql = format!(
3180                "INSERT INTO {} (c0, c1, ts) \
3181                 SELECT GREATEST(v1.c{id_col}, v2.c{id_col}), \
3182                        LEAST(v1.c{id_col}, v2.c{id_col}), \
3183                        ?2 \
3184                 FROM {} v1, {} v2 \
3185                 WHERE v1.ts >= ?1 AND v1.ts < ?2 AND v2.ts < ?2 \
3186                   AND v1.c{id_col} <> v2.c{id_col} \
3187                   AND {} \
3188                 ON CONFLICT DO NOTHING",
3189                q(&pname),
3190                q(&view),
3191                q(&view),
3192                join_preds.join(" AND "),
3193            );
3194            let n = exec_bound(&self.conn, &sql, last_at, cur)?;
3195            if n > 0 {
3196                self.rules_affected = self.rules_affected.wrapping_add(n as u64);
3197                self.bump_watermark(&pname, cur);
3198                total += n;
3199            }
3200            self.last_inline_cong_at.insert(view, cur);
3201        }
3202        Ok(total)
3203    }

3339/// Build the `ON CONFLICT ...` clause for an INSERT into the given
3340/// table. Used by both seeding (`EGraph::insert`) and rule-action
3341/// codegen (`compile.rs`).
3342pub(crate) fn conflict_clause(info: &FunctionInfo) -> String {
3343    // `declare` emits an empty PK clause iff the PK width is 0.
3344    // Without a PK, DuckDB rejects `ON CONFLICT` entirely. For
3345    // these tables (nullary functions with output) we just emit
3346    // plain INSERTs; second-write semantics are then "duplicate
3347    // rows accumulate", which matches `:no-merge` if the user
3348    // never re-sets, and is wrong otherwise. Term encoding's use
3349    // of nullary `:no-merge` functions is for `let`-binding
3350    // globals, and those are set exactly once at declaration time,
3351    // so this is safe in practice.
3352    let pk_width = match (&info.merge, info.eq_sort_ctor) {
3353        (Some(_), false) => info.inputs_len,
3354        _ => info.cols.len(),
3355    };
3356    if pk_width == 0 {
3357        return String::new();
3358    }
3359    if info.eq_sort_ctor {
3360        // EqSort constructor inserts come with a freshly-allocated
3361        // ID column, so collisions on the all-cols PK shouldn't
3362        // happen in practice. If they do (caller passed a literal
3363        // ID), drop quietly.
3364        return "ON CONFLICT DO NOTHING".to_string();
3365    }
3366    match info.merge {
3367        None => "ON CONFLICT DO NOTHING".to_string(),
3368        Some(MergeMode::Old) => "ON CONFLICT DO NOTHING".to_string(),
3369        Some(MergeMode::New) => {
3370            let out_col = info.inputs_len;
3371            format!("ON CONFLICT DO UPDATE SET c{out_col} = EXCLUDED.c{out_col}, ts = EXCLUDED.ts")
3372        }
3373        Some(MergeMode::Min) => {
3374            let out_col = info.inputs_len;
3375            // Unprefixed `c{out_col}` on the RHS refers to the
3376            // existing target row; `EXCLUDED.c{out_col}` is the new
3377            // value being inserted.
3378            format!(
3379                "ON CONFLICT DO UPDATE SET c{out_col} = LEAST(c{out_col}, EXCLUDED.c{out_col}), ts = EXCLUDED.ts"
3380            )
3381        }
3382    }
3383}
