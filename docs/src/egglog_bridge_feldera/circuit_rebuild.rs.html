1//! Stage-3 integration: route the term encoder's rebuild rulesets onto the
2//! persistent congruence-shuttle circuit ([`crate::rebuild_circuit`]) instead of
3//! the host interpreter, behind the `FELDERA_CIRCUIT_REBUILD` flag.
4//!
5//! ## What this replaces
6//!
7//! The term encoder (`src/proofs/proof_encoding.rs`) emits, per eq-sort:
8//!
9//! * `@UF_<sort>(child, parent, _)` — union (parent) edges, a relation.
10//! * `@UF_<sort>f(child) = leader`  — the uf-index function, `:merge ordering-min`.
11//! * `@<ctor>View(args…, out)`      — one view table per constructor.
12//!
13//! and rebuild rulesets (`parent` / `single_parent` / `uf_function_index` /
14//! `rebuilding` / `rebuilding_cleanup`) that the frontend saturates. The host
15//! interpreter re-reads the full mirror on every `run_rules` call, which is
16//! O(state) per call and blows up super-linearly (the migration's target bug).
17//!
18//! ## What this does instead
19//!
20//! When `run_rules` is invoked with a set of rules that are *all* recognized
21//! rebuild rules, we run the **whole** rebuild fixpoint in one shot on the
22//! persistent circuit:
23//!
24//! 1. Feed the circuit the current `@UF` edges (as unions) and every view-table
25//!    row (as view rows, args packed).
26//! 2. Run the §2.4 congruence shuttle to fixpoint.
27//! 3. Read leaders + canonical views, and SYNTHESIZE the canonical mirror:
28//!    `@UFf = {(x, leader(x)) : leader(x) defined}` and
29//!    `@UF  = {(x, leader(x)) : leader(x) != x}` (child→leader edges, exactly
30//!    what the interpreter's path-compressed `@UF` converges to), plus each view
31//!    table rewritten to canonical outputs.
32//!
33//! Because the result is the *converged* rebuild, the FIRST such call applies
34//! the full fixpoint and a SECOND call is a no-op (`changed=false`), so the
35//! frontend's saturation loop terminates after one effective pass — replacing
36//! the interpreter's many O(state) passes with one circuit fixpoint.
37//!
38//! ## Safety / fallback
39//!
40//! Recognition is conservative: if the rule set contains anything we don't
41//! recognize as a pure rebuild rule, or the relation roles can't be resolved,
42//! we return `None` and the caller falls back to the interpreter. So this can
43//! never *regress* a program — at worst it declines to accelerate it.
44

53/// The relation roles recognized for one eq-sort's rebuild.
54#[derive(Default)]
55pub struct RebuildRoles {
56    /// The `@UF` parent relation(s) (child, parent, _).
57    pub uf: HashSet<FunctionId>,
58    /// The `@UFf` uf-index function(s) (child) -> leader.
59    pub uff: HashSet<FunctionId>,
60    /// The `@…View` view tables (args…, out).
61    pub views: HashSet<FunctionId>,
62}
63
64/// Classify a single rebuild rule, contributing to `roles`. Returns `false` if
65/// the rule is not a recognized rebuild rule.
66///
67/// Recognized term-mode rebuild rules (no proofs):
68///   * uf_index:      body `(@uf a b _)`              head `set @uff(a)=b`
69///   * path_compress: body `(@uf a b _)(@uf b c _)(!=)` head `delete/set @uf`
70///   * single_parent: body `(@uf a b _)(@uff a c)(!=)` head `delete/set @uf,@uff`
71///   * congruence:    body two `(@view args o1)(@view args o2)(!=)` head set @uf
72///   * cleanup:       body `(@view args old)(@view args new)` head delete/set view
73fn classify_rule(rule: &RuleIr, roles: &mut RebuildRoles) -> bool {
74    // Collect body table atoms and head set/remove targets.
75    let body_atoms: Vec<&crate::compile::BodyAtom> = rule
76        .body
77        .iter()
78        .filter_map(|op| match op {
79            BodyOp::Atom(a) => Some(a),
80            BodyOp::Prim { .. } => None,
81        })
82        .collect();
83    if body_atoms.is_empty() {
84        return false;
85    }
86
87    let name = rule.name.as_str();
88    // The encoder names rebuild rules predictably (see proof_encoding.rs).
89    let is_uf_index = name.contains("uf_function_index");
90    let is_path = !is_uf_index && (name.contains("uf_update") || name.starts_with("uf_update"));
91    let is_single = name.contains("singleparent");
92    let is_congru = name.contains("congruence_rule");
93    let is_cleanup = name.contains("merge_cleanup") || name.contains("merge_rule");
94
95    if is_uf_index {
96        // body (@uf a b _) ; head set (@uff a) b
97        roles.uf.insert(body_atoms[0].func);
98        for op in &rule.head {
99            if let HeadOp::Set { func, .. } = op {
100                roles.uff.insert(*func);
101            }
102        }
103        return true;
104    }
105    if is_path {
106        // body (@uf a b _)(@uf b c _)
107        roles.uf.insert(body_atoms[0].func);
108        return true;
109    }
110    if is_single {
111        // body (@uf a b _)(@uff a c)
112        roles.uf.insert(body_atoms[0].func);
113        if body_atoms.len() >= 2 {
114            roles.uff.insert(body_atoms[1].func);
115        }
116        return true;
117    }
118    if is_congru {
119        // body (@view args o1)(@view args o2) ; head set @uf
120        roles.views.insert(body_atoms[0].func);
121        for op in &rule.head {
122            if let HeadOp::Set { func, .. } = op {
123                roles.uf.insert(*func);
124            }
125        }
126        return true;
127    }
128    if is_cleanup {
129        roles.views.insert(body_atoms[0].func);
130        return true;
131    }
132    false
133}

135/// Attempt to recognize `rule_idxs` as a set of rebuild rules. Returns `None`
136/// (interpreter fallback) if any rule in this call is not a recognized rebuild
137/// rule.
138///
139/// On success, returns the FULL set of rebuild roles learned across **every**
140/// registered rebuild rule (not just this call's) — because rebuild relations
141/// (`@UF` / `@UFf` / views) are persistent mirror state that any rebuild-ruleset
142/// invocation must read in full, even when the frontend schedules only a subset
143/// of the rebuild rules in a given `run_rules` call.
144pub fn recognize(eg: &EGraph, rule_idxs: &[usize]) -> Option<RebuildRoles> {
145    // First: every rule in THIS call must be a recognized rebuild rule.
146    {
147        let mut probe = RebuildRoles::default();
148        for &i in rule_idxs {
149            let rule = eg.rules.get(i).and_then(|r| r.as_ref())?;
150            if !classify_rule(rule, &mut probe) {
151                return None;
152            }
153        }
154        if probe.uf.is_empty() && probe.views.is_empty() {
155            return None;
156        }
157    }
158    // Then: gather the FULL role set from all registered rebuild rules.
159    let mut roles = RebuildRoles::default();
160    for rule in eg.rules.iter().flatten() {
161        let mut r = RebuildRoles::default();
162        if classify_rule(rule, &mut r) {
163            roles.uf.extend(r.uf);
164            roles.uff.extend(r.uff);
165            roles.views.extend(r.views);
166        }
167    }
168    // The circuit canonicalizes up to TWO arguments (arity <= 3 view tables:
169    // 2 args + output). Decline (interpreter fallback) if any recognized view
170    // table is wider — never a regression, just no acceleration.
171    for &v in &roles.views {
172        let arity = eg.info(v).arity;
173        if arity > 3 {
174            return None;
175        }
176    }
177    Some(roles)
178}

180/// The key columns of a `@UF` parent relation: 0 (child), 1 (parent).
181const UF_CHILD: usize = 0;
182const UF_PARENT: usize = 1;
183
184/// Persistent rebuild-circuit cache: the congruence-shuttle circuit built ONCE
185/// and the bookkeeping needed to feed it only per-call DELTAS (so rebuild is
186/// O(delta), not O(state) — Stage-1 persistence in service of Stage 3).
187pub struct RebuildCache {
188    circuit: RebuildCircuit,
189    /// Congruence/seed edges already pushed into the circuit's `union_in`
190    /// (normalized lo<=hi), so each is shuttled exactly once over the lifetime.
191    pushed_unions: std::collections::BTreeSet<(u64, u64)>,
192    /// The view rows last fed to the circuit (so we push only the +/- diff).
193    /// Keyed by `(table_tag, arg0, arg1, out)` real-id tuple.
194    fed_views: HashSet<(u64, u64, u64, u64)>,
195    /// The raw `@UF` edges last fed as seed unions (normalized lo<=hi).
196    fed_uf_seed: std::collections::BTreeSet<(u64, u64)>,
197    /// Stable per-view-table tag (assigned once; index in a sorted role list).
198    tag_of: HashMap<FunctionId, u64>,
199}
200
201/// Run the entire rebuild fixpoint for `roles` on the PERSISTENT circuit, fed
202/// only the per-call delta of raw view rows / `@UF` seed edges, and fold the
203/// canonical result back into the mirror. Returns whether the mirror changed.
204pub fn run_rebuild(eg: &mut EGraph, roles: &RebuildRoles) -> anyhow::Result<bool> {
205    // Snapshot the pre-rebuild mirror for the change check.
206    let before: HashMap<FunctionId, HashSet<crate::compile::Row>> = eg
207        .mirror
208        .iter()
209        .filter(|(f, _)| {
210            roles.uf.contains(*f) || roles.uff.contains(*f) || roles.views.contains(*f)
211        })
212        .map(|(f, s)| (*f, (**s).clone()))
213        .collect();
214
215    // Assign stable per-table tags (sorted for determinism) and lazily build the
216    // persistent circuit + cache.
217    let mut view_list: Vec<FunctionId> = roles.views.iter().copied().collect();
218    view_list.sort_by_key(|f| f.rep());
219
220    if eg.rebuild_cache.is_none() {
221        let circuit = RebuildCircuit::build()?;
222        let mut tag_of: HashMap<FunctionId, u64> = HashMap::new();
223        for (i, &view) in view_list.iter().enumerate() {
224            tag_of.insert(view, (1u64 << 40) + i as u64);
225        }
226        eg.rebuild_cache = Some(RebuildCache {
227            circuit,
228            pushed_unions: std::collections::BTreeSet::new(),
229            fed_views: HashSet::new(),
230            fed_uf_seed: std::collections::BTreeSet::new(),
231            tag_of,
232        });
233    }
234    // Any newly-appeared view tables get a fresh tag.
235    {
236        let cache = eg.rebuild_cache.as_mut().unwrap();
237        for (i, &view) in view_list.iter().enumerate() {
238            cache
239                .tag_of
240                .entry(view)
241                .or_insert_with(|| (1u64 << 40) + i as u64);
242        }
243    }
244
245    // Compute the CURRENT raw view-row set and @UF seed set from the mirror.
246    let mut cur_views: HashSet<(u64, u64, u64, u64)> = HashSet::new();
247    for &view in &view_list {
248        let arity = eg.info(view).arity;
249        if arity == 0 {
250            continue;
251        }
252        let nargs = arity - 1;
253        let tag = eg.rebuild_cache.as_ref().unwrap().tag_of[&view];
254        if let Some(set) = eg.mirror.get(&view) {
255            for row in set.iter() {
256                let arg0 = row_col(row, 0) as u64;
257                let arg1 = if nargs >= 2 {
258                    row_col(row, 1) as u64
259                } else {
260                    NO_ARG
261                };
262                let out = row_col(row, nargs) as u64;
263                cur_views.insert((tag, arg0, arg1, out));
264            }
265        }
266    }
267    let mut cur_uf_seed: std::collections::BTreeSet<(u64, u64)> = std::collections::BTreeSet::new();
268    for &uf in &roles.uf {
269        if let Some(set) = eg.mirror.get(&uf) {
270            for row in set.iter() {
271                let child = row_col(row, UF_CHILD) as u64;
272                let parent = row_col(row, UF_PARENT) as u64;
273                cur_uf_seed.insert((child.min(parent), child.max(parent)));
274            }
275        }
276    }
277
278    // Push only the DELTA to the persistent circuit.
279    let canon_rows;
280    let leaders;
281    {
282        let cache = eg.rebuild_cache.as_mut().unwrap();
283        // View-row diff (+1 new, -1 retracted).
284        for &(tag, a0, a1, out) in &cur_views {
285            if !cache.fed_views.contains(&(tag, a0, a1, out)) {
286                cache.circuit.push_view(tag, a0, a1, out, 1);
287            }
288        }
289        for &(tag, a0, a1, out) in &cache.fed_views {
290            if !cur_views.contains(&(tag, a0, a1, out)) {
291                cache.circuit.push_view(tag, a0, a1, out, -1);
292            }
293        }
294        cache.fed_views = cur_views;
295        // @UF seed-union diff (additions only matter — unions are monotone; a
296        // removed seed edge stays implied by the leaders it already produced,
297        // matching the union-find's monotone "same class" semantics).
298        for &(a, b) in &cur_uf_seed {
299            if cache.fed_uf_seed.insert((a, b)) && cache.pushed_unions.insert((a, b)) {
300                cache.circuit.push_union(a, b);
301            }
302        }
303
304        let mut pushed = std::mem::take(&mut cache.pushed_unions);
305        cache.circuit.run_to_fixpoint(&mut pushed)?;
306        cache.pushed_unions = pushed;
307
308        leaders = cache.circuit.read_leaders();
309        canon_rows = cache.circuit.read_canon();
310    }
311
312    // Synthesize canonical @UFf: (x, leader(x)) for every x with a leader.
313    for &uff in &roles.uff {
314        let mut resolved: HashSet<crate::compile::Row> = HashSet::new();
315        for (&x, &l) in &leaders {
316            resolved.insert(vec![x as u32, l as u32].into_boxed_slice());
317        }
318        eg.mirror.insert(uff, std::rc::Rc::new(resolved));
319    }
320
321    // Synthesize canonical @UF: child->leader edge for each non-leader child.
322    // This matches the interpreter's path-compressed @UF (every child points
323    // directly at its leader; leaders have no edge).
324    for &uf in &roles.uf {
325        let arity = eg.info(uf).arity;
326        let mut resolved: HashSet<crate::compile::Row> = HashSet::new();
327        for (&x, &l) in &leaders {
328            if x != l {
329                let mut row = vec![x as u32, l as u32];
330                // Pad the (unit/proof) output column(s) with 0.
331                while row.len() < arity {
332                    row.push(0);
333                }
334                resolved.insert(row.into_boxed_slice());
335            }
336        }
337        eg.mirror.insert(uf, std::rc::Rc::new(resolved));
338    }
339
340    // Rewrite each view table to the circuit's canonical rows (args + output
341    // both canonicalized, congruence-collapsed). Group canon_rows by table tag.
342    let mut by_tag: HashMap<u64, Vec<(u64, u64, u64)>> = HashMap::new();
343    for (tag, c0, c1, out) in &canon_rows {
344        by_tag.entry(*tag).or_default().push((*c0, *c1, *out));
345    }
346    for &view in &view_list {
347        let arity = eg.info(view).arity;
348        if arity == 0 {
349            continue;
350        }
351        let nargs = arity - 1;
352        let tag = eg.rebuild_cache.as_ref().unwrap().tag_of[&view];
353        let mut resolved: HashSet<crate::compile::Row> = HashSet::new();
354        if let Some(rows) = by_tag.get(&tag) {
355            for &(c0, c1, out) in rows {
356                let mut full = vec![c0 as u32];
357                if nargs >= 2 {
358                    full.push(c1 as u32);
359                }
360                full.push(out as u32);
361                resolved.insert(full.into_boxed_slice());
362            }
363        }
364        eg.mirror.insert(view, std::rc::Rc::new(resolved));
365    }
366
367    // No per-rule `seen` to forget: the persistent rules' circuits diff the
368    // (now canonicalized) mirror against their last-fed view at the next
369    // `run_rules`, so the rebuild's rewrites of the uf / uff / view relations are
370    // picked up as proper signed deltas against each circuit's actual integral.
371    // (Forgetting the fed view would double-count, since the integral still holds
372    // the pre-rebuild rows.)
373
374    // Change check: did any touched relation's contents differ?
375    let mut changed = false;
376    for (f, b) in &before {
377        let after = eg.mirror.get(f);
378        let same = match after {
379            Some(a) => a.len() == b.len() && b.iter().all(|r| a.contains(r)),
380            None => b.is_empty(),
381        };
382        if !same {
383            changed = true;
384            break;
385        }
386    }
387    Ok(changed)
388}
