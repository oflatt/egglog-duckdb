1//! Per-iteration driver for the Feldera backend (#23 Stage C complete).
2//!
3//! One `run_rules` call = **one bounded egglog iteration**. The driver:
4//!
5//! 1. snapshots the relation mirror (the read view for this iteration — all
6//!    rules see the same pre-iteration state, matching egglog's semi-naive
7//!    "match against the old database, then apply" model for a single hop);
8//! 2. for each rule, runs its body join on the rule's **persistent DBSP
9//!    circuit** ([`crate::dbsp_join::PersistentJoin`]) — the ONLY join path. The
10//!    circuit is fed the per-relation signed delta vs the rows last fed to it;
11//!    its incremental join + integral do the seminaive bookkeeping and handle
12//!    retraction natively (signed weights), so there is no host nested-loop and
13//!    no per-rule host `seen` set. Pure value-computing body prims are evaluated
14//!    ON-CIRCUIT through the shared [`crate::PrimEngine`] (rep-comparable ones
15//!    are inlined as `filter`/`map`; everything else — base-value
16//!    `ordering-min/max`, f64 `!=`, `+`/`int-div`/`string-concat`, … — runs the
17//!    REAL prim on actual values via the call-prim path). Atom-less rules
18//!    (`(rule () …)` / `eval_actions`) fire their single unconditional binding
19//!    once. A genuinely-ineligible rule (an IMPURE body prim, or a shape above
20//!    the [`crate::dbsp_join::JOIN_WIDTH`] row cap) PANICS — there is no host
21//!    fallback;
22//! 3. for every surviving binding, executes the head ops in order — `set` /
23//!    `delete` / `subsume` writes, RHS `lookup` (eq-sort constructor: create on
24//!    miss), RHS primitive `call` (with side effects + result binding),
25//!    `union`, `panic`;
26//! 4. applies all collected writes/removes to the mirror and resolves
27//!    functional-dependency conflicts per each touched function's merge mode.
28//!
29//! Primitives are invoked through `Database::with_execution_state`, so they see
30//! the same interned base `Value`s the frontend created — giving the Feldera
31//! backend bit-for-bit value parity with the reference backend.
32

43/// Binding environment: variable id → bound `u32` value.
44type Env = HashMap<u32, u32>;

46/// A pending write to apply after all matches are computed.
47enum Write {
48    /// Insert/overwrite a full row.
49    Set(FunctionId, Row),
50    /// Retract by key (the slots address inputs for a function, whole row for a
51    /// relation).
52    Remove(FunctionId, Vec<u32>),
53}

55/// Run one bounded iteration of `rule_idxs` against the egraph. Returns whether
56/// the mirror changed.
57pub fn run_iteration(eg: &mut EGraph, rule_idxs: &[usize]) -> Result<bool> {
58    let prof = std::env::var("FELDERA_PROFILE").is_ok();
59    let t_read = std::time::Instant::now();
60    // Snapshot the read view: rules match against the pre-iteration mirror.
61    //
62    // The snapshot shares each function's row set by `Rc` rather than deep-cloning
63    // every row: this is O(#functions), not O(state). Mutations to the mirror this
64    // call (head writes, hash-cons in `lookup_or_create`, merge resolution) go
65    // through `Rc::make_mut`, which copy-on-writes only the functions actually
66    // changed while this snapshot is alive — so `read` keeps the start-of-call
67    // contents and rules all match the pre-iteration state, exactly as before, but
68    // without paying for a full-mirror clone every call.
69    let read: HashMap<FunctionId, std::rc::Rc<HashSet<Row>>> = eg
70        .mirror
71        .iter()
72        .map(|(f, set)| (*f, std::rc::Rc::clone(set)))
73        .collect();
74    if prof {
75        let n: usize = read.values().map(|v| v.len()).sum();
76        eg.prof_read_clone += t_read.elapsed();
77        eg.prof_read_rows += n as u64;
78    }
79
80    // Snapshot the fresh-id counter: any hash-cons (`lookup_or_create`) this call
81    // advances it, which is the O(1) signal that a new term row was created.
82    let next_id_at_start = eg.next_id;
83
84    let mut writes: Vec<Write> = Vec::new();
85    let mut touched: HashSet<FunctionId> = HashSet::new();
86    // Iteration-scoped `key -> output` index for `lookup_or_create` (eq-sort
87    // constructor hash-cons). Built lazily per function so repeated lookups in
88    // one iteration are O(1) instead of rescanning the growing mirror each time.
89    let mut lookup_index: HashMap<FunctionId, HashMap<Box<[u32]>, u32>> = HashMap::new();
90
91    // Collect each rule's index + IR up front (clone to avoid borrow conflicts
92    // while we also mutate the db / mirror via lookups).
93    let rules: Vec<(usize, RuleIr)> = rule_idxs
94        .iter()
95        .filter_map(|&i| eg.rules.get(i).and_then(|r| r.clone()).map(|r| (i, r)))
96        .collect();
97
98    // The persistent DBSP circuit is the ONLY join path (the host nested-loop +
99    // `seen` fallback was retired): every rule runs its body join on a per-rule
100    // persistent circuit fed signed deltas, which does the seminaive bookkeeping
101    // and retraction natively (its integral IS the `seen`). Atom-less rules fire
102    // their single unconditional binding once. A genuinely-ineligible rule (an
103    // IMPURE body prim, or a shape above the row-width cap) PANICS — there is no
104    // longer a graceful fallback.
105    for (idx, rule) in &rules {
106        let envs = persistent_bindings(eg, &read, *idx, rule)?;
107        for mut env in envs {
108            apply_head(
109                eg,
110                &rule.head,
111                &mut env,
112                &mut writes,
113                &mut touched,
114                &mut lookup_index,
115            )?;
116        }
117    }
118
119    // Apply collected writes to the mirror.
120    //
121    // Removes are BATCHED per function: a `Write::Remove(f, key)` retracts every
122    // row of `f` whose key columns equal `key`. Applying each remove with its own
123    // `set.retain` scan is O(|removes| · |state|) — quadratic, and the dominant
124    // per-iteration cost during rebuild (which retracts many rows at once). We
125    // instead collect all retraction keys per function into a hash set keyed by
126    // the function's key arity, then do a SINGLE `retain` pass per touched
127    // function: O(|state|) total regardless of how many rows are retracted.
128    //
129    // Removes are applied FIRST (batched), then Sets — preserving the term
130    // encoder's `(@uf)` "delete old leader row, set new leader row" ordering
131    // (delete-then-set: the set must win). A Set whose key was also retracted
132    // this iteration is re-inserted afterward, which is the intended result.
133    let t_apply = std::time::Instant::now();
134    // `changed` is computed INCREMENTALLY as writes land (O(delta)), not via a
135    // full before/after content compare of every function (O(state)). A hash-cons
136    // in `lookup_or_create` always allocates a fresh id, so any term created this
137    // call advances `next_id` — that alone is a real mirror change.
138    let mut changed = eg.next_id != next_id_at_start;
139    let mut removes_by_func: HashMap<FunctionId, (usize, HashSet<Box<[u32]>>)> = HashMap::new();
140    let mut sets: Vec<(FunctionId, Row)> = Vec::new();
141    for w in writes {
142        match w {
143            Write::Set(f, row) => {
144                sets.push((f, row));
145            }
146            Write::Remove(f, key) => {
147                let entry = removes_by_func
148                    .entry(f)
149                    .or_insert_with(|| (key.len(), HashSet::new()));
150                entry.1.insert(key.into_boxed_slice());
151            }
152        }
153    }
154    for (f, (keylen, keys)) in removes_by_func {
155        if let Some(set) = eg.mirror.get_mut(&f) {
156            let before_len = set.len();
157            std::rc::Rc::make_mut(set).retain(|row| {
158                let k: Box<[u32]> = (0..keylen)
159                    .map(|i| crate::compile::row_col(row, i))
160                    .collect();
161                !keys.contains(&k)
162            });
163            // A retraction that actually removed a row is a real change.
164            changed |= set.len() != before_len;
165        }
166    }
167    // Per-function set of input keys touched by a `set` this call — the only
168    // keys that can newly conflict and need merge resolution.
169    let mut touched_keys: HashMap<FunctionId, HashSet<Vec<u32>>> = HashMap::new();
170    for (f, row) in sets {
171        let inputs_len = eg.info(f).arity.saturating_sub(1);
172        let key: Vec<u32> = (0..inputs_len)
173            .map(|i| crate::compile::row_col(&row, i))
174            .collect();
175        // `insert` returns true iff the row was genuinely new (set a row that
176        // already exists ⇒ no content change, so don't flag `changed`).
177        let inserted = std::rc::Rc::make_mut(eg.mirror.entry(f).or_default()).insert(row);
178        changed |= inserted;
179        touched_keys.entry(f).or_default().insert(key);
180    }
181
182    if prof {
183        eg.prof_apply += t_apply.elapsed();
184    }
185
186    // Resolve FD conflicts on every function that a head action wrote to (a
187    // `set` can introduce two rows sharing a key that must be merged per the
188    // function's merge mode). Resolution is INCREMENTAL — only the keys whose
189    // rows were touched this call can have a new conflict, and `resolve_merge`
190    // reports whether it actually collapsed/changed any row (another real
191    // change source for the saturation loop).
192    let t_merge = std::time::Instant::now();
193    let empty_keys: HashSet<Vec<u32>> = HashSet::new();
194    for &f in &touched {
195        let keys = touched_keys.get(&f).unwrap_or(&empty_keys);
196        changed |= eg.resolve_merge(f, keys);
197    }
198    if prof {
199        eg.prof_merge += t_merge.elapsed();
200    }
201
202    // `change_detect` is now folded incrementally into apply/merge (O(delta));
203    // there is no separate full before/after compare to time.
204    Ok(changed)
205}

207/// Evaluate one body PRIMITIVE op against the current binding envs, extending /
208/// pruning each. Used ONLY by the atom-less rule path (`persistent_bindings`):
209/// an atom-less rule has no table atoms, so its body is exclusively primitives
210/// (a guard prunes, a value prim binds a var the head reads). A body `Atom` is
211/// unreachable here — the persistent DBSP circuit handles all atom-bearing
212/// rules.
213fn eval_body_prim(eg: &mut EGraph, op: &BodyOp, envs: Vec<Env>) -> Result<Vec<Env>> {
214    let BodyOp::Prim { id, args, ret } = op else {
215        unreachable!("atom-less body must contain only primitives");
216    };
217    let mut out = Vec::new();
218    for env in envs {
219        // Resolve args; an unbound arg means this primitive can't fire for this
220        // binding (shouldn't happen for well-formed rules).
221        let resolved: Option<Vec<Value>> = args
222            .iter()
223            .map(|s| slot_lookup(s, &|v| env.get(&v).copied()).map(Value::new))
224            .collect();
225        let Some(argv) = resolved else { continue };
226        let result = eg.eval_prim_internal(*id, &argv);
227        let Some(result) = result else {
228            // Primitive failed (e.g. `!=` of equal args) — prune.
229            continue;
230        };
231        // The `ret` slot binds the result or asserts equality.
232        match ret {
233            Slot::Var(v) => {
234                let mut next = env.clone();
235                match next.get(v) {
236                    Some(&existing) if existing != result.rep() => continue,
237                    _ => {
238                        next.insert(*v, result.rep());
239                    }
240                }
241                out.push(next);
242            }
243            Slot::Const(c) => {
244                if *c == result.rep() {
245                    out.push(env);
246                }
247            }
248        }
249    }
250    Ok(out)
251}

253/// Run rule `idx`'s body join on its PERSISTENT DBSP circuit (the ONLY join
254/// path — the host nested-loop fallback has been retired), returning the
255/// positive binding deltas as envs ready for head application.
256///
257/// The circuit's integral is kept equal to the start-of-call read view by
258/// pushing, per body relation, only the `+/-` diff vs the rows last fed to this
259/// rule. DBSP's incremental join then emits exactly the binding delta: positive
260/// weights are new matches (fire the head once), negative weights are matches
261/// retracted because a body row was retracted — integral bookkeeping only, no
262/// head re-firing (egglog heads are monotone-fire; explicit `delete`s in the
263/// head, not binding disappearance, remove rows). `.distinct()` in the circuit
264/// makes every weight `±1`.
265///
266/// Atom-less rules fire their single unconditional binding once. A genuinely
267/// ineligible rule (an IMPURE body prim, or a shape above the row-width cap)
268/// PANICS — there is no fallback.
269// The `FELDERA_DEBUG_COUNTS` block below is an env-gated stderr diagnostic (off
270// in normal runs); `eprintln!` is intentional.
271#[allow(clippy::disallowed_macros)]
272fn persistent_bindings(
273    eg: &mut EGraph,
274    read: &HashMap<FunctionId, std::rc::Rc<HashSet<Row>>>,
275    idx: usize,
276    rule: &RuleIr,
277) -> Result<Vec<Env>> {
278    // ATOM-LESS rules (`(rule () …)` / `eval_actions` / `eval_resolved_expr`):
279    // there is no body relation to drive a join, so the (trivially-satisfied)
280    // empty binding fires EXACTLY ONCE. Handle it here without the host
281    // nested-loop: evaluate the body prims once over the empty env (a guard may
282    // prune; a value prim binds a var the head reads), fire, and record that the
283    // rule has fired so it never re-fires. This is a tiny dedicated "fire
284    // unconditional rule once" helper, NOT a seminaive nested-loop join.
285    let has_atoms = rule.body.iter().any(|op| matches!(op, BodyOp::Atom(_)));
286    if !has_atoms {
287        if eg.atomless_fired.contains(&idx) {
288            return Ok(Vec::new());
289        }
290        // Evaluate the body prims once on the empty binding. An impure body prim
291        // here is still safe to evaluate (it runs exactly once, like the host
292        // path would), so we do not gate on purity for the atom-less case.
293        let mut envs: Vec<Env> = vec![Env::new()];
294        for op in &rule.body {
295            envs = eval_body_prim(eg, op, envs)?;
296            if envs.is_empty() {
297                break;
298            }
299        }
300        eg.atomless_fired.insert(idx);
301        return Ok(envs);
302    }
303
304    // Plan the body join. The persistent circuit is the ONLY join path, so an
305    // ineligible rule (an IMPURE body prim, or a shape above the row-width cap)
306    // PANICS with the reason — there is no graceful fallback. Pure prims never
307    // reach here as a rejection: rep-comparable ones inline, all others route to
308    // the on-circuit call-prim engine (`plan_join` docs).
309    let plan = match dbsp_join::plan_join(eg, rule) {
310        Ok(p) => p,
311        Err(reason) => panic!(
312            "unsupported on feldera persistent engine: {reason} (rule {:?})",
313            rule.name
314        ),
315    };
316    let var_order = plan.var_order().to_vec();
317
318    // Build the persistent circuit once for this rule. The shared primitive
319    // engine (Stage C) is captured into the circuit's call-prim closures so
320    // value-computing body prims evaluate on-circuit; it is built lazily here,
321    // after all primitives are registered.
322    if !eg.persistent.contains_key(&idx) {
323        let engine = eg.prim_engine();
324        let pj = dbsp_join::PersistentJoin::build(&plan, &engine)?;
325        eg.persistent.insert(idx, pj);
326    }
327
328    let body_funcs: Vec<FunctionId> = rule
329        .body
330        .iter()
331        .filter_map(|op| match op {
332            BodyOp::Atom(a) => Some(a.func),
333            BodyOp::Prim { .. } => None,
334        })
335        .collect();
336
337    // Compute the +/- delta vs the last-fed view per body relation, and advance
338    // the fed view to the current read view. The fed view is the `Rc` snapshot
339    // handle from the previous call: if the current read view shares that same
340    // `Rc` (the function was untouched since this rule last ran), the delta is
341    // empty and we skip the O(state) set diff entirely — the O(delta) fast path.
342    let prof = std::env::var("FELDERA_PROFILE").is_ok();
343    let diag = std::env::var("FELDERA_DEBUG_COUNTS").is_ok();
344    let t_diff = std::time::Instant::now();
345    let mut delta: HashMap<FunctionId, Vec<(Vec<u32>, ZWeight)>> = HashMap::new();
346    // DIAGNOSTIC ONLY (gated): the +1 (added) rows per func, as a host delta set.
347    let mut added_delta: HashMap<FunctionId, HashSet<Row>> = HashMap::new();
348    let empty_set: std::rc::Rc<HashSet<Row>> = std::rc::Rc::new(HashSet::new());
349    {
350        let fed = eg.fed.entry(idx).or_default();
351        for &f in &body_funcs {
352            let cur = read.get(&f).cloned().unwrap_or_else(|| empty_set.clone());
353            let prev = fed.entry(f).or_insert_with(|| empty_set.clone());
354            // Fast path: same `Rc` ⇒ identical contents ⇒ no delta.
355            if std::rc::Rc::ptr_eq(&cur, prev) {
356                *prev = cur;
357                continue;
358            }
359            let mut rows: Vec<(Vec<u32>, ZWeight)> = Vec::new();
360            let mut added: HashSet<Row> = HashSet::new();
361            for r in cur.iter() {
362                if !prev.contains(r) {
363                    rows.push((r.to_vec(), 1));
364                    if diag {
365                        added.insert(r.clone());
366                    }
367                }
368            }
369            for r in prev.iter() {
370                if !cur.contains(r) {
371                    rows.push((r.to_vec(), -1));
372                }
373            }
374            if !rows.is_empty() {
375                delta.insert(f, rows);
376            }
377            if !added.is_empty() {
378                added_delta.insert(f, added);
379            }
380            *prev = cur;
381        }
382    }
383
384    if prof {
385        eg.prof_fed_diff += t_diff.elapsed();
386    }
387
388    eg.dbsp_rule_runs += 1;
389    let t_step = std::time::Instant::now();
390    let bindings = {
391        let pj = eg
392            .persistent
393            .get_mut(&idx)
394            .expect("persistent circuit present");
395        pj.step(&delta)?
396    };
397    if prof {
398        eg.prof_circuit_step += t_step.elapsed();
399    }
400
401    // Eligible rules carry only inlined pure prims (guards applied in-circuit,
402    // value prims kept symbolic and never escaping to the head — verified by
403    // `plan_join`), so the env is fully determined by `var_order`.
404    let mut envs: Vec<Env> = Vec::new();
405    let mut neg_count = 0usize;
406    for (bind, w) in &bindings {
407        if *w <= 0 {
408            neg_count += 1;
409            continue; // retracted binding: integral bookkeeping only
410        }
411        let mut env: Env = Env::new();
412        for (i, &v) in var_order.iter().enumerate() {
413            env.insert(v, bind[i]);
414        }
415        envs.push(env);
416    }
417    if std::env::var("FELDERA_DEBUG_COUNTS").is_ok() {
418        let added_n: usize = added_delta.values().map(|s| s.len()).sum();
419        eprintln!(
420            "[CNT] rule={} added={} pos_emit={} neg_emit={}",
421            rule.name,
422            added_n,
423            envs.len(),
424            neg_count
425        );
426    }
427
428    Ok(envs)
429}

431/// Execute the head ops for one binding, accumulating writes.
432fn apply_head(
433    eg: &mut EGraph,
434    head: &[HeadOp],
435    env: &mut Env,
436    writes: &mut Vec<Write>,
437    touched: &mut HashSet<FunctionId>,
438    lookup_index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
439) -> Result<()> {
440    for op in head {
441        match op {
442            HeadOp::Set { func, slots } => {
443                let row = build_row(slots, env)?;
444                touched.insert(*func);
445                writes.push(Write::Set(*func, row));
446            }
447            HeadOp::Remove { func, slots } => {
448                let key: Vec<u32> = slots
449                    .iter()
450                    .map(|s| resolve(s, env))
451                    .collect::<Result<_>>()?;
452                touched.insert(*func);
453                writes.push(Write::Remove(*func, key));
454            }
455            HeadOp::Subsume { func, .. } => {
456                // Subsumption is not tracked; treat as a no-op (the row stays
457                // present). `supports_subsumption()` is false, so the frontend
458                // never relies on subsumed-row filtering on this backend.
459                touched.insert(*func);
460            }
461            HeadOp::Lookup { func, args, ret } => {
462                let key: Vec<Value> = args
463                    .iter()
464                    .map(|s| resolve(s, env).map(Value::new))
465                    .collect::<Result<_>>()?;
466                let val = lookup_or_create(eg, *func, &key, lookup_index);
467                env.insert(*ret, val.rep());
468            }
469            HeadOp::Call { id, args, ret } => {
470                let argv: Vec<Value> = args
471                    .iter()
472                    .map(|s| resolve(s, env).map(Value::new))
473                    .collect::<Result<_>>()?;
474                let result = eg.eval_prim_internal(*id, &argv);
475                if let Some(v) = result {
476                    env.insert(*ret, v.rep());
477                }
478                // A `None` result (primitive failure) in an action is a no-op
479                // here; the binding simply has no value for `ret`. Real failures
480                // surface as panics through `PanicFunc`.
481            }
482            HeadOp::Union { .. } => {
483                // Term-encoding mode emits unions as `(set (@uf …))` writes, not
484                // trait `union` calls (mirrors the DuckDB backend). A direct
485                // `union` therefore never reaches a tractable program; ignore.
486            }
487            HeadOp::Panic(msg) => {
488                return Err(anyhow!("{msg}"));
489            }
490        }
491    }
492    Ok(())
493}

495/// Build a full row from head-action slots under `env`.
496fn build_row(slots: &[Slot], env: &Env) -> Result<Row> {
497    let vals: Vec<u32> = slots
498        .iter()
499        .map(|s| resolve(s, env))
500        .collect::<Result<_>>()?;
501    Ok(vals.into_boxed_slice())
502}

504/// Resolve a slot to a concrete value, erroring if it is an unbound variable.
505fn resolve(s: &Slot, env: &Env) -> Result<u32> {
506    slot_lookup(s, &|v| env.get(&v).copied())
507        .ok_or_else(|| anyhow!("unbound variable {s:?} in rule head"))
508}

510/// Look up the output of `func` for input `key`. If absent, create the row with
511/// a fresh id (eq-sort constructor semantics — mirrors `add_term`). The created
512/// row is written directly into the mirror so subsequent lookups in the same
513/// iteration see it (hash-cons).
514///
515/// `index` is an iteration-scoped `key -> output` map per function. A naive
516/// implementation rescans the whole (growing) mirror set on every call, which is
517/// O(bindings · state) per iteration — a super-linear blowup on eqsat workloads
518/// that create many terms per round. We instead build the index lazily once per
519/// function (O(state) the first time it is touched this iteration) and keep it
520/// updated as new rows are hash-consed, so each subsequent lookup is O(1).
521fn lookup_or_create(
522    eg: &mut EGraph,
523    func: FunctionId,
524    key: &[Value],
525    index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
526) -> Value {
527    let info = eg.info(func);
528    let inputs_len = info.arity.saturating_sub(1);
529    // Lazily populate the key->output index for this function from the mirror.
530    let idx = index.entry(func).or_insert_with(|| {
531        let mut m: HashMap<Box<[u32]>, u32> = HashMap::new();
532        if let Some(set) = eg.mirror.get(&func) {
533            for row in set.iter() {
534                let k: Box<[u32]> = (0..inputs_len)
535                    .map(|i| crate::compile::row_col(row, i))
536                    .collect();
537                m.insert(k, crate::compile::row_col(row, inputs_len));
538            }
539        }
540        m
541    });
542    let k: Box<[u32]> = key.iter().map(|v| v.rep()).collect();
543    if let Some(&out) = idx.get(&k) {
544        return Value::new(out);
545    }
546    // Not present: allocate a fresh id, insert the row, and update the index.
