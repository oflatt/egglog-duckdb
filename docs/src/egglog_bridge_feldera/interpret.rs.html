1//! Per-iteration driver for the Feldera backend (#23 Stage C complete).
2//!
3//! One `run_rules` call = **one bounded egglog iteration**. The driver:
4//!
5//! 1. snapshots the relation mirror (the read view for this iteration — all
6//!    rules see the same pre-iteration state, matching egglog's semi-naive
7//!    "match against the old database, then apply" model for a single hop);
8//! 2. for each rule, runs its body join on the rule's **persistent DBSP
9//!    circuit** ([`crate::dbsp_join::PersistentJoin`]) — the ONLY join path. The
10//!    circuit is fed the per-relation signed delta vs the rows last fed to it;
11//!    its incremental join + integral do the seminaive bookkeeping and handle
12//!    retraction natively (signed weights), so there is no host nested-loop and
13//!    no per-rule host `seen` set. Pure value-computing body prims are evaluated
14//!    ON-CIRCUIT through the shared [`crate::PrimEngine`] (rep-comparable ones
15//!    are inlined as `filter`/`map`; everything else — base-value
16//!    `ordering-min/max`, f64 `!=`, `+`/`int-div`/`string-concat`, … — runs the
17//!    REAL prim on actual values via the call-prim path). Atom-less rules
18//!    (`(rule () …)` / `eval_actions`) fire their single unconditional binding
19//!    once. A genuinely-ineligible rule (an IMPURE body prim, or a shape above
20//!    the [`crate::dbsp_join::JOIN_WIDTH`] row cap) PANICS — there is no host
21//!    fallback;
22//! 3. for every surviving binding, executes the head ops in order — `set` /
23//!    `delete` / `subsume` writes, RHS `lookup` (eq-sort constructor: create on
24//!    miss), RHS primitive `call` (with side effects + result binding),
25//!    `union`, `panic`;
26//! 4. applies all collected writes/removes to the mirror and resolves
27//!    functional-dependency conflicts per each touched function's merge mode.
28//!
29//! Primitives are invoked through `Database::with_execution_state`, so they see
30//! the same interned base `Value`s the frontend created — giving the Feldera
31//! backend bit-for-bit value parity with the reference backend.
32

43/// Binding environment: variable id → bound `u32` value.
44type Env = HashMap<u32, u32>;

46/// A pending write to apply after all matches are computed.
47enum Write {
48    /// Insert/overwrite a full row.
49    Set(FunctionId, Row),
50    /// Retract by key (the slots address inputs for a function, whole row for a
51    /// relation).
52    Remove(FunctionId, Vec<u32>),
53}

55/// Run one bounded iteration of `rule_idxs` against the egraph. Returns whether
56/// the mirror changed.
57pub fn run_iteration(eg: &mut EGraph, rule_idxs: &[usize]) -> Result<bool> {
58    let prof = std::env::var("FELDERA_PROFILE").is_ok();
59
60    // Feldera runs the fast RELATIONAL term-encoding: EVERY rule — including the
61    // `@rebuild_rule*` canonicalization rules — lowers as an ordinary rule and
62    // joins on the fused DBSP circuit (`view ⋈ @UF_Sf` + full-recanon action). No
63    // native-UF host-pass, no synthetic `@DispΔ` relation, no rebuild-rule
64    // recognizer, no δuf-scoping special-casing.
65
66    let t_read = std::time::Instant::now();
67    // Snapshot the read view: rules match against the pre-iteration mirror.
68    //
69    // The snapshot shares each function's row set by `Rc` rather than deep-cloning
70    // every row: this is O(#functions), not O(state). Mutations to the mirror this
71    // call (head writes, hash-cons in `lookup_or_create`, merge resolution) go
72    // through `Rc::make_mut`, which copy-on-writes only the functions actually
73    // changed while this snapshot is alive — so `read` keeps the start-of-call
74    // contents and rules all match the pre-iteration state, exactly as before, but
75    // without paying for a full-mirror clone every call.
76    let read: HashMap<FunctionId, std::rc::Rc<HashSet<Row>>> = eg
77        .mirror
78        .iter()
79        .map(|(f, set)| (*f, std::rc::Rc::clone(set)))
80        .collect();
81    if prof {
82        let n: usize = read.values().map(|v| v.len()).sum();
83        eg.prof_read_clone += t_read.elapsed();
84        eg.prof_read_rows += n as u64;
85    }
86
87    // Snapshot the fresh-id counter: any hash-cons (`lookup_or_create`) this call
88    // advances it, which is the O(1) signal that a new term row was created.
89    let next_id_at_start = eg.next_id;
90
91    let mut writes: Vec<Write> = Vec::new();
92    let mut touched: HashSet<FunctionId> = HashSet::new();
93    // Iteration-scoped `key -> output` index for `lookup_or_create` (eq-sort
94    // constructor hash-cons). Built lazily per function so repeated lookups in
95    // one iteration are O(1) instead of rescanning the growing mirror each time.
96    let mut lookup_index: HashMap<FunctionId, HashMap<Box<[u32]>, u32>> = HashMap::new();
97
98    // Collect each rule's index + IR up front (clone to avoid borrow conflicts
99    // while we also mutate the db / mirror via lookups).
100    let rules: Vec<(usize, RuleIr)> = rule_idxs
101        .iter()
102        .filter_map(|&i| eg.rules.get(i).and_then(|r| r.clone()).map(|r| (i, r)))
103        .collect();
104
105    // The persistent DBSP circuit is the ONLY join path (the host nested-loop +
106    // `seen` fallback was retired): every rule runs its body join on a per-rule
107    // persistent circuit fed signed deltas, which does the seminaive bookkeeping
108    // and retraction natively (its integral IS the `seen`). Atom-less rules fire
109    // their single unconditional binding once. A genuinely-ineligible rule (an
110    // IMPURE body prim, or a shape above the row-width cap) PANICS — there is no
111    // longer a graceful fallback.
112    //
113    // Partition the ruleset into ATOM-LESS rules (fired once via the dedicated
114    // helper) and ATOM-BEARING rules (joined on the FUSED per-ruleset circuit,
115    // ONE transaction for the whole set — see `fused_bindings`). The
116    // `@rebuild_rule*` canonicalization rules take NO special path: they are
117    // ordinary atom-bearing rules (`view ⋈ @UF_Sf` body + full-recanon head) and
118    // join on the fused circuit like any other rule.
119    let mut atom_rules: Vec<(usize, &RuleIr)> = Vec::new();
120    for (idx, rule) in &rules {
121        let has_atoms = rule.body.iter().any(|op| matches!(op, BodyOp::Atom(_)));
122        if has_atoms {
123            atom_rules.push((*idx, rule));
124        } else {
125            let envs = atomless_bindings(eg, *idx, rule)?;
126            for mut env in envs {
127                apply_head(
128                    eg,
129                    &rule.head,
130                    &mut env,
131                    &mut writes,
132                    &mut touched,
133                    &mut lookup_index,
134                )?;
135            }
136        }
137    }
138
139    // Run all atom-bearing rules' joins in ONE fused transaction, then apply
140    // each rule's head over its bindings.
141    let fused = fused_bindings(eg, &read, &atom_rules)?;
142    for ((_idx, rule), envs) in atom_rules.iter().zip(fused) {
143        for mut env in envs {
144            apply_head(
145                eg,
146                &rule.head,
147                &mut env,
148                &mut writes,
149                &mut touched,
150                &mut lookup_index,
151            )?;
152        }
153    }
154
155    // Apply collected writes to the mirror.
156    //
157    // Removes are BATCHED per function: a `Write::Remove(f, key)` retracts every
158    // row of `f` whose key columns equal `key`. Applying each remove with its own
159    // `set.retain` scan is O(|removes| · |state|) — quadratic, and the dominant
160    // per-iteration cost during rebuild (which retracts many rows at once). We
161    // instead collect all retraction keys per function into a hash set keyed by
162    // the function's key arity, then do a SINGLE `retain` pass per touched
163    // function: O(|state|) total regardless of how many rows are retracted.
164    //
165    // Removes are applied FIRST (batched), then Sets — preserving the term
166    // encoder's `(@uf)` "delete old leader row, set new leader row" ordering
167    // (delete-then-set: the set must win). A Set whose key was also retracted
168    // this iteration is re-inserted afterward, which is the intended result.
169    let t_apply = std::time::Instant::now();
170    // `changed` is computed INCREMENTALLY as writes land (O(delta)), not via a
171    // full before/after content compare of every function (O(state)). A hash-cons
172    // in `lookup_or_create` always allocates a fresh id, so any term created this
173    // call advances `next_id` — that alone is a real mirror change.
174    let mut changed = eg.next_id != next_id_at_start;
175    let mut removes_by_func: HashMap<FunctionId, (usize, HashSet<Box<[u32]>>)> = HashMap::new();
176    let mut sets: Vec<(FunctionId, Row)> = Vec::new();
177    for w in writes {
178        match w {
179            Write::Set(f, row) => {
180                sets.push((f, row));
181            }
182            Write::Remove(f, key) => {
183                let entry = removes_by_func
184                    .entry(f)
185                    .or_insert_with(|| (key.len(), HashSet::new()));
186                entry.1.insert(key.into_boxed_slice());
187            }
188        }
189    }
190    for (f, (keylen, keys)) in removes_by_func {
191        if let Some(set) = eg.mirror.get_mut(&f) {
192            let before_len = set.len();
193            std::rc::Rc::make_mut(set).retain(|row| {
194                let k: Box<[u32]> = (0..keylen)
195                    .map(|i| crate::compile::row_col(row, i))
196                    .collect();
197                !keys.contains(&k)
198            });
199            // A retraction that actually removed a row is a real change.
200            changed |= set.len() != before_len;
201        }
202    }
203    // Per-function set of input keys touched by a `set` this call — the only
204    // keys that can newly conflict and need merge resolution.
205    let mut touched_keys: HashMap<FunctionId, HashSet<Vec<u32>>> = HashMap::new();
206    for (f, row) in sets {
207        let inputs_len = eg.info(f).arity.saturating_sub(1);
208        let key: Vec<u32> = (0..inputs_len)
209            .map(|i| crate::compile::row_col(&row, i))
210            .collect();
211        // `insert` returns true iff the row was genuinely new (set a row that
212        // already exists ⇒ no content change, so don't flag `changed`).
213        let inserted = std::rc::Rc::make_mut(eg.mirror.entry(f).or_default()).insert(row.clone());
214        changed |= inserted;
215        touched_keys.entry(f).or_default().insert(key);
216    }
217
218    if prof {
219        eg.prof_apply += t_apply.elapsed();
220    }
221
222    // Resolve FD conflicts on every function that a head action wrote to (a
223    // `set` can introduce two rows sharing a key that must be merged per the
224    // function's merge mode). Resolution is INCREMENTAL — only the keys whose
225    // rows were touched this call can have a new conflict, and `resolve_merge`
226    // reports whether it actually collapsed/changed any row (another real
227    // change source for the saturation loop).
228    let t_merge = std::time::Instant::now();
229    // `--native-merge` TERM-BUILD scratch: cleared before the resolves so the view
230    // rows a term-build merge mints this iteration are recorded fresh.
231    eg.term_build_view_keys.clear();
232    let empty_keys: HashSet<Vec<u32>> = HashSet::new();
233    for &f in &touched {
234        let keys = touched_keys.get(&f).unwrap_or(&empty_keys);
235        changed |= eg.resolve_merge(f, keys);
236    }
237    // `--native-merge` TERM-BUILD: a term-build merge minted `@<C>View` rows
238    // (recorded in `term_build_view_keys`). Those views carry their OWN congruence
239    // merge, so reconcile their FD conflicts now — exactly the "TableInsert respects
240    // the target table's merge" semantics of the bridge. Drain to a fixpoint:
241    // resolving one view writes no new view rows, so this terminates in one pass in
242    // practice; the loop is defensive.
243    while !eg.term_build_view_keys.is_empty() {
244        let pending: Vec<(FunctionId, HashSet<Vec<u32>>)> =
245            eg.term_build_view_keys.drain().collect();
246        for (view, keys) in pending {
247            changed |= eg.resolve_merge(view, &keys);
248        }
249    }
250    if prof {
251        eg.prof_merge += t_merge.elapsed();
252    }
253
254    // `change_detect` is now folded incrementally into apply/merge (O(delta));
255    // there is no separate full before/after compare to time.
256    // Fold any proof-mode `Pair<sort, proof>` containers the head/on-circuit
257    // prims interned in the prim engine's (deep-cloned) container store back
258    // into `self.db`, the single store proof extraction reads. See
259    // `EGraph::sync_prim_engine_containers`. Cheap when no container was
260    // interned this iteration.
261    eg.sync_prim_engine_containers();
262    Ok(changed)
263}

265/// Evaluate one body PRIMITIVE op against the current binding envs, extending /
266/// pruning each. Used ONLY by the atom-less rule path (`persistent_bindings`):
267/// an atom-less rule has no table atoms, so its body is exclusively primitives
268/// (a guard prunes, a value prim binds a var the head reads). A body `Atom` is
269/// unreachable here — the persistent DBSP circuit handles all atom-bearing
270/// rules.
271fn eval_body_prim(eg: &mut EGraph, op: &BodyOp, envs: Vec<Env>) -> Result<Vec<Env>> {
272    let BodyOp::Prim { id, args, ret } = op else {
273        unreachable!("atom-less body must contain only primitives");
274    };
275    let mut out = Vec::new();
276    for env in envs {
277        // Resolve args; an unbound arg means this primitive can't fire for this
278        // binding (shouldn't happen for well-formed rules).
279        let resolved: Option<Vec<Value>> = args
280            .iter()
281            .map(|s| slot_lookup(s, &|v| env.get(&v).copied()).map(Value::new))
282            .collect();
283        let Some(argv) = resolved else { continue };
284        let result = eg.eval_prim_internal(*id, &argv);
285        let Some(result) = result else {
286            // Primitive failed (e.g. `!=` of equal args) — prune.
287            continue;
288        };
289        // The `ret` slot binds the result or asserts equality.
290        match ret {
291            Slot::Var(v) => {
292                let mut next = env.clone();
293                match next.get(v) {
294                    Some(&existing) if existing != result.rep() => continue,
295                    _ => {
296                        next.insert(*v, result.rep());
297                    }
298                }
299                out.push(next);
300            }
301            Slot::Const(c) => {
302                if *c == result.rep() {
303                    out.push(env);
304                }
305            }
306        }
307    }
308    Ok(out)
309}

311/// Fire an ATOM-LESS rule (`(rule () …)` / `eval_actions` / `eval_resolved_expr`)
312/// once: there is no body relation to drive a join, so the (trivially-satisfied)
313/// empty binding fires EXACTLY ONCE. Evaluate the body prims once over the empty
314/// env (a guard may prune; a value prim binds a var the head reads) and record
315/// that the rule has fired so it never re-fires. NOT a join.
316fn atomless_bindings(eg: &mut EGraph, idx: usize, rule: &RuleIr) -> Result<Vec<Env>> {
317    if eg.atomless_fired.contains(&idx) {
318        return Ok(Vec::new());
319    }
320    // An impure body prim here is still safe (it runs exactly once), so we do
321    // not gate on purity for the atom-less case.
322    let mut envs: Vec<Env> = vec![Env::new()];
323    for op in &rule.body {
324        envs = eval_body_prim(eg, op, envs)?;
325        if envs.is_empty() {
326            break;
327        }
328    }
329    eg.atomless_fired.insert(idx);
330    Ok(envs)
331}

333/// Run ALL atom-bearing rules of a ruleset on ONE FUSED DBSP circuit (#23
334/// transaction-count fix): a single circuit per ruleset, one shared input
335/// z-set per body relation, each rule a parallel join sub-stream with its own
336/// output handle — clocked in a SINGLE `transaction()` per `run_rules` call.
337/// This collapses the ruleset's R per-rule transactions (the dominant fixed
338/// per-transaction clocking cost) into one.
339///
340/// Returns one env vector per rule, in the SAME order as `atom_rules`. Semantics
341/// are identical to running each rule's join on its own persistent circuit: the
342/// fused circuit's shared-input integrals do the seminaive bookkeeping and
343/// retraction (signed weights); positive binding weights are new matches,
344/// negatives are integral bookkeeping (no head re-fire). A genuinely-ineligible
345/// rule (an IMPURE body prim, or a shape above the row-width cap) PANICS — there
346/// is no host fallback.
347// The `FELDERA_DEBUG_COUNTS` block below is an env-gated stderr diagnostic (off
348// in normal runs); `eprintln!` is intentional.
349#[allow(clippy::disallowed_macros)]
350fn fused_bindings(
351    eg: &mut EGraph,
352    read: &HashMap<FunctionId, std::rc::Rc<HashSet<Row>>>,
353    atom_rules: &[(usize, &RuleIr)],
354) -> Result<Vec<Vec<Env>>> {
355    if atom_rules.is_empty() {
356        return Ok(Vec::new());
357    }
358
359    // The fused circuit is keyed by the sorted rule-index list of this ruleset.
360    let key: Vec<usize> = {
361        let mut k: Vec<usize> = atom_rules.iter().map(|(i, _)| *i).collect();
362        k.sort_unstable();
363        k
364    };
365
366    // Build the fused circuit once for this ruleset. Plan every rule's join; an
367    // ineligible rule PANICS (the persistent circuit is the only join path).
368    if !eg.fused.contains_key(&key) {
369        let mut plans: Vec<(usize, dbsp_join::JoinPlan)> = Vec::with_capacity(atom_rules.len());
370        for (idx, rule) in atom_rules {
371            let plan = match dbsp_join::plan_join(eg, rule) {
372                Ok(p) => p,
373                Err(reason) => panic!(
374                    "unsupported on feldera persistent engine: {reason} (rule {:?})",
375                    rule.name
376                ),
377            };
378            plans.push((*idx, plan));
379        }
380        // Every rule — including the `@rebuild_rule*` canonicalization rules —
381        // runs through the general fused join. There is NO rebuild-specific
382        // δuf-scoping split: the rebuild rules join `view ⋈ @UF_Sf` on the DBSP
383        // circuit exactly like any other rule.
384        let engine = eg.prim_engine();
385        let fj = dbsp_join::FusedJoin::build(&plans, &engine)?;
386        eg.fused.insert(key.clone(), fj);
387    }
388
389    // The fused circuit shares one input per relation across all its rules, so
390    // the body relations are the UNION of every rule's body relations.
391    let mut body_funcs: Vec<FunctionId> = Vec::new();
392    for (_idx, rule) in atom_rules {
393        for op in &rule.body {
394            if let BodyOp::Atom(a) = op {
395                if !body_funcs.contains(&a.func) {
396                    body_funcs.push(a.func);
397                }
398            }
399        }
400    }
401
402    // Compute the +/- delta vs the last-fed view per body relation (one fed view
403    // per fused circuit), and advance it to the current read view. Same
404    // `Rc`-ptr-eq fast path as the per-rule code: an untouched function is
405    // skipped in O(1).
406    let prof = std::env::var("FELDERA_PROFILE").is_ok();
407    let diag = std::env::var("FELDERA_DEBUG_COUNTS").is_ok();
408    let t_diff = std::time::Instant::now();
409    let mut delta: HashMap<FunctionId, Vec<(Vec<u32>, ZWeight)>> = HashMap::new();
410    let mut added_delta: HashMap<FunctionId, HashSet<Row>> = HashMap::new();
411    let empty_set: std::rc::Rc<HashSet<Row>> = std::rc::Rc::new(HashSet::new());
412    {
413        let fed = eg.fed_fused.entry(key.clone()).or_default();
414        for &f in &body_funcs {
415            let cur = read.get(&f).cloned().unwrap_or_else(|| empty_set.clone());
416            let prev = fed.entry(f).or_insert_with(|| empty_set.clone());
417            if std::rc::Rc::ptr_eq(&cur, prev) {
418                *prev = cur;
419                continue;
420            }
421            let mut rows: Vec<(Vec<u32>, ZWeight)> = Vec::new();
422            let mut added: HashSet<Row> = HashSet::new();
423            for r in cur.iter() {
424                if !prev.contains(r) {
425                    rows.push((r.to_vec(), 1));
426                    if diag {
427                        added.insert(r.clone());
428                    }
429                }
430            }
431            for r in prev.iter() {
432                if !cur.contains(r) {
433                    rows.push((r.to_vec(), -1));
434                }
435            }
436            if !rows.is_empty() {
437                delta.insert(f, rows);
438            }
439            if !added.is_empty() {
440                added_delta.insert(f, added);
441            }
442            *prev = cur;
443        }
444    }
445    if prof {
446        eg.prof_fed_diff += t_diff.elapsed();
447    }
448
449    eg.dbsp_rule_runs += atom_rules.len() as u64;
450    let t_step = std::time::Instant::now();
451    let per_rule = {
452        let fj = eg.fused.get_mut(&key).expect("fused circuit present");
453        fj.step(&delta)?
454    };
455    if prof {
456        eg.prof_circuit_step += t_step.elapsed();
457        // ONE transaction per fused step (and only when a delta was pushed).
458        if !delta.is_empty() {
459            eg.prof_transactions += 1;
460        }
461    }
462
463    // Route each fused rule's bindings back to its env vector. The fused circuit
464    // reports rules in the SAME order as the `plans`/`atom_rules` slice it was
465    // built from (its `rule_indices()` mirror that), and `atom_rules` here is the
466    // same slice in the same order, so positions line up directly.
467    debug_assert_eq!(
468        eg.fused.get(&key).map(|f| f.rule_indices()),
469        Some(atom_rules.iter().map(|(i, _)| *i).collect::<Vec<_>>()),
470        "fused rule order must match atom_rules order"
471    );
472    let mut out: Vec<Vec<Env>> = Vec::with_capacity(atom_rules.len());
473    for (pos, (_idx, rule)) in atom_rules.iter().enumerate() {
474        let var_order = eg
475            .fused
476            .get(&key)
477            .expect("fused circuit present")
478            .var_order_at(pos)
479            .to_vec();
480        let bindings = &per_rule[pos];
481        let mut envs: Vec<Env> = Vec::new();
482        let mut neg_count = 0usize;
483        for (bind, w) in bindings {
484            if *w <= 0 {
485                neg_count += 1;
486                continue;
487            }
488            let mut env: Env = Env::new();
489            for (i, &v) in var_order.iter().enumerate() {
490                env.insert(v, bind[i]);
491            }
492            envs.push(env);
493        }
494        if diag {
495            let added_n: usize = added_delta.values().map(|s| s.len()).sum();
496            eprintln!(
497                "[CNT] rule={} added={} pos_emit={} neg_emit={}",
498                rule.name,
499                added_n,
500                envs.len(),
501                neg_count
502            );
503        }
504        out.push(envs);
505    }
506
507    Ok(out)
508}

510/// Execute the head ops for one binding, accumulating writes.
511fn apply_head(
512    eg: &mut EGraph,
513    head: &[HeadOp],
514    env: &mut Env,
515    writes: &mut Vec<Write>,
516    touched: &mut HashSet<FunctionId>,
517    lookup_index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
518) -> Result<()> {
519    for op in head {
520        match op {
521            HeadOp::Set { func, slots } => {
522                let row = build_row(slots, env)?;
523                // A union `(set (@UF_Sf lhs) rhs)` is an ordinary relational SET on
524                // the `@UF_Sf` flat-index table (`Min` merge picks the leader) — no
525                // native-UF interception (Feldera is off native-UF).
526                touched.insert(*func);
527                writes.push(Write::Set(*func, row));
528            }
529            HeadOp::Remove { func, slots } => {
530                let key: Vec<u32> = slots
531                    .iter()
532                    .map(|s| resolve(s, env))
533                    .collect::<Result<_>>()?;
534                touched.insert(*func);
535                writes.push(Write::Remove(*func, key));
536            }
537            HeadOp::Subsume { func, .. } => {
538                // Subsumption is not tracked; treat as a no-op (the row stays
539                // present). `supports_subsumption()` is false, so the frontend
540                // never relies on subsumed-row filtering on this backend.
541                touched.insert(*func);
542            }
543            HeadOp::Lookup { func, args, ret } => {
544                let key: Vec<Value> = args
545                    .iter()
546                    .map(|s| resolve(s, env).map(Value::new))
547                    .collect::<Result<_>>()?;
548                let val = lookup_or_create(eg, *func, &key, lookup_index);
549                env.insert(*ret, val.rep());
550            }
551            HeadOp::Call { id, args, ret } => {
552                let argv: Vec<Value> = args
553                    .iter()
554                    .map(|s| resolve(s, env).map(Value::new))
555                    .collect::<Result<_>>()?;
556                // Route head prim calls through the SAME shared primitive engine the
557                // on-circuit body call-prims use, so any container (proof mode's
558                // `Pair<sort, proof>`) a head `pair` action interns is later visible
559                // to the body `pair-first` / `pair-second` reads. `eval_prim_internal`
560                // uses `self.db` directly, whose ContainerValues store is a DIFFERENT
561                // instance from the engine's clone — container ids interned in one are
562                // absent from the other (base values re-intern identically; containers
563                // do not). Sharing one engine keeps them consistent.
564                let result = eg.prim_engine().eval(*id, &argv);
565                if let Some(v) = result {
566                    env.insert(*ret, v.rep());
567                }
568                // A `None` result (primitive failure) in an action is a no-op
569                // here; the binding simply has no value for `ret`. Real failures
570                // surface as panics through `PanicFunc`.
571            }
572            HeadOp::Union { .. } => {
573                // Term-encoding mode emits unions as `(set (@uf …))` writes, not
574                // trait `union` calls (mirrors the DuckDB backend). A direct
575                // `union` therefore never reaches a tractable program; ignore.
576            }
577            HeadOp::Panic(msg) => {
578                return Err(anyhow!("{msg}"));
579            }
580        }
581    }
582    Ok(())
583}

585/// Build a full row from head-action slots under `env`.
586fn build_row(slots: &[Slot], env: &Env) -> Result<Row> {
587    let vals: Vec<u32> = slots
588        .iter()
589        .map(|s| resolve(s, env))
590        .collect::<Result<_>>()?;
591    Ok(vals.into_boxed_slice())
592}
593
594/// Resolve a slot to a concrete value, erroring if it is an unbound variable.
595fn resolve(s: &Slot, env: &Env) -> Result<u32> {
596    slot_lookup(s, &|v| env.get(&v).copied())
597        .ok_or_else(|| anyhow!("unbound variable {s:?} in rule head"))
598}

600/// Look up the output of `func` for input `key`. If absent, create the row with
601/// a fresh id (eq-sort constructor semantics — mirrors `add_term`). The created
602/// row is written directly into the mirror so subsequent lookups in the same
603/// iteration see it (hash-cons).
604///
605/// `index` is an iteration-scoped `key -> output` map per function. A naive
606/// implementation rescans the whole (growing) mirror set on every call, which is
607/// O(bindings · state) per iteration — a super-linear blowup on eqsat workloads
608/// that create many terms per round. We instead build the index lazily once per
609/// function (O(state) the first time it is touched this iteration) and keep it
610/// updated as new rows are hash-consed, so each subsequent lookup is O(1).
611pub(crate) fn lookup_or_create(
612    eg: &mut EGraph,
613    func: FunctionId,
614    key: &[Value],
615    index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
616) -> Value {
617    let info = eg.info(func);
618    let inputs_len = info.arity.saturating_sub(1);
619    let identity_on_miss = info.identity_on_miss;
620    // Lazily populate the key->output index for this function from the mirror.
621    let idx = index.entry(func).or_insert_with(|| {
622        let mut m: HashMap<Box<[u32]>, u32> = HashMap::new();
623        if let Some(set) = eg.mirror.get(&func) {
624            for row in set.iter() {
625                let k: Box<[u32]> = (0..inputs_len)
626                    .map(|i| crate::compile::row_col(row, i))
627                    .collect();
628                m.insert(k, crate::compile::row_col(row, inputs_len));
629            }
630        }
631        m
632    });
633    let k: Box<[u32]> = key.iter().map(|v| v.rep()).collect();
634    if let Some(&out) = idx.get(&k) {
635        return Value::new(out);
636    }
637    // Identity-on-miss ("lookup-or-self"): a missing key resolves to the key
638    // itself and inserts no row. Used by the canonicalize-at-creation encoding
639    // for the flat UF-index `@UF_Sf`, where this is exactly `find_UFold(x)=x`
640    // for an id with no recorded leader.
641    if identity_on_miss {
642        debug_assert_eq!(key.len(), 1, "identity-on-miss expects a single key column");
643        return key[0];
644    }
645    // Not present: allocate a fresh id, insert the row, and update the index.
646    let id = eg.fresh_id_internal();
647    idx.insert(k, id);
648    let mut full: Vec<u32> = key.iter().map(|v| v.rep()).collect();
649    full.push(id);
650    let row: Row = full.into_boxed_slice();
651    std::rc::Rc::make_mut(eg.mirror.entry(func).or_default()).insert(row);
652    Value::new(id)
653}
