1//! Base-value pool for the Feldera/DBSP backend.
2//!
3//! The Feldera backend stores base values exactly the way the reference bridge
4//! does: a real [`egglog_core_relations::BaseValues`] registry. As of Milestone
5//! 3 that registry lives **inside the embedded [`Database`]** (see
6//! `lib.rs`), so primitives invoked through `Database::with_execution_state`
7//! observe the same interned `Value`s the frontend created — giving bit-for-bit
8//! `Value` parity with the reference backend.
9//!
10//! This module provides the layout-preserving reference cast from
11//! `&BaseValues` / `&mut BaseValues` to `&dyn BaseValuePool` (the same trick the
12//! bridge uses with its `BaseValuesAsPool` newtype).
13//!
14//! [`Database`]: egglog_core_relations::Database
15
16use std::any::{Any, TypeId};
17
18use egglog_backend_trait::{BaseValueId, BaseValuePool, Value};
19use egglog_core_relations::{BaseValues, DynamicInternTable};
20
21/// `#[repr(transparent)]` view of [`BaseValues`] implementing [`BaseValuePool`].
22///
23/// Obtained from a `&BaseValues` (which we borrow from the embedded
24/// [`egglog_core_relations::Database`]) via a layout-preserving reference cast,
25/// identical to the bridge's `BaseValuesAsPool` trick.
26#[repr(transparent)]
27pub struct BaseValuesAsPool(BaseValues);
28
29/// View `&BaseValues` as `&dyn BaseValuePool`.
30pub fn base_values_as_pool(bv: &BaseValues) -> &dyn BaseValuePool {
31    // SAFETY: `BaseValuesAsPool` is `#[repr(transparent)]` over `BaseValues`,
32    // so the reference cast is sound.
33    let as_pool: &BaseValuesAsPool =
34        unsafe { &*(bv as *const BaseValues as *const BaseValuesAsPool) };
35    as_pool
36}
37
38/// View `&mut BaseValues` as `&mut dyn BaseValuePool`.
39pub fn base_values_as_pool_mut(bv: &mut BaseValues) -> &mut dyn BaseValuePool {
40    // SAFETY: as above.
41    let as_pool: &mut BaseValuesAsPool =
42        unsafe { &mut *(bv as *mut BaseValues as *mut BaseValuesAsPool) };
43    as_pool
44}
45
46impl BaseValuePool for BaseValuesAsPool {
47    fn register_type_dyn(
48        &mut self,
49        type_id: TypeId,
50        factory: Box<dyn FnOnce() -> Box<dyn DynamicInternTable>>,
51    ) -> BaseValueId {
52        self.0.register_type_dyn(type_id, factory)
53    }
54
55    fn get_ty_by_type_id(&self, type_id: TypeId) -> BaseValueId {
56        self.0.get_ty_by_id(type_id)
57    }
58
59    fn intern_dyn(&self, ty: BaseValueId, value: Box<dyn Any + Send + Sync>) -> Value {
60        let any_ref: &dyn Any = &*value;
61        self.0.intern_dyn_by_id(ty, any_ref)
62    }
63
64    fn unwrap_dyn(&self, ty: BaseValueId, val: Value) -> Box<dyn Any + Send + Sync> {
65        self.0.unwrap_dyn_by_id(ty, val)
66    }
67
68    fn has_ty(&self, type_id: TypeId) -> bool {
69        self.0.has_ty_by_id(type_id)
70    }
71}
