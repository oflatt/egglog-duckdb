1//! Persistent rebuild circuit (Stage 3 of the interpreter-deprecation
2//! migration; see `docs/feldera_persistent_circuit_design.md`).
3//!
4//! This is the production port of the validated feasibility spike
5//! (`spike-dbsp/src/rebuild.rs`): egglog **rebuild** (union-find find +
6//! congruence, non-monotone delete-and-rewrite) expressed as DBSP recursive
7//! views in a **persistent circuit fed input deltas across transactions**, with
8//! the §2.4 **congruence-shuttle** encoding that is *required* for convergence
9//! (the naive co-recursive `(uf, reach)` tuple silently loses closure across
10//! transactions; routing congruence-discovered edges back through the INPUT
11//! boundary fixes it — proven in the spike on all 120 iterations of a deep
12//! deep-chaining eqsat).
13//!
14//! ## What the term encoder's rebuild actually is
15//!
16//! For each eq-sort, the proof/term encoder (`src/proofs/proof_encoding.rs`)
17//! emits:
18//!
19//! * `@uf(a, b)` — a relation of union (parent) edges. `path_compress` /
20//!   `single_parent` mutate it (delete stale edge, set new), `@uf_function_index`
21//!   mirrors it into `@uff`.
22//! * `@uff(a) = leader` — the uf-index function, `:merge (ordering-min …)`, i.e.
23//!   `leader(a) = min` over the connected component of `a`.
24//! * `@XView(args…, out)` — one view table per constructor; the `@rebuilding`
25//!   ruleset's congruence rule unions the outputs of two view rows with equal
26//!   args, and `@rebuilding_cleanup` rewrites each view row to its canonical
27//!   (leader) output, deleting the stale row.
28//!
29//! The whole point of the migration is that this fixpoint, when scheduled by
30//! the host interpreter, re-reads the full relation mirror on **every**
31//! `run_rules` call — so it is O(state) per call and blows up super-linearly on
32//! `math-microbenchmark` (`(run 11)` ≈ 56 s vs. `(run 9)` ≈ 4 s). The
33//! persistent circuit feeds only per-transaction deltas, so the rebuild cost is
34//! sub-linear in state.
35//!
36//! ## The encoding (mirrors the spike, generalized over view arity)
37//!
38//! `leader(x)` = the minimum id reachable from `x` over undirected union edges,
39//! computed with pure join/antijoin (fully incremental, no typed `Min`):
40//!
41//! ```text
42//!   reach(x,y)      :- self-reach; reach(x,z), sym(z,y).      (closure)
43//!   dominated(x,y)  :- reach(x,y), reach(x,z), z < y.         (y not the min)
44//!   leader(x,y)     :- reach(x,y), not dominated(x,y).        (the unique min)
45//! ```
46//!
47//! `uf` is kept SYMMETRIC inside the circuit so reachability = connected
48//! component. `reach` is a SINGLE recursive closure over the `uf` INPUT z-set
49//! (the shape that provably persists across transactions). Congruence is
50//! computed at root from the converged `reach` + canonical views and emitted as
51//! an output delta; the host **shuttle** pushes new congruence edges back into
52//! the `uf` input and re-steps until no new edge appears.
53//!
54//! ## Generality vs. the spike
55//!
56//! The spike fixed view arity at 2 with **leaf-constant** args (already
57//! canonical). Production view-table arguments are themselves e-class ids that
58//! get unioned, so congruence must be over *leader-canonicalized* arguments.
59//! This circuit therefore carries each view row as `Tup4(table_tag, arg0, arg1,
60//! out)` of **real ids** and canonicalizes BOTH arguments and the output via
61//! `leader_of` joins before computing congruence — the production extension over
62//! the spike. A `table_tag` distinguishes view tables so a single shared circuit
63//! handles them all without cross-constructor congruence collisions. View tables
64//! of arity > 3 (more than two arguments) are not handled here; the integration
65//! (`circuit_rebuild.rs`) falls back to the interpreter for them. See
66//! [`RebuildCircuit`].
67
68use anyhow::Result;
69use dbsp::{
70    typed_batch::OrdZSet, utils::Tup2, utils::Tup4, CircuitHandle, OutputHandle, RootCircuit,
71    Stream, ZSetHandle, ZWeight,
72};
73
74type Id = u64;
75type UfEdge = Tup2<Id, Id>;
76/// A view row flowing through the circuit, carrying REAL ids so the circuit can
77/// canonicalize the arguments (not just the output) by the union-find leader —
78/// the production extension over the spike (whose view args were leaf
79/// constants, already canonical). Layout: `(table_tag, arg0, arg1, out)`.
80///
81/// `table_tag` distinguishes view tables so congruence never collides across
82/// constructors. `arg1` is `NO_ARG` (a reserved sentinel) for arity-1
83/// constructors. Constructors with >2 arguments are not handled here (the
84/// integration falls back to the interpreter for them).
85type View = Tup4<Id, Id, Id, Id>; // (table_tag, arg0, arg1, output-id)
86
87/// Sentinel for "no second argument" (arity-1 constructors). Chosen above any
88/// real egglog id (`u32` reps), and never a leader, so it is left unchanged by
89/// the leader canonicalization (it maps to itself).
90pub const NO_ARG: Id = u64::MAX - 1;
91
92type ZUf = OrdZSet<UfEdge>;
93type ZView = OrdZSet<View>;
94type ZPair = OrdZSet<Tup2<Id, Id>>;
95/// A canonical view row read out: (table_tag, arg0, arg1, out).
96type CanonRow = (Id, Id, Id, Id);
97
98/// Handles exposed by the persistent rebuild circuit (one per view table).
99pub struct RebuildCircuit {
100    handle: CircuitHandle,
101    /// Push (+1 / -1) raw view rows produced by user rules / seeding.
102    view_in: ZSetHandle<View>,
103    /// Push (+1) union facts. Symmetric closure is built inside the circuit.
104    union_in: ZSetHandle<UfEdge>,
105    /// The canonicalized view relation (each row uses leader outputs).
106    canon_out: OutputHandle<ZView>,
107    /// leader(x) = representative of x.
108    leader_out: OutputHandle<ZPair>,
109    /// Congruence-discovered unions emitted this run (shuttled by the host).
110    congru_out: OutputHandle<OrdZSet<UfEdge>>,
111}
112
113/// leader(x) = min y with reach(x,y). NON-monotone: a pair (x,y) is "dominated"
114/// (not the minimum) if some (x,z) with z<y exists; antijoin removes those,
115/// leaving the unique minimum per x.
116///
117/// A macro, not a fn: DBSP's typed operators are implemented per concrete
118/// circuit-nesting depth, so this must be expanded at the root scope.
119macro_rules! leaders_from_reach {
120    ($reach:expr) => {{
121        let reach = $reach;
122        let reach_by_x = reach.map_index(|Tup2(x, y): &Tup2<Id, Id>| (*x, *y));
123        let dominated = reach_by_x
124            .join(&reach_by_x, |x, y, z| {
125                if z < y {
126                    Tup2(*x, *y)
127                } else {
128                    Tup2(u64::MAX, u64::MAX)
129                }
130            })
131            .filter(|Tup2(x, _): &Tup2<Id, Id>| *x != u64::MAX)
132            .distinct();
133        let reach_keyed = reach
134            .distinct()
135            .map_index(|Tup2(x, y): &Tup2<Id, Id>| (Tup2(*x, *y), ()));
136        let dom_keyed = dominated.map_index(|Tup2(x, y): &Tup2<Id, Id>| (Tup2(*x, *y), ()));
137        reach_keyed
138            .antijoin(&dom_keyed)
139            .map(|(Tup2(x, y), ())| Tup2(*x, *y))
140    }};
141}

143impl RebuildCircuit {
144    /// Build the persistent rebuild circuit ONCE (for one view table). Mirrors
145    /// `spike-dbsp/src/rebuild.rs::build` exactly — the §2.4 shuttle encoding.
146    pub fn build() -> Result<RebuildCircuit> {
147        let (handle, (view_in, union_in, canon_out, leader_out, congru_out)) =
148            RootCircuit::build(|root| {
149                let (raw_view, view_in): (Stream<_, ZView>, ZSetHandle<View>) =
150                    root.add_input_zset::<View>();
151                let (raw_union, union_in): (Stream<_, ZUf>, ZSetHandle<UfEdge>) =
152                    root.add_input_zset::<UfEdge>();
153
154                let raw_view_acc = raw_view.integrate();
155                let uf_acc = raw_union.integrate();
156                let sym = uf_acc
157                    .flat_map(|Tup2(a, b): &UfEdge| vec![Tup2(*a, *b), Tup2(*b, *a)])
158                    .distinct();
159                let sym_for_scope = sym.clone();
160                // Every real id appearing anywhere (args + output) needs a
161                // self-reach seed so it has a leader (itself if in no union).
162                // NO_ARG is a sentinel, not a real id — it self-reaches so the
163                // arg-canon join leaves it unchanged.
164                let self_reach0 = raw_view_acc
165                    .flat_map(|Tup4(_tag, a0, a1, id): &View| {
166                        vec![Tup2(*a0, *a0), Tup2(*a1, *a1), Tup2(*id, *id)]
167                    })
168                    .plus(&sym.map(|Tup2(a, _): &UfEdge| Tup2(*a, *a)))
169                    .distinct();
170                let reach: Stream<_, ZPair> = root
171                    .recursive(move |child, reach: Stream<_, ZPair>| {
172                        let self_reach = self_reach0.delta0(child);
173                        let sym = sym_for_scope.delta0(child);
174                        let reach_by_2nd = reach.map_index(|Tup2(x, y): &Tup2<Id, Id>| (*y, *x));
175                        let sym_by_src = sym.map_index(|Tup2(y, z): &UfEdge| (*y, *z));
176                        let step = reach_by_2nd.join(&sym_by_src, |_y, x, z| Tup2(*x, *z));
177                        Ok(self_reach.plus(&step).distinct())
178                    })
179                    .unwrap();
180
181                let leader_pairs: Stream<_, ZPair> = leaders_from_reach!(&reach);
182                let leader_of = leader_pairs.map_index(|Tup2(x, l): &Tup2<Id, Id>| (*x, *l));
183                // Canonicalize BOTH arguments AND the output to their leaders.
184                // This is the production extension: view-table arguments are
185                // themselves e-class ids that get unioned, so congruence must be
186                // over leader-canonicalized arguments (the spike's args were leaf
187                // constants and skipped this). Three successive joins canonicalize
188                // arg0, arg1, then out; NO_ARG maps to itself (self-reach seed).
189                let canon_a0 = raw_view_acc
190                    .map_index(|Tup4(tag, a0, a1, id): &View| (*a0, Tup4(*tag, *a0, *a1, *id)))
191                    .join(&leader_of, |_key, Tup4(tag, _a0, a1, id), l0| {
192                        Tup4(*tag, *l0, *a1, *id)
193                    });
194                let canon_a1 = canon_a0
195                    .map_index(|Tup4(tag, c0, a1, id): &View| (*a1, Tup4(*tag, *c0, *a1, *id)))
196                    .join(&leader_of, |_key, Tup4(tag, c0, _a1, id), l1| {
197                        Tup4(*tag, *c0, *l1, *id)
198                    });
199                let canon = canon_a1
200                    .map_index(|Tup4(tag, c0, c1, id): &View| (*id, Tup4(*tag, *c0, *c1, *id)))
201                    .join(&leader_of, |_key, Tup4(tag, c0, c1, _id), lo| {
202                        Tup4(*tag, *c0, *c1, *lo)
203                    })
204                    .distinct();
205                // congruence: same (tag, canon_arg0, canon_arg1), different
206                // canonical output ⟹ union the outputs.
207                let canon_by_args =
208                    canon.map_index(|Tup4(tag, c0, c1, out): &View| ((*tag, *c0, *c1), *out));
209                let congru = canon_by_args
210                    .join(&canon_by_args, |_args, o1, o2| {
211                        if o1 < o2 {
212                            Tup2(*o1, *o2)
213                        } else {
214                            Tup2(u64::MAX, u64::MAX)
215                        }
216                    })
217                    .filter(|Tup2(a, _): &UfEdge| *a != u64::MAX)
218                    .distinct();
219
220                Ok((
221                    view_in,
222                    union_in,
223                    canon.integrate().output(),
224                    leader_pairs.integrate().output(),
225                    congru.integrate().output(),
226                ))
227            })?;
228
229        Ok(RebuildCircuit {
230            handle,
231            view_in,
232            union_in,
233            canon_out,
234            leader_out,
235            congru_out,
236        })
237    }

239    /// Push a +1/-1 view row `(table_tag, arg0, arg1, out)` with REAL ids.
240    /// Use [`NO_ARG`] for `arg1` on arity-1 constructors.
241    pub fn push_view(&self, table_tag: Id, arg0: Id, arg1: Id, out: Id, weight: ZWeight) {
242        self.view_in.push(Tup4(table_tag, arg0, arg1, out), weight);
243    }
244
245    /// Push a +1 union edge (seed union from `@uf`).
246    pub fn push_union(&self, a: Id, b: Id) {
247        self.union_in.push(Tup2(a, b), 1);
248    }
249
250    /// Run the rebuild to fixpoint via the §2.4 congruence shuttle: step once,
251    /// read the congruence output, push any not-yet-applied congruence edge into
252    /// `union_in`, and step again — until no new congruence edge appears. Each
253    /// step is O(delta). `already_pushed` tracks edges (normalized lo<=hi)
254    /// pushed across the *lifetime* of the circuit so each is shuttled once.
255    pub fn run_to_fixpoint(
256        &self,
257        already_pushed: &mut std::collections::BTreeSet<(Id, Id)>,
258    ) -> Result<()> {
259        self.handle.transaction()?;
260        let mut steps = 1usize;
261        loop {
262            let mut new_edges: Vec<(Id, Id)> = Vec::new();
263            for (Tup2(a, b), (), w) in self.congru_out.consolidate().iter() {
264                if w > 0 {
265                    let key = (a.min(b), a.max(b));
266                    if already_pushed.insert(key) {
267                        new_edges.push((a, b));
268                    }
269                }
270            }
271            if new_edges.is_empty() {
272                break;
273            }
274            for (a, b) in new_edges {
275                self.union_in.push(Tup2(a, b), 1);
276            }
277            self.handle.transaction()?;
278            steps += 1;
279            if steps > 100_000 {
280                anyhow::bail!("congruence shuttle did not converge");
281            }
282        }
283        Ok(())
284    }

286    /// Read the canonical view rows as `(table_tag, canon_arg0, canon_arg1,
287    /// canon_out)` with positive weight.
288    pub fn read_canon(&self) -> Vec<CanonRow> {
289        let mut out = Vec::new();
290        for (Tup4(tag, a, b, c), (), w) in self.canon_out.consolidate().iter() {
291            let w: ZWeight = w;
292            if w > 0 {
293                out.push((tag, a, b, c));
294            }
295        }
296        out
297    }
298
299    /// Read leader(x) for every known x.
300    pub fn read_leaders(&self) -> std::collections::BTreeMap<Id, Id> {
301        let mut m = std::collections::BTreeMap::new();
302        for (Tup2(x, l), (), w) in self.leader_out.consolidate().iter() {
303            let w: ZWeight = w;
304            if w > 0 {
305                m.insert(x, l);
306            }
307        }
308        m
309    }
