1//! Rule IR and DBSP circuit assembly for the Feldera backend.
2//!
3//! ## The load-bearing design choice (per the milestone brief)
4//!
5//! egglog's `(run N)` applies a ruleset **N times with rebuild between
6//! iterations** — so a transitive-closure rule extends **N hops, NOT to full
7//! closure**. The Phase-0 spike used DBSP's RECURSIVE scope, which saturates to
8//! a fixed point inside one transaction; that is *wrong* for egglog's bounded
9//! iteration.
10//!
11//! So this backend compiles user rules into a **non-recursive** circuit: one
12//! `transaction()` performs exactly **one round** of rule application over the
13//! current relation contents. The frontend's existing iteration loop drives
14//! `(run N)` by calling `run_rules` N times — and because the circuit does one
15//! hop per call, `(run 1)` and `(run 3)` produce different, bounded results.
16//!
17//! ## Milestone 2: ruleset-scoped execution + retraction-rebuild
18//!
19//! Each `run_rules(&[RuleId])` call runs a **subset** of rules. A DBSP circuit
20//! is a static monolithic graph, so we build a *per-subset* circuit keyed by the
21//! sorted rule-id list (cached). The host pushes the current mirror as the
22//! circuit's input, runs one transaction, and reads back two diff streams per
23//! relation: an **insert** stream and a **delete** stream. The host then folds
24//! `(old ∪ inserts) \ deletes` into the mirror and resolves FD conflicts via the
25//! relation's merge mode. Deletes are the term encoder's `(delete ...)` actions
26//! (rebuild's retraction half); inserts are `(set ...)`.
27//!
28//! ## Row representation
29//!
30//! Every relation uses a single uniform row type [`Row`] = `Tup8<u32,…>` (eight
31//! `u32` slots). egglog values are `u32` ([`Value`] reps); columns beyond a
32//! relation's arity are padded with 0.
33
34use egglog_backend_trait::{ExternalFunctionId, FunctionId, QueryEntry, Value};
35use egglog_numeric_id::NumericId;

37/// Max number of columns a relation may have.
38/// Upper bound on relation arity (sanity check; the host row representation is
39/// variable-width so this is generous — the term encoder's widest view tables
40/// stay well under it).
41pub const MAX_ARITY: usize = 64;
42
43/// Uniform row type for every relation: a variable-width boxed slice of `u32`
44/// (egglog `Value` reps), exactly `arity` columns wide. Milestones 1–2 used a
45/// fixed `Tup8` because rows flowed through DBSP zsets; Milestone 3 evaluates
46/// rules host-side (primitives must invoke through the embedded `Database`), so
47/// the row is a plain hashable slice with no arity cap of 8.
48pub type Row = Box<[u32]>;
49
50/// How a function resolves a functional-dependency conflict (two rows sharing
51/// the same key columns with different output columns). Recognized from the
52/// trait `MergeFn` / the term encoder's `:merge` clause (see `lib.rs`).
53#[derive(Clone, Copy, Debug, PartialEq, Eq)]
54pub enum MergeMode {
55    /// Plain relation: the whole row is the key, no output column to resolve.
56    Relation,
57    /// `:merge old` / `AssertEq`: keep the existing value on conflict.
58    Old,
59    /// `:merge new`: keep the new value on conflict.
60    New,
61    /// `:merge (ordering-min old new)`: keep the numerically smallest value.
62    /// This is the term encoder's `@uff` (uf-index) merge — load-bearing for
63    /// rebuild: the function must hold the *minimum* representative per child.
64    Min,
65}
66
67/// Pack a slice of `Value`s into a [`Row`] (exactly `vals.len()` columns).
68pub fn pack_row(vals: &[Value]) -> Row {
69    assert!(
70        vals.len() <= MAX_ARITY,
71        "row arity {} exceeds {MAX_ARITY}",
72        vals.len()
73    );
74    vals.iter().map(|v| v.rep()).collect()
75}
76
77/// Read column `i` (0-based) out of a [`Row`].
78#[inline]
79pub fn row_col(r: &Row, i: usize) -> u32 {
80    r[i]
81}
82
83/// Unpack the first `arity` columns of a [`Row`] into a `Vec<Value>`.
84pub fn unpack_row(r: &Row, arity: usize) -> Vec<Value> {
85    (0..arity).map(|i| Value::new(row_col(r, i))).collect()
86}

88/// One column reference in a rule body atom or head action: either a bound
89/// variable (identified by its [`egglog_backend_trait::VariableId`] rep) or a
90/// constant value.
91#[derive(Clone, Debug)]
92pub enum Slot {
93    Var(u32),
94    Const(u32),
95}
96
97impl Slot {
98    pub fn from_entry(e: &QueryEntry) -> Self {
99        match e {
100            QueryEntry::Var(v) => Slot::Var(v.id.rep()),
101            QueryEntry::Const { val, .. } => Slot::Const(val.rep()),
102        }
103    }
104}
105
106/// A body atom: a function table reference with one [`Slot`] per column.
107#[derive(Clone, Debug)]
108pub struct BodyAtom {
109    pub func: FunctionId,
110    pub slots: Vec<Slot>,
111}

117/// One operation in a rule **body**, in emission order. The host interpreter
118/// (`lib.rs::interpret_rule`) processes these left to right, threading a
119/// variable→value binding environment.
120#[derive(Clone, Debug)]
121pub enum BodyOp {
122    /// Match a table/relation atom: bind its variables (or constrain already-
123    /// bound vars / constants) against every row of `func`.
124    Atom(BodyAtom),
125    /// Evaluate a primitive `func(args..)`. The last entry (`ret`) is the
126    /// primitive's return slot: if it is an as-yet-unbound variable it is bound
127    /// to the result; otherwise the result must equal the slot's value (a
128    /// guard). `None` result (primitive failure, e.g. `!=` of equal args) prunes
129    /// the match.
130    Prim {
131        id: ExternalFunctionId,
132        args: Vec<Slot>,
133        ret: Slot,
134    },
135}
136
137/// One operation in a rule **head**, in emission order.
138#[derive(Clone, Debug)]
139pub enum HeadOp {
140    /// `(set func(key..) = val)` / relation `insert` — the slots are the full
141    /// row (inputs then output).
142    Set { func: FunctionId, slots: Vec<Slot> },
143    /// `(delete func(key..))` — slots address the row to retract by key.
144    Remove { func: FunctionId, slots: Vec<Slot> },
145    /// `(subsume func(key..))`.
146    Subsume { func: FunctionId, slots: Vec<Slot> },
147    /// RHS function lookup `(let v (func args..))`: look up the output column of
148    /// `func` for the given input slots, binding `ret` to it (creating the row
149    /// with a fresh id if absent — eq-sort constructor semantics).
150    Lookup {
151        func: FunctionId,
152        args: Vec<Slot>,
153        ret: u32,
154    },
155    /// RHS primitive call `(let v (prim args..))`: invoke the primitive and bind
156    /// `ret` to the result.
157    Call {
158        id: ExternalFunctionId,
159        args: Vec<Slot>,
160        ret: u32,
161    },
162    /// `(union l r)` — merge two eq-sort ids.
163    Union { l: Slot, r: Slot },
164    /// `(panic msg)`.
165    Panic(String),
166}
167
168/// The compiled IR for one egglog rule (Milestone 3 ordered form).
169///
170/// `body` is an ordered list of table-atom matches and primitive evaluations;
171/// `head` is an ordered list of writes / lookups / unions. The host interpreter
172/// runs the body as a nested-loop join over the current relation mirror,
173/// evaluating primitives via the embedded `Database`, then applies the head
174/// actions for each surviving binding.
175#[derive(Clone, Debug, Default)]
176pub struct RuleIr {
177    pub name: String,
178    pub body: Vec<BodyOp>,
179    pub head: Vec<HeadOp>,
180}
181
182impl BodyAtom {
183    pub fn from_entries(func: FunctionId, entries: &[QueryEntry]) -> Self {
184        BodyAtom {
185            func,
186            slots: entries.iter().map(Slot::from_entry).collect(),
187        }
188    }
189}

191/// Resolve a [`Slot`] to a concrete `u32` value given a variable→value lookup.
192/// Returns `None` if the slot is an unbound variable.
193pub fn slot_lookup(s: &Slot, get: &dyn Fn(u32) -> Option<u32>) -> Option<u32> {
194    match s {
195        Slot::Var(v) => get(*v),
196        Slot::Const(c) => Some(*c),
197    }
