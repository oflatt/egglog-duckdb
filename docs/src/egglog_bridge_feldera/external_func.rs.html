1//! External-function (primitive) registry for the Feldera/DBSP backend.
2//!
3//! ## Milestone 3 — primitives actually evaluate
4//!
5//! The canonical home of every registered primitive is the embedded
6//! [`egglog_core_relations::Database`] (see `lib.rs`), which is what gets
7//! *invoked* via `Database::with_execution_state`. This module is a thin
8//! side-table indexed by the **same** [`ExternalFunctionId`] the Database
9//! assigns, used for two things the Database doesn't surface conveniently:
10//!
11//! - **Names.** The rule builder recognizes built-in predicates (`!=`) by name
12//!   to lower them to circuit filters; the frontend records names via
13//!   `rename_prim`.
14//! - **Panic sentinels.** `Backend::new_panic` registers a `PanicFunc` in the
15//!   Database (so it is invokable) and mirrors the message here.
16
17use egglog_backend_trait::{ExecutionState, ExternalFunction, ExternalFunctionId, Value};
18use egglog_numeric_id::NumericId;
19
20/// Either a real registered primitive, a panic sentinel, or a freed slot.
21enum Slot {
22    Func,
23    Panic(#[allow(dead_code)] String),
24    Free,
25}
26
27/// A side-table of external-function metadata, indexed by the
28/// [`ExternalFunctionId`] the embedded `Database` assigned. `add_*_at` keeps
29/// this `Vec` aligned with the Database's `DenseIdMapWithReuse` (ids advance in
30/// lockstep). Names are tracked so the rule builder can recognize built-in
31/// predicates like `!=`.
32#[derive(Default)]
33pub struct ExternalFuncRegistry {
34    slots: Vec<Slot>,
35    names: Vec<Option<String>>,
36}
37
38impl ExternalFuncRegistry {
39    fn ensure_len(&mut self, idx: usize) {
40        while self.slots.len() <= idx {
41            self.slots.push(Slot::Free);
42            self.names.push(None);
43        }
44    }
45
46    /// Record a primitive at the id the Database assigned.
47    pub fn add_func_at(
48        &mut self,
49        id: ExternalFunctionId,
50        _func: Box<dyn ExternalFunction + 'static>,
51    ) {
52        let idx = id.rep() as usize;
53        self.ensure_len(idx);
54        self.slots[idx] = Slot::Func;
55        // (the func itself lives in the Database; we keep only metadata)
56    }
57
58    /// Record a panic sentinel at the id the Database assigned.
59    pub fn add_panic_at(&mut self, id: ExternalFunctionId, message: String) {
60        let idx = id.rep() as usize;
61        self.ensure_len(idx);
62        self.slots[idx] = Slot::Panic(message);
63    }
64
65    /// Record a primitive's display name (the frontend calls this via
66    /// `Backend`/`RuleBuilderOps::rename_prim`).
67    pub fn set_name(&mut self, id: ExternalFunctionId, name: String) {
68        let idx = id.rep() as usize;
69        self.ensure_len(idx);
70        self.names[idx] = Some(name);
71    }
72
73    /// The display name of a primitive, if recorded.
74    #[allow(dead_code)]
75    pub fn name(&self, id: ExternalFunctionId) -> Option<&str> {
76        self.names.get(id.rep() as usize).and_then(|n| n.as_deref())
77    }
78
79    /// Tombstone a slot. Idempotent.
80    pub fn free(&mut self, id: ExternalFunctionId) {
81        if let Some(slot) = self.slots.get_mut(id.rep() as usize) {
82            *slot = Slot::Free;
83        }
84    }
85
86    /// The panic message for a deferred-panic id, if any.
87    #[allow(dead_code)]
88    pub fn panic_message(&self, id: ExternalFunctionId) -> Option<&str> {
89        match self.slots.get(id.rep() as usize) {
90            Some(Slot::Panic(m)) => Some(m.as_str()),
91            _ => None,
92        }
93    }
94}
95
96/// A real, invokable panic sentinel registered into the Database by
97/// `Backend::new_panic`. Invoking it panics with the recorded message — the
98/// same observable behavior as the reference backend's `Action::Panic`.
99#[derive(Clone)]
100pub struct PanicFunc {
101    message: String,
102}
103
104impl PanicFunc {
105    pub fn new(message: String) -> Self {
106        PanicFunc { message }
107    }
108}
109
110impl ExternalFunction for PanicFunc {
111    fn invoke(&self, _state: &mut ExecutionState, _args: &[Value]) -> Option<Value> {
112        panic!("{}", self.message);
113    }
114}
