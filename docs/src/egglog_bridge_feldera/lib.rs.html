1//! # egglog-bridge-feldera
2//!
3//! A Feldera/DBSP-backed executor for egglog's resolved IR, behind the
4//! [`egglog_backend_trait::Backend`] interface.
5//!
6//! ## Milestone 1 — bounded per-iteration stepping
7//!
8//! - Register relations and *non-recursive* single-/double-join rules.
9//! - Run them so that `(run 1)` and `(run 3)` produce **different, bounded**
10//!   results — N rounds of extension, **NOT** full saturation in one call.
11//! - Read results back through `for_each` / `lookup_id` / `table_size`.
12//!
13//! ## Milestone 2 — union-find + rebuild (the research crux)
14//!
15//! The backend now faithfully RUNS multi-ruleset term-encoded programs that do
16//! unions + rebuild, matching the reference backend's per-function tuple counts
17//! (see `tests/rebuild_proof.rs`). Three new pieces of machinery:
18//!
19//! - **Ruleset-scoped execution.** `run_rules(&[RuleId])` runs an arbitrary
20//!   *subset* of rules. A DBSP circuit is static once built, so we build and
21//!   cache one circuit per distinct sorted rule subset (keyed in
22//!   [`EGraph::circuits`]). This lets the frontend schedule the term encoder's
23//!   `(saturate single_parent)` / `(saturate path_compress)` /
24//!   `(saturate uf_index)` rulesets in order.
25//! - **Retraction-rebuild.** `(delete …)` / the trait `remove` becomes a
26//!   negative-weight DBSP diff. The `@uf` rewrites (delete `a→b`, set `a→c`)
27//!   are retraction + insertion; the host folds the integrated insert/delete
28//!   diff streams into the mirror as `(old ∪ inserts) \ deletes`.
29//! - **Merge recognition.** `:merge (ordering-min old new)` (the term encoder's
30//!   `@uff` uf-index) is recognized DuckDB-style and implemented as a
31//!   lattice-min upsert at fold time. `Old` / `New` / relation modes too.
32//!
33//! ## The per-iteration model (load-bearing)
34//!
35//! The brief's central correction: egglog's `(run N)` is N bounded rounds with
36//! rebuild between them, so the backend must do **ONE egglog iteration per
37//! `run_rules` call**, not saturate. We therefore compile a **non-recursive**
38//! circuit (NO `recursive` scope): one `transaction()` is one round of rule
39//! application over the *current* relation contents. The frontend's existing
40//! loop drives `(run N)` / `(saturate R)` by calling `run_rules` repeatedly.
41//! See `compile.rs` for the circuit shape and `run_rules` below for the
42//! host-side feedback that takes the next hop.

107/// A shareable handle to a [`Database`] used purely as the primitive engine, so
108/// that the `Send + 'static` DBSP circuit closures can evaluate value-computing
109/// body primitives ON-CIRCUIT (Stage C of #23), instead of forcing a host
110/// nested-loop fallback.
111///
112/// The handle wraps `Arc<Mutex<Database>>`. The wrapped `Database` is a **clone**
113/// of [`EGraph::db`]; crucially, a cloned `Database` *shares* its base-value
114/// intern tables (`InternTable` clones the inner `Arc<ConcurrentVec>` /
115/// `Arc<Mutex<HashTable>>`), so a pure primitive evaluated through this clone
116/// interns its result into the SAME tables the host `Value`s came from — the
117/// returned handle is bit-for-bit identical to a host evaluation. Pure prims are
118/// stateless beyond interning, so re-evaluating them on-circuit is idempotent.
119///
120/// The Feldera backend is single-threaded (DBSP runs one worker synchronously
121/// during `step()`), so the `Mutex` is uncontended and cannot deadlock: the host
122/// never holds the lock while the circuit steps (it locks, evaluates, unlocks).
123/// The `unsafe impl Send + Sync` mirrors `EGraph`'s own assertion (the egraph is
124/// only ever driven from a single thread); a `Database`'s `Rc`-free internals are
125/// `Send`-safe under that single-thread invariant.
126#[derive(Clone)]
127pub struct PrimEngine(Arc<Mutex<Database>>);
128
129// SAFETY: see `EGraph`'s `unsafe impl Send/Sync` below — the egraph (and hence
130// this handle) is only ever used from a single thread; DBSP runs its worker
131// synchronously on that same thread during `step()`.
132unsafe impl Send for PrimEngine {}
133unsafe impl Sync for PrimEngine {}
134
135impl PrimEngine {
136    /// Wrap a `Database` as a shared primitive-engine handle.
137    pub(crate) fn new(db: Database) -> Self {
138        PrimEngine(Arc::new(Mutex::new(db)))
139    }
140
141    /// Evaluate primitive `id` on `args` through the shared engine, returning
142    /// the interned result handle (or `None` if the prim "fails", e.g. `!=` of
143    /// equal args — which prunes the match). Used by the on-circuit call-prim
144    /// node; reuses the exact host eval path (`call_external_func`) so prim
145    /// semantics are never reimplemented.
146    pub(crate) fn eval(&self, id: ExternalFunctionId, args: &[Value]) -> Option<Value> {
147        self.0
148            .lock()
149            .unwrap()
150            .with_execution_state(|st| st.call_external_func(id, args))
151    }
152
153    /// A clone of the engine's [`ContainerValues`] store. A cloned `Database`
154    /// shares its base-value intern tables (via `Arc`) but DEEP-CLONES its
155    /// container registry, so any container (proof mode's `Pair<sort, proof>`)
156    /// a prim interns through this engine lives ONLY in this clone's store —
157    /// invisible to `EGraph::db`. We snapshot it here so the host can fold it
158    /// back into `db` after each iteration, keeping the single authoritative
159    /// store (`db`, read by proof extraction) consistent with the ids the
160    /// on-circuit / head prims actually produced. See
161    /// [`EGraph::sync_prim_engine_containers`].
162    pub(crate) fn container_values_snapshot(&self) -> egglog_core_relations::ContainerValues {
163        self.0.lock().unwrap().container_values().clone()
164    }
165}

171/// What we remember about each registered relation/function.
172struct RelationInfo {
173    name: String,
174    /// Number of columns (including the output column for functions).
175    pub(crate) arity: usize,
176    /// The column types (one per column), used by `dbsp_join::plan_join` to
177    /// type-gate prim inlining: rep-arithmetic (`ordering-min/max`, `<`) is only
178    /// valid on `ColumnTy::Id` columns (rep IS the union-find id) and on bool
179    /// columns; on other base values the rep is an intern handle whose order
180    /// differs from the logical value.
181    pub(crate) schema: Vec<ColumnTy>,
182    /// True for functions/constructors that have an output column.
183    has_output: bool,
184    /// How functional-dependency conflicts are resolved at flush time. For a
185    /// function the key is the input columns (`arity - 1`) and the output column
186    /// is resolved per this mode; for a relation it is [`MergeMode::Relation`]
187    /// (whole row is the key, nothing to resolve).
188    merge: MergeMode,
189    /// True iff this function uses identity-on-miss lookup semantics
190    /// (`DefaultVal::Identity`): an action-position lookup of an absent key
191    /// resolves to the key itself, with no row inserted. Used by the
192    /// canonicalize-at-creation encoding for the flat UF-index `@UF_Sf`.
193    /// Only valid for a single-key function whose key and output share a type.
194    identity_on_miss: bool,
195    /// `--native-merge` NATIVE CUSTOM `:merge`: the retained [`MergeFn`] tree for a
196    /// custom `:merge` whose body either BUILDS TERMS (a top-level
197    /// `IfEq{canon(old), canon(new), …, Seq[<Construct/TableInsert>…, <value>]}`
198    /// from `translate_term_build_merge_to_seq`) or is a pure VALUE-FOLD (a fold of
199    /// primitives over `old`/`new`, e.g. `(from-string (to-string (* new old)))`).
200    /// `add_table` retains it here — instead of collapsing to a 4-variant
201    /// [`MergeMode`] — so `resolve_merge` can EXECUTE it host-side
202    /// (`resolve_term_build_merge` / `eval_term_build_node`): the value fold is a
203    /// side-effect-free pairwise fold, the term build mints e-nodes via
204    /// `lookup_or_create`, writes the `@<C>View` rows, and returns the merged
205    /// eclass. `None` for every other merge (the [`merge`](Self::merge)
206    /// `MergeMode` drives those).
207    merge_tree: Option<MergeFn>,
208}

599    /// Resolve functional-dependency conflicts in a relation's mirror set: for
600    /// each key (input columns) keep a single output column chosen by the merge
601    /// mode. Relations (whole-row key) are left untouched. Returns whether any
602    /// row was actually changed/collapsed.
603    ///
604    /// INCREMENTAL: the pre-call mirror is already FD-resolved (this runs every
605    /// `run_rules` call), so only the keys whose rows were `set` this call (in
606    /// `keys`) can newly conflict. We re-resolve just those keys instead of
607    /// rebuilding the whole function — O(touched-keys), not O(state). The
608    /// deterministic fold order (sort) is preserved PER KEY, which gives the same
609    /// chosen output as the old whole-set sort+fold (the fold is independent
610    /// across distinct keys).
611    fn resolve_merge(&mut self, f: FunctionId, keys: &HashSet<Vec<u32>>) -> bool {
612        let arity = self.info(f).arity;
613        // `--native-merge` NATIVE CUSTOM `:merge`: a custom `:merge` whose body
614        // builds terms (or is a pure value fold) runs its retained `MergeFn` tree
615        // host-side, minting e-nodes and writing `@<C>View` rows. Dispatch to the
616        // dedicated interpreter, which folds the conflicting candidate rows pairwise
617        // through the tree. Checked FIRST so the retained tree drives the merge even
618        // though the function still carries a (sentinel `New`) `MergeMode`.
619        if self.info(f).merge_tree.is_some() {
620            return self.resolve_term_build_merge(f, keys);
621        }
622        // `--native-merge` PROOF mode (2-table): a single-output eclass view whose
623        // FD conflict means the two eclasses are congruent. Compose the congruence
624        // edge proof from the side-table and write it (with the union edge) into the
625        // relational proof-UF `@UF_S`. Dispatch to the dedicated resolver.
626        if self.native_merge_proof.contains_key(&f) {
627            return self.resolve_proof_congruence_merge(f, keys);
628        }
629        let merge = self.merge_mode(f);
630        if !matches!(merge, MergeMode::Old | MergeMode::New | MergeMode::Min)
631            || arity == 0
632            || keys.is_empty()
633        {
634            return false;
635        }
636        let Some(set) = self.mirror.get(&f) else {
637            return false;
638        };
639        let inputs_len = arity - 1;
640        // Gather the candidate rows for the touched keys only.
641        let mut by_key: HashMap<&[u32], Vec<&Row>> = HashMap::new();
642        for row in set.iter() {
643            let key: Vec<u32> = (0..inputs_len).map(|i| row_col(row, i)).collect();
644            if keys.contains(&key) {
645                by_key.entry(&row[..inputs_len]).or_default().push(row);
646            }
647        }
648        // Resolve each touched key; collect the rows to remove and the winner to
649        // insert. Only keys with >1 candidate row can change.
650        let mut new_rows: Vec<Row> = Vec::new();
651        let mut drop_rows: HashSet<Row> = HashSet::new();
652        for (_key, mut cands) in by_key {
653            if cands.len() < 2 {
654                continue;
655            }
656            // Deterministic fold order (mirror is a HashSet — arbitrary order).
657            cands.sort();
658            let mut chosen = row_col(cands[0], inputs_len);
659            for row in &cands[1..] {
660                let out = row_col(row, inputs_len);
661                chosen = match merge {
662                    MergeMode::Old => chosen,
663                    MergeMode::New => out,
664                    MergeMode::Min => chosen.min(out),
665                    MergeMode::Relation => unreachable!(),
666                };
667            }
668            // The winning row.
669            let mut winner: Vec<u32> = cands[0][..inputs_len].to_vec();
670            winner.push(chosen);
671            let winner: Row = winner.into_boxed_slice();
672            for row in &cands {
673                if ***row != *winner {
674                    drop_rows.insert((**row).clone());
675                }
676            }
677            new_rows.push(winner);
678        }
679        if drop_rows.is_empty() {
680            return false;
681        }
682        let set = std::rc::Rc::make_mut(self.mirror.get_mut(&f).unwrap());
683        for r in &drop_rows {
684            set.remove(r);
685        }
686        for r in &new_rows {
687            set.insert(r.clone());
688        }
689        true
690    }

692    /// `--native-merge` PROOF mode (2-table): resolve an FD conflict on a
693    /// single-output eclass view `(children) -> eclass` by composing the congruence
694    /// edge proof from the proof SIDE-TABLE and writing it (with the union edge)
695    /// into the relational proof-UF `@UF_S`.
696    ///
697    /// For each touched key with >1 candidate eclass, fold the candidates pairwise
698    /// (matching the bridge's `UnionIntoUfWithProof`, egglog-bridge `run`):
699    ///   * read each row's view proof from the side-table (`children..., eclass`);
700    ///   * `larger = max(cur, new)`, `smaller = min(cur, new)`, with the matching
701    ///     `larger_pf` / `smaller_pf`;
702    ///   * `edge = Trans(larger_pf, Sym(smaller_pf))` — EXACTLY the orientation of
703    ///     the rule-encoded `@congruence_rule`
704    ///     (`(set (@UF_S larger smaller) (Trans larger_pf (Sym smaller_pf)))`);
705    ///   * write `[larger, smaller, edge]` into `@UF_S` (respecting its `:merge old`
706    ///     — first proof per `(larger, smaller)` key wins, so the insert is
707    ///     idempotent), and keep `smaller` as the surviving (min) view eclass.
708    ///
709    /// The relational singleparent / path_compress rules then canonicalize the
710    /// proof-UF on the next iteration; the view's rebuild rules re-canonicalize any
711    /// other rows holding the loser eclass.
712    fn resolve_proof_congruence_merge(&mut self, f: FunctionId, keys: &HashSet<Vec<u32>>) -> bool {
713        let arity = self.info(f).arity;
714        if arity == 0 || keys.is_empty() {
715            return false;
716        }
717        let cfg = *self.native_merge_proof.get(&f).expect(
718            "resolve_proof_congruence_merge called on a function with no proof-congruence config",
719        );
720        let inputs_len = arity - 1;
721        // Gather candidate rows for the touched keys only (clone out — the proof
722        // composition below mints `Trans`/`Sym` e-nodes and writes `@UF_S`, so it
723        // cannot hold a borrow of the view mirror).
724        let mut by_key: HashMap<Vec<u32>, Vec<Row>> = HashMap::new();
725        if let Some(set) = self.mirror.get(&f) {
726            for row in set.iter() {
727                let key: Vec<u32> = (0..inputs_len).map(|i| row_col(row, i)).collect();
728                if keys.contains(&key) {
729                    by_key.entry(key).or_default().push(row.clone());
730                }
731            }
732        }
733        // A per-call hash-cons index for `lookup_or_create` (the `Trans`/`Sym`
734        // mints), lazily built per function — an optimisation, identical results.
735        let mut lookup_index: HashMap<FunctionId, HashMap<Box<[u32]>, u32>> = HashMap::new();
736
737        let mut new_rows: Vec<Row> = Vec::new();
738        let mut drop_rows: HashSet<Row> = HashSet::new();
739        // Stable iteration over touched keys so the `Trans`/`Sym` mint order is
740        // deterministic (run-to-run reproducible byte image).
741        let mut key_list: Vec<Vec<u32>> = by_key.keys().cloned().collect();
742        key_list.sort();
743        for key in &key_list {
744            let mut cands = by_key.remove(key).unwrap();
745            if cands.len() < 2 {
746                continue;
747            }
748            // Deterministic fold order (mirror is a HashSet — arbitrary order).
749            cands.sort();
750            // Fold pairwise: `cur_eclass`/`cur_pf` start at the first candidate; for
751            // each later candidate compose the congruence edge and keep the min.
752            let mut cur_eclass = row_col(&cands[0], inputs_len);
753            let mut cur_pf = self.read_side_table_proof(cfg.proof_side_table, key, cur_eclass);
754            for next in &cands[1..] {
755                let new_eclass = row_col(next, inputs_len);
756                if cur_eclass == new_eclass {
757                    continue;
758                }
759                let new_pf = self.read_side_table_proof(cfg.proof_side_table, key, new_eclass);
760                // Orientation byte-for-byte with the bridge's `UnionIntoUfWithProof`
761                // and the rule-encoded `@congruence_rule`.
762                let (larger, smaller, larger_pf, smaller_pf) = if cur_eclass > new_eclass {
763                    (cur_eclass, new_eclass, cur_pf, new_pf)
764                } else {
765                    (new_eclass, cur_eclass, new_pf, cur_pf)
766                };
767                // edge_proof = Trans(larger_pf, Sym(smaller_pf)).
768                let sym_smaller = interpret::lookup_or_create(
769                    self,
770                    cfg.sym,
771                    &[Value::new(smaller_pf)],
772                    &mut lookup_index,
773                )
774                .rep();
775                let edge_proof = interpret::lookup_or_create(
776                    self,
777                    cfg.trans,
778                    &[Value::new(larger_pf), Value::new(sym_smaller)],
779                    &mut lookup_index,
780                )
781                .rep();
782                // Write the proof-carrying union edge into `@UF_S` (`:merge old` —
783                // skip if a row for this `(larger, smaller)` key already exists).
784                self.insert_proof_uf_edge(cfg.parent_table, larger, smaller, edge_proof);
785                // The surviving eclass is the min; its proof is the min's proof.
786                cur_eclass = smaller;
787                cur_pf = smaller_pf;
788            }
789            // The winner row: same children key, surviving (min) eclass.
790            let mut winner: Vec<u32> = key.clone();
791            winner.push(cur_eclass);
792            let winner: Row = winner.into_boxed_slice();
793            for row in &cands {
794                if **row != *winner {
795                    drop_rows.insert((*row).clone());
796                }
797            }
798            new_rows.push(winner);
799        }
800
801        // Reconcile the view mirror: drop losers, insert winners.
802        let mut changed = false;
803        if !drop_rows.is_empty() || !new_rows.is_empty() {
804            let set = std::rc::Rc::make_mut(self.mirror.entry(f).or_default());
805            for r in &drop_rows {
806                changed |= set.remove(r);
807            }
808            for r in &new_rows {
809                changed |= set.insert(r.clone());
810            }
811        }
812        // Returning `true` whenever an edge was composed keeps the saturate loop
813        // iterating until the proof-UF maintenance rules + rebuild reach a fixpoint
814        // (the relational signal for the dropped `@congruence_rule`). A no-op key
815        // (single candidate) changes nothing and returns `changed` (false).
816        changed
817    }

819    /// Read the view proof for `(children..., eclass)` from the proof SIDE-TABLE.
820    /// CANON-AT-CREATION guarantees a hit: every view row's proof is written to the
821    /// side-table when the row is born (see `update_view` in the encoder). A miss
822    /// would mean a broken invariant — `debug_assert` it rather than fabricate a
823    /// proof. NEVER routes through `lookup_or_create` (which would MINT a junk row
824    /// on a miss), so a release build returns the (sentinel) eclass id on a miss
825    /// instead of silently corrupting the proof-UF.
826    fn read_side_table_proof(&self, side_table: FunctionId, children: &[u32], eclass: u32) -> u32 {
827        let arity = self.info(side_table).arity;
828        let inputs_len = arity - 1; // children..., eclass
829        if let Some(set) = self.mirror.get(&side_table) {
830            for row in set.iter() {
831                if row.len() != arity {
832                    continue;
833                }
834                // Key columns are `children..., eclass`; the proof is the output.
835                if row_col(row, inputs_len - 1) == eclass
836                    && (0..children.len()).all(|i| row_col(row, i) == children[i])
837                {
838                    return row_col(row, inputs_len);
839                }
840            }
841        }
842        debug_assert!(
843            false,
844            "2-table proof-congruence: side-table proof miss for key (children={children:?}, \
845             eclass={eclass}) — canon-at-creation invariant broken (would fabricate a proof)"
846        );
847        eclass
848    }

850    /// Write a proof-carrying congruence edge `[larger, smaller, edge_proof]` into
851    /// the relational proof-UF `@UF_S` (`(S S) -> Proof :merge old`). Respects
852    /// `:merge old`: if a row for the `(larger, smaller)` key already exists, keep
853    /// it (idempotent — the first composed proof wins, exactly as the rule-encoded
854    /// `(set (@UF_S larger smaller) …)` with `:merge old` would).
855    fn insert_proof_uf_edge(
856        &mut self,
857        parent_table: FunctionId,
858        larger: u32,
859        smaller: u32,
860        edge_proof: u32,
861    ) {
862        if let Some(set) = self.mirror.get(&parent_table) {
863            // `:merge old` — an existing edge for this key wins; do not overwrite.
864            let exists = set.iter().any(|row| {
865                row.len() >= 2 && row_col(row, 0) == larger && row_col(row, 1) == smaller
866            });
867            if exists {
868                return;
869            }
870        }
871        let row: Row = vec![larger, smaller, edge_proof].into_boxed_slice();
872        self.mirror_insert(parent_table, row);
873    }

875    /// `--native-merge`: resolve FD conflicts on a custom `:merge` function with a
876    /// retained `MergeFn` tree, by EXECUTING the tree host-side. Handles BOTH native
877    /// merge classes:
878    ///   * VALUE-FOLD — a pure fold of primitives over `old`/`new` (e.g.
879    ///     `(from-string (to-string (* new old)))`): a side-effect-free pairwise
880    ///     fold returning a value.
881    ///   * TERM-BUILD — a body that mints e-nodes (the `IfEq{…, Seq[…]}` tree from
882    ///     `translate_term_build_merge_to_seq`): mints helper e-nodes via
883    ///     `lookup_or_create` and writes the `@<C>View` rows as a side effect.
884    ///
885    /// For each touched key with >1 candidate row, fold the candidates pairwise
886    /// through the tree (`cur` = the chosen-so-far row, `new` = the next candidate),
887    /// taking the tree's returned value as the new chosen output. Drops the losing
888    /// rows and inserts the single winner — the FD shape the rest of the engine
889    /// expects. Mirrors the bridge's `ResolvedMergeFn::run`. The value-fold class
890    /// never touches `term_build_view_keys` (no `TableInsert`/`Construct`), so the
891    /// `run_iteration` view-congruence drain is a no-op for it.
892    ///
893    /// The deterministic per-key fold order (sort) matches the `Old`/`New`/`Min`
894    /// path, so the chosen output is stable across runs. View rows a term-build fold
895    /// mints are reconciled for congruence by the `term_build_view_keys` drain in
896    /// `run_iteration`.
897    fn resolve_term_build_merge(&mut self, f: FunctionId, keys: &HashSet<Vec<u32>>) -> bool {
898        let arity = self.info(f).arity;
899        if arity == 0 || keys.is_empty() {
900            return false;
901        }
902        let inputs_len = arity - 1;
903        // Gather the candidate rows for the touched keys only (matching the
904        // incremental `Old`/`New`/`Min` resolver). Clone out so the tree
905        // interpreter can mutate the egraph (mint / view-write) without holding a
906        // borrow of the mirror.
907        let mut by_key: HashMap<Vec<u32>, Vec<Row>> = HashMap::new();
908        if let Some(set) = self.mirror.get(&f) {
909            for row in set.iter() {
910                let key: Vec<u32> = (0..inputs_len).map(|i| row_col(row, i)).collect();
911                if keys.contains(&key) {
912                    by_key.entry(key).or_default().push(row.clone());
913                }
914            }
915        }
916        // The merge tree (cloned out — the interpreter borrows `&mut self`).
917        let tree = self
918            .info(f)
919            .merge_tree
920            .clone()
921            .expect("resolve_term_build_merge called on a non-term-build function");
922        // A per-call hash-cons index for `lookup_or_create` (lazily built per
923        // function from the mirror — an optimisation, identical results without it).
924        let mut lookup_index: HashMap<FunctionId, HashMap<Box<[u32]>, u32>> = HashMap::new();
925
926        let mut new_rows: Vec<Row> = Vec::new();
927        let mut drop_rows: HashSet<Row> = HashSet::new();
928        // Stable iteration over the touched keys so the mint order (and thus any
929        // run-to-run id assignment) is deterministic. Hash-cons makes the
930        // STRUCTURAL result insensitive to the integer ids, but a stable order
931        // keeps the byte image reproducible across runs.
932        let mut key_list: Vec<Vec<u32>> = by_key.keys().cloned().collect();
933        key_list.sort();
934        for key in &key_list {
935            let mut cands = by_key.remove(key).unwrap();
936            if cands.len() < 2 {
937                continue;
938            }
939            // Deterministic fold order (mirror is a HashSet — arbitrary order).
940            cands.sort();
941            // Fold pairwise: `cur` starts as the first candidate; for each later
942            // candidate run the tree with (cur, new) and take its returned value as
943            // the new `cur` output. The key columns are identical across all
944            // candidates (same FD key), so reuse them.
945            let mut cur: Row = cands[0].clone();
946            for next in &cands[1..] {
947                let merged =
948                    self.eval_term_build_node(&tree, &cur, next, inputs_len, &mut lookup_index);
949                // Build the new `cur` row: same key, merged output.
950                let mut row: Vec<u32> = key.clone();
951                row.push(merged);
952                cur = row.into_boxed_slice();
953            }
954            // The winner is the final folded row; every candidate that differs is
955            // dropped.
956            for row in &cands {
957                if **row != *cur {
958                    drop_rows.insert((*row).clone());
959                }
960            }
961            new_rows.push(cur);
962        }
963
964        // Reconcile the mirror: drop losers, insert winners.
965        let mut changed = false;
966        if !drop_rows.is_empty() || !new_rows.is_empty() {
967            let set = std::rc::Rc::make_mut(self.mirror.entry(f).or_default());
968            for r in &drop_rows {
969                changed |= set.remove(r);
970            }
971            for r in &new_rows {
972                changed |= set.insert(r.clone());
973            }
974        }
975        // Any e-node minted / view row written this resolve is a real change — keep
976        // the saturate loop iterating until congruence + rebuild reach a fixpoint.
977        if !self.term_build_view_keys.is_empty() {
978            changed = true;
979        }
980        changed
981    }
982
983    /// Evaluate one node of a term-build `MergeFn` tree against the two conflicting
984    /// FD rows (`cur` = chosen-so-far, `new` = next candidate), performing the
985    /// node's side effects (mint via `lookup_or_create`, `@<C>View` row writes,
986    /// native-UF self-unions) and returning its value. `inputs_len` is the number
987    /// of key columns, so value column `i` is at `row[inputs_len + i]`; bare
988    /// `Old`/`New` read the sole output column (index 0). Mirrors the bridge's
989    /// `ResolvedMergeFn::run` for the term-build subset. Columns are read via the
990    /// uniform `Row` accessor (`row[..]`), so the packed-vs-unpacked detail does not
991    /// leak here — the rows are already `u32` reps (see `compile::Row`).
992    fn eval_term_build_node(
993        &mut self,
994        node: &MergeFn,
995        cur: &[u32],
996        new: &[u32],
997        inputs_len: usize,
998        lookup_index: &mut HashMap<FunctionId, HashMap<Box<[u32]>, u32>>,
999    ) -> u32 {
1000        match node {
1001            MergeFn::Const(v) => v.rep(),
1002            // A single-output term-build function: `Old`/`New` are the sole output
1003            // column (index 0), so `Old == OldCol(0)`.
1004            MergeFn::Old | MergeFn::OldCol(0) => cur[inputs_len],
1005            MergeFn::New | MergeFn::NewCol(0) => new[inputs_len],
1006            MergeFn::OldCol(i) => cur[inputs_len + i],
1007            MergeFn::NewCol(i) => new[inputs_len + i],
1008            MergeFn::Primitive(ext, args) => {
1009                let argv: Vec<Value> = args
1010                    .iter()
1011                    .map(|a| {
1012                        Value::new(self.eval_term_build_node(a, cur, new, inputs_len, lookup_index))
1013                    })
1014                    .collect();
1015                self.eval_prim_internal(*ext, &argv)
1016                    .unwrap_or_else(|| Value::new(cur[inputs_len]))
1017                    .rep()
1018            }
1019            MergeFn::Function(func, args) => {
1020                let argv: Vec<u32> = args
1021                    .iter()
1022                    .map(|a| self.eval_term_build_node(a, cur, new, inputs_len, lookup_index))
1023                    .collect();
1024                // A `Function` lookup mints on miss (`lookup_or_create`), except for
1025                // an identity-on-miss function (the relational `@UF_Sf` flat index),
1026                // where `lookup_or_create` returns the key itself on miss — the
1027                // find-or-self canon read, no mint, no write.
1028                let key: Vec<Value> = argv.into_iter().map(Value::new).collect();
1029                interpret::lookup_or_create(self, *func, &key, lookup_index).rep()
1030            }
1031            MergeFn::Construct(table, key_args, value_args) => {
1032                // Mint a constructor e-node (hash-consed; existing key -> its id,
1033                // missing -> fresh id + insert), returning its output eclass. In
1034                // non-proof TERM mode `value_args` is empty (no extra value
1035                // columns), so this is exactly a single-output `lookup_or_create`.
1036                let key: Vec<Value> = key_args
1037                    .iter()
1038                    .map(|a| {
1039                        Value::new(self.eval_term_build_node(a, cur, new, inputs_len, lookup_index))
1040                    })
1041                    .collect();
1042                // A non-empty `value_args` would be a multi-value constructor, which
1043                // Feldera does not yet support (the single-output `lookup_or_create`
1044                // only mints the output column).
1045                assert!(
1046                    value_args.is_empty(),
1047                    "Feldera term-build merge: multi-value Construct (value_args) not supported"
1048                );
1049                interpret::lookup_or_create(self, *table, &key, lookup_index).rep()
1050            }
1051            MergeFn::TableInsert(table, args) => {
1052                let row: Vec<u32> = args
1053                    .iter()
1054                    .map(|a| self.eval_term_build_node(a, cur, new, inputs_len, lookup_index))
1055                    .collect();
1056                // A `@<C>View` (or relational `@UF_Sf`) row: mirror-insert it and
1057                // record its key so `run_iteration` reconciles the target's own
1058                // congruence / `Min` merge. (The native-UF WRINKLE — routing a
1059                // `@UF_Sf` insert into an in-core UF — is gone: Feldera is off
1060                // native-UF, so `@UF_Sf` is a relational table like any other.)
1061                let view_arity = self.info(*table).arity;
1062                let view_inputs = view_arity.saturating_sub(1);
1063                let key: Vec<u32> = row.iter().take(view_inputs).copied().collect();
1064                self.mirror_insert(*table, row.into_boxed_slice());
1065                self.term_build_view_keys
1066                    .entry(*table)
1067                    .or_default()
1068                    .insert(key);
1069                // The bridge returns the OLD value of the resolving column; the
1070                // enclosing `Seq` discards it.
1071                cur[inputs_len]
1072            }
1073            MergeFn::Seq(items) => {
1074                let mut result = cur[inputs_len];
1075                for item in items {
1076                    result = self.eval_term_build_node(item, cur, new, inputs_len, lookup_index);
1077                }
1078                result
1079            }
1080            MergeFn::IfEq { a, b, then, els } => {
1081                let av = self.eval_term_build_node(a, cur, new, inputs_len, lookup_index);
1082                let bv = self.eval_term_build_node(b, cur, new, inputs_len, lookup_index);
1083                if av == bv {
1084                    self.eval_term_build_node(then, cur, new, inputs_len, lookup_index)
1085                } else {
1086                    self.eval_term_build_node(els, cur, new, inputs_len, lookup_index)
1087                }
1088            }
1089            MergeFn::Columns(cols) => match cols.as_slice() {
1090                [single] => self.eval_term_build_node(single, cur, new, inputs_len, lookup_index),
1091                _ => panic!(
1092                    "Feldera term-build merge: multi-value Columns ({} cols) not supported",
1093                    cols.len()
1094                ),
1095            },
1096            MergeFn::AssertEq
1097            | MergeFn::UnionId
1098            | MergeFn::UnionIntoUf(_)
1099            | MergeFn::UnionIntoParentTable { .. }
1100            | MergeFn::UnionIntoParentTableWithProof { .. }
1101            | MergeFn::UnionIntoUfWithProof { .. }
1102            | MergeFn::EclassMinProof { .. }
1103            // `KeyCol` is emitted only by the bridge-only proof-mode term-build
1104            // merge; Feldera never runs that path (gated on
1105            // `supports_complex_merge()`).
1106            | MergeFn::KeyCol(_) => {
1107                panic!(
1108                    "Feldera term-build merge: unexpected congruence/proof merge variant inside a \
1109                     term-build tree (these are not nested in a term build)"
1110                )
1111            }
1112        }
1113    }
1114}

1177/// Map a *scalar* (single value column) [`MergeFn`] to the Feldera [`MergeMode`] that resolves one
1178/// output column's FD conflict. Mirrors the DuckDB/FlowLog recognition:
1179///   - `AssertEq` / `Old`         => keep the old value
1180///   - `New`                      => keep the new value
1181///   - `UnionId` / `UnionIntoUf`  => lattice-min (the union-find leader). `UnionIntoUf` is the
1182///     `--native-merge` form; the *target UF* it names is read separately (see
1183///     [`native_merge_target_uf`]) and the union itself is injected by `resolve_merge`, so here it
1184///     only governs which value survives the FD collapse — identical to `UnionId` (`Min`).
1185///   - `Primitive(_)`             => lattice-min (the term encoder's `@uff` `:merge ordering-min`
1186///     and user `:merge (min old new)`).
1187///   - `OldCol(_)`                => keep the old value
1188///   - `NewCol(_)`                => keep the new value
1189///   - `Function`/`Const`         => fall back to `Old`
1190fn merge_mode_for_scalar(merge: &egglog_backend_trait::MergeFn) -> MergeMode {
1191    use egglog_backend_trait::MergeFn;
1192    match merge {
1193        MergeFn::AssertEq | MergeFn::Old => MergeMode::Old,
1194        MergeFn::New => MergeMode::New,
1195        // `UnionIntoParentTable` is the bridge-only RELATIONAL native-merge merge
1196        // (`--native-merge` without `--native-uf`); Feldera always uses the
1197        // UF-backed `UnionIntoUf`, but the surviving value is the min either way.
1198        MergeFn::UnionId | MergeFn::UnionIntoUf(_) | MergeFn::UnionIntoParentTable { .. } => {
1199            MergeMode::Min
1200        }
1201        MergeFn::Primitive(_, _) => MergeMode::Min,
1202        // For a single-output table `OldCol(i)`/`NewCol(i)` can only reference column 0, which is
1203        // this table's sole value column — i.e. plain old/new.
1204        MergeFn::OldCol(_) => MergeMode::Old,
1205        MergeFn::NewCol(_) => MergeMode::New,
1206        MergeFn::Function(_, _) | MergeFn::Const(_) => MergeMode::Old,
1207        // `--native-merge` PROOF mode 2-table congruence (Feldera's single-output
1208        // proof view): the surviving value is the min eclass (the UF leader),
1209        // identical to the term-mode `UnionIntoUf`. The proof side-effect (reading
1210        // the side-table, composing the edge proof, writing `@UF_S`) is handled by
1211        // `resolve_proof_congruence_merge` via the `native_merge_proof` map, which
1212        // is captured separately at `add_table`; here we only govern which value
1213        // survives the FD collapse.
1214        MergeFn::UnionIntoParentTableWithProof { .. } => MergeMode::Min,
1215        // The TUPLE-output proof merges (`UnionIntoUfWithProof` / `EclassMinProof`)
1216        // and the proof-mode term-build `KeyCol` are bridge-only; Feldera uses the
1217        // 2-table form above instead, so these never reach it.
1218        MergeFn::UnionIntoUfWithProof { .. }
1219        | MergeFn::EclassMinProof { .. }
1220        | MergeFn::KeyCol(_) => {
1221            panic!("Feldera backend does not support tuple-output proof-mode native-merge merges")
1222        }
1223        MergeFn::Columns(_) => unreachable!("merge_mode_for_scalar called on a Columns merge"),
1224        // `Seq` / `TableInsert` / `Construct` / `IfEq` are the bridge-only
1225        // `:merge`-multiple-actions variants (PR #933 + the term-build guard);
1226        // Feldera has no multi-action merge support, so they never reach this
1227        // backend.
1228        MergeFn::Seq(_)
1229        | MergeFn::TableInsert(..)
1230        | MergeFn::Construct(..)
1231        | MergeFn::IfEq { .. } => {
1232            panic!(
1233                "Feldera backend does not support multi-action (Seq/TableInsert/Construct/IfEq) merges"
1234            )
1235        }
1236    }
1237}

1239/// Resolve the [`MergeMode`] for a function table with a single output column.
1240///
1241/// A top-level [`MergeFn::Columns`] of length 1 is the multi-value spelling of a single-output
1242/// merge and collapses to its one inner column's mode. Feldera's relation model carries exactly
1243/// one output-column merge mode, so a genuine multi-value `Columns` (len > 1) — a tuple-output
1244/// function — cannot be expressed and is a clean, documented error. (Tuple-output views are a
1245/// later step; the term encoder does not emit them on Feldera today.)
1246fn merge_mode_for_table(merge: &egglog_backend_trait::MergeFn, name: &str) -> MergeMode {
1247    use egglog_backend_trait::MergeFn;
1248    match merge {
1249        MergeFn::Columns(cols) => match cols.as_slice() {
1250            [single] => merge_mode_for_scalar(single),
1251            _ => panic!(
1252                "Feldera backend does not support tuple-output (multi-value) merges yet \
1253                 (function `{name}` has a {}-column `Columns` merge)",
1254                cols.len()
1255            ),
1256        },
1257        other => merge_mode_for_scalar(other),
1258    }
1259}

1261/// `--native-merge`: the per-sort UF a view's congruence union targets, named UNIFORMLY by the
1262/// `UnionIntoUf(uf_func)` merge variant (top-level, or a single-column `Columns` wrapping it).
1263/// `None` for any other merge.
1264fn native_merge_target_uf(merge: &egglog_backend_trait::MergeFn) -> Option<FunctionId> {
1265    use egglog_backend_trait::MergeFn;
1266    match merge {
1267        MergeFn::UnionIntoUf(uf_func) => Some(*uf_func),
1268        MergeFn::Columns(cols) => match cols.as_slice() {
1269            [MergeFn::UnionIntoUf(uf_func)] => Some(*uf_func),

1272        _ => None,

1276/// `--native-merge` PROOF mode (2-table): the proof-congruence config a view's
1277/// FD-conflict merge targets, named by the `UnionIntoParentTableWithProof` merge
1278/// variant (top-level, or a single-column `Columns` wrapping it). `None` for any
1279/// other merge.
1280fn native_merge_proof_target(merge: &egglog_backend_trait::MergeFn) -> Option<NativeMergeProof> {
1281    use egglog_backend_trait::MergeFn;
1282    fn from_variant(merge: &MergeFn) -> Option<NativeMergeProof> {
1283        match merge {
1284            MergeFn::UnionIntoParentTableWithProof {
1285                parent_table,
1286                proof_side_table,
1287                trans,
1288                sym,
1289            } => Some(NativeMergeProof {
1290                parent_table: *parent_table,
1291                proof_side_table: *proof_side_table,
1292                trans: *trans,
1293                sym: *sym,
1294            }),
1295            _ => None,
1296        }
1297    }
1298    match merge {
1299        MergeFn::Columns(cols) => match cols.as_slice() {
1300            [single] => from_variant(single),
1301            _ => None,
1302        },
1303        other => from_variant(other),
1304    }
1305}

1307/// `--native-merge` TERM-BUILD: whether `merge` is a term-building custom `:merge`
1308/// tree the Feldera host interpreter runs natively (`resolve_term_build_merge`).
1309///
1310/// The frontend lowers a term-building body (e.g. `(C2 (C1 old new) (C2 old new))`)
1311/// to a top-level `IfEq{canon(old), canon(new), then, els: Seq[…]}` guard (see
1312/// `translate_term_build_merge_to_seq`); a bare `Seq` (no guard) is also accepted
1313/// defensively. These are the only multi-action shapes Feldera executes; the other
1314/// multi-action variants (`TableInsert`/`Construct` at top level) are never emitted
1315/// as a top-level merge by the frontend and stay unsupported.
1316fn is_term_build_tree(merge: &MergeFn) -> bool {
1317    matches!(merge, MergeFn::IfEq { .. } | MergeFn::Seq(_))
1318}
1319
1320/// `--native-merge` VALUE-FOLD: whether `merge` is a pure value-fold custom
1321/// `:merge` tree the Feldera host interpreter runs natively as a pairwise fold
1322/// (`eval_term_build_node` over leaves with NO side effects).
1323///
1324/// The frontend lowers a value-fold body (a fold of primitives / literals over
1325/// `old`/`new`, e.g. `(from-string (to-string (* new old)))`) to a `MergeFn` of
1326/// `Primitive`/`Const`/`Old`/`New`/`OldCol`/`NewCol`; a single-output function may
1327/// wrap it in a `Columns([single])`. A value-fold body contains NO `Function` call
1328/// (a `ResolvedCall::Func` disqualifies it on the frontend — that is a term build),
1329/// and none of the multi-action variants (`Seq`/`TableInsert`/`Construct`/`IfEq`),
1330/// so we recognise it structurally as "a tree built only from those pure leaves".
1331/// `Old`/`New`/`OldCol`/`NewCol`/`Const` ALONE are degenerate merges the 4-variant
1332/// `MergeMode` already expresses (e.g. a bare `old`/`new`), so they are NOT retained
1333/// as a tree — only a fold containing at least one `Primitive` is (everything the
1334/// plain `MergeMode` cannot express).
1335fn is_value_fold_tree(merge: &MergeFn) -> bool {
1336    fn is_pure_leaf(m: &MergeFn) -> bool {
1337        match m {
1338            MergeFn::Const(_)
1339            | MergeFn::Old
1340            | MergeFn::New
1341            | MergeFn::OldCol(_)
1342            | MergeFn::NewCol(_) => true,
1343            MergeFn::Primitive(_, args) => args.iter().all(is_pure_leaf),
1344            _ => false,
1345        }
1346    }
1347    /// Whether the top of the tree is a `Primitive` (i.e. a genuine fold, not a
1348    /// bare `old`/`new`/`Const` the plain `MergeMode` already handles). A value-fold
1349    /// body's root is the outermost primitive application, so testing the root
1350    /// suffices to distinguish a fold from a degenerate leaf merge.
1351    fn has_primitive(m: &MergeFn) -> bool {
1352        matches!(m, MergeFn::Primitive(..))
1353    }
1354    match merge {
1355        // A single-output function may present the merge wrapped in a one-column
1356        // `Columns`; unwrap it (a multi-column `Columns` is a tuple-output merge,
1357        // which Feldera does not support — left to `merge_mode_for_table`).
1358        MergeFn::Columns(cols) => match cols.as_slice() {
1359            [single] => has_primitive(single) && is_pure_leaf(single),
1360            _ => false,
1361        },
1362        other => has_primitive(other) && is_pure_leaf(other),
1363    }
1364}

1373    fn add_table(&mut self, config: FunctionConfig) -> FunctionId {
1374        let id = FunctionId::new(self.relations.len() as u32);
1375        let arity = config.schema.len();
1376        assert!(
1377            arity <= compile::MAX_ARITY,
1378            "Feldera backend (milestone 1) supports relations of arity <= {} (got {} for `{}`)",
1379            compile::MAX_ARITY,
1380            arity,
1381            config.name
1382        );
1383        // Relation vs function: a table is a **relation** (whole row is the key,
1384        // no output column to merge) iff it is nullary OR its last column is
1385        // `Unit` — this is the term encoder's view-table pattern
1386        // `(function @XView (...) Unit :merge old)` AND ordinary relations.
1387        // Otherwise the last column is a function OUTPUT, resolved by the merge
1388        // mode. This mirrors the DuckDB backend's Unit-detection
1389        // (`backend_impl.rs` ~593) — NOT `DefaultVal`, which is `Fail` for every
1390        // custom function regardless of whether it has a real output column.
1391        let output_is_unit = config.schema.last().is_some_and(|t| match t {
1392            ColumnTy::Base(bv) => {
1393                let bvs = self.db.base_values();
1394                bvs.has_ty_by_id(std::any::TypeId::of::<()>())
1395                    && *bv == bvs.get_ty_by_id(std::any::TypeId::of::<()>())
1396            }
1397            _ => false,
1398        });
1399        let has_output = arity > 0 && !output_is_unit;
1400        // `--native-merge` PROOF mode (2-table): a single-output eclass view whose
1401        // merge is `UnionIntoParentTableWithProof` carries the proof-congruence
1402        // config (relational `@UF_S`, the `@<F>ViewProof` side-table, Trans/Sym).
1403        // Captured here so `resolve_proof_congruence_merge` reads it from the same
1404        // single source of truth as every other native-merge association.
1405        let native_merge_proof = native_merge_proof_target(&config.merge);
1406        // `--native-merge` NATIVE CUSTOM `:merge`: a custom merge whose body either
1407        // BUILDS TERMS (a top-level `IfEq{…, Seq[…]}` from
1408        // `translate_term_build_merge_to_seq`) or is a pure VALUE-FOLD (a fold of
1409        // primitives over `old`/`new`, e.g. `(from-string (to-string (* new old)))`)
1410        // cannot collapse to a 4-variant `MergeMode` — the term build mints e-nodes
1411        // + writes view rows, and the value fold computes a value no
1412        // `Old`/`New`/`Min`/`Relation` mode can express. Retain the lowered tree and
1413        // EXECUTE it host-side in `resolve_merge` (`resolve_term_build_merge`, which
1414        // handles both: the value fold is the side-effect-free degenerate case). The
1415        // function still carries a `MergeMode`, but it is unused for a retained tree
1416        // (`resolve_merge` dispatches on `merge_tree.is_some()` first); the sentinel
1417        // `New` keeps a key with >1 candidate eligible (and `merge_mode_for_table`
1418        // would otherwise panic on a term-build `Seq`/`IfEq`).
1419        let retain_tree =
1420            has_output && (is_term_build_tree(&config.merge) || is_value_fold_tree(&config.merge));
1421        // Recognize the merge mode (FD-conflict resolution); see `merge_mode_for_table`.
1422        // A relation (no output column, or Unit output) needs no FD resolution.
1423        let merge = if !has_output {
1424            MergeMode::Relation
1425        } else if retain_tree {
1426            MergeMode::New
1427        } else {
1428            merge_mode_for_table(&config.merge, &config.name)
1429        };
1430        let merge_tree = retain_tree.then(|| config.merge.clone());
1431        // Identity-on-miss ("lookup-or-self"): an action-position lookup of an
1432        // absent key resolves to the key itself, inserting no row. Only valid for
1433        // a single-key function (2 columns) whose key and output share a type —
1434        // the term encoder's flat UF-index `@UF_Sf`.
1435        let identity_on_miss = matches!(config.default, egglog_backend_trait::DefaultVal::Identity);
1436        if identity_on_miss {
1437            assert_eq!(
1438                arity, 2,
1439                "DefaultVal::Identity (`{}`) expects a single key column (2-column key->output function)",
1440                config.name
1441            );
1442        }
1443        self.relations.push(RelationInfo {
1444            name: config.name,
1445            arity,
1446            schema: config.schema.clone(),
1447            has_output,
1448            merge,
1449            identity_on_miss,
1450            merge_tree,
1451        });
1452        self.mirror.insert(id, std::rc::Rc::new(HashSet::new()));
1453        // A `UnionIntoUf` merge (the native-UF native-merge encoding) is no longer
1454        // supported on Feldera (migrated off native-UF; the CLI rejects it), so it
1455        // must never reach here. Panic loudly if it does rather than silently drop
1456        // the congruence routing.
1457        assert!(
1458            native_merge_target_uf(&config.merge).is_none(),
1459            "Feldera was migrated off native-UF: a `UnionIntoUf` (native-UF native-merge) \
1460             view reached add_table (`{}`); non-proof --native-merge/--native-uf are rejected",
1461            self.info(id).name
1462        );
1463        if let Some(proof) = native_merge_proof {
1464            self.native_merge_proof.insert(id, proof);
1465        }
1466        self.invalidate_circuit();
1467        id
1468    }

1513    fn for_each_while(
1514        &self,
1515        table: FunctionId,
1516        f: &mut dyn for<'r> FnMut(FunctionRow<'r>) -> bool,
1517    ) {
1518        let arity = self.info(table).arity;
1519        let Some(set) = self.mirror.get(&table) else {
1520            return;
1521        };
1522        for row in set.iter() {
1523            let vals = unpack_row(row, arity);
1524            let frow = FunctionRow {
1525                vals: &vals,
1526                subsumed: false,
1527            };
1528            if !f(frow) {
1529                break;
1530            }
1531        }
1532    }

1536    fn lookup_id(&self, func: FunctionId, key: &[Value]) -> Option<Value> {
1537        let info = self.info(func);
1538        if !info.has_output {
1539            return None;
1540        }
1541        let inputs_len = info.arity - 1;
1542        if key.len() != inputs_len {
1543            return None;
1544        }
1545        // Identity-on-miss (the relational `@UF_Sf` flat index): a single-key
1546        // lookup of an absent key resolves to the key itself (find-or-self). This is
1547        // the extractor's `find_canonical` path (`backend.lookup_id`).
1548        let set = self.mirror.get(&func)?;
1549        for row in set.iter() {
1550            if (0..inputs_len).all(|i| compile::row_col(row, i) == key[i].rep()) {
1551                return Some(Value::new(compile::row_col(row, inputs_len)));
1552            }
1553        }
1554        if info.identity_on_miss && key.len() == 1 {
1555            return Some(key[0]);
1556        }
1557        None
1558    }

1560    fn add_values(&mut self, values: Box<dyn Iterator<Item = (FunctionId, Vec<Value>)> + '_>) {
1561        for (func, row) in values {
1562            let arity = self.info(func).arity;
1563            assert_eq!(row.len(), arity, "add_values: row arity mismatch");
1564            self.mirror_insert(func, pack_row(&row));
1565        }
1566        // Seed inserts are reflected in the mirror immediately and pushed into
1567        // the circuit at the next `run_rules`; no separate flush needed.
1568        self.invalidate_circuit();
1569    }

1572        // Allocate a fresh id and store `(inputs..., fresh_id)`.
1573        let id = self.fresh_id();

1577        assert_eq!(full.len(), arity, "add_term: arity mismatch");
1578        self.mirror_insert(func, pack_row(&full));
1579        self.invalidate_circuit();
1580        id
1581    }

1647    fn free_rule(&mut self, id: RuleId) {
1648        if let Some(slot) = self.rules.get_mut(id.rep() as usize) {
1649            *slot = None;
1650            let i = id.rep() as usize;
1651            // Drop all per-rule persistent state so a reused rule index (egglog's
1652            // `eval_actions` builds + frees a fresh rule every command) starts
1653            // clean — otherwise a stale `atomless_fired` / circuit would suppress
1654            // the new rule's firing.
1655            self.persistent.remove(&i);
1656            self.fed.remove(&i);
1657            self.atomless_fired.remove(&i);
1658            // Drop any FUSED circuit whose ruleset includes this freed rule, so
1659            // a reused index never reuses a stale fused circuit.
1660            self.fused.retain(|key, _| !key.contains(&i));
1661            self.fed_fused.retain(|key, _| !key.contains(&i));
1662            self.invalidate_circuit();
1663        }
1664    }

1666    fn run_rules(&mut self, rules: &[RuleId]) -> Result<IterationReport> {
1667        // ONE egglog iteration = one round of rule application over the current
1668        // relations, running ONLY the rules in `rules`. The frontend calls this
1669        // N times for `(run N)` and with different rule subsets to schedule
1670        // distinct rulesets.
1671        //
1672        // Milestone 3 executes rules with a **host-side interpreter**
1673        // (`interpret.rs`): a nested-loop join of the rule's table body atoms
1674        // over the current mirror, primitive body atoms / RHS lookups / RHS
1675        // primitive calls evaluated against the embedded `Database`, then the
1676        // head actions applied. This is one bounded hop (the round reads the
1677        // mirror as it was at entry — semi-naive-equivalent for one iteration),
1678        // matching the per-iteration model M1/M2 established with DBSP.
1679        if rules.is_empty() {
1680            return Ok(IterationReport::default());
1681        }
1682
1683        let live: Vec<usize> = rules
1684            .iter()
1685            .map(|r| r.rep() as usize)
1686            .filter(|&i| self.rules.get(i).map(|s| s.is_some()).unwrap_or(false))
1687            .collect();
1688        if live.is_empty() {
1689            return Ok(IterationReport::default());
1690        }
1691
1692        // Stage-3: when `FELDERA_CIRCUIT_REBUILD=1` and this call is a pure
1693        // rebuild ruleset, run the whole rebuild fixpoint on the persistent
1694        // congruence-shuttle circuit instead of the interpreter. Recognition is
1695        // conservative — anything unrecognized falls back to the interpreter, so
1696        // this can never regress a program, only accelerate rebuild.
1697        if self.circuit_rebuild {
1698            if let Some(roles) = circuit_rebuild::recognize(self, &live) {
1699                self.circuit_rebuild_runs += 1;
1700                let changed = circuit_rebuild::run_rebuild(self, &roles)?;
1701                let mut report = IterationReport::default();
1702                report.rule_set_report.changed = changed;
1703                return Ok(report);
1704            }
1705        }
1706
1707        let changed = interpret::run_iteration(self, &live)?;
1708
1709        let mut report = IterationReport::default();
1710        report.rule_set_report.changed = changed;
1711        Ok(report)
1712    }

1722    fn register_external_func(
1723        &mut self,
1724        func: Box<dyn ExternalFunction + 'static>,
1725    ) -> ExternalFunctionId {
1726        // Register into the embedded Database so the primitive is *invokable*
1727        // through `Database::with_execution_state` during rule firing (the
1728        // frontend's query rules call `collect_matches` this way). The local
1729        // registry mirrors the same id for name tracking + panic sentinels;
1730        // both advance in lockstep so the ids stay aligned.
1731        let func2 = dyn_clone::clone_box(&*func);
1732        let id = self.db.add_external_function(func);
1733        self.external_funcs.add_func_at(id, func2);
1734        // The cached prim-engine clone (if any) predates this registration;
1735        // drop it so the next on-circuit eval rebuilds a clone holding this prim.
1736        self.prim_engine = None;
1737        id
1738    }

1744        self.prim_engine = None;

1747    fn new_panic(&mut self, message: String) -> ExternalFunctionId {
1748        // A panic sentinel needs an id aligned with the Database's external-func
1749        // table (the frontend references it via `call_external_func`). Register
1750        // a real panicking `ExternalFunction` so invoking it surfaces the
1751        // message, and mirror it in the local registry.
1752        let panic_fn = external_func::PanicFunc::new(message.clone());

1755        self.prim_engine = None;
