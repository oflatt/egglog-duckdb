1use std::{collections::HashSet, convert::TryFrom};
2
3use egglog::{
4    Core, Primitive, Read, ReadPrim, ReadState, Value,
5    constraint::{AllEqualTypeConstraint, TypeConstraint},
6    prelude::BaseSort,
7    prelude::{I64Sort, Span, StringSort},
8    sort::S,
9    util::INTERNAL_SYMBOL_PREFIX,
10};
11
12#[derive(Clone)]
13pub struct GetSizePrimitive;
14
15impl Primitive for GetSizePrimitive {
16    fn name(&self) -> &str {
17        "get-size!"
18    }
19
20    fn get_type_constraints(&self, span: &Span) -> Box<dyn TypeConstraint> {
21        AllEqualTypeConstraint::new(self.name(), span.clone())
22            .with_output_sort(I64Sort.to_arcsort())
23            .with_all_arguments_sort(StringSort.to_arcsort())
24            .into_box()
25    }
26}
27
28impl ReadPrim for GetSizePrimitive {
29    fn apply<'a, 'db>(&self, state: ReadState<'a, 'db>, args: &[Value]) -> Option<Value> {
30        let filters: Option<HashSet<String>> = if args.is_empty() {
31            None
32        } else {
33            Some(
34                args.iter()
35                    .map(|value| state.base_values().unwrap::<S>(*value).0)
36                    .collect::<HashSet<_>>(),
37            )
38        };
39
40        let total_size: usize = state
41            .table_sizes()
42            .into_iter()
43            .filter_map(|(name, size)| {
44                if name.starts_with(INTERNAL_SYMBOL_PREFIX) {
45                    return None;
46                }
47                if let Some(filter) = &filters
48                    && !filter.contains(name)
49                {
50                    return None;
51                }
52                Some(size)
53            })
54            .sum();
55        let total_size = i64::try_from(total_size).ok()?;
56        Some(state.base_values().get::<i64>(total_size))
57    }
58}
