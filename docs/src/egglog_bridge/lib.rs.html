54/// A live registry of action handles for use by typed primitives.
55///
56/// Maps table name to [`TableAction`] (plus the shared [`UnionAction`]
57/// and the default-panic external-function id) and is owned by the
58/// bridge `EGraph`. The state wrappers (`PureState`/`ReadState`/
59/// `WriteState`/`FullState`) live in the `egglog` crate; they read
60/// from this registry at invoke time to back name-indexed action
61/// methods. Held by the bridge `EGraph` inside an `Arc<RwLock<_>>`.
62#[derive(Clone)]
63pub struct ActionRegistry {
64    table_actions: hashbrown::HashMap<String, TableAction>,
65    union_action: UnionAction,
66    default_panic_id: ExternalFunctionId,
67}

385    /// Load the given values into the database.
386    ///
387    /// # Panics
388    /// This method panics if the values do not match the arity of the function.
389    ///
390    /// NB: this is not an efficient interface for bulk loading. We should add
391    /// one that allows us to pass through a series of RowBuffers before
392    /// incrementing the timestamp.
393    pub fn add_values(&mut self, values: impl IntoIterator<Item = (FunctionId, Vec<Value>)>) {
394        let mut extended_row = Vec::<Value>::new();
395        let mut bufs = DenseIdMap::default();
396        for (func, row) in values.into_iter() {
397            let table_info = &self.funcs[func];
398            let schema_math = table_info.schema_math();
399            let table_id = table_info.table;
400            extended_row.extend_from_slice(&row);
401            schema_math.write_table_row(
402                &mut extended_row,
403                RowVals {
404                    timestamp: self.next_ts().to_value(),
405                    subsume: schema_math.subsume.then_some(NOT_SUBSUMED),
406                    ret_val: None, // already filled in.
407                },
408            );
409            let buf = bufs.get_or_insert(table_id, || self.db.new_buffer(table_id));
410            buf.stage_insert(&extended_row);
411            extended_row.clear();
412        }
413        // Flush the buffers.
414        mem::drop(bufs);
415        self.flush_updates();
416    }

418    /// A term-oriented means of adding data to the database: hand back a "term
419    /// id" for the given function and keys for the function.
420    ///
421    /// # Panics
422    /// This method panics if the values do not match the arity of the function.
423    pub fn add_term(&mut self, func: FunctionId, inputs: &[Value]) -> Value {
424        let info = &self.funcs[func];
425        let schema_math = info.schema_math();
426        let mut extended_row = Vec::new();
427        extended_row.extend_from_slice(inputs);
428        let res = self.fresh_id();
429        schema_math.write_table_row(
430            &mut extended_row,
431            RowVals {
432                timestamp: self.next_ts().to_value(),
433                ret_val: Some(res),
434                subsume: schema_math.subsume.then_some(NOT_SUBSUMED),
435            },
436        );
437        extended_row[schema_math.ret_val_col()] = res;
438        let table_id = self.funcs[func].table;
439        self.db.new_buffer(table_id).stage_insert(&extended_row);
440        self.flush_updates();
441        self.get_canon_in_uf(res)
442    }

490    /// Iterate over the rows of a function table, calling `f` on each row. If `f` returns `false`
491    /// the function returns early and stops reading rows from the table.
492    pub fn for_each_while(&self, table: FunctionId, mut f: impl FnMut(FunctionRow<'_>) -> bool) {
493        let info = &self.funcs[table];
494        let table = self.funcs[table].table;
495        let schema_math = info.schema_math();
496        let imp = self.db.get_table(table);
497        let all = imp.all();
498        let mut cur = Offset::new(0);
499        let mut buf = TaggedRowBuffer::new(imp.spec().arity());
500        // This somewhat awkward iteration strategy is forced on us by the `scan_bounded` API. We
501        // should look into ways to avoid this cludge where the loop body effectively must be
502        // repeated at the end. The obvious and idiomatic ways to do this all require
503        // `dyn`-compatibility on `Table` or dynamic dispatch per row.
504        macro_rules! drain_buf {
505            ($buf:expr) => {
506                for (_, row) in $buf.non_stale() {
507                    let subsumed =
508                        schema_math.subsume && row[schema_math.subsume_col()] == SUBSUMED;
509                    if !f(FunctionRow {
510                        vals: &row[0..schema_math.func_cols],
511                        subsumed,
512                    }) {
513                        return;
514                    }
515                }
516                $buf.clear();
517            };
518        }
519        while let Some(next) = imp.scan_bounded(all.as_ref(), cur, 32, &mut buf) {
520            drain_buf!(buf);
521            cur = next;
522        }
523        drain_buf!(buf);
524    }

543    /// A helper for scanning the entries in a table.
544    fn scan_table(&self, table: &WrappedTable, mut f: impl FnMut(&[Value])) {
545        const BATCH_SIZE: usize = 128;
546        let all = table.all();
547        let mut cur = Offset::new(0);
548        let mut out = TaggedRowBuffer::new(table.spec().arity());
549        while let Some(next) = table.scan_bounded(all.as_ref(), cur, BATCH_SIZE, &mut out) {
550            out.non_stale().for_each(|(_, row)| f(row));
551            out.clear();
552            cur = next;
553        }
554        out.non_stale().for_each(|(_, row)| f(row));
555    }

576    /// Register a function in this EGraph.
577    pub fn add_table(&mut self, config: FunctionConfig) -> FunctionId {
578        let FunctionConfig {
579            schema,
580            default,
581            merge,
582            name,
583            can_subsume,
584        } = config;
585        assert!(
586            !schema.is_empty(),
587            "must have at least one column in schema"
588        );
589        let to_rebuild: Vec<ColumnId> = schema
590            .iter()
591            .enumerate()
592            .filter(|(_, ty)| matches!(ty, ColumnTy::Id))
593            .map(|(i, _)| ColumnId::from_usize(i))
594            .collect();
595        // The number of value (return) columns is determined by the merge: a `Columns` merge has
596        // one entry per value column; every other (scalar) merge applies to a single value column.
597        let n_vals = match &merge {
598            MergeFn::Columns(cols) => cols.len(),
599            _ => 1,
600        };
601        assert!(
602            schema.len() >= n_vals,
603            "function {name} has fewer columns ({}) than value columns ({n_vals})",
604            schema.len()
605        );
606        let n_keys = schema.len() - n_vals;
607        let schema_math = SchemaMath {
608            subsume: can_subsume,
609            n_keys,
610            func_cols: schema.len(),
611        };
612        let n_args = schema_math.num_keys();
613        let n_cols = schema_math.table_columns();
614        let next_func_id = self.funcs.next_id();
615        // `--native-merge`: remember which per-sort `@UF_Sf` this view's
616        // FD-conflict congruence union targets (named by the `UnionIntoUf` merge),
617        // so a later `register_native_merge_view` can assert the registered UF
618        // matches the one already baked into the merge.
619        let native_merge_target = match &merge {
620            MergeFn::UnionIntoUf(uf_func) => Some(*uf_func),
621            // Proof-mode tuple-output view: the UF target is named by the `col0`
622            // `UnionIntoUfWithProof` inside the top-level `Columns`.
623            MergeFn::Columns(cols) => cols.iter().find_map(|c| match c {
624                MergeFn::UnionIntoUf(uf_func) => Some(*uf_func),
625                MergeFn::UnionIntoUfWithProof { uf, .. } => Some(*uf),
626                _ => None,
627            }),
628            _ => None,
629        };
630        // Knot-tying for a SELF-REFERENTIAL merge (the term-encoder's
631        // `@UF_Sf : (S) -> S`, whose `:merge` does a `MergeFn::TableInsert` into
632        // ITS OWN table for the recursive parent-union): the merge resolution
633        // (`merge_fn_fill_deps` / `merge_fn_to_callback` → `TableAction::new`)
634        // reads `self.funcs[self_id].table` etc., so the new `FunctionInfo` must
635        // already be in `self.funcs` BEFORE the merge is built. We therefore
636        // reserve the table id (deterministic — the next `add_table_named` will
637        // assign exactly this id) and push the `FunctionInfo` up front, then
638        // build the merge and create the backing table. Tables whose merge does
639        // not self-reference resolve identically either way; this ordering is a
640        // strict superset.
641        let name: Arc<str> = name.into();
642        let table_id = self.db.next_table_id();
643        let res = self.funcs.push(FunctionInfo {
644            table: table_id,
645            schema: schema.clone(),
646            n_keys,
647            incremental_rebuild_rules: Default::default(),
648            nonincremental_rebuild_rule: RuleId::new(!0),
649            default_val: default,
650            can_subsume,
651            name: name.clone(),
652            kind: FunctionKind::Table,
653            uf_rebuild_rule: None,
654        });
655        debug_assert_eq!(res, next_func_id);
656
657        let mut read_deps = IndexSet::<TableId>::new();
658        let mut write_deps = IndexSet::<TableId>::new();
659        merge_fn_fill_deps(&merge, self, &mut read_deps, &mut write_deps);
660        let merge_fn = merge_fn_to_callback(&merge, schema_math, &name, self);
661        let table = SortedWritesTable::new(
662            n_args,
663            n_cols,
664            Some(ColumnId::from_usize(schema.len())),
665            to_rebuild,
666            merge_fn,
667        );
668        let assigned_table_id = self.db.add_table_named(
669            table,
670            name.clone(),
671            read_deps.iter().copied(),
672            write_deps.iter().copied(),
673        );
674        debug_assert_eq!(assigned_table_id, table_id);
675        if let Some(uf_func) = native_merge_target {
676            self.native_merge_view_target.insert(res, uf_func);
677        }
678        let incremental_rebuild_rules = self.incremental_rebuild_rules(res, &schema);
679        let nonincremental_rebuild_rule = self.nonincremental_rebuild(res, &schema);
680        let info = &mut self.funcs[res];
681        info.incremental_rebuild_rules = incremental_rebuild_rules;
682        info.nonincremental_rebuild_rule = nonincremental_rebuild_rule;
683        let action = TableAction::new(self, res);
684        let table_name = self.funcs[res].name.to_string();
685        self.action_registry
686            .write()
687            .unwrap()
688            .register_table(table_name, action);
689        res
690    }

716    /// `--native-merge` (term-encoding native-UF, non-proof, native bridge):
717    /// associate an FD-keyed constructor `@<F>View` view function (`view_func`)
718    /// with the per-sort `@UF_Sf` UF-backed function (`uf_func`) that owns its
719    /// eclass (OUTPUT) column. This is the uniform contract the dataflow/SQL
720    /// backends use to route the FD-conflict congruence union into the right
721    /// union-find. On the native bridge the routing is already baked into the
722    /// view's [`MergeFn::UnionIntoUf`] merge (which stages the union directly into
723    /// this `@UF_Sf`), so this method records the association and ASSERTS the
724    /// view's merge actually targets the registered UF — a defensive consistency
725    /// check that the frontend's declare-time UF resolution and its
726    /// schedule-time registration agree. `uf_func` must be a UF-backed function.

730            "register_native_merge_view: {uf_func:?} is not a UF-backed function"
731        );
732        if let Some(target) = self.native_merge_view_target.get(&view_func) {
733            assert_eq!(
734                *target, uf_func,
735                "register_native_merge_view: view {view_func:?} merge targets UF {target:?} \
736                 but is being registered against UF {uf_func:?}"
737            );
738        }
739        self.native_merge_uf.insert(view_func, uf_func);
740    }

742    /// Register a UF-backed function in this EGraph.
743    ///
744    /// UF Functions record the series of "leader changes" in the e-graph, where
745    /// the leader is always 'up to date' with the latest value. It has two
746    /// columns: one key (the "displaced" value), mapping to its current leader.
747    /// Queries of such a function, coupled with semi-naive evaluation, allow for
748    /// rules to "listen" on 'leader changes' in the underlying union-find. This
749    /// makes it more efficient, if less convenient, than a standard
750    /// quadratic-sized equivalence relation.
751    ///
752    /// The returned external function canonicalizes its single argument against
753    /// the underlying union-find (find-or-self).
754    ///
755    /// Values in this union find are "coherent" with those of the underlying
756    /// e-graph union-find: rebuilding the e-graph also forwards global unions
757    /// into these union-finds. What makes these useful is that they can store
758    /// _more_ equalities that are themselves "visible" to egglog rules.
759    pub fn add_uf_function(
760        &mut self,
761        config: UfFunctionConfig,
762    ) -> Result<(FunctionId, ExternalFunctionId)> {
763        let UfFunctionConfig {
764            name,
765            on_leader_change,
766            read_deps,
767            write_deps,
768            proof_wiring,
769        } = config;
770        let name: Arc<str> = name.into();
771        // In proof mode the function is backed by a `DisplacedTableWithProvenance`
772        // (4-column writes carrying a per-edge proof, exposing `get_proof`); in
773        // term mode the plain `DisplacedTable` (3-column writes) is used.
774        let table_id = if let Some(wiring) = proof_wiring {
775            // TableActions to intern the composed proof and write the onchange row.
776            let onchange_action = TableAction::new(self, wiring.onchange);
777            let trans_action = TableAction::new(self, wiring.trans);
778            let sym_action = TableAction::new(self, wiring.sym);
779            let onchange_tid = self.table_id(wiring.onchange);
780            let trans_tid = self.table_id(wiring.trans);
781            let sym_tid = self.table_id(wiring.sym);
782            // The leader-change callback both reads (hash-cons dedup via
783            // `predict_val`'s `get_row`) and writes (`stage_insert`) the onchange
784            // and Trans/Sym tables. They must therefore be both read- and
785            // write-dependencies: the write dep pre-allocates a mutation buffer
786            // (so writes don't touch the moved-out table info), and the read dep
787            // forces them into an earlier merge stratum (so they stay in the
788            // database — readable — while this UF table is merging). Without the
789            // read dep, the Trans/Sym tables (which are also written by ordinary
790            // proof rules) can be merged in the same stratum as the UF table and
791            // get moved out from under the callback's `get_row`.
792            let mut read_deps = read_deps;
793            read_deps.push(onchange_tid);
794            read_deps.push(trans_tid);
795            read_deps.push(sym_tid);
796            let mut write_deps = write_deps;
797            write_deps.push(onchange_tid);
798            write_deps.push(trans_tid);
799            write_deps.push(sym_tid);
800
801            // The provenance table reconstructs the proof path `displaced ->
802            // new_leader` itself (it can't be read back from the database during
803            // the table's own merge) and hands it to the callback, which folds
804            // it into a single `Proof` term and records the onchange row.
805            let table = DisplacedTableWithProvenance::with_leader_change_callback(
806                move |state: &mut ExecutionState, change: LeaderChange, steps: &[ProofStep]| {
807                    let new_leader = change.new_leader();
808                    let proof = compose_uf_proof(state, &trans_action, &sym_action, steps);
809                    onchange_action.lookup_or_insert(
810                        state,
811                        &[
812                            change.write_lhs,
813                            change.write_rhs,
814                            change.lhs_leader,
815                            change.rhs_leader,
816                            new_leader,
817                            proof,
818                        ],
819                    );
820                },
821            );
822            self.db.add_table_named(
823                table,
824                name.clone(),
825                read_deps.iter().copied(),
826                write_deps.iter().copied(),
827            )
828        } else {
829            let mut table = DisplacedTable::default();
830            if let Some(callback) = on_leader_change {
831                table.set_leader_change_callback(move |state, change| callback(state, change));
832            }
833            self.db.add_table_named(
834                table,
835                name.clone(),
836                read_deps.iter().copied(),
837                write_deps.iter().copied(),
838            )
839        };
840        let canon =
841            self.register_external_func(Box::new(make_external_func(move |state, vals| {
842                let [val] = vals else {
843                    panic!("uf canonicalizer expected 1 value, got {vals:?}")
844                };
845                let table = state.get_table(table_id);
846                let canon = table
847                    .get_row_column(&[*val], ColumnId::new(1))
848                    .unwrap_or(*val);
849                Some(canon)
850            })));
851        // In proof mode the UF function is `(S S) Proof`: writes carry the
852        // per-edge proof as a third column (`set (@UF_Sf a b) proof` →
853        // `[a, b, proof, ts]`), which the provenance table consumes. In term
854        // mode it is the usual `(S) S` (writes `[a, b, ts]`).
855        let schema = if proof_wiring.is_some() {
856            vec![ColumnTy::Id, ColumnTy::Id, ColumnTy::Id]
857        } else {
858            vec![ColumnTy::Id, ColumnTy::Id]
859        };
860        // UF-backed functions are single-value (keyed on the displaced id; the leader, plus an
861        // optional proof column, are the value columns). Mirror the single-output convention.
862        let n_keys = schema.len() - 1;
863        let res = self.funcs.push(FunctionInfo {
864            table: table_id,
865            schema,
866            n_keys,
867            incremental_rebuild_rules: Vec::new(),
868            nonincremental_rebuild_rule: RuleId::new(!0),
869            default_val: DefaultVal::Fail,
870            can_subsume: false,
871            name,
872            kind: FunctionKind::Uf,
873            uf_rebuild_rule: None,
874        });
875        let uf_rebuild_rule = self.uf_rebuild_rule(res);
876        self.funcs[res].uf_rebuild_rule = Some(uf_rebuild_rule);
877        Ok((res, canon))
878    }

880    /// Build a rule that forwards global uf_table unions into a UF-backed table
881    /// so it stays consistent with the main union-find.
882    fn uf_rebuild_rule(&mut self, table: FunctionId) -> RuleId {
883        let uf_table = self.uf_table;
884        // Proof-mode UF functions are `(S S) Proof` (3 columns): the set must
885        // supply a proof column. Forwarded global e-graph unions don't carry an
886        // egglog-level proof here (the term encoder routes unions through
887        // `(set @UF_Sf ...)` directly, so this rule is normally inert), so use a
888        // placeholder proof value.
889        let proof_col = self.funcs[table].schema.len() == 3;
890        let mut rb = self.new_rule(&format!("uf rebuild {table:?}"), true);
891        let lhs: QueryEntry = rb.new_var(ColumnTy::Id).into();
892        let rhs: QueryEntry = rb.new_var(ColumnTy::Id).into();
893        rb.add_atom_with_timestamp_and_func(uf_table, None, None, &[lhs.clone(), rhs.clone()]);
894        if proof_col {
895            let placeholder = QueryEntry::Const {
896                val: Value::new(0),
897                ty: ColumnTy::Id,
898            };
899            rb.set(table, &[lhs, rhs, placeholder]);
900        } else {
901            rb.set(table, &[lhs, rhs]);
902        }
903        rb.build()
904    }

929    /// Total rows across the union-find tables that drive a rebuild: the global
930    /// `$uf`, plus (under `--nativerb`) every registered per-sort `@UF_Sf`. In
931    /// term-encoding native-UF mode unions go through `@UF_Sf`, not `$uf`, so
932    /// the rebuild trigger must watch those tables too.
933    fn rebuild_uf_size(&self) -> usize {
934        let mut total = self.db.get_table(self.uf_table).len();
935        for uf_func in self.nativerb_views.keys() {
936            total += self.db.get_table(self.funcs[*uf_func].table).len();
937        }
938        total

941    fn run_rules_inner(&mut self, rules: &[RuleId]) -> Result<IterationReport> {
942        let ts = self.next_ts();
943
944        let uf_size_before = self.rebuild_uf_size();
945        let rule_set_report =
946            run_rules_impl(&mut self.db, &mut self.rules, rules, ts, self.report_level)?;
947        if let Some(message) = self.panic_message.lock().unwrap().take() {
948            return Err(PanicError(message).into());
949        }
950
951        let mut iteration_report = IterationReport {
952            rule_set_report,
953            rebuild_time: Duration::ZERO,
954        };
955        let uf_size_after = self.rebuild_uf_size();
956        if uf_size_before == uf_size_after {
957            // No new unions: skip the full rebuild but still advance the
958            // timestamp so that seminaive evaluation sees a fresh epoch.
959            // Rebuilding is only necessary when new unions have been made because ids may need to be updated.
960            // Adding terms doesn't necessarily touch the union-find, only doing a union between existing ids does.
961            self.inc_ts();
962            return Ok(iteration_report);
963        }
964
965        let rebuild_timer = Instant::now();
966        self.rebuild()?;
967        iteration_report.rebuild_time = rebuild_timer.elapsed();
968
969        if let Some(message) = self.panic_message.lock().unwrap().take() {
970            return Err(PanicError(message).into());
971        }
972
973        Ok(iteration_report)
974    }

976    fn rebuild(&mut self) -> Result<()> {
977        let uf_rules = self.uf_rebuild_rules();
978        let do_parallel = rayon::current_num_threads() > 1;
979        if self.db.get_table(self.uf_table).rebuilder(&[]).is_some() {
980            // The UF implementation supports "native"  rebuilding.
981            let mut tables = Vec::with_capacity(self.funcs.next_id().index());
982            for (_, func) in self.funcs.iter() {
983                if func.is_uf() {
984                    continue;
985                }
986                tables.push(func.table);
987            }
988            loop {
989                // Order matters here: we need to rebuild containers first and then rebuild the
990                // tables. Why?
991                //
992                // Say we have a sort that can map to and from a vector containing only itself:
993                // (sort X)
994                // (function to-vec (X) (Vec X) :no-merge)
995                // (constructor from-vec (Vec X) X)
996                // (constructor Num (i64) X)
997                // (constructor Add (X X) X)
998                //
999                // Along with rules:
1000                // (rule ((= x (Num i))) ((set (to-vec x) (vec-of x))))
1001                // (rule ((= x (Add i j))) ((set (to-vec x) (vec-of x))))
1002                // (rule ((= x (from-vec v))) ((set (to-vec x) v))
1003                // (rewrite (Add (Num i) (Num j)) (Num (+ i j)))
1004                //
1005                // These rules, while redundant, should be safe. However, if we rebuild tables
1006                // before containers some schedules can cause us to violate the `:no-merge`
1007                // directive, which asserts that all values written for a key are equal.
1008                //
1009                // Suppose we start off with x1=(Num 1), x2=(Num 3), and x3=(Add (Num 1) (Num 2)) as
1010                // expressions, with `to-vec` and `from-vec` entries for all three expressions.
1011                // We'll call (to-vec xi) vi for all i.
1012                //
1013                // Now suppose we run the `rewrite` above: now, x3 = x2. But v3 will only equal v2
1014                // _after_ we rebuild the `Vec` container. That means that if we rebuild `to-vec`
1015                // we will collapse the the rows for x3 and x2, but then fail to merge v3 and v2
1016                // because they are not (yet) equal.
1017                //
1018                // Rebuilding containers first will find that v3 and v2 are equal, and the rest of
1019                // the rules can proceed.
1020                let container_rebuild = self.db.rebuild_containers(self.uf_table);
1021                let next_ts = self.next_ts().to_value();
1022                let table_rebuild = self.db.apply_rebuild(self.uf_table, &tables, next_ts);
1023                // Container rebuild can make a parent row newly matchable without
1024                // changing the row's stored id. Re-timestamp those parents so
1025                // seminaive sees the newly enabled match on the next pass.
1026                let dirty_ids: Vec<Value> = container_rebuild.dirty_ids().iter().copied().collect();
1027                let refreshed_rows = self
1028                    .db
1029                    .refresh_rows_for_values(&tables, &dirty_ids, next_ts);
1030                // Forward any global e-graph unions into the UF-backed tables so
1031                // they stay coherent with the main union-find.
1032                let uf_rebuild = if uf_rules.is_empty() {
1033                    false
1034                } else {
1035                    let uf_ts = self.next_ts();
1036                    run_rules_impl(
1037                        &mut self.db,
1038                        &mut self.rules,
1039                        &uf_rules,
1040                        uf_ts,
1041                        ReportLevel::TimeOnly,
1042                    )?
1043                    .changed
1044                };
1045                // `--nativerb`: re-canonicalize the registered `@<F>View` view
1046                // tables against each per-sort `@UF_Sf` using the ENGINE's
1047                // native table rebuild, replacing the encoding-level
1048                // `@rebuild_rule*` rules. This runs inside the same saturating
1049                // rebuild loop as the `$uf` rebuild so the two reach a JOINT
1050                // fixpoint: congruence (still emitted as `@congruence_rule*`)
1051                // issues unions on `@UF_Sf` from outside (in the `@rebuilding`
1052                // ruleset), and the surrounding `(saturate @rebuilding)` re-runs
1053                // congruence whenever this engine rebuild collapses a view row.
1054                // Container rebuild first (same ordering rationale as above), in
1055                // case a container holds eq-sort ids of this sort.
1056                let nativerb_changed = self.nativerb_rebuild_pass()?;
1057                self.inc_ts();
1058                if !table_rebuild
1059                    && !refreshed_rows
1060                    && !container_rebuild.changed()
1061                    && !uf_rebuild
1062                    && !nativerb_changed
1063                {
1064                    break;
1065                }
1066            }
1067            return Ok(());
1068        }
1069        if do_parallel {
1070            return self.rebuild_parallel();
1071        }
1072        let start = Instant::now();
1073
1074        // The database changed. Rebuild. New entries should land after the given rules.
1075        let mut changed = true;
1076        while changed {
1077            changed = false;
1078            // We need to iterate rebuilding to a fixed point. Future scans
1079            // should look only at the latest updates.
1080            self.inc_ts();
1081            let ts = self.next_ts();
1082            for (_, info) in self.funcs.iter_mut() {
1083                if info.is_uf() {
1084                    continue;
1085                }
1086                let last_rebuilt_at = self.rules[info.nonincremental_rebuild_rule].last_run_at;
1087                let table_size = self.db.estimate_size(info.table, None);
1088                let uf_size = self.db.estimate_size(
1089                    self.uf_table,
1090                    Some(Constraint::GeConst {
1091                        col: ColumnId::new(2),
1092                        val: last_rebuilt_at.to_value(),
1093                    }),
1094                );
1095                if incremental_rebuild(uf_size, table_size, false) {
1096                    marker_incremental_rebuild(|| -> Result<()> {
1097                        // Run each of the incremental rules serially.
1098                        //
1099                        // This is to avoid recanonicalizing the same row multiple
1100                        // times.
1101                        for rule in &info.incremental_rebuild_rules {
1102                            changed |= run_rules_impl(
1103                                &mut self.db,
1104                                &mut self.rules,
1105                                &[*rule],
1106                                ts,
1107                                ReportLevel::TimeOnly,
1108                            )?
1109                            .changed;
1110                        }
1111                        // Reset the rule we did not run. These two should be equivalent.
1112                        self.rules[info.nonincremental_rebuild_rule].last_run_at = ts;
1113                        Ok(())
1114                    })?;
1115                } else {
1116                    marker_nonincremental_rebuild(|| -> Result<()> {
1117                        changed |= run_rules_impl(
1118                            &mut self.db,
1119                            &mut self.rules,
1120                            &[info.nonincremental_rebuild_rule],
1121                            ts,
1122                            ReportLevel::TimeOnly,
1123                        )?
1124                        .changed;
1125                        for rule in &info.incremental_rebuild_rules {
1126                            self.rules[*rule].last_run_at = ts;
1127                        }
1128                        Ok(())
1129                    })?;
1130                }
1131            }
1132            if !uf_rules.is_empty() {
1133                changed |= run_rules_impl(
1134                    &mut self.db,
1135                    &mut self.rules,
1136                    &uf_rules,
1137                    ts,
1138                    ReportLevel::TimeOnly,
1139                )?
1140                .changed;
1141            }
1142        }
1143        log::info!("rebuild took {:?}", start.elapsed());
1144        Ok(())
1145    }

1147    /// `--nativerb`: re-canonicalize every registered `@<F>View` view table
1148    /// against its per-sort `@UF_Sf` using the engine's native table rebuild.
1149    /// Returns whether anything changed (a view row was re-canonicalized or
1150    /// collapsed by `:merge`), so the caller's rebuild loop can iterate to a
1151    /// fixpoint. A no-op (returns `false`) when no UF is registered.
1152    fn nativerb_rebuild_pass(&mut self) -> Result<bool> {
1153        if self.nativerb_views.is_empty() {
1154            return Ok(false);
1155        }
1156        // Snapshot the registry so we can take `&mut self.db` in the loop.
1157        let entries: Vec<(FunctionId, Vec<TableId>)> = self
1158            .nativerb_views
1159            .iter()
1160            .map(|(uf, views)| {
1161                (
1162                    *uf,
1163                    views
1164                        .iter()
1165                        .map(|v| self.funcs[*v].table)
1166                        .collect::<Vec<_>>(),
1167                )
1168            })
1169            .collect();
1170        let mut changed = false;
1171        for (uf_func, view_tables) in &entries {
1172            let uf_table = self.funcs[*uf_func].table;
1173            // Container rebuild first (same ordering rationale as the `$uf`
1174            // path): a container value holding eq-sort ids of this sort must be
1175            // re-canonicalized before the view rows that key on it.
1176            let container_rebuild = self.db.rebuild_containers(uf_table);
1177            let next_ts = self.next_ts().to_value();
1178            let table_rebuild = self.db.apply_rebuild(uf_table, view_tables, next_ts);
1179            let dirty_ids: Vec<Value> = container_rebuild.dirty_ids().iter().copied().collect();
1180            let refreshed_rows = self
1181                .db
1182                .refresh_rows_for_values(view_tables, &dirty_ids, next_ts);
1183            changed |= table_rebuild || refreshed_rows || container_rebuild.changed();
1184        }
1185        Ok(changed)
1186    }

1188    /// A variant of `rebuild` that attempts to combine rebuild rules into
1189    /// larger rulesets to increase parallelism. This kind of preprocessing can
1190    /// slow processing down in a single-threaded setting, so it is only used
1191    /// when the number of active threads is greater than 1.
1192    fn rebuild_parallel(&mut self) -> Result<()> {
1193        let start = Instant::now();
1194        let uf_rules = self.uf_rebuild_rules();
1195        #[derive(Default)]
1196        struct RebuildState {
1197            nonincremental: Vec<FunctionId>,
1198            incremental: DenseIdMap<usize, SmallVec<[FunctionId; 2]>>,
1199        }
1200
1201        impl RebuildState {
1202            fn clear(&mut self) {
1203                self.nonincremental.clear();
1204                self.incremental.iter_mut().for_each(|(_, v)| v.clear());
1205            }
1206        }
1207
1208        let mut changed = true;
1209        let mut state = RebuildState::default();
1210        let mut scratch = Vec::new();
1211        while changed {
1212            changed = false;
1213            state.clear();
1214            self.inc_ts();
1215            // First, figure out which functions will be rebuilt nonincrementally,
1216            // vs. incrementally. Group them together.
1217            for (func, info) in self.funcs.iter_mut() {
1218                if info.is_uf() {
1219                    continue;
1220                }
1221                let last_rebuilt_at = self.rules[info.nonincremental_rebuild_rule].last_run_at;
1222                let table_size = self.db.estimate_size(info.table, None);
1223                let uf_size = self.db.estimate_size(
1224                    self.uf_table,
1225                    Some(Constraint::GeConst {
1226                        col: ColumnId::new(2),
1227                        val: last_rebuilt_at.to_value(),
1228                    }),
1229                );
1230                if incremental_rebuild(uf_size, table_size, true) {
1231                    for (i, _) in info.incremental_rebuild_rules.iter().enumerate() {
1232                        state.incremental.get_or_default(i).push(func);
1233                    }
1234                } else {
1235                    state.nonincremental.push(func);
1236                }
1237            }
1238            let ts = self.next_ts();
1239            for func in state.nonincremental.iter().copied() {
1240                scratch.push(self.funcs[func].nonincremental_rebuild_rule);
1241                for rule in &self.funcs[func].incremental_rebuild_rules {
1242                    self.rules[*rule].last_run_at = ts;
1243                }
1244            }
1245            changed |= run_rules_impl(
1246                &mut self.db,
1247                &mut self.rules,
1248                &scratch,
1249                ts,
1250                ReportLevel::TimeOnly,
1251            )?
1252            .changed;
1253            scratch.clear();
1254            let ts = self.next_ts();
1255            for (i, funcs) in state.incremental.iter() {
1256                for func in funcs.iter().copied() {
1257                    let info = &mut self.funcs[func];
1258                    scratch.push(info.incremental_rebuild_rules[i]);
1259                    self.rules[info.nonincremental_rebuild_rule].last_run_at = ts;
1260                }
1261                changed |= run_rules_impl(
1262                    &mut self.db,
1263                    &mut self.rules,
1264                    &scratch,
1265                    ts,
1266                    ReportLevel::TimeOnly,
1267                )?
1268                .changed;
1269                scratch.clear();
1270            }
1271            if !uf_rules.is_empty() {
1272                changed |= run_rules_impl(
1273                    &mut self.db,
1274                    &mut self.rules,
1275                    &uf_rules,
1276                    ts,
1277                    ReportLevel::TimeOnly,
1278                )?
1279                .changed;
1280            }
1281        }
1282        log::info!("rebuild took {:?}", start.elapsed());
1283        Ok(())
1284    }

1299    fn incremental_rebuild_rule(
1300        &mut self,
1301        table: FunctionId,
1302        schema: &[ColumnTy],
1303        col: ColumnId,
1304    ) -> RuleId {
1305        let subsume = self.funcs[table].can_subsume;
1306        let table_id = self.funcs[table].table;
1307        let uf_table = self.uf_table;
1308        // Two atoms, one binding a whole tuple, one binding a displaced column
1309        let mut rb = self.new_rule(&format!("incremental rebuild {table:?}, {col:?}"), true);
1310        rb.set_plan_strategy(PlanStrategy::MinCover);
1311        let mut vars = Vec::<QueryEntry>::with_capacity(schema.len());
1312        for ty in schema {
1313            vars.push(rb.new_var(*ty).into());
1314        }
1315        let canon_val: QueryEntry = rb.new_var(ColumnTy::Id).into();
1316        let subsume_var = subsume.then(|| rb.new_var(ColumnTy::Id));
1317        rb.add_atom_with_timestamp_and_func(
1318            table_id,
1319            Some(table),
1320            subsume_var.clone().map(QueryEntry::from),
1321            &vars,
1322        );
1323        rb.add_atom_with_timestamp_and_func(
1324            uf_table,
1325            None,
1326            None,
1327            &[vars[col.index()].clone(), canon_val.clone()],
1328        );
1329        rb.set_focus(1); // Set the uf atom as the sole focus.
1330
1331        // Now canonicalize the entire row.
1332        let mut canon = Vec::<QueryEntry>::with_capacity(schema.len());
1333        for (i, (var, ty)) in vars.iter().zip(schema.iter()).enumerate() {
1334            canon.push(if i == col.index() {
1335                canon_val.clone()
1336            } else if let ColumnTy::Id = ty {
1337                rb.lookup_uf(var.clone()).unwrap().into()
1338            } else {
1339                var.clone()
1340            })
1341        }
1342
1343        // Remove the old row and insert the new one.
1344        rb.rebuild_row(table, &vars, &canon, subsume_var);
1345        rb.build()
1346    }

1348    fn nonincremental_rebuild(&mut self, table: FunctionId, schema: &[ColumnTy]) -> RuleId {
1349        let can_subsume = self.funcs[table].can_subsume;
1350        let table_id = self.funcs[table].table;
1351        let mut rb = self.new_rule(&format!("nonincremental rebuild {table:?}"), false);
1352        rb.set_plan_strategy(PlanStrategy::MinCover);
1353        let mut vars = Vec::<QueryEntry>::with_capacity(schema.len());
1354        for ty in schema {
1355            vars.push(rb.new_var(*ty).into());
1356        }
1357        let subsume_var = can_subsume.then(|| rb.new_var(ColumnTy::Id));
1358        rb.add_atom_with_timestamp_and_func(
1359            table_id,
1360            Some(table),
1361            subsume_var.clone().map(QueryEntry::from),
1362            &vars,
1363        );
1364        let mut lhs = SmallVec::<[QueryEntry; 4]>::new();
1365        let mut rhs = SmallVec::<[QueryEntry; 4]>::new();
1366        let mut canon = Vec::<QueryEntry>::with_capacity(schema.len());
1367        for (var, ty) in vars.iter().zip(schema.iter()) {
1368            canon.push(if let ColumnTy::Id = ty {
1369                lhs.push(var.clone());
1370                let canon_var = QueryEntry::from(rb.lookup_uf(var.clone()).unwrap());
1371                rhs.push(canon_var.clone());
1372                canon_var
1373            } else {
1374                var.clone()
1375            })
1376        }
1377        rb.check_for_update(&lhs, &rhs).unwrap();
1378        rb.rebuild_row(table, &vars, &canon, subsume_var);
1379        rb.build()
1380    }

1401    /// Returns `true` if the database is updated.

1411        }
1412        updated
1413    }

1476// `DefaultVal` and `MergeFn` are now defined in `egglog-backend-trait`; the
1477// re-export above brings them back into this module's path. The bridge-only
1478// helpers (`fill_deps`, `to_callback`, `resolve`) live as free functions
1479// here so that they can keep using `EGraph` internals.
1480fn merge_fn_fill_deps(
1481    merge: &MergeFn,
1482    egraph: &EGraph,
1483    read_deps: &mut IndexSet<TableId>,
1484    write_deps: &mut IndexSet<TableId>,
1485) {
1486    use MergeFn::*;
1487    match merge {
1488        Primitive(_, args) => {
1489            args.iter()
1490                .for_each(|arg| merge_fn_fill_deps(arg, egraph, read_deps, write_deps));
1491            write_deps.insert(egraph.uf_table);
1492        }
1493        Function(func, args) => {
1494            assert!(
1495                !egraph.funcs[*func].is_uf(),
1496                "{}",
1497                FunctionConfigError::MergeUsesUf(egraph.funcs[*func].name.to_string())
1498            );
1499            read_deps.insert(egraph.funcs[*func].table);
1500            write_deps.insert(egraph.funcs[*func].table);
1501            args.iter()
1502                .for_each(|arg| merge_fn_fill_deps(arg, egraph, read_deps, write_deps));
1503        }
1504        UnionId => {
1505            write_deps.insert(egraph.uf_table);
1506        }
1507        UnionIntoUf(uf_func) => {
1508            // `--native-merge`: the FD-conflict congruence union is staged into
1509            // the named per-sort `@UF_Sf` (a UF-backed function), so that table —
1510            // not the global `$uf` — is the write dependency.
1511            write_deps.insert(egraph.funcs[*uf_func].table);
1512        }
1513        UnionIntoParentTable { parent_table, .. } => {
1514            // `--native-merge` relational path: the FD-conflict congruence edge is
1515            // written into the per-sort relational `@UF_S` parent table, so that
1516            // table is the write dependency.
1517            write_deps.insert(egraph.funcs[*parent_table].table);
1518        }
1519        UnionIntoUfWithProof { uf, trans, sym, .. } => {
1520            // `--native-merge` PROOF mode: the proof-carrying congruence edge is
1521            // staged into the proof-mode `@UF_Sf`, and composing its edge proof
1522            // hash-conses `Trans` / `Sym` rows. All three are write deps; the
1523            // Trans/Sym tables are also read deps (the `lookup_or_insert`
1524            // hash-cons dedup reads them — same dependency shape as the
1525            // leader-change callback in `add_uf_function`).
1526            write_deps.insert(egraph.funcs[*uf].table);
1527            write_deps.insert(egraph.funcs[*trans].table);
1528            write_deps.insert(egraph.funcs[*sym].table);
1529            read_deps.insert(egraph.funcs[*trans].table);
1530            read_deps.insert(egraph.funcs[*sym].table);
1531        }
1532        UnionIntoParentTableWithProof { .. } => {
1533            // The 2-table proof-congruence merge is FlowLog/Feldera-only (it reads a
1534            // proof side-table and writes the relational `@UF_S`); the native bridge
1535            // never builds or resolves it (it uses the A2 tuple proof view). So it
1536            // has no dependency footprint here.
1537            panic!(
1538                "MergeFn::UnionIntoParentTableWithProof is a single-output-backend \
1539                 (FlowLog/Feldera) merge and is not resolved on the native bridge"
1540            )
1541        }
1542        EclassMinProof { .. } => {}
1543        Columns(cols) => {
1544            cols.iter()
1545                .for_each(|col| merge_fn_fill_deps(col, egraph, read_deps, write_deps));
1546        }
1547        TableInsert(func, args) => {
1548            // The side write makes the target table a write-dependency, so its
1549            // buffer is pre-allocated during batched merges (the whole reason
1550            // this exists rather than a by-name primitive insert). Ported from
1551            // PR #933.
1552            write_deps.insert(egraph.funcs[*func].table);
1553            args.iter()
1554                .for_each(|arg| merge_fn_fill_deps(arg, egraph, read_deps, write_deps));
1555        }
1556        Seq(items) => {
1557            items
1558                .iter()
1559                .for_each(|item| merge_fn_fill_deps(item, egraph, read_deps, write_deps));
1560        }
1561        Construct(func, args, value_args) => {
1562            // A nested constructor mint inside the merge reads (hash-cons dedup)
1563            // and writes the target view table. Ported from PR #933.
1564            read_deps.insert(egraph.funcs[*func].table);
1565            write_deps.insert(egraph.funcs[*func].table);
1566            args.iter()
1567                .chain(value_args.iter())
1568                .for_each(|arg| merge_fn_fill_deps(arg, egraph, read_deps, write_deps));
1569        }
1570        IfEq { a, b, then, els } => {
1571            // Both branches may run (the condition is data-dependent), so union
1572            // the deps of every sub-expression.
1573            for m in [a, b, then, els] {
1574                merge_fn_fill_deps(m, egraph, read_deps, write_deps);
1575            }
1576        }
1577        AssertEq | Old | New | OldCol(..) | NewCol(..) | KeyCol(..) | Const(..) => {}
1578    }
1579}

1581fn merge_fn_to_callback(
1582    merge: &MergeFn,
1583    schema_math: SchemaMath,
1584    function_name: &str,
1585    egraph: &mut EGraph,
1586) -> Box<core_relations::MergeFn> {
1587    let resolved = merge_fn_resolve_columns(merge, function_name, egraph);
1588    assert_eq!(
1589        resolved.len(),
1590        schema_math.n_vals(),
1591        "merge for {function_name} must have one entry per value column"
1592    );
1593
1594    // Fast path: the overwhelmingly common single-output function. Avoids the per-column loop
1595    // and the `SmallVec` of merged values, keeping the hot path identical to the pre-multi-value
1596    // form. A single-output merge always resolves to a one-element `Vec` (a bare scalar merge or
1597    // `Columns` of length 1), so this branch is exhaustive for `n_vals == 1`.
1598    if let [resolved] = resolved.as_slice() {
1599        let resolved = resolved.clone();
1600        let ret_val_col = schema_math.ret_val_col();
1601        return Box::new(move |state, cur, new, out| {
1602            let timestamp = new[schema_math.ts_col()];
1603
1604            let mut changed = false;
1605
1606            let ret_val = {
1607                let out = resolved.run(state, cur, new, schema_math.n_keys, 0, timestamp);
1608                changed |= cur[ret_val_col] != out;
1609                out
1610            };
1611
1612            let subsume = schema_math.subsume.then(|| {
1613                let cur = cur[schema_math.subsume_col()];
1614                let new = new[schema_math.subsume_col()];
1615                let out = combine_subsumed(cur, new);
1616                changed |= cur != out;
1617                out
1618            });
1619            if changed {
1620                out.extend_from_slice(new);
1621                schema_math.write_table_row(
1622                    out,
1623                    RowVals {
1624                        timestamp,
1625                        subsume,
1626                        ret_val: Some(ret_val),
1627                    },
1628                );
1629            }
1630
1631            changed
1632        });
1633    }
1634
1635    Box::new(move |state, cur, new, out| {
1636        let timestamp = new[schema_math.ts_col()];
1637
1638        let mut changed = false;
1639
1640        // Compute each merged value column. `cur` and `new` are full rows, so a tuple-output
1641        // column's merge may reference any output column via `OldCol`/`NewCol`.
1642        let mut merged_vals = SmallVec::<[Value; 4]>::new();
1643        for (i, col_merge) in resolved.iter().enumerate() {
1644            let out_val = col_merge.run(state, cur, new, schema_math.n_keys, i, timestamp);
1645            changed |= cur[schema_math.val_col(i)] != out_val;
1646            merged_vals.push(out_val);
1647        }
1648
1649        let subsume = schema_math.subsume.then(|| {
1650            let cur = cur[schema_math.subsume_col()];
1651            let new = new[schema_math.subsume_col()];
1652            let out = combine_subsumed(cur, new);
1653            changed |= cur != out;
1654            out
1655        });
1656        if changed {
1657            out.extend_from_slice(new);
1658            for (i, val) in merged_vals.iter().enumerate() {
1659                out[schema_math.val_col(i)] = *val;
1660            }
1661            schema_math.write_table_row(
1662                out,
1663                RowVals {
1664                    timestamp,
1665                    subsume,
1666                    ret_val: None,
1667                },
1668            );
1669        }
1670
1671        changed
1672    })
1673}

1691fn merge_fn_resolve(merge: &MergeFn, function_name: &str, egraph: &mut EGraph) -> ResolvedMergeFn {
1692    match merge {
1693        MergeFn::Const(v) => ResolvedMergeFn::Const(*v),
1694        MergeFn::Old => ResolvedMergeFn::Old,
1695        MergeFn::New => ResolvedMergeFn::New,
1696        MergeFn::OldCol(i) => ResolvedMergeFn::OldCol(*i),
1697        MergeFn::NewCol(i) => ResolvedMergeFn::NewCol(*i),
1698        MergeFn::KeyCol(i) => ResolvedMergeFn::KeyCol(*i),
1699        MergeFn::Columns(_) => {
1700            panic!("nested Columns merge is not supported (Columns must be top-level)")
1701        }
1702        MergeFn::AssertEq => ResolvedMergeFn::AssertEq {
1703            panic: egraph.new_panic(format!(
1704                "Illegal merge attempted for function {function_name}"
1705            )),
1706        },
1707        MergeFn::UnionId => ResolvedMergeFn::UnionId {
1708            uf_table: egraph.uf_table,
1709        },
1710        // `--native-merge`: stage the FD-conflict congruence union directly into
1711        // the named per-sort `@UF_Sf`'s union-find (the table that owns this
1712        // view's eclass column) instead of the global `$uf`. Resolves to the same
1713        // `ResolvedMergeFn::UnionId` runtime form — only the target table differs
1714        // — so the staged `[cur, new, ts]` lands in the per-sort UF, whose leader
1715        // change drives the view's relational `@rebuild_rule*` re-canonicalization.
1716        MergeFn::UnionIntoUf(uf_func) => ResolvedMergeFn::UnionId {
1717            uf_table: egraph.funcs[*uf_func].table,
1718        },
1719        // `--native-merge` relational path: stage the FD-conflict congruence edge
1720        // as a row into the per-sort relational `@UF_S` parent table (a plain
1721        // 2-key `(S S) -> Unit` function with `:merge old`), matching the
1722        // rule-encoded `(set (@UF_S larger smaller) ())`. We capture the parent
1723        // table's schema math so the runtime can shape the `[larger, smaller,
1724        // unit, ts]` row.
1725        MergeFn::UnionIntoParentTable { parent_table, unit } => {
1726            let info = &egraph.funcs[*parent_table];
1727            ResolvedMergeFn::UnionIntoParentTable {
1728                parent_table: info.table,
1729                parent_math: info.schema_math(),
1730                unit: *unit,
1731            }
1732        }
1733        // `--native-merge` PROOF mode (`col0`): stage a proof-carrying congruence
1734        // edge into the proof-mode `@UF_Sf`. The `Trans`/`Sym` proof constructors
1735        // are interned via `TableAction`s, mirroring `compose_uf_proof`.
1736        MergeFn::UnionIntoUfWithProof {
1737            uf,
1738            trans,
1739            sym,
1740            eclass_col,
1741            proof_col,
1742        } => ResolvedMergeFn::UnionIntoUfWithProof {
1743            uf_table: egraph.funcs[*uf].table,
1744            trans: TableAction::new(egraph, *trans),
1745            sym: TableAction::new(egraph, *sym),
1746            eclass_col: *eclass_col,
1747            proof_col: *proof_col,
1748        },
1749        // `--native-merge` PROOF mode on a single-output backend (FlowLog/Feldera):
1750        // the 2-table proof-congruence merge is resolved by that backend's host
1751        // interpreter, never by the native bridge.
1752        MergeFn::UnionIntoParentTableWithProof { .. } => {
1753            panic!(
1754                "MergeFn::UnionIntoParentTableWithProof is a single-output-backend \
1755                 (FlowLog/Feldera) merge and is not resolved on the native bridge"
1756            )
1757        }
1758        // `--native-merge` PROOF mode (`col1`): proof of the surviving min eclass.
1759        MergeFn::EclassMinProof {
1760            eclass_col,
1761            proof_col,
1762        } => ResolvedMergeFn::EclassMinProof {
1763            eclass_col: *eclass_col,
1764            proof_col: *proof_col,
1765        },
1766        // NB: The primitive and function-based merge functions heap allocate a single callback
1767        // for each layer of nesting. This introduces a bit of overhead, particularly for cases
1768        // that look like `(f old new)` or `(f new old)`. We could special-case common cases in
1769        // this function if that overhead shows up.
1770        MergeFn::Primitive(prim, args) => ResolvedMergeFn::Primitive {
1771            prim: *prim,
1772            args: args
1773                .iter()
1774                .map(|arg| merge_fn_resolve(arg, function_name, egraph))
1775                .collect::<Vec<_>>(),
1776            panic: egraph.new_panic(format!(
1777                "Merge function for {function_name} primitive call failed"
1778            )),
1779        },
1780        MergeFn::Function(func, args) => {
1781            let func_info = &egraph.funcs[*func];
1782            assert_eq!(
1783                func_info.schema.len(),
1784                args.len() + 1,
1785                "Merge function for {function_name} must match function arity for {}",
1786                func_info.name
1787            );
1788            let identity_on_miss = matches!(func_info.default_val, DefaultVal::Identity);
1789            ResolvedMergeFn::Function {
1790                func: TableAction::new(egraph, *func),
1791                panic: egraph.new_panic(format!(
1792                    "Lookup on {} failed in the merge function for {function_name}",
1793                    func_info.name
1794                )),
1795                args: args
1796                    .iter()
1797                    .map(|arg| merge_fn_resolve(arg, function_name, egraph))
1798                    .collect::<Vec<_>>(),
1799                identity_on_miss,
1800            }
1801        }
1802        // `:merge`-multiple-actions variants ported from PR #933.
1803        MergeFn::TableInsert(func, args) => ResolvedMergeFn::TableInsert {
1804            table: TableAction::new(egraph, *func),
1805            args: args
1806                .iter()
1807                .map(|arg| merge_fn_resolve(arg, function_name, egraph))
1808                .collect::<Vec<_>>(),
1809        },
1810        MergeFn::Seq(items) => ResolvedMergeFn::Seq(
1811            items
1812                .iter()
1813                .map(|item| merge_fn_resolve(item, function_name, egraph))
1814                .collect::<Vec<_>>(),
1815        ),
1816        MergeFn::Construct(func, args, value_args) => {
1817            let func_info = &egraph.funcs[*func];
1818            let num_values = func_info.schema.len() - func_info.n_keys;
1819            debug_assert_eq!(
1820                func_info.schema.len(),
1821                args.len() + num_values,
1822                "Construct for {function_name}: key arity must be schema minus value columns for {}",
1823                func_info.name
1824            );
1825            debug_assert_eq!(
1826                value_args.len() + 1,
1827                num_values,
1828                "Construct for {function_name}: value_args must fill every value column \
1829                 except the minted output for {}",
1830                func_info.name
1831            );
1832            ResolvedMergeFn::Construct {
1833                table: TableAction::new(egraph, *func),
1834                args: args
1835                    .iter()
1836                    .map(|arg| merge_fn_resolve(arg, function_name, egraph))
1837                    .collect::<Vec<_>>(),
1838                value_args: value_args
1839                    .iter()
1840                    .map(|arg| merge_fn_resolve(arg, function_name, egraph))
1841                    .collect::<Vec<_>>(),
1842            }
1843        }
1844        MergeFn::IfEq { a, b, then, els } => ResolvedMergeFn::IfEq {
1845            a: Box::new(merge_fn_resolve(a, function_name, egraph)),
1846            b: Box::new(merge_fn_resolve(b, function_name, egraph)),
1847            then: Box::new(merge_fn_resolve(then, function_name, egraph)),
1848            els: Box::new(merge_fn_resolve(els, function_name, egraph)),
1849        },
1850    }

1940impl ResolvedMergeFn {
1941    /// Compute the merged value for value column `self_col`.
1942    ///
1943    /// `cur` and `new` are the full conflicting rows. `n_keys` is the number of key columns, so
1944    /// value column `i` lives at `cur[n_keys + i]`. `Old`/`New` refer to `self_col`'s value, while
1945    /// `OldCol`/`NewCol` reference an explicit column — this lets a tuple-output column's merge
1946    /// read any other output column.
1947    fn run(
1948        &self,
1949        state: &mut ExecutionState,
1950        cur: &[Value],
1951        new: &[Value],
1952        n_keys: usize,
1953        self_col: usize,
1954        ts: Value,
1955    ) -> Value {
1956        match self {
1957            ResolvedMergeFn::Const(v) => *v,
1958            ResolvedMergeFn::Old => cur[n_keys + self_col],
1959            ResolvedMergeFn::New => new[n_keys + self_col],
1960            ResolvedMergeFn::OldCol(i) => cur[n_keys + i],
1961            ResolvedMergeFn::NewCol(i) => new[n_keys + i],
1962            // Key (input) column `i`. The two FD-conflicting rows share the same
1963            // key, so `cur[i] == new[i]`; read `cur`.
1964            ResolvedMergeFn::KeyCol(i) => cur[*i],
1965            ResolvedMergeFn::AssertEq { panic } => {
1966                let (cur, new) = (cur[n_keys + self_col], new[n_keys + self_col]);
1967                if cur != new {
1968                    let res = state.call_external_func(*panic, &[]);
1969                    assert_eq!(res, None);
1970                }
1971                cur
1972            }
1973            ResolvedMergeFn::UnionId { uf_table } => {
1974                let (cur, new) = (cur[n_keys + self_col], new[n_keys + self_col]);
1975                if cur != new {
1976                    state.stage_insert(*uf_table, &[cur, new, ts]);
1977                    // We pick the minimum when unioning. This matches the original egglog
1978                    // behavior. THIS MUST MATCH THE UNION-FIND IMPLEMENTATION!
1979                    std::cmp::min(cur, new)
1980                } else {
1981                    cur
1982                }
1983            }
1984            ResolvedMergeFn::UnionIntoParentTable {
1985                parent_table,
1986                parent_math,
1987                unit,
1988            } => {
1989                let (cur, new) = (cur[n_keys + self_col], new[n_keys + self_col]);
1990                if cur != new {
1991                    // Write the union edge into the relational `@UF_S` parent
1992                    // table, EXACTLY as the rule-encoded `union()` helper does:
1993                    // `(set (@UF_S larger smaller) ())`, where the key is the
1994                    // larger id and the value is the smaller id. The relational
1995                    // maintenance rulesets (singleparent / path_compress /
1996                    // uf_function_index) then propagate this into `@UF_Sf` and the
1997                    // rule-based rebuild re-canonicalizes the views. Keeping the
1998                    // min as the surviving eclass matches the union-by-min the
1999                    // `:merge (ordering-min old new)` index relies on.
2000                    let larger = std::cmp::max(cur, new);
2001                    let smaller = std::cmp::min(cur, new);
2002                    // Row layout for the 2-key `(S S) -> Unit` parent table:
2003                    // `[larger, smaller, unit, ts]` (no subsume column).
2004                    let mut row: SmallVec<[Value; 4]> = smallvec::smallvec![larger, smaller];
2005                    parent_math.write_table_row(
2006                        &mut row,
2007                        RowVals {
2008                            timestamp: ts,
2009                            subsume: None,
2010                            ret_val: Some(*unit),
2011                        },
2012                    );
2013                    state.stage_insert(*parent_table, &row);
2014                    smaller
2015                } else {
2016                    cur
2017                }
2018            }
2019            ResolvedMergeFn::UnionIntoUfWithProof {
2020                uf_table,
2021                trans,
2022                sym,
2023                eclass_col,
2024                proof_col,
2025            } => {
2026                let cur_eclass = cur[n_keys + *eclass_col];
2027                let new_eclass = new[n_keys + *eclass_col];
2028                if cur_eclass == new_eclass {
2029                    return cur_eclass;
2030                }
2031                let cur_proof = cur[n_keys + *proof_col];
2032                let new_proof = new[n_keys + *proof_col];
2033                // Match the rule-encoded `@congruence_rule`'s orientation EXACTLY:
2034                // it writes `(set (@UF_Sf larger smaller) (Trans larger_pf (Sym
2035                // smaller_pf)))`, where `larger = (ordering-max old new)` and the
2036                // edge proof carries the larger row's proof first. The proof-mode
2037                // UF's `get_proof` consumes the per-edge proof verbatim for a
2038                // forward step (`larger -> smaller` IS the leader-change
2039                // direction, since the UF tie-break makes `smaller` the leader),
2040                // so the stored proof must be the un-Sym'd `Trans(larger, Sym(smaller))`.
2041                let (larger, smaller, larger_pf, smaller_pf) = if cur_eclass > new_eclass {
2042                    (cur_eclass, new_eclass, cur_proof, new_proof)
2043                } else {
2044                    (new_eclass, cur_eclass, new_proof, cur_proof)
2045                };
2046                // edge_proof = Trans(larger_pf, Sym(smaller_pf)), proving
2047                // `larger = smaller` (larger = f(children) = smaller).
2048                let sym_smaller = sym
2049                    .lookup_or_insert(state, &[smaller_pf])
2050                    .expect("Sym constructor lookup_or_insert failed in congruence merge");
2051                let edge_proof = trans
2052                    .lookup_or_insert(state, &[larger_pf, sym_smaller])
2053                    .expect("Trans constructor lookup_or_insert failed in congruence merge");
2054                // Proof-mode UF writes are 4-column `[lhs, rhs, proof, ts]`.
2055                state.stage_insert(*uf_table, &[larger, smaller, edge_proof, ts]);
2056                smaller
2057            }
2058            ResolvedMergeFn::EclassMinProof {
2059                eclass_col,
2060                proof_col,
2061            } => {
2062                // The proof of the surviving (min) eclass: matches the eclass kept
2063                // by `UnionIntoUfWithProof` (which returns `min`).
2064                if cur[n_keys + *eclass_col] <= new[n_keys + *eclass_col] {
2065                    cur[n_keys + *proof_col]
2066                } else {
2067                    new[n_keys + *proof_col]
2068                }
2069            }
2070            // NB: The primitive and function-based merge functions heap allocate a single callback
2071            // for each layer of nesting. This introduces a bit of overhead, particularly for cases
2072            // that look like `(f old new)` or `(f new old)`. We could special-case common cases in
2073            // this function if that overhead shows up.
2074            ResolvedMergeFn::Primitive { prim, args, panic } => {
2075                let args = args
2076                    .iter()
2077                    .map(|arg| arg.run(state, cur, new, n_keys, self_col, ts))
2078                    .collect::<Vec<_>>();
2079
2080                match state.call_external_func(*prim, &args) {
2081                    Some(result) => result,
2082                    None => {
2083                        let res = state.call_external_func(*panic, &[]);
2084                        assert_eq!(res, None);
2085                        cur[n_keys + self_col]
2086                    }
2087                }
2088            }
2089            ResolvedMergeFn::Function {
2090                func,
2091                args,
2092                panic,
2093                identity_on_miss,
2094            } => {
2095                // see github.com/egraphs-good/egglog/pull/287
2096                //
2097                // The `cur == new` short-circuit is only valid when this
2098                // `Function` IS the merge resolving `self_col` (the classic
2099                // `(f old new)` value-fold case). For an `identity_on_miss`
2100                // canon lookup `(__UF_Sf x)` nested inside a `Construct`/`Seq`
2101                // (PR #933 term build), the lookup must run unconditionally —
2102                // `cur`/`new` here are the conflicting FD rows, not this lookup's
2103                // argument.
2104                if !identity_on_miss && cur[n_keys + self_col] == new[n_keys + self_col] {
2105                    return cur[n_keys + self_col];
2106                }
2107
2108                let args = args
2109                    .iter()
2110                    .map(|arg| arg.run(state, cur, new, n_keys, self_col, ts))
2111                    .collect::<Vec<_>>();
2112
2113                // Merge functions dispatch to another function that may be
2114                // a constructor (mint fresh id on miss) or a custom function
2115                // (return `None` → panic). `lookup_or_insert` preserves
2116                // both behaviors; the pure-read `lookup` would skip
2117                // constructor minting.
2118                func.lookup_or_insert(state, &args).unwrap_or_else(|| {
2119                    if *identity_on_miss {
2120                        // Find-or-self: a missing key in an `Identity`-default
2121                        // `@UF_Sf` index returns the (single) lookup argument
2122                        // unchanged, NOT a panic — the canonicalize-at-creation
2123                        // `(__UF_Sf x)` semantics.
2124                        debug_assert_eq!(
2125                            args.len(),
2126                            1,
2127                            "identity_on_miss canon lookup must be single-arg"
2128                        );
2129                        args[0]
2130                    } else {
2131                        let res = state.call_external_func(*panic, &[]);
2132                        assert_eq!(res, None);
2133                        cur[n_keys + self_col]
2134                    }
2135                })
2136            }
2137            // `:merge`-multiple-actions runtime (ported from PR #933).
2138            ResolvedMergeFn::TableInsert { table, args } => {
2139                let row = args
2140                    .iter()
2141                    .map(|arg| arg.run(state, cur, new, n_keys, self_col, ts))
2142                    .collect::<Vec<_>>();
2143                // Insert respects the target table's own merge; the timestamp
2144                // column is appended by `TableAction::insert`.
2145                table.insert(state, row.into_iter());
2146                // Return value is discarded by the enclosing `Seq`.
2147                cur[n_keys + self_col]
2148            }
2149            ResolvedMergeFn::Seq(items) => {
2150                let mut result = cur[n_keys + self_col];
2151                for item in items {
2152                    result = item.run(state, cur, new, n_keys, self_col, ts);
2153                }
2154                result
2155            }
2156            ResolvedMergeFn::Construct {
2157                table,
2158                args,
2159                value_args,
2160            } => {
2161                let key = args
2162                    .iter()
2163                    .map(|arg| arg.run(state, cur, new, n_keys, self_col, ts))
2164                    .collect::<SmallVec<[Value; 4]>>();
2165                let vals = value_args
2166                    .iter()
2167                    .map(|arg| arg.run(state, cur, new, n_keys, self_col, ts))
2168                    .collect::<SmallVec<[Value; 4]>>();
2169                // Constructor: always mints on miss, so this is `Some`.
2170                table
2171                    .lookup_or_insert_multi(state, &key, &vals)
2172                    .unwrap_or(cur[n_keys + self_col])
2173            }
2174            ResolvedMergeFn::IfEq { a, b, then, els } => {
2175                // Evaluate the (side-effect-free) condition operands first, then
2176                // run ONLY the taken branch — so a guarded term-building `Seq`
2177                // mints nothing when the values are already equal.
2178                let av = a.run(state, cur, new, n_keys, self_col, ts);
2179                let bv = b.run(state, cur, new, n_keys, self_col, ts);
2180                if av == bv {
2181                    then.run(state, cur, new, n_keys, self_col, ts)
2182                } else {
2183                    els.run(state, cur, new, n_keys, self_col, ts)
2184                }
2185            }
2186        }
2187    }

2200
2201impl TableAction {
2202    /// Create a new `TableAction` to be used later.
2203    /// This requires access to the `egglog_bridge::EGraph`.
2204    pub fn new(egraph: &EGraph, func: FunctionId) -> TableAction {
2205        let func_info = &egraph.funcs[func];
2206        assert!(
2207            !func_info.is_uf(),
2208            "{}",
2209            FunctionConfigError::TableActionUsesUf(func_info.name.to_string())
2210        );
2211        TableAction {
2212            table: func_info.table,
2213            table_math: func_info.schema_math(),
2214            default: match &func_info.default_val {
2215                DefaultVal::FreshId => Some(MergeVal::Counter(egraph.id_counter)),
2216                // `Identity` is consumed only by the rule-compilation lookup
2217                // path (see `lookup_with_subsumed`); the `TableAction` exec path
2218                // never looks up an Identity-default function, so fall back to
2219                // plain `lookup` semantics here.
2220                DefaultVal::Fail | DefaultVal::Identity => None,
2221                DefaultVal::Const(val) => Some(MergeVal::Constant(*val)),
2222            },
2223            timestamp: egraph.timestamp_counter,
2224        }
2225    }
2226
2227    /// Look up a row and return its return-value column, or `None` if the
2228    /// key is not present. **This is a pure read**: it never inserts a row,
2229    /// regardless of the table's configured [`DefaultVal`].
2230    ///
2231    /// For the lookup-or-insert behavior that mints fresh eclass IDs for
2232    /// constructors, use [`TableAction::lookup_or_insert`].
2233    pub fn lookup(&self, state: &ExecutionState, key: &[Value]) -> Option<Value> {
2234        state
2235            .get_table(self.table)

2246    /// For constructor tables this mints a fresh eclass ID; for custom
2247    /// functions (no default) this behaves identically to
2248    /// [`TableAction::lookup`].
2249    ///
2250    /// This is a write operation — only safe in action contexts. See
2251    /// issue #772.

2253        match self.default {
2254            Some(default) => {

2256                    MergeVal::Constant(Value::from_usize(state.read_counter(self.timestamp)));
2257                let mut merge_vals = SmallVec::<[MergeVal; 3]>::new();

2259                // `lookup_or_insert` is only used for single-value functions/constructors.

2261                    n_keys: 0,
2262                    func_cols: 1,
2263                    ..self.table_math
2264                }
2265                .write_table_row(
2266                    &mut merge_vals,
2267                    RowVals {
2268                        timestamp,
2269                        subsume: self

2271                            .subsume
2272                            .then_some(MergeVal::Constant(NOT_SUBSUMED)),
2273                        ret_val: Some(default),
2274                    },
2275                );

2277                    state.predict_val(self.table, key, merge_vals.iter().copied())
2278                        [self.table_math.ret_val_col()],
2279                )
2280            }
2281            None => self.lookup(state, key),
2282        }

2286    /// constructor `(children) -> (output, extra...)`: the first value column
2287    /// (`output`) is minted (the configured `FreshId` default) and the rest are
2288    /// written from `provided_vals` (e.g. the proof). Returns the minted `output`.
2289    ///
2290    /// Idempotent: an already-present key returns its existing `output` and writes
2291    /// nothing, so it can be evaluated more than once and yield the same id. This is
2292    /// a write operation, only safe in action/merge contexts. Ported from PR #933.
2293    pub fn lookup_or_insert_multi(
2294        &self,
2295        state: &mut ExecutionState,
2296        key: &[Value],
2297        provided_vals: &[Value],
2298    ) -> Option<Value> {
2299        match self.default {
2300            Some(default) => {
2301                debug_assert_eq!(
2302                    self.table_math.n_vals(),
2303                    1 + provided_vals.len(),
2304                    "lookup_or_insert_multi: provided_vals must fill every value \
2305                     column except the minted first one"
2306                );
2307                let timestamp =
2308                    MergeVal::Constant(Value::from_usize(state.read_counter(self.timestamp)));
2309                // Non-key columns, in order: [output (minted), provided.., ts, subsume?].
2310                let mut merge_vals = SmallVec::<[MergeVal; 4]>::new();
2311                merge_vals.push(default);
2312                merge_vals.extend(provided_vals.iter().map(|v| MergeVal::Constant(*v)));
2313                merge_vals.push(timestamp);
2314                if self.table_math.subsume {
2315                    merge_vals.push(MergeVal::Constant(NOT_SUBSUMED));
2316                }
2317                // The first value column (the minted output) is at `ret_val_col()`.
2318                Some(
2319                    state.predict_val(self.table, key, merge_vals.iter().copied())
2320                        [self.table_math.ret_val_col()],
2321                )
2322            }
2323            None => self.lookup(state, key),
2324        }
2325    }

2390/// Compose a `Proof` term for a native-UF leader change from a reconstructed
2391/// proof path (`steps`, produced by the union-find's provenance graph).
2392///
2393/// Folds the path into a single `Proof` value: each step contributes its
2394/// per-edge proof (the value stored in the write's proof column), wrapped in
2395/// `Sym` for a backward step, and consecutive steps are chained with `Trans`.
2396/// The `Trans`/`Sym` constructor rows are minted/hash-consed via
2397/// `TableAction::lookup_or_insert`, so the resulting proof is a real, interned
2398/// term visible to egglog rules.
2399fn compose_uf_proof(
2400    state: &mut ExecutionState,
2401    trans_action: &TableAction,
2402    sym_action: &TableAction,
2403    steps: &[ProofStep],
2404) -> Value {
2405    let mut acc: Option<Value> = None;
2406    for step in steps {
2407        // Each step's per-edge proof proves `write_lhs = write_rhs`.
2408        let step_proof = match step.reason {
2409            ProofReason::Forward(p) => p,
2410            ProofReason::Backward(p) => sym_action
2411                .lookup_or_insert(state, &[p])
2412                .expect("Sym constructor lookup_or_insert failed"),
2413        };
2414        acc = Some(match acc {
2415            None => step_proof,
2416            Some(prev) => trans_action
2417                .lookup_or_insert(state, &[prev, step_proof])
2418                .expect("Trans constructor lookup_or_insert failed"),
2419        });
2420    }
2421    acc.expect("get_proof returned an empty path for a real leader change")
2422}

2560/// A struct helping with some calculations of where some information is stored at the
2561/// core-relations Table level for a given function.
2562///
2563/// Functions can have multiple "output columns" in the underlying core-relations layer depending
2564/// on whether different features are enabled. Roughly, tables are laid out as:
2565///
2566/// > `[key0, ..., keyn, value0, ..., valuem, timestamp, subsume?]`
2567///
2568/// Where there are `n+1` key columns, `m+1` value (return) columns, and columns marked with a
2569/// question mark are optional, depending on the egraph and table-level configuration.
2570///
2571/// Most functions have a single value column (`m == 0`); tuple-output (multi-value) functions
2572/// have more than one. The functional dependency is always from the key columns to the value
2573/// columns.
2574#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
2575struct SchemaMath {
2576    /// Whether or not the table is enabled for subsumption.
2577    subsume: bool,
2578    /// The number of key (input) columns.
2579    n_keys: usize,
2580    /// The number of columns in the function (keys plus all value/return columns).
2581    func_cols: usize,
2582}

2584/// A struct containing possible non-key portions of a table row. To be used with
2585/// [`SchemaMath::write_table_row`].
2586///
2587/// This is not to be confused with [`FunctionRow`], which is higher-level and for public uses.
2588struct RowVals<T> {
2589    /// The timestamp for the row.
2590    timestamp: T,
2591    /// The subsumption tag for the row. Only relevant if the table has subsumption enabled.
2592    subsume: Option<T>,
2593    /// The return value of the row. Return values are mandatory but callers may have already
2594    /// filled it in.
2595    ret_val: Option<T>,
2596}
2597
2598// `FunctionRow` is now defined in `egglog-backend-trait`; the re-export at
2599// the top of this module brings it back into `egglog_bridge::FunctionRow`.
2600
2601impl SchemaMath {
2602    fn write_table_row<T: Clone>(
2603        &self,
2604        row: &mut impl HasResizeWith<T>,
2605        RowVals {
2606            timestamp,
2607            subsume,
2608            ret_val,
2609        }: RowVals<T>,
2610    ) {
2611        row.resize_with(self.table_columns(), || timestamp.clone());
2612        row[self.ts_col()] = timestamp;
2613        if let Some(ret_val) = ret_val {
2614            row[self.ret_val_col()] = ret_val;
2615        }
2616        if let Some(subsume) = subsume {
2617            row[self.subsume_col()] = subsume;
2618        } else {
2619            assert!(
2620                !self.subsume,
2621                "subsume flag must be provided if subsumption is enabled"
2622            );
2623        }
2624    }
2625
2626    fn num_keys(&self) -> usize {
2627        self.n_keys
2628    }
2629
2630    /// The number of value (return) columns.
2631    fn n_vals(&self) -> usize {
2632        self.func_cols - self.n_keys
2633    }
2634
2635    fn table_columns(&self) -> usize {
2636        self.func_cols + 1 /* timestamp */ + if self.subsume { 1 } else { 0 }
2637    }
2638
2639    /// The column index of the `i`th value (return) column.
2640    fn val_col(&self, i: usize) -> usize {
2641        self.n_keys + i
2642    }
2643
2644    /// The first value column. For single-output functions this is *the* return value; for
2645    /// tuple-output functions it is value column 0.
2646    fn ret_val_col(&self) -> usize {
2647        self.n_keys
2648    }
2649
2650    fn ts_col(&self) -> usize {
2651        self.func_cols
2652    }
2653
2654    #[track_caller]
2655    fn subsume_col(&self) -> usize {
2656        assert!(self.subsume);
2657        self.func_cols + 1
2658    }
2659}
