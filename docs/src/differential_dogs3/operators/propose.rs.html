6/// Proposes extensions to a prefix stream.
7///
8/// This method takes a collection `prefixes` and an arrangement `arrangement` and for each
9/// update in the collection joins with the accumulated arranged records at a time less or
10/// equal to that of the update. Note that this is not a join by itself, but can be used to
11/// create a join if the `prefixes` collection is also arranged and responds to changes that
12/// `arrangement` undergoes. More complicated patterns are also appropriate, as in the case
13/// of delta queries.
14pub fn propose<'scope, Tr, K, F, P, V>(
15    prefixes: VecCollection<'scope, Tr::Time, P, Tr::Diff>,
16    arrangement: Arranged<'scope, Tr>,
17    key_selector: F,
18) -> VecCollection<'scope, Tr::Time, (P, V), Tr::Diff>
19where
20    Tr: for<'a> TraceReader<
21        ValOwn = V,
22        Time: std::hash::Hash,
23        Diff: Monoid+Multiply<Output = Tr::Diff>+ExchangeData+Semigroup<Tr::DiffGat<'a>>,
24    >+Clone+'static,
25    Tr::KeyContainer: differential_dataflow::trace::implementations::BatchContainer<Owned=K>,
26    K: Hashable + Default + Ord + 'static,
27    F: Fn(&P)->K+Clone+'static,
28    P: ExchangeData,
29    V: Clone + 'static,
30{
31    crate::operators::lookup_map(
32        prefixes,
33        arrangement,
34        move |p: &P, k: &mut K | { *k = key_selector(p); },
35        move |prefix, diff, value, sum| ((prefix.clone(), Tr::owned_val(value)), diff.clone().multiply(sum)),
36        Default::default(),
37        Default::default(),
38        Default::default(),
39    )
40}

42/// Proposes distinct extensions to a prefix stream.
43///
44/// Unlike `propose`, this method does not scale the multiplicity of matched
45/// prefixes by the number of matches in `arrangement`. This can be useful to
46/// avoid the need to prepare an arrangement of distinct extensions.
47pub fn propose_distinct<'scope, Tr, K, F, P, V>(
48    prefixes: VecCollection<'scope, Tr::Time, P, Tr::Diff>,
49    arrangement: Arranged<'scope, Tr>,
50    key_selector: F,
51) -> VecCollection<'scope, Tr::Time, (P, V), Tr::Diff>
52where
53    Tr: for<'a> TraceReader<
54        ValOwn = V,
55        Time: std::hash::Hash,
56        Diff : Semigroup<Tr::DiffGat<'a>>+Monoid+Multiply<Output = Tr::Diff>+ExchangeData,
57    >+Clone+'static,
58    Tr::KeyContainer: differential_dataflow::trace::implementations::BatchContainer<Owned=K>,
59    K: Hashable + Default + Ord + 'static,
60    F: Fn(&P)->K+Clone+'static,
61    P: ExchangeData,
62    V: Clone + 'static,
63{
64    crate::operators::lookup_map(
65        prefixes,
66        arrangement,
67        move |p: &P, k: &mut K| { *k = key_selector(p); },
68        move |prefix, diff, value, _sum| ((prefix.clone(), Tr::owned_val(value)), diff.clone()),
69        Default::default(),
70        Default::default(),
71        Default::default(),
72    )
