6/// Reports a number of extensions to a stream of prefixes.
7///
8/// This method takes as input a stream of `(prefix, count, index)` triples.
9/// For each triple, it extracts a key using `key_selector`, and finds the
10/// associated count in `arrangement`. If the found count is less than `count`,
11/// the `count` and `index` fields are overwritten with their new values.
12pub fn count<'scope, Tr, K, R, F, P>(
13    prefixes: VecCollection<'scope, Tr::Time, (P, usize, usize), R>,
14    arrangement: Arranged<'scope, Tr>,
15    key_selector: F,
16    index: usize,
17) -> VecCollection<'scope, Tr::Time, (P, usize, usize), R>
18where
19    Tr: TraceReader<Time: std::hash::Hash, Diff=isize>+Clone+'static,
20    Tr::KeyContainer: differential_dataflow::trace::implementations::BatchContainer<Owned=K>,
21    for<'a> Tr::Diff : Semigroup<Tr::DiffGat<'a>>,
22    K: Hashable + Ord + Default + 'static,
23    R: Monoid+Multiply<Output = R>+ExchangeData,
24    F: Fn(&P)->K+Clone+'static,
25    P: ExchangeData,
26{
27    crate::operators::lookup_map(
28        prefixes,
29        arrangement,
30        move |p: &(P,usize,usize), k: &mut K| { *k = key_selector(&p.0); },
31        move |(p,c,i), r, _, s| {
32            let s = *s as usize;
33            if *c < s { ((p.clone(), *c, *i), r.clone()) }
34            else      { ((p.clone(), s, index), r.clone()) }
35        },
36        Default::default(),
37        Default::default(),
38        Default::default(),
39    )
40}
