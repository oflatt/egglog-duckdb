1//! Dataflow operator for delta joins over partially ordered timestamps.
2//!
3//! Given multiple streams of updates `(data, time, diff)` that are each
4//! defined over the same partially ordered `time`, we want to form the
5//! full cross-join of all relations (we will *later* apply some filters
6//! and instead equijoin on keys).
7//!
8//! The "correct" output is the outer join of these triples, where
9//!   1. The `data` entries are just tuple'd up together,
10//!   2. The `time` entries are subjected to the lattice `join` operator,
11//!   3. The `diff` entries are multiplied.
12//!
13//! One way to produce the correct output is to form independent dataflow
14//! fragments for each input stream, such that each intended output is then
15//! produced by exactly one of these input streams.
16//!
17//! There are several incorrect ways one might do this, but here is one way
18//! that I hope is not incorrect:
19//!
20//! Each input stream of updates is joined with each other input collection,
21//! where each input update is matched against each other input update that
22//! has a `time` that is less-than the input update's `time`, *UNDER A TOTAL
23//! ORDER ON `time`*. The output are the `(data, time, diff)` entries that
24//! follow the rules above, except that we additionally preserve the input's
25//! initial `time` as well, for use in subsequent joins with the other input
26//! collections.
27//!
28//! There are some caveats about ties, and we should treat each `time` for
29//! each input as occurring at distinct times, one after the other (so that
30//! ties are resolved by the index of the input). There is also the matter
31//! of logical compaction, which should not be done in a way that prevents
32//! the correct determination of the total order comparison.
33

55/// A binary equijoin that responds to updates on only its first input.
56///
57/// This operator responds to inputs of the form
58///
59/// ```ignore
60/// ((key, val1, time1), initial_time, diff1)
61/// ```
62///
63/// where `initial_time` is less or equal to `time1`, and produces as output
64///
65/// ```ignore
66/// ((output_func(key, val1, val2), lub(time1, time2)), initial_time, diff1 * diff2)
67/// ```
68///
69/// for each `((key, val2), time2, diff2)` present in `arrangement`, where
70/// `time2` is less than `initial_time` *UNDER THE TOTAL ORDER ON TIMES*.
71/// This last constraint is important to ensure that we correctly produce
72/// all pairs of output updates across multiple `half_join` operators.
73///
74/// Notice that the time is hoisted up into data. The expectation is that
75/// once out of the "delta flow region", the updates will be `delay`d to the
76/// times specified in the payloads.
77pub fn half_join<'scope, K, V, R, Tr, FF, CF, DOut, S>(
78    stream: VecCollection<'scope, Tr::Time, (K, V, Tr::Time), R>,
79    arrangement: Arranged<'scope, Tr>,
80    frontier_func: FF,
81    comparison: CF,
82    mut output_func: S,
83) -> VecCollection<'scope, Tr::Time, (DOut, Tr::Time), <R as Mul<Tr::Diff>>::Output>
84where
85    K: Hashable + ExchangeData,
86    V: ExchangeData,
87    R: ExchangeData + Monoid,
88    Tr: TraceReader<Time: std::hash::Hash>+Clone+'static,
89    Tr::KeyContainer: BatchContainer<Owned=K>,
90    R: Mul<Tr::Diff, Output: Semigroup>,
91    FF: Fn(&Tr::Time, &mut Antichain<Tr::Time>) + 'static,
92    CF: Fn(Tr::TimeGat<'_>, &Tr::Time) -> bool + 'static,
93    DOut: Clone+'static,
94    S: FnMut(&K, &V, Tr::Val<'_>)->DOut+'static,
95{
96    let output_func = move |session: &mut SessionFor<Tr::Time, _>, k: &K, v1: &V, v2: Tr::Val<'_>, initial: &Tr::Time, diff1: &R, output: &mut Vec<(Tr::Time, Tr::Diff)>| {
97        for (time, diff2) in output.drain(..) {
98            let diff = diff1.clone() * diff2.clone();
99            let dout = (output_func(k, v1, v2), time.clone());
100            session.give((dout, initial.clone(), diff));
101        }
102    };
103    half_join_internal_unsafe::<_, _, _, _, _, _,_,_, CapacityContainerBuilder<Vec<_>>>(stream, arrangement, frontier_func, comparison, |_timer, _count| false, output_func)
104        .as_collection()
105}

117/// An unsafe variant of `half_join` where the `output_func` closure takes
118/// additional arguments a vector of `time` and `diff` tuples as input and
119/// writes its outputs at a container builder. The container builder
120/// can, but isn't required to, accept `(data, time, diff)` triplets.
121/// This allows for more flexibility, but is more error-prone.
122///
123/// This operator responds to inputs of the form
124///
125/// ```ignore
126/// ((key, val1, time1), initial_time, diff1)
127/// ```
128///
129/// where `initial_time` is less or equal to `time1`, and produces as output
130///
131/// ```ignore
132/// output_func(session, key, val1, val2, initial_time, diff1, &[lub(time1, time2), diff2])
133/// ```
134///
135/// for each `((key, val2), time2, diff2)` present in `arrangement`, where
136/// `time2` is less than `initial_time` *UNDER THE TOTAL ORDER ON TIMES*.
137///
138/// The `yield_function` allows the caller to indicate when the operator should
139/// yield control, as a function of the elapsed time and the number of matched
140/// records. Note this is not the number of *output* records, owing mainly to
141/// the number of matched records being easiest to record with low overhead.
142pub fn half_join_internal_unsafe<'scope, K, V, R, Tr, FF, CF, Y, S, CB>(
143    stream: VecCollection<'scope, Tr::Time, (K, V, Tr::Time), R>,
144    mut arrangement: Arranged<'scope, Tr>,
145    frontier_func: FF,
146    comparison: CF,
147    yield_function: Y,
148    mut output_func: S,
149) -> Stream<'scope, Tr::Time, CB::Container>
150where
151    K: Hashable + ExchangeData,
152    V: ExchangeData,
153    R: ExchangeData + Monoid,
154    Tr: TraceReader<Time: std::hash::Hash>+Clone+'static,
155    Tr::KeyContainer: BatchContainer<Owned=K>,
156    FF: Fn(&Tr::Time, &mut Antichain<Tr::Time>) + 'static,
157    CF: Fn(Tr::TimeGat<'_>, &Tr::Time) -> bool + 'static,
158    Y: Fn(std::time::Instant, usize) -> bool + 'static,
159    S: FnMut(&mut SessionFor<Tr::Time, CB>, &K, &V, Tr::Val<'_>, &Tr::Time, &R, &mut Vec<(Tr::Time, Tr::Diff)>) + 'static,
160    CB: ContainerBuilder,
161{
162    // No need to block physical merging for this operator.
163    arrangement.trace.set_physical_compaction(Antichain::new().borrow());
164    let mut arrangement_trace = Some(arrangement.trace);
165    let arrangement_stream = arrangement.stream;
166
167    let mut stash = HashMap::new();
168
169    let exchange = Exchange::new(move |update: &((K, V, Tr::Time),Tr::Time,R)| (update.0).0.hashed().into());
170
171    // Stash for (time, diff) accumulation.
172    let mut output_buffer = Vec::new();
173
174    let scope = stream.scope();
175    stream.inner.binary_frontier(arrangement_stream, exchange, Pipeline, "HalfJoin", move |_,info| {
176
177        // Acquire an activator to reschedule the operator when it has unfinished work.
178        let activator = scope.activator_for(info.address);
179
180        move |(input1, frontier1), (input2, frontier2), output| {
181
182            // drain the first input, stashing requests.
183            input1.for_each(|capability, data| {
184                stash.entry(capability.retain(0))
185                    .or_insert(Vec::new())
186                    .append(data)
187            });
188
189            // Drain input batches; although we do not observe them, we want access to the input
190            // to observe the frontier and to drive scheduling.
191            input2.for_each(|_, _| { });
192
193            // Local variables to track if and when we should exit early.
194            // The rough logic is that we fully process inputs and set their differences to zero,
195            // stopping at any point. We clean up all of the zeros in buffers that did any work,
196            // and reactivate at the end if the yield function still says so.
197            let mut yielded = false;
198            let timer = std::time::Instant::now();
199            let mut work = 0;
200
201            // New entries to introduce to the stash after processing.
202            let mut stash_additions = HashMap::new();
203
204            if let Some(ref mut trace) = arrangement_trace {
205
206                for (capability, proposals) in stash.iter_mut() {
207
208                    // Avoid computation if we should already yield.
209                    // TODO: Verify this is correct for TOTAL ORDER.
210                    yielded = yielded || yield_function(timer, work);
211                    if !yielded && !frontier2.less_equal(capability.time()) {
212
213                        let frontier = frontier2.frontier();
214
215                        // Update yielded: We can only go from false to {false, true} as
216                        // we're checking that `!yielded` holds before entering this block.
217                        yielded = process_proposals::<_, _, _, _, _, _, _, _>(
218                            &comparison,
219                            &yield_function,
220                            &mut output_func,
221                            &mut output_buffer,
222                            timer,
223                            &mut work,
224                            trace,
225                            proposals,
226                            output.session_with_builder(capability),
227                            frontier
228                        );
229
230                        proposals.retain(|ptd| !ptd.2.is_zero());
231
232                        // Determine the lower bound of remaining update times.
233                        let mut antichain = Antichain::new();
234                        for (_, initial, _) in proposals.iter() {
235                            antichain.insert(initial.clone());
236                        }
237                        // Fast path: there is only one element in the antichain.
238                        // All times in `proposals` must be greater or equal to it.
239                        if antichain.len() == 1 && !antichain.less_equal(capability.time()) {
240                            stash_additions
241                                .entry(capability.delayed(&antichain[0]))
242                                .or_insert(Vec::new())
243                                .append(proposals);
244                        }
245                        else if antichain.len() > 1 {
246                            // Any remaining times should peel off elements from `proposals`.
247                            let mut additions = vec![Vec::new(); antichain.len()];
248                            for (data, initial, diff) in proposals.drain(..) {
249                                let position = antichain.iter().position(|t| t.less_equal(&initial)).unwrap();
250                                additions[position].push((data, initial, diff));
251                            }
252                            for (time, addition) in antichain.into_iter().zip(additions) {
253                                stash_additions
254                                    .entry(capability.delayed(&time))
255                                    .or_insert(Vec::new())
256                                    .extend(addition);
257                            }
258                        }
259                    }
260                }
261            }
262
263            // If we yielded, re-activate the operator.
264            if yielded {
265                activator.activate();
266            }
267
268            // drop fully processed capabilities.
269            stash.retain(|_,proposals| !proposals.is_empty());
270
271            for (capability, proposals) in stash_additions.into_iter() {
272                stash.entry(capability).or_insert(Vec::new()).extend(proposals);
273            }
274
275            // The logical merging frontier depends on both input1 and stash.
276            let mut frontier = timely::progress::frontier::Antichain::new();
277            for time in frontier1.frontier().iter() {
278                frontier_func(time, &mut frontier);
279            }
280            for time in stash.keys() {
281                frontier_func(time, &mut frontier);
282            }
283            arrangement_trace.as_mut().map(|trace| trace.set_logical_compaction(frontier.borrow()));
284
285            if frontier1.is_empty() && stash.is_empty() {
286                arrangement_trace = None;
287            }
288        }
289    })
290}

292/// Outlined inner loop for `half_join_internal_unsafe` for reasons of performance.
293///
294/// Gives Rust/LLVM the opportunity to inline the loop body instead of inlining the loop and
295/// leaving all calls in the loop body outlined.
296///
297/// Consumes proposals until the yield function returns `true` or all proposals are processed.
298/// Leaves a zero diff in place for all proposals that were processed.
299///
300/// Returns `true` if the operator should yield.
301fn process_proposals<Tr, CF, Y, S, CB, K, V, R>(
302    comparison: &CF,
303    yield_function: &Y,
304    output_func: &mut S,
305    mut output_buffer: &mut Vec<(Tr::Time, Tr::Diff)>,
306    timer: Instant,
307    work: &mut usize,
308    trace: &mut Tr,
309    proposals: &mut Vec<((K, V, Tr::Time), Tr::Time, R)>,
310    mut session: SessionFor<Tr::Time, CB>,
311    frontier: AntichainRef<Tr::Time>
312) -> bool
313where
314    Tr: TraceReader,
315    Tr::KeyContainer: BatchContainer<Owned=K>,
316    CF: Fn(Tr::TimeGat<'_>, &Tr::Time) -> bool + 'static,
317    Y: Fn(Instant, usize) -> bool + 'static,
318    S: FnMut(&mut SessionFor<Tr::Time, CB>, &K, &V, Tr::Val<'_>, &Tr::Time, &R, &mut Vec<(Tr::Time, Tr::Diff)>) + 'static,
319    CB: ContainerBuilder,
320    K: Ord,
321    V: Ord,
322    R: Monoid,
323{
324    // Sort requests by key for in-order cursor traversal.
325    consolidate_updates(proposals);
326
327    let (mut cursor, storage) = trace.cursor();
328    let mut yielded = false;
329
330    let mut key_con = Tr::KeyContainer::with_capacity(1);
331    let mut time_con = Tr::TimeContainer::with_capacity(1);
332    for time in frontier.iter() {
333        time_con.push_own(time);
334    }
335
336    // Process proposals one at a time, stopping if we should yield.
337    for ((ref key, ref val1, ref time), ref initial, ref mut diff1) in proposals.iter_mut() {
338        // Use TOTAL ORDER to allow the release of `time`.
339        yielded = yielded || yield_function(timer, *work);
340
341        if !yielded && !(0 .. time_con.len()).any(|i| comparison(time_con.index(i), initial)) {
342            key_con.clear(); key_con.push_own(&key);
343            cursor.seek_key(&storage, key_con.index(0));
344            if cursor.get_key(&storage) == key_con.get(0) {
345                while let Some(val2) = cursor.get_val(&storage) {
346                    cursor.map_times(&storage, |t, d| {
347                        if comparison(t, initial) {
348                            let mut t = Tr::owned_time(t);
349                            t.join_assign(time);
350                            output_buffer.push((t, Tr::owned_diff(d)))
351                        }
352                    });
353                    consolidate(&mut output_buffer);
354                    *work += output_buffer.len();
355                    output_func(&mut session, key, val1, val2, initial, diff1, &mut output_buffer);
356                    // Defensive clear; we'd expect `output_func` to clear the buffer.
357                    // TODO: Should we assert it is empty?
358                    output_buffer.clear();
359                    cursor.step_val(&storage);
360                }
361                cursor.rewind_vals(&storage);
362            }
363            *diff1 = R::zero();
364        }
365    }
366
367    yielded
