1//! Phase 0 spike: transitive closure on (edge, path) using DuckDB.
2//!
3//! Goal: measure per-iteration overhead of issuing seminaive SQL
4//! against an in-process DuckDB connection, so we can compare against
5//! egglog's existing backend on the equivalent program.
6//!
7//! What this is: a hand-translation of the term-encoding-free user
8//! program below into DuckDB SQL + a Rust driver loop that mimics
9//! the schedule/seminaive bookkeeping.
10//!
11//! ```text
12//! (relation edge (i64 i64))
13//! (relation path (i64 i64))
14//! (rule ((edge a b))            ((path a b))         :name "base")
15//! (rule ((path a b) (edge b c)) ((path a c))         :name "step")
16//! ; ... seed edges ...
17//! (run-schedule (saturate (run)))
18//! (check (path 1 N))
19//! ```
20//!
21//! What this is NOT: the term-encoded version. We're measuring the
22//! base "DuckDB can run Datalog seminaive at all" cost. Term encoding
23//! adds UF tables, view tables, deletion deferral — orthogonal cost
24//! we'll layer in Phase 1.
25
26use duckdb::{Connection, Result, params};
27use std::time::Instant;
28
29fn main() -> Result<()> {
30    // Configurable problem size: chain of N edges 0→1→2→…→N.
31    // Saturated path table has N*(N+1)/2 rows.
32    let chain_len: i64 = std::env::args()
33        .nth(1)
34        .and_then(|s| s.parse().ok())
35        .unwrap_or(100);
36
37    let conn = Connection::open_in_memory()?;
38    setup_schema(&conn)?;
39
40    let setup_start = Instant::now();
41    seed_edges(&conn, chain_len, 1)?;
42    let setup_ms = setup_start.elapsed().as_secs_f64() * 1000.0;
43
44    let run_start = Instant::now();
45    let (iterations, ts_final) = run_to_saturation(&conn)?;
46    let run_ms = run_start.elapsed().as_secs_f64() * 1000.0;
47
48    // Verify: there should be a path from 0 to chain_len.
49    let final_path: i64 = conn.query_row(
50        "SELECT COUNT(*) FROM path WHERE a = 0 AND b = ?",
51        params![chain_len],
52        |row| row.get(0),
53    )?;
54    let total_paths: i64 = conn.query_row("SELECT COUNT(*) FROM path", [], |row| row.get(0))?;
55    let expected_total = chain_len * (chain_len + 1) / 2;
56
57    println!("chain_len: {chain_len}");
58    println!("setup (CREATE + seed): {setup_ms:.3} ms");
59    println!("saturate:              {run_ms:.3} ms ({iterations} iterations, ts={ts_final})");
60    println!("paths from 0 to {chain_len}: {final_path} (want 1)");
61    println!("total path rows:       {total_paths} (want {expected_total})");
62
63    assert_eq!(final_path, 1, "expected path 0→{chain_len}");
64    assert_eq!(total_paths, expected_total, "wrong path count");
65
66    Ok(())
67}
68
69fn setup_schema(conn: &Connection) -> Result<()> {
70    // Each table carries a `ts BIGINT` column for seminaive
71    // bookkeeping. PRIMARY KEY enforces functional-dependency
72    // semantics (egglog's "no duplicate row for same key") and is
73    // what we ON CONFLICT DO NOTHING against.
74    conn.execute_batch(
75        "CREATE TABLE edge (
76            a  BIGINT NOT NULL,
77            b  BIGINT NOT NULL,
78            ts BIGINT NOT NULL,
79            PRIMARY KEY (a, b)
80         );
81         CREATE TABLE path (
82            a  BIGINT NOT NULL,
83            b  BIGINT NOT NULL,
84            ts BIGINT NOT NULL,
85            PRIMARY KEY (a, b)
86         );",
87    )?;
88    Ok(())
89}
90
91fn seed_edges(conn: &Connection, chain_len: i64, ts: i64) -> Result<()> {
92    let mut app = conn.appender("edge")?;
93    for i in 0..chain_len {
94        app.append_row(params![i, i + 1, ts])?;
95    }
96    app.flush()?;
97    Ok(())
98}

100/// Run the schedule to saturation. Returns (iterations, final ts).
101fn run_to_saturation(conn: &Connection) -> Result<(usize, i64)> {
102    // Per-rule "last run at" — the ts at which each rule last ran.
103    // For seminaive: variant i of rule R requires the focused atom's
104    // ts >= last_run_at_R. After running R in iteration t, set
105    // last_run_at_R := t.
106    let mut last_run_at_base: i64 = 0;
107    let mut last_run_at_step: i64 = 0;
108    let mut next_ts: i64 = 1;
109    let mut iterations = 0;
110
111    loop {
112        iterations += 1;
113        next_ts += 1;
114        let cur = next_ts;
115
116        // Rule "base": (rule ((edge a b)) ((path a b)))
117        // 1 body atom → 1 seminaive variant focused on `edge`.
118        let base_inserted: usize = conn
119            .prepare_cached(
120                "INSERT INTO path (a, b, ts)
121                 SELECT a, b, ?2 FROM edge
122                 WHERE ts >= ?1
123                 ON CONFLICT DO NOTHING",
124            )?
125            .execute(params![last_run_at_base, cur])?;
126        last_run_at_base = cur;
127
128        // Rule "step": (rule ((path a b) (edge b c)) ((path a c)))
129        // 2 body atoms → 2 seminaive variants.
130        // Variant 1: focus on `path`.
131        let step1: usize = conn
132            .prepare_cached(
133                "INSERT INTO path (a, b, ts)
134                 SELECT p.a, e.b, ?2
135                 FROM path p JOIN edge e ON p.b = e.a
136                 WHERE p.ts >= ?1
137                 ON CONFLICT DO NOTHING",
138            )?
139            .execute(params![last_run_at_step, cur])?;
140        // Variant 2: focus on `edge`.
141        let step2: usize = conn
142            .prepare_cached(
143                "INSERT INTO path (a, b, ts)
144                 SELECT p.a, e.b, ?2
145                 FROM path p JOIN edge e ON p.b = e.a
146                 WHERE e.ts >= ?1
147                 ON CONFLICT DO NOTHING",
148            )?
149            .execute(params![last_run_at_step, cur])?;
150        last_run_at_step = cur;
151
152        let total = base_inserted + step1 + step2;
153        if total == 0 {
154            break;
155        }
156    }
157
158    Ok((iterations, next_ts))
159}
